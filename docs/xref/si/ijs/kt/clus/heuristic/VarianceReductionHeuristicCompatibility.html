73      @Override
74      public double calcHeuristic(ClusStatistic tstat, ClusStatistic pstat, ClusStatistic missing) {
75          // Acceptable?
76          if (stopCriterion(tstat, pstat, missing)) { return Double.NEGATIVE_INFINITY; }
77          // Compute |S|Var[S]
78          double ss_tot = tstat.getSVarS(m_ClusteringWeights);
79          double ss_pos = pstat.getSVarS(m_ClusteringWeights);
80          double ss_neg = tstat.getSVarSDiff(m_ClusteringWeights, pstat);
81          double value = FTest.calcVarianceReductionHeuristic(tstat.getTotalWeight(), ss_tot, ss_pos + ss_neg);
82          if (getSettings().getGeneral().getVerbose() >= 10) {
83              ClusLogger.info("TOT: " + tstat.getDebugString());
84              ClusLogger.info("POS: " + pstat.getDebugString());
85              ClusLogger.info("-> (" + ss_tot + ", " + ss_pos + ", " + ss_neg + ") " + value);
86          }
87          // NOTE: This is here for compatibility reasons only
88          if (value < 1e-6)
89              return Double.NEGATIVE_INFINITY;
90          return value;
91      }

95      public double LanLongdistance(double lat1, double lon1, double lat2, double lon2) {
96          double dist;
97          if (lat1 == lat2 && lon1 == lon2) {
98              dist = 0;
99          }
100         else {
101             double theta = lon1 - lon2;
102             dist = Math.sin(deg2rad(lat1)) * Math.sin(deg2rad(lat2)) + Math.cos(deg2rad(lat1)) * Math.cos(deg2rad(lat2)) * Math.cos(deg2rad(theta));
103             dist = rad2deg(Math.acos(dist));
104             dist = dist * 60 * 1.1515 * 1.609344;
105         }
106         return (dist);
107     }

129     public void generateMatrix(RowData data) throws IOException {
130         //PrintWriter pw = new PrintWriter(new FileWriter("distances.txt"));            
131         m_distances.clear();
132         ClusSchema schema = data.getSchema();
133         int N = data.getNbRows();
134         double maxdist = 0;
135         double d = 0;
136         //double W = 0;
137         double minDistLine = Double.POSITIVE_INFINITY;
138         double maxOnMinDistLine = 0;
139         double[][] w = new double[N][N];
140         //max distance in the file
141         for (int i = 0; i < N; i++) {
142             minDistLine = Double.POSITIVE_INFINITY;
143             for (int j = 0; j < N; j++) {
144                 DataTuple exi = data.getTuple(i);
145                 DataTuple exj = data.getTuple(j);
146                 ClusAttrType xt = schema.getNumericAttrUse(AttributeUseType.GIS)[0];
147                 ClusAttrType yt = schema.getNumericAttrUse(AttributeUseType.GIS)[1];
148                 double xi = xt.getNumeric(exi);
149                 double yi = yt.getNumeric(exi);
150                 double xj = xt.getNumeric(exj);
151                 double yj = yt.getNumeric(exj);
152                 Boolean longlat = schema.getSettings().getTree().isLonglat();
153                 if (longlat == true)
154                     d = LanLongdistance(xi, yi, xj, yj);
155                 else
156                     d = Math.sqrt((xi - xj) * (xi - xj) + (yi - yj) * (yi - yj));
157                 if (d > maxdist)
158                     maxdist = d;
159                 if (d != 0 && d < minDistLine)
160                     minDistLine = d;
161             }
162             if (minDistLine > maxOnMinDistLine)
163                 maxOnMinDistLine = minDistLine;
164         }
165         double bandwidth = schema.getSettings().getTree().getBandwidth();
166         double b = bandwidth * maxdist;
167         //ClusLogger.info(maxdist);
168         if (maxOnMinDistLine != Double.POSITIVE_INFINITY && b < maxOnMinDistLine)
169             b = maxOnMinDistLine;
170         //calculate the weights 
171         for (int i = 0; i < N; i++) {
172             for (int j = 0; j < N; j++) {
173                 if (i > j) {
174                     w[i][j] = w[j][i];
175                    // W += w[i][j];
176                     continue;
177                 }
178 
179                 DataTuple exi = data.getTuple(i); //example i
180                 DataTuple exj = data.getTuple(j); //example j               
181                 ClusAttrType xti = schema.getNumericAttrUse(AttributeUseType.GIS)[0];
182                 ClusAttrType yti = schema.getNumericAttrUse(AttributeUseType.GIS)[1];
183                 double xii = xti.getNumeric(exi);
184                 double yii = yti.getNumeric(exi);
185                 double xji = xti.getNumeric(exj);
186                 double yji = yti.getNumeric(exj);
187                 Boolean longlat = schema.getSettings().getTree().isLonglat();
188                 if (longlat == true)
189                     d = LanLongdistance(xii, yii, xji, yji);
190                 else
191                     d = Math.sqrt((xii - xji) * (xii - xji) + (yii - yji) * (yii - yji));
192                 /*
193                  * distance over descriptive and all atributes
194                  * NumericAttrType[] arr = schema.getNumericAttrUse(AttributeUseType.Descriptive);
195                  * int countA=arr.length; double dd=0;double d=0;
196                  * double xi[] = new double[countA];double xj[] = new double[countA];
197                  * for (int ii = 0; ii < countA; ii++) {
198                  * ClusAttrType xt = schema.getNumericAttrUse(AttributeUseType.Descriptive)[ii];
199                  * xi[ii] = xt.getNumeric(exi); xj[ii] = xt.getNumeric(exj);
200                  * dd += (xi[ii]-xj[ii])*(xi[ii]-xj[ii]); //euc over all atrb}
201                  */
202                 SpatialMatrixType spatialMatrix = schema.getSettings().getTree().getSpatialMatrix();
203                 if (d >= b)
204                     w[i][j] = 0;
205                 else {
206                     if (d == 0)
207                         w[i][j] = 1;
208                     else {
209                         switch (spatialMatrix) {
210                             case Binary:
211                                 w[i][j] = 1; //0;   break;  //binary 
212                             case Euclidean:
213                                 w[i][j] = 1 - d / b;
214                                 break; //euclidian
215                             case Modified:
216                                 w[i][j] = (1 - (d * d) / (b * b)) * (1 - (d * d) / (b * b));
217                                 break; //modified
218                             case Gaussian:
219                                 w[i][j] = Math.exp(-(d * d) / (b * b));
220                                 break; //gausian
221                             default:
222                                 w[i][j] = 0;
223                                 break;
224                         }
225                         //W += w[i][j]; //ClusLogger.info(i+"\t"+j+"\t"+exi+"\t"+exj+"\t"+(long)(i*N+j)+"\t"+w[i][j]);
226                     }
227                     m_distances.put((long) (i * N + j), w[i][j]); //write to hasp map only, not in file
228                     //pw.println(i+"#"+j+" "+d);
229                     //pw.println(i+"\t"+W);     //write to file
230                 }
231             }
232         }
233         //ClusLogger.info("spatialMatrix end: "+ System.currentTimeMillis());
234         //pw.close();
235     }

238     public void readMatrixFromFile(RowData data) throws IOException {
239         //PrintWriter pw = new PrintWriter(new FileWriter("distancesDistance.txt"));
240         m_distancesS.clear();
241         m_distancesN.clear();
242         ClusSchema schema = data.getSchema();
243         //int N = data.getNbRows();
244         double maxdist = 0;
245         double minDistLine = Double.POSITIVE_INFINITY;
246         double maxOnMinDistLine = 0;
247         double iEqual = Double.POSITIVE_INFINITY;
248         String filename = "distances.csv";
249         BufferedReader br = new BufferedReader(new FileReader(filename));
250         double bandwidth = schema.getSettings().getTree().getBandwidth();
251         double w;
252         if (bandwidth != 100.0) {
253             while (br.ready()) {
254                 String s = br.readLine();
255                 StringTokenizer st = new StringTokenizer(s, ",");
256                 Long ii = Long.parseLong(st.nextToken());
257                 Long jj = Long.parseLong(st.nextToken());
258                 double d = Double.parseDouble(st.nextToken());
259                 if (iEqual != ii) {
260                     m_distancesN.put((ii + "#" + ii), 0.0);
261                     iEqual = ii; // pw.println(ii+" "+ii+" "+0.0);
262                 }
263                 if (ii > jj) {
264                     m_distancesN.put((jj + "#" + ii), d); //pw.println(jj+" "+ii+" "+d);
265                 }
266                 else {
267                     m_distancesN.put((ii + "#" + jj), d); //pw.println(ii+" "+jj+" "+d);
268                 }
269             }
270 
271             for (double d : m_distancesN.values()) {
272                 if (d > maxdist)
273                     maxdist = d;
274                 if (d != 0 && d < minDistLine)
275                     minDistLine = d;
276             }
277             if (minDistLine > maxOnMinDistLine)
278                 maxOnMinDistLine = minDistLine;
279             double b = bandwidth * maxdist;
280             if (maxOnMinDistLine != Double.POSITIVE_INFINITY && b < maxOnMinDistLine)
281                 b = maxOnMinDistLine;
282             SpatialMatrixType spatialMatrix = schema.getSettings().getTree().getSpatialMatrix();
283             for (Map.Entry<String, Double> entry : m_distancesN.entrySet()) {
284                 String i = entry.getKey();
285                 double d = entry.getValue();
286                 if (d >= b)
287                     w = 0;
288                 else {
289                     if (d == 0)
290                         w = 1;
291                     else {
292                         switch (spatialMatrix) {
293                             case Binary:
294                                 w = 0;
295                                 break; //binary 
296                             case Euclidean:
297                                 w = 1 - d / b;
298                                 break; //euclidian
299                             case Modified:
300                                 w = (1 - (d * d) / (b * b)) * (1 - (d * d) / (b * b));
301                                 break; //modified
302                             case Gaussian:
303                                 w = Math.exp(-(d * d) / (b * b));
304                                 break; //gausian
305                             default:
306                                 w = 0;
307                                 break;
308                         }
309                     }
310                     m_distancesS.put((i), w); //write to hasp map non-zero only, not in file
311                     //ClusLogger.info((i)+" "+w);
312                 }
313             }
314         }
315         else {
316             while (br.ready()) {
317                 String s = br.readLine();
318                 StringTokenizer st = new StringTokenizer(s, ",");
319                 //Long ii=Long.parseLong(st.nextToken());
320                 //Long jj=Long.parseLong(st.nextToken());
321                 String ii = st.nextToken();
322                 String jj = st.nextToken();
323                 double d = Double.parseDouble(st.nextToken());
324                 SpatialMatrixType spatialMatrix = schema.getSettings().getTree().getSpatialMatrix();
325                 switch (spatialMatrix) {
326                     case Binary:
327                         if (d == 1)
328                             w = 1;
329                         else
330                             w = 0;
331                         break; //binary 
332                     case Euclidean:
333                         if (d == 1)
334                             w = 0.01;
335                         else
336                             w = 1 - 1 / d;
337                         break; //euclidian
338                     default:
339                         w = 0;
340                         break;
341                 }
342                 //if (ii>jj) m_distancesS.put((jj+"#"+ii), w); else m_distancesS.put((ii+"#"+jj), w); 
343                 m_distancesS.put((ii + "#" + jj), w);
344                 m_distancesS.put((jj + "#" + ii), w);
345             }
346         }
347         br.close();
348     }

351     public double calcI(ClusStatistic t_stat, ClusStatistic p_stat, ClusStatistic missing, Integer[] permutation) throws Exception {
352         //try{
353         //ClusStatistic tstat = (ClusStatistic)t_stat;
354         ClusStatistic pstat = p_stat; // Acceptable?
355         if (stopCriterion(t_stat, p_stat, missing)) { return Double.NEGATIVE_INFINITY; }
356 
357         double ss_pos = 0; //pstat.calcItotal();             
358         ClusSchema schema = m_Data.getSchema();
359         SpatialMeasure sm = schema.getSettings().getTree().getSpatialMeasure();
360         if (!m_WarningGiven && !sm.equals(SpatialMeasure.GlobalMoran)) {
361             m_WarningGiven = true;
362             System.err.println("Warning: your spatial measure was not tested. Be careful.");
363         }
364         switch (sm) {
365             case GlobalMoran:
366                 ss_pos = pstat.calcItotal(permutation);
367                 break; //"Global Moran"
368             case GlobalGeary:
369                 ss_pos = pstat.calcGtotal(permutation);
370                 break; //"Global Geary"
371             case GlobalGetis:
372                 ss_pos = pstat.calcGetisTotal(permutation);
373                 break;//"Global Getis"
374             case LocalMoran:
375                 ss_pos = pstat.calcLISAtotal(permutation);
376                 break; //"Local Moran"
377             case LocalGeary:
378                 ss_pos = pstat.calcGLocalTotal(permutation);
379                 break; //"Local Geary"
380             case LocalGetis:
381                 ss_pos = pstat.calcLocalGetisTotal(permutation);
382                 break;//"Local Getis"
383             case StandardizedGetis:
384                 ss_pos = pstat.calcGETIStotal(permutation);
385                 break;//Local Standardized Getis
386             case EquvalentI:
387                 ss_pos = pstat.calcEquvalentItotal(permutation);
388                 break;//EquvalentI 
389             case IwithNeighbours:
390                 ss_pos = pstat.calcIwithNeighbourstotal(permutation);
391                 break; //Global Moran with neigh
392             case EquvalentIwithNeighbours:
393                 ss_pos = pstat.calcEquvalentIwithNeighbourstotal(permutation);
394                 break; // EquvalentI Global Moran with neigh
395             case GlobalMoranDistance:
396                 ss_pos = pstat.calcItotalD(permutation);
397                 break; //"Global Moran with a separate distance file"
398             case GlobalGearyDistance:
399                 ss_pos = pstat.calcGtotalD(permutation);
400                 break; //"Global Geary with a separate distance file"
401             case CI:
402                 ss_pos = pstat.calcCItotal(permutation);
403                 break; // Connectivity Index (CI) for Graph Data
404             case MultiVariateMoranI:
405                 ss_pos = pstat.calcMutivariateItotal(permutation);
406                 break; //Cross Moran (Wartenberg, 1985)
407             case CwithNeighbours:
408                 ss_pos = pstat.calcCwithNeighbourstotal(permutation);
409                 break; //Global Geary with neigh
410             case Lee:
411                 ss_pos = pstat.calcBivariateLee(permutation);
412                 break; //Bivariate Lee's measure: integration of Moran&Pearson coef. (Lee, 2001)
413             case MultiIwithNeighbours:
414                 ss_pos = pstat.calcMultiIwithNeighbours(permutation);
415                 break; //Cross Moran (Wartenberg, 1985) with neigh
416             case CIwithNeighbours:
417                 ss_pos = pstat.calcCIwithNeighbours(permutation);
418                 break; // Connectivity Index (CI) for Graph Data with neigh
419             case LeewithNeighbours:
420                 ss_pos = pstat.calcLeewithNeighbours(permutation);
421                 break; //Bivariate Lee's measure with neigh
422             case Pearson:
423                 ss_pos = pstat.calcPtotal(permutation);
424                 break; //Global Moran without weights i.e Pearson c. coef.
425             case CIDistance:
426                 ss_pos = pstat.calcCItotalD(permutation);
427                 break; // Connectivity Index (CI) for Graph Data with a separate distance file
428             case DH:
429                 ss_pos = pstat.calcDHtotalD(permutation);
430                 break; // Dyadicity and Heterophilicity for Graph Data with a separate distance file
431             case EquvalentIDistance:
432                 ss_pos = pstat.calcEquvalentIDistance(permutation);
433                 break; // EquvalentI with a separate distance file
434             case PearsonDistance:
435                 ss_pos = pstat.calcPDistance(permutation);
436                 break; //Pearson c. coef. with a separate distance file
437             case EquvalentG:
438                 ss_pos = pstat.calcEquvalentGtotal(permutation);
439                 break; //EquvalentG, (Global Geary)
440             case EquvalentGDistance:
441                 ss_pos = pstat.calcEquvalentGDistance(permutation);
442                 break; //EquvalentGDistance, EquvalentG with a separate distance file
443             case EquvalentPDistance:
444             	ss_pos = pstat.calcEquvalentPDistance(permutation);
445             	break;
446         }
447 
448         return ss_pos;
449     }
