109     @Override
110     public void addPrediction(ClusStatistic other, double weight) {
111         TupleStatistic or = (TupleStatistic) other;
112         m_SumWeight += weight * or.m_SumWeight;
113         Tuple pred = new Tuple(or.getTuplePred());
114         pred.setTSWeight(weight);
115         m_TupleStack.add(pred);
116     }

154     @Override
155     public double getAbsoluteDistance(DataTuple tuple, ClusAttributeWeights weights) throws ClusException {
156         int idx = m_Attr.getIndex();

158         return calcDistance(m_RepresentativeMean, actual) * weights.getWeight(idx);

162     public void initNormalizationWeights(ClusAttributeWeights weights, boolean[] shouldNormalize) {
163         int idx = m_Attr.getIndex();
164         if (shouldNormalize[idx]) {
165             double var = m_SVarS / getTotalWeight();
166             double norm = var > 0 ? 1 / var : 1; // No normalization if variance = 0;
167             weights.setWeight(m_Attr, norm);
168         }
169     }

172     public void calcSumAndSumSqDistances(Tuple prototype) throws ClusException {
173         m_AvgDistances = 0.0;
174         int count = m_TupleStack.size();
175         for (int i = 0; i < count; i++) {
176             double dist = calcDistance(prototype, m_TupleStack.get(i));
177             m_AvgDistances += dist;
178         }
179         m_AvgDistances /= count;
180     }

183     /*
184      * [Aco]
185      * this is executed in the end
186      * @see clus.statistic.ClusStatistic#calcMean()
187      */
188     @Override
189     public void calcMean() throws ClusException {
190         // Medoid
191         m_RepresentativeMedoid = null;
192         double minDistance = Double.POSITIVE_INFINITY;
193         for (int i = 0; i < m_TupleStack.size(); i++) {
194             double crDistance = 0.0;
195             Tuple t1 = m_TupleStack.get(i);
196             for (int j = 0; j < m_TupleStack.size(); j++) {
197                 Tuple t2 = m_TupleStack.get(j);
198                 double dist = calcDistance(t1, t2);
199                 crDistance += dist * t2.geTSWeight();
200                 if (Double.isNaN(dist)) {
201                     System.err.println("ooops!!!");
202 
203                 }
204             }
205             if (crDistance < minDistance) {
206                 m_RepresentativeMedoid = m_TupleStack.get(i);
207                 minDistance = crDistance;
208             }
209         }
210         calcSumAndSumSqDistances(m_RepresentativeMedoid);
211 
212         double sumwi = 0.0;
213         for (int j = 0; j < m_TupleStack.size(); j++) {
214             Tuple t1 = m_TupleStack.get(j);
215             sumwi += t1.geTSWeight();
216         }
217         double diff = Math.abs(m_SumWeight - sumwi);
218         if (diff > 1e-6) {
219             System.err.println("Error: Sanity check failed! - " + diff);
220         }
221     }

238         ClusNumberFormat fr = ClusFormat.SIX_AFTER_DOT;

241         buf.append(m_RepresentativeMean.toString());
242         if (info.SHOW_EXAMPLE_COUNT) {
243             buf.append(": ");
244             buf.append(fr.format(m_SumWeight));
245         }
246         buf.append("; ");
247 
248         buf.append("Medoid: ");
249         buf.append(m_RepresentativeMedoid.toString());
250         if (info.SHOW_EXAMPLE_COUNT) {
251             buf.append(": ");
252             buf.append(fr.format(m_SumWeight));
253             buf.append(", ");
254             buf.append(fr.format(m_AvgDistances));
255         }
256         buf.append("; ");

261          * buf.append(": ");
262          * buf.append(fr.format(m_SumWeight));
263          * }
264          * buf.append("; ");
