27  package si.ijs.kt.clus.ext.beamsearch;
28  
29  import si.ijs.kt.clus.algo.tdidt.ClusNode;
30  import si.ijs.kt.clus.heuristic.ClusHeuristic;
31  import si.ijs.kt.clus.main.settings.Settings;
32  import si.ijs.kt.clus.statistic.ClusStatistic;
33  
34  
35  public abstract class ClusBeamHeuristic extends ClusHeuristic {
36  
37      protected double m_NbTrain;
38      protected double m_TreeOffset;
39      protected ClusStatistic m_Pos, m_Neg;
40      protected ClusHeuristic m_AttrHeuristic;
41  
42  
43      public ClusBeamHeuristic(ClusStatistic stat, Settings sett) {
44          super(sett);
45          
46          m_Pos = stat;
47          m_Neg = stat.cloneStat();
48      }
49  
50  
51      public abstract double estimateBeamMeasure(ClusNode tree);
52  
53  
54      public abstract double computeLeafAdd(ClusNode leaf);
55  
56  
57      public void setTreeOffset(double value) {
58          m_TreeOffset = value;
59      }
60  
61  
62      @Override
63      public void setRootStatistic(ClusStatistic stat) {
64          m_NbTrain = stat.getTotalWeight();
65      }
66  
67  
68      public void setAttrHeuristic(ClusHeuristic heur) {
69          m_AttrHeuristic = heur;
70      }
71  
72  
73      public String getAttrHeuristicString() {
74          if (m_AttrHeuristic == null)
75              return "";
76          else
77              return ", attribute heuristic = " + m_AttrHeuristic.getName();
78      }
79  }
