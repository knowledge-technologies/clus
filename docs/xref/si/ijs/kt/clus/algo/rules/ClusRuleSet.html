259 	public ClusStatistic predictWeightedSLS(DataTuple tuple) throws ClusException {
260 		boolean covered = false;
261 
262 		// Unordered rules (with different weighting schemes)
263 		ClusStatistic stat = m_TargetStat.cloneSimple();
264 		double weight_sum = 0.0;
265 
266 		// Get the overall sum of weights for all the rules that cover the example
267 		for (int i = 0; i < getModelSize(); i++) {
268 			ClusRule rule = getRule(i);
269 			if (rule.m_CoverageBits.get(tuple.getIndex())) {
270 				weight_sum += getAppropriateWeight(rule);
271 			}
272 		}
273 
274 		// Take the prediction. Normalise
275 		for (int i = 0; i < getModelSize(); i++) {
276 			ClusRule rule = getRule(i);
277 			if (rule.m_CoverageBits.get(tuple.getIndex())) {
278 				ClusStatistic rulestat = rule.predictWeighted(tuple);
279 				double weight = getAppropriateWeight(rule) / weight_sum;
280 				ClusStatistic norm_rulestat = rulestat.normalizedCopy();
281 				stat.addPrediction(norm_rulestat, weight);
282 				covered = true;
283 			}
284 		}
285 		if (covered) {
286 			stat.computePrediction();
287 			return stat;
288 		} else {
289 			// If not covered, return default prediction
290 			return m_TargetStat;
291 		}
292 
293 	}

295 	/**
296 	 * Returns the statistic (prediction) for a given tuple.
297 	 * 
298 	 * @throws ClusException
299 	 */
300 	@Override
301 	public ClusStatistic predictWeighted(DataTuple tuple) throws ClusException {
302 		boolean covered = false;
303 		RulePredictionMethod pred_method = getSettings().getRules().getRulePredictionMethod();
304 		if (pred_method.equals(RulePredictionMethod.DecisionList)) {
305 			for (int i = 0; i < getModelSize(); i++) {
306 				ClusRule rule = getRule(i);
307 				if (rule.covers(tuple)) {
308 					return rule.getTargetStat();
309 				}
310 			}
311 			return m_TargetStat;
312 		} else if (pred_method.equals(RulePredictionMethod.Union)) {
313 			// In multi-label classification: predicted set of classes is
314 			// union of predictions by individual rules
315 			// TODO: Check if this is obsolete/move/reuse for hierarchical MLC ...
316 			ClusStatistic stat = m_TargetStat.cloneSimple();
317 			stat.unionInit();
318 			for (int i = 0; i < getModelSize(); i++) {
319 				ClusRule rule = getRule(i);
320 				if (rule.covers(tuple)) {
321 					stat.union(rule.getTargetStat());
322 					covered = true;
323 				}
324 			}
325 			stat.unionDone();
326 
327 			// If covered, return stat, else default prediction
328 			return covered ? stat : m_TargetStat;
329 
330 		} else if (getSettings().getRules().isRulePredictionOptimized()) {
331 			// Optimized weights
332 			// The rules are considered as base predictors of ensemble
333 
334 			// Get the weighted prediction of the rule that covers all the
335 			// instances (former default rule)
336 			ClusRule firstRule = getRule(0);
337 			ClusStatistic prediction = (firstRule.predictWeighted(tuple))
338 					.copyNormalizedWeighted(getAppropriateWeight(firstRule));
339 
340 			for (int iBaseRule = 1; iBaseRule < getModelSize(); iBaseRule++) {
341 				ClusRule rule = getRule(iBaseRule);
342 
343 				// if (iBaseRule == getModelSize()-1) {
344 				// boolean blah = false;
345 				// }
346 				// If the rule covers tuple (indicator function is nonzero)
347 				if (rule.covers(tuple)) {
348 					ClusStatistic rulestat = rule.predictWeighted(tuple);
349 
350 					// For regression normalization does not do anything.
351 					// For classification the values are normalized over the the counts.
352 					ClusStatistic norm_rulestat = rulestat.normalizedCopy();
353 					prediction.addPrediction(norm_rulestat, getAppropriateWeight(rule));
354 				}
355 			}
356 
357 			// Is always covered because default rule is used
358 			return prediction;
359 
360 		} else { // Unordered rules (with different weighting schemes)
361 			ClusStatistic stat = m_TargetStat.cloneSimple();
362 			double weight_sum = 0.0;
363 
364 			// Get the overall sum of weights for all the rules that cover the example
365 			for (int i = 0; i < getModelSize(); i++) {
366 				ClusRule rule = getRule(i);
367 				if (rule.covers(tuple)) {
368 					weight_sum += getAppropriateWeight(rule);
369 				}
370 			}
371 
372 			// Take the prediction. Normalise
373 			for (int i = 0; i < getModelSize(); i++) {
374 				ClusRule rule = getRule(i);
375 				if (rule.covers(tuple)) {
376 					ClusStatistic rulestat = rule.predictWeighted(tuple);
377 					double weight = getAppropriateWeight(rule) / weight_sum;
378 					ClusStatistic norm_rulestat = rulestat.normalizedCopy();
379 					stat.addPrediction(norm_rulestat, weight);
380 					covered = true;
381 				}
382 			}
383 			if (covered) {
384 				stat.computePrediction();
385 				return stat;
386 			} else {
387 				// If not covered, return default prediction
388 				return m_TargetStat;
389 			}
390 		}
391 	}

393 	public double getAppropriateWeight(ClusRule rule) {
394 		switch (getSettings().getRules().getRulePredictionMethod()) {
395 		case CoverageWeighted:
396 			return rule.m_TargetStat.getTotalWeight();
397 
398 		case TotCoverageWeighted:
399 			return rule.getCoverage()[ClusModel.TRAIN];
400 
401 		case AccCovWeighted:
402 			return rule.m_TargetStat.getTotalWeight() * (1 - rule.getTrainErrorScore());
403 
404 		case AccuracyWeighted:
405 			return 1 - rule.getTrainErrorScore();
406 
407 		case Optimized:
408 		case GDOptimized:
409 		case GDOptimizedBinary:
410 			return rule.getOptWeight();
411 		default:
412 			throw new RuntimeException(
413 					"si.ijs.kt.clus.algo.rules.ClusRuleSet.getAppropriateWeight(ClusRule): Unknown weighted prediction method!");
414 		}
415 	}

425 	/**
426 	 * Remove rules that have less weight than OptRuleWeightThreshold set in .s
427 	 * file.
428 	 * 
429 	 * @return Original index of the last handled weight.
430 	 */
431 	public int removeLowWeightRules() {
432 		double threshold = getSettings().getRules().getOptRuleWeightThreshold();
433 		boolean binarizeWeights = getSettings().getRules().getOptRuleWeightBinarization();
434 
435 		if (binarizeWeights) {
436 			System.err.println("Weight binarization will be used.");
437 		}
438 
439 		int nb_rules = getModelSize();
440 		ClusRule r = null;
441 
442 		for (int i = nb_rules - 1; i >= 0; i--) {
443 
444 			// If the first rule is all covering, it is not removed (can predict just 0 if
445 			// that is the weight)
446 			// This is used instead of default rule if the weights are optimized.
447 			if (i == 0 && m_allCoveringRuleExists) {
448 				continue;
449 			}
450 
451 			r = getRule(i);
452 
453 			if (binarizeWeights) {
454 				if (Math.abs(r.getOptWeight()) > threshold) {
455 					r.setOptWeight(1.0);
456 				} else {
457 					m_Rules.remove(i);
458 				}
459 			} else {
460 				if (Math.abs(getRule(i).getOptWeight()) < threshold || getRule(i).getOptWeight() == 0.0) {
461 					// If optimization is used, the last first covers all the instances. This should
462 					// not be removed
463 					m_Rules.remove(i);
464 				}
465 			}
466 		}
467 
468 		ClusLogger.info("Rules left: " + getModelSize() + " out of " + nb_rules);
469 
470 		return nb_rules - 1;
471 	}

473 	/**
474 	 * Update rule sets so, that the offset and scaling factors of linear terms are
475 	 * included In the weights. Thus linear terms stay untouched in the end. NOTE:
476 	 * Missing values cause this function to reduce accuracy! See settings file for
477 	 * more information.
478 	 */
479 	public void convertToPlainLinearTerms() {
480 		RegressionStat firstRuleStat = (RegressionStat) getRule(0).m_TargetStat;
481 
482 		// If w0 = 0, we have to make some changes
483 		if (getRule(0).getOptWeight() == 0) {
484 			// Equals the previous prediction, but no dividing with zero now
485 			for (int iTarget = 0; iTarget < firstRuleStat.getNbAttributes(); iTarget++) {
486 				firstRuleStat.m_Means[iTarget] = 0d;
487 			}
488 			getRule(0).setOptWeight(1d); //
489 		}
490 
491 		double defaultWeight = getRule(0).getOptWeight();
492 		ClusStatistic tar_stat = m_StatManager.getStatistic(AttributeUseType.Target);
493 		// What is to be added to default pred
494 		double[] addToDefaultPred = new double[tar_stat.getNbAttributes()];
495 
496 		for (int iRule = 0; iRule < getModelSize(); iRule++) {
497 
498 			ClusRule cursor = getRule(iRule);
499 			if (!cursor.isRegularRule()) {
500 				addToDefaultPred = ((ClusRuleLinearTerm) cursor).convertToPlainTerm(addToDefaultPred, defaultWeight);
501 			}
502 		}
503 
504 		for (int iTarget = 0; iTarget < firstRuleStat.getNbAttributes(); iTarget++) {
505 			firstRuleStat.m_Means[iTarget] += addToDefaultPred[iTarget];
506 			firstRuleStat.m_SumValues[iTarget] = firstRuleStat.m_Means[iTarget];
507 			firstRuleStat.m_SumWeights[iTarget] = 1;
508 		}
509 	}

531 	public void printModel(PrintWriter wrt, StatisticPrintInfo info) {
532 		ClusNumberFormat fr = ClusFormat.SIX_AFTER_DOT;
533 		// boolean headers = getSettings().computeDispersion() || hasRuleErrors();
534 		boolean headers = true;
535 		// [train/test][comb/num/nom]
536 		double[][] avg_dispersion = new double[2][3];
537 		double[] avg_coverage = new double[2];
538 		double[][] avg_prod = new double[2][3];
539 
540 		if (!getSettings().getRules().isPrintAllRules())
541 			wrt.println("Set of " + m_Rules.size() + " rules." + System.lineSeparator());
542 
543 		if (getSettings().getRules().getRulePredictionMethod().equals(RulePredictionMethod.GDOptimized)
544 				&& getSettings().getRules().getOptGDNbOfTParameterTry() > 1) {
545 			// Print the T value that won the game.
546 			wrt.println("Gradient descent optimization: Smallest test fitness of " + fr.format(m_optWeightBestFitness)
547 					+ " with T value: " + fr.format(m_optWeightBestTValue) + System.lineSeparator());
548 		}
549 
550 		for (int i = 0; i < m_Rules.size(); i++) {
551 			ClusRule rule = m_Rules.get(i);
552 			if (headers) {
553 				if (getSettings().getRules().isPrintAllRules()) {
554 					String head = new String("Rule " + (i + 1) + ":");
555 					char[] underline = new char[head.length()];
556 					for (int j = 0; j < head.length(); j++) {
557 						underline[j] = '=';
558 					}
559 
560 					wrt.println(head);
561 					wrt.println(new String(underline));
562 				}
563 				// Added this test so that PrintRuleWiseErrors also works in HMC setting
564 				// (02/01/06)
565 				if (getSettings().getRules().computeDispersion()) {
566 					avg_dispersion[0][0] += rule.m_CombStat[0].dispersionCalc();
567 					avg_dispersion[0][1] += rule.m_CombStat[0].dispersionNum(1);
568 					avg_dispersion[0][2] += rule.m_CombStat[0].dispersionNom(1);
569 					avg_coverage[0] += rule.m_Coverage[0];
570 					avg_prod[0][0] += rule.m_CombStat[0].dispersionCalc() * rule.m_Coverage[0];
571 					avg_prod[0][1] += rule.m_CombStat[0].dispersionNum(1) * rule.m_Coverage[0];
572 					avg_prod[0][2] += rule.m_CombStat[0].dispersionNom(1) * rule.m_Coverage[0];
573 					if (rule.m_CombStat[1] != null) {
574 						avg_dispersion[1][0] += rule.m_CombStat[1].dispersionCalc();
575 						avg_dispersion[1][1] += rule.m_CombStat[1].dispersionNum(1);
576 						avg_dispersion[1][2] += rule.m_CombStat[1].dispersionNom(1);
577 						avg_coverage[1] += rule.m_Coverage[1];
578 						avg_prod[1][0] += rule.m_CombStat[1].dispersionCalc() * rule.m_Coverage[1];
579 						avg_prod[1][1] += rule.m_CombStat[1].dispersionNum(1) * rule.m_Coverage[1];
580 						avg_prod[1][2] += rule.m_CombStat[1].dispersionNom(1) * rule.m_Coverage[1];
581 					}
582 				}
583 			}
584 			if (getSettings().getRules().isPrintAllRules()) {
585 				rule.printModel(wrt, info);
586 				wrt.println();
587 			}

782 	 * Computes the error score of the rule set. To be used for deciding whether to
783 	 * add more rules to the rule set or not.
784 	 * 
785 	 * @throws ClusException
786 	 */
787 	// TODO: finish
788 	public double computeErrorScore(RowData data) throws ClusException {
789 		ClusStatistic tar_stat = m_StatManager.getStatistic(AttributeUseType.Target);
790 		// Average error rate over all target attributes
791 		if (tar_stat instanceof ClassificationStat) {
792 			double result = 0;
793 			int nb_rows = data.getNbRows();
794 			int nb_tar = tar_stat.getNbNominalAttributes();
795 			int[] nb_right = new int[nb_tar];
796 			for (int i = 0; i < nb_rows; i++) {
797 				DataTuple tuple = data.getTuple(i);
798 				int[] predictions = predictWeighted(tuple).getNominalPred();
799 				int true_value;
800 				NominalAttrType[] targetAttrs = data.getSchema().getNominalAttrUse(AttributeUseType.Target);
801 				for (int j = 0; j < nb_tar; j++) {
802 					true_value = targetAttrs[j].getNominal(tuple);
803 					if (predictions[j] == true_value) {
804 						nb_right[j]++;
805 					}
806 				}
807 			}
808 			for (int j = 0; j < nb_tar; j++) {
809 				result += (1.0 * nb_rows - nb_right[j]) / nb_rows;
810 			}
811 			result /= nb_tar;
812 			return result;
813 			// Average RMSE over all target attributes
814 		} else if (tar_stat instanceof RegressionStat) {
815 			double result = 0;
816 			int nb_rows = data.getNbRows();
817 			int nb_tar = tar_stat.getNbNumericAttributes();
818 			double[] sum_sqr_err = new double[nb_tar];
819 			NumericAttrType[] targetAttrs = data.getSchema().getNumericAttrUse(AttributeUseType.Target);
820 			for (int i = 0; i < nb_rows; i++) {
821 				DataTuple tuple = data.getTuple(i);
822 				double[] predictions = ((RegressionStat) predictWeighted(tuple)).getNumericPred();
823 				for (int j = 0; j < nb_tar; j++) {
824 					double diff = predictions[j] - targetAttrs[j].getNumeric(tuple);
825 					sum_sqr_err[j] += diff * diff;
826 				}
827 			}
828 			for (int j = 0; j < nb_tar; j++) {
829 				result += Math.sqrt(sum_sqr_err[j] / nb_rows);
830 			}
831 			result /= nb_tar;
832 			return result;
833 		} else { // Mixed classification and regression not yet implemented
834 			return -1;
835 		}

953 	/**
954 	 * Returns the form of this rule set that can be used for weight optimization.
955 	 * The predictions should be always a value or NaN. NaN occurs if the rule does
956 	 * not cover the example OR the linear term predicts undefined attribute.
957 	 * 
958 	 * @param outLogFile File stream for outputs.
959 	 * @param data       Data the optimization is based on.
960 	 * @return Parameters for optimization. Include true values and predictions for
961 	 *         each of the data instances.
962 	 * @throws ClusException
963 	 */
964 	public OptimizationProblem.OptimizationParameter giveFormForWeightOptimization(PrintWriter outLogFile, RowData data)
965 			throws ClusException {
966 		// data = Clus.returnNormalizedData(data);
967 
968 		ClusSchema schema = data.getSchema();
969 
970 		/** Default prediction. This may be used to alter other predictions. */
971 		double[] defaultPred = null;
972 
973 		// We need to transform the default rule to ordinary rule. We want to optimize
974 		// it also.
975 		// Also if set, modify the other rules according to this
976 		defaultPred = addDefaultRuleToRuleSet();
977 
978 		/** Data normalization steps */
979 		if (getSettings().getRules().isOptDefaultShiftPred()) {
980 			// Shift predictions according to the default rule
981 			shiftRulePredictions(defaultPred);
982 		}
983 
984 		if (getSettings().getRules().isOptOmitRulePredictions()) {
985 			/*
986 			 * after creating default rule, before adding linear terms or weighting
987 			 * generality
988 			 */
989 			omitRulePredictions();
990 		}
991 
992 		if (getSettings().getRules().isOptWeightGenerality()) {
993 			/*
994 			 * Weight the rules on generality. Do this after adding the default rule! Also
995 			 * do this after omitting rule predictions!
996 			 */
997 			weightGeneralityForPredictions(data.getNbRows());
998 		}
999 
1000 		// Add linear terms to rule set
1001 		if (getSettings().getRules().isOptAddLinearTerms()) {
1002 			// This should be done after default rule adding.
1003 
1004 			ClusRuleLinearTerm.initializeClass(data, m_StatManager);
1005 
1006 			if (getSettings().getRules().getOptAddLinearTerms().equals(OptimizationGDAddLinearTerms.Yes)) {
1007 				// Add linear terms only separately
1008 				addLinearTermsToRuleSet();
1009 			} // if memory saving is used, adding is done in the optimization phase
1010 		}
1011 
1012 		// Generate optimization input
1013 		ClusStatistic tar_stat = m_StatManager.getStatistic(AttributeUseType.Target);
1014 		int nb_target = tar_stat.getNbAttributes();
1015 		int nb_baseFunctions = getModelSize();
1016 		int nb_rows = data.getNbRows();
1017 
1018 		/** TRUE VALUES */
1019 		ClusAttrType[] trueValuesTemp = new ClusAttrType[nb_target];
1020 		switch (m_StatManager.getTargetMode()) {
1021 			case CLASSIFY:
1022                 trueValuesTemp = schema.getNominalAttrUse(AttributeUseType.Target);
1023                 break;
1024 
1025 			case REGRESSION:
1026                 trueValuesTemp = schema.getNumericAttrUse(AttributeUseType.Target);
1027                 break;
1028 
1029             default:
1030                 throw new ClusException(
1031                         "si.ijs.kt.clus.algo.rules.ClusRuleSet.giveFormForWeightOptimization(PrintWriter, RowData): not implemented");
1032 		}
1033 
1034 		/**
1035 		 * True values for each target and instance
1036 		 */
1037 		// double[][] trueValues = new double[nb_rows][nb_target];
1038 		OptimizationProblem.TrueValues[] trueValues = new OptimizationProblem.TrueValues[nb_rows];
1039 		// Index over the instances of data
1040 		for (int iRows = 0; iRows < nb_rows; iRows++) {
1041 			DataTuple tuple = data.getTuple(iRows);
1042 
1043 			OptimizationProblem.TrueValues newTrueTargets = null;
1044 			// Give the tuple to attribute line only if it is needed for implicit linear
1045 			// prediction.
1046 			if (getSettings().getRules().getOptAddLinearTerms().equals(OptimizationGDAddLinearTerms.YesSaveMemory)) {
1047 				newTrueTargets = new OptimizationProblem.TrueValues(nb_target, tuple);
1048 			} else {
1049 				newTrueTargets = new OptimizationProblem.TrueValues(nb_target, null);
1050 			}
1051 
1052 			trueValues[iRows] = newTrueTargets;
1053 
1054 			// Take the true values from the data target by target
1055 			for (int kTargets = 0; kTargets < nb_target; kTargets++) {
1056 
1057 				switch (m_StatManager.getTargetMode()) {
1058 					case CLASSIFY:
1059                         trueValues[iRows].m_targets[kTargets] = ((NominalAttrType) trueValuesTemp[kTargets])
1060                                 .getNominal(tuple);
1061                         break;
1062 
1063 					case REGRESSION:
1064                         trueValues[iRows].m_targets[kTargets] = ((NumericAttrType) trueValuesTemp[kTargets])
1065                                 .getNumeric(tuple);
1066                         break;
1067 
1068                     default:
1069                         throw new ClusException(
1070                                 "si.ijs.kt.clus.algo.rules.ClusRuleSet.giveFormForWeightOptimization(PrintWriter, RowData): not implemented");
1071 				}
1072 			}
1073 		}
1074 
1075 		/**
1076 		 * PREDICTIONS: We have two different prediction arrays. First one for rules,
1077 		 * second one for all other types of predictors (e.g. linear terms). To get the
1078 		 * order right, the rules have to be the first ones. Let's check this.
1079 		 */
1080 
1081 		boolean allRulesScrolled = false;
1082 		int nbOfRegularRules = 0;
1083 		for (int jRules = 0; jRules < nb_baseFunctions; jRules++) {
1084 			ClusRule rule = getRule(jRules);
1085 			if (rule.isRegularRule()) {
1086 				nbOfRegularRules++;
1087 				if (allRulesScrolled) {
1088 					// Error! There is a regular rule after something else
1089 					throw new ClusException(
1090 							"Error: Order of rule set is wrong. Rules have to be before linear terms etc.");
1091 				}
1092 			} else {
1093 				allRulesScrolled = true; // Last rule has gone already
1094 			}
1095 
1096 		}
1097 
1098 		// Number of nominal values for each target. For regression, no number of
1099 		// nominal values needed, i.e. 1
1100 		int nb_values[] = new int[nb_target];
1101 
1102 		for (int iTarget = 0; iTarget < nb_target; iTarget++) {
1103 
1104 			switch (m_StatManager.getTargetMode()) {
1105 				case CLASSIFY:
1106                     nb_values[iTarget] = ((ClassificationStat) m_TargetStat).getAttribute(0).getNbValues();
1107                     break;
1108 
1109 				case REGRESSION:
1110                     nb_values[iTarget] = 1; // Nominal values not needed
1111                     break;
1112 
1113                 default:
1114                     throw new ClusException(
1115                             "si.ijs.kt.clus.algo.rules.ClusRuleSet.giveFormForWeightOptimization(PrintWriter, RowData): not implemented");
1116 			}
1117 		}
1118 
1119 		// DecimalFormat mf = new DecimalFormat("###.000");
1120 		DecimalFormat mf = new DecimalFormat("#00.000");
1121 		mf.setPositivePrefix("+"); // DEBUG
1122 
1123 		// [nonrule predictor][instance][target][class_value]
1124 		// For regression, class_value = 0 always.
1125 		// double[][][][] rule_pred = new
1126 		// double[nb_rules][nb_rows][nb_target][nb_values];
1127 		double[][][][] nonrule_pred = new double[nb_baseFunctions - nbOfRegularRules][nb_rows][nb_target][];
1128 
1129 		// Rule predictions.
1130 
1131 		OptimizationProblem.RulePred[] rule_pred = new OptimizationProblem.RulePred[nbOfRegularRules];
1132 		for (int jRules = 0; jRules < nbOfRegularRules; jRules++) {
1133 			rule_pred[jRules] = new OptimizationProblem.RulePred(nb_rows, nb_target);
1134 
1135 			ClusRule rule = getRule(jRules);
1136 
1137 			// Works only for regression
1138 			double[] targets = ((RegressionStat) rule.predictWeighted(null)).normalizedCopy().getNumericPred();
1139 
1140 			for (int iTarget = 0; iTarget < nb_target; iTarget++) {
1141 				rule_pred[jRules].m_prediction[iTarget][0] = targets[iTarget];
1142 			}
1143 		}
1144 
1145 		// If printing predictions to file, print the name of the rule here

1318 	 * Changes the rule predictions such that they always predict 1. This is by
1319 	 * Friedman 2005;
1320 	 */
1321 	private void omitRulePredictions() {
1322 		ClusLogger.info("Omitting rule predictions for optimization.");
1323 
1324 		/**
1325 		 * It depends on inner normalization if rule0 is omitted If normalization is
1326 		 * done in such a way that x-avg for all the targets, do not omit because it
1327 		 * will be omitted later anyway.
1328 		 * 
1329 		 * This procedure equalizes the maximal target prediction value for all rules,
1330 		 * i.e., the highest target prediction or a single rule is in the same range
1331 		 * across all rules.
1332 		 * 
1333 		 * Our aim is to equalize the rules with respect to the optimization problem.
1334 		 */
1335 		int iRule = 0;
1336 		if (getSettings().getRules().isOptNormalization()
1337 				&& !getSettings().getRules().getOptNormalization().equals(OptimizationNormalization.OnlyScaling)) {
1338 			iRule = 1;
1339 		}
1340 
1341 		double tmp;
1342 
1343 		for (; iRule < getModelSize(); iRule++) {
1344 			ClusRule rule = getRule(iRule);
1345 			if (rule.isRegularRule()) {// If this is linear term, do not touch it
1346 				if (!(rule.m_TargetStat instanceof RegressionStat)) {
1347 					throw new RuntimeException("Error: GD optimization is implemented regression only.");
1348 				}
1349 
1350 				RegressionStat stat = (RegressionStat) rule.m_TargetStat;
1351 
1352 				if (stat.getNbAttributes() > 1) {
1353 					// Multi-target
1354 
1355 					Double scalingValue = null;
1356 					/*
1357 					 * We try to find the value that is greatest compared to std dev. However, in
1358 					 * actual scaling, we may not be using std dev information.
1359 					 */
1360 					double chi = 0d;
1361 					for (int iTarget = 0; iTarget < stat.getNbAttributes(); iTarget++) {
1362 						tmp = stat.m_Means[iTarget] / (2 * getTargetStdDev(iTarget));
1363 						if (Math.abs(tmp) > Math.abs(chi)) {
1364 
1365 							chi = tmp;
1366 							scalingValue = stat.m_Means[iTarget]; // Scales values to abs(x)<=1
1367 
1368 							if (getSettings().getRules().isOptNormalization()) {
1369 								// After normalization, the greatest value is 1
1370 								scalingValue /= (2 * getTargetStdDev(iTarget));
1371 							}
1372 						}
1373 					}
1374 
1375 					if (scalingValue == null) {
1376 						// all targets have the same prediction/2*stdev
1377 						// do not scale in this case
1378 						scalingValue = 1d;
1379 					}
1380 
1381 					for (int iTarget = 0; iTarget < stat.getNbAttributes(); iTarget++) {
1382 						stat.m_Means[iTarget] /= scalingValue;
1383 						// Bigger one (on absolute value) is 1
1384 						stat.m_SumValues[iTarget] = stat.m_Means[iTarget];
1385 						stat.m_SumWeights[iTarget] = 1;
1386 
1387 						// ClusLogger.info(stat.m_Means[iTarget]/(2*getTargStd(iTarget)));
1388 					}
1389 				} else {
1390 					// Single target
1391 					if (getSettings().getRules().isOptNormalization()) {
1392 						stat.m_Means[0] = 2 * getTargetStdDev(0);
1393 					} else {
1394 						stat.m_Means[0] = 1;
1395 					}
1396 
1397 					stat.m_SumValues[0] = stat.m_Means[0];
