23  	@Test
24  	public void smallTest() {
25  		double[] numbers = new double[]{0.987654, 0.0987654, 0.00987654, 0.000987654, 0.0000987654, 0.00000987654, 0.000000987654,
26  										0.98, 0.098, 0.0098, 0.00098, 0.000098, 0.0000098, 0.00000098};
27  		String[] numbersString = new String[] {"0.987654", "0.0987654", "0.00987654", "0.000987654", "0.0000987654", "0.00000987654", "0.000000987654",
28  											   "0.98", "0.098", "0.0098", "0.00098", "0.000098", "0.0000098", "0.00000098"};
29  		String[][] solutions = new String[][]{
30  			new String[] {
31  							"1E0", "1E-1", "1E-2", "1E-3", "1E-4", "1E-5", "1E-6",
32  							"1E0", "1E-1", "1E-2", "1E-3", "1E-4", "1E-5", "1E-6"
33  						 },
34  			new String[] {
35  					"9.9E-1", "9.9E-2", "9.9E-3", "9.9E-4", "9.9E-5", "9.9E-6", "9.9E-7",
36  					"9.8E-1", "9.8E-2", "9.8E-3", "9.8E-4", "9.8E-5", "9.8E-6", "9.8E-7"
37  				 		 },
38  			new String[] {
39  					"9.8765E-1", "9.8765E-2", "9.8765E-3", "9.8765E-4", "9.8765E-5", "9.8765E-6", "9.8765E-7",
40  					"9.8000E-1", "9.8000E-2", "9.8000E-3", "9.8000E-4", "9.8000E-5", "9.8000E-6", "9.8000E-7"
41  				 		 },
42  		};
43  		for(int i = 0; i < m_Formatters.length; i++) {
44  			for(int j = 0; j < numbers.length; j++) {
45  				double[] positiveNegative = new double[] {numbers[j], -numbers[j]};
46  				String[] solutionPositiveNegative = new String[] {solutions[i][j], "-" + solutions[i][j]};
47  				for(int k = 0; k < 2; k++) {
48  					String number = k == 0 ? numbersString[j] : "-" + numbersString[j];
49  					assertEquals(solutionPositiveNegative[k], m_Formatters[i].format(positiveNegative[k]), "Wrong format for " + number + " with precision " + m_Precisions[i]);
50  				}
51  			}
52  		}
53  	}
