70      public int[][] createConstraintsIndex() {
71          /* create index as array lists */
72          ArrayList[] crIndex = new ArrayList[m_NbTrain];
73          for (int i = 0; i < m_Constraints.size(); i++) {
74              ILevelConstraint ic = (ILevelConstraint) m_Constraints.get(i);
75              int t1 = ic.getT1().getIndex();
76              int t2 = ic.getT2().getIndex();
77              if (crIndex[t1] == null)
78                  crIndex[t1] = new ArrayList();
79              if (crIndex[t2] == null)
80                  crIndex[t2] = new ArrayList();
81              crIndex[t1].add(new Integer(i));
82              crIndex[t2].add(new Integer(i));
83          }
84          /* copy it to final int matrix */
85          int[][] index = new int[m_NbTrain][];
86          for (int i = 0; i < m_NbTrain; i++) {
87              if (crIndex[i] != null) {
88                  int nb = crIndex[i].size();
89                  index[i] = new int[nb];
90                  for (int j = 0; j < nb; j++) {
91                      Integer value = (Integer) crIndex[i].get(j);
92                      index[i][j] = value.intValue();
93                  }
94              }
95          }
96          return index;
97      }

100     public ILevelConstraint[] getSubsetConstraints(RowData data) {
101         int count = 0;
102         boolean[] constr = new boolean[m_Constraints.size()];
103         for (int i = 0; i < data.getNbRows(); i++) {
104             DataTuple tuple = data.getTuple(i);
105             int[] index = m_ConstraintsIndex[tuple.getIndex()];
106             if (index != null) {
107                 for (int j = 0; j < index.length; j++) {
108                     int cid = index[j];
109                     if (!constr[cid]) {
110                         constr[cid] = true;
111                         count++;
112                     }
113                 }
114             }
115         }
116         int pos = 0;
117         ILevelConstraint[] result = new ILevelConstraint[count];
118         for (int i = 0; i < m_Constraints.size(); i++) {
119             if (constr[i])
120                 result[pos++] = (ILevelConstraint) m_Constraints.get(i);
121         }
122         return result;
123     }

135     public double computeHeuristic(int violated, double ss) {
136         double ss_norm = ss / m_GlobalSS;
137         double violated_norm = 1.0 * violated / m_Constraints.size();
138         double alpha = getSettings().getILevelC().getILevelCAlpha();
139         double heur = (1.0 - alpha) * ss_norm + alpha * violated_norm;
140         // ClusLogger.info("Violated: "+violated+" SS: "+ss+" -> "+heur);
141         return heur;
142     }

145     public double computeHeuristic(ILevelCHeurStat pos, ILevelCHeurStat neg, ILevelCStatistic ps, boolean use_p_lab, double ss_offset, int nb_violated) {
146         if (pos.getTotalWeight() < m_MinLeafWeight)
147             return Double.POSITIVE_INFINITY;
148         if (neg.getTotalWeight() < m_MinLeafWeight)
149             return Double.POSITIVE_INFINITY;
150         int pLabel = -1;
151         int nbLabels = m_NbClasses;
152         if (use_p_lab) {
153             /* if does not occur elsewhere then it becomes available for reuse */
154             nbLabels = m_NbClasses - 1;
155             pLabel = ps.getClusterID();
156         }
157         int v1 = tryLabel(pos, neg, pLabel, nbLabels);
158         int v2 = tryLabel(neg, pos, pLabel, nbLabels);
159         if (v1 == -1) {
160             if (v2 != -1)
161                 nb_violated += v2;
162             else
163                 return Double.POSITIVE_INFINITY;
164         }
165         else if (v2 == -1) {
166             if (v1 != -1)
167                 nb_violated += v1;
168             else
169                 return Double.POSITIVE_INFINITY;
170         }
171         else {
172             nb_violated += Math.min(v1, v2);
173         }
174         double ss_pos = pos.getSVarS(m_Scale);
175         double ss_neg = neg.getSVarS(m_Scale);
176         double ss = ss_offset + ss_pos + ss_neg;
177         return computeHeuristic(nb_violated, ss);
178     }

181     public void findNumericConstraints(NumericAttrType at, ClusNode leaf, boolean use_p_lab, double ss_offset, int violated_offset, int violated_leaf, int[] clusters) throws ClusException {
182         RowData data = (RowData) leaf.getVisitor();
183         ILevelCStatistic tot = (ILevelCStatistic) leaf.getClusteringStat();
184         // ClusLogger.info("Trying: "+at.getName());
185         int idx = at.getArrayIndex();
186         if (at.isSparse()) {
187             data.sortSparse(at, m_SortHelper);
188         }
189         else {
190             data.sort(at);
191         }
192         int nb_rows = data.getNbRows();
193         if (at.hasMissing()) { throw new ClusException("Does not support attributes with missing values: " + at.getName()); }
194         /* test is of form A > vi */
195         /* vi is sorted from large to small */
196         /* pos statistic is values larger than current threshold */
197         /* neg statistic is values smaller than current threshold */
198         ILevelCStatistic cs = (ILevelCStatistic) leaf.getClusteringStat();
199         ILevelCHeurStat pos = new ILevelCHeurStat(getSettings(), cs, m_NbClasses);
200         ILevelCHeurStat neg = new ILevelCHeurStat(getSettings(), cs, m_NbClasses);
201         /* create internal/external index */
202         int[] ie = createIE(data);
203         /* pass required indices to statistics */
204         pos.setIndices(m_ConstraintsIndex, m_Constraints, ie, clusters);
205         neg.setIndices(m_ConstraintsIndex, m_Constraints, ie, clusters);
206         /* initially all data is in negative statistic */
207         for (int i = 0; i < nb_rows; i++) {
208             DataTuple tuple = data.getTuple(i);
209             neg.updateWeighted(tuple, tuple.getWeight());
210         }
211         int nb_violated = violated_leaf;
212         /* then loop over all examples */
213         double prev = Double.NaN;
214         for (int i = 0; i < nb_rows; i++) {
215             DataTuple tuple = data.getTuple(i);
216             double value = tuple.getDoubleVal(idx);
217             if (value != prev && !Double.isNaN(prev)) {
218                 // m_Selector.updateNumeric(value, at);
219                 double heuristic = computeHeuristic(pos, neg, tot, use_p_lab, ss_offset, violated_offset + nb_violated);
220                 if (heuristic < m_BestHeur) {
221                     m_BestHeur = heuristic;
222                     m_BestLeaf = leaf;
223                     double pos_freq = pos.getTotalWeight() / tot.getTotalWeight();
224                     double splitpoint = (value + prev) / 2.0;
225                     m_BestTest = new NumericTest(at, splitpoint, pos_freq);
226                 }
227                 prev = value;
228             }
229             pos.updateWeighted(tuple, tuple.getWeight());
230             neg.removeWeighted(tuple, tuple.getWeight());
231             int tidx = tuple.getIndex();
232             int[] cidx = m_ConstraintsIndex[tidx];
233             if (cidx != null) {
234                 for (int j = 0; j < cidx.length; j++) {
235                     ILevelConstraint cons = (ILevelConstraint) m_Constraints.get(cidx[j]);
236                     int otidx = cons.getOtherTupleIdx(tuple);
237                     if (ie[otidx] != ILevelCHeurStat.EXT) {
238                         boolean was_violated = false;
239                         if (cons.getType() == ILevelConstraint.ILevelCMustLink) {
240                             if (ie[tidx] != ie[otidx])
241                                 was_violated = true;
242                         }
243                         else {
244                             if (ie[tidx] == ie[otidx])
245                                 was_violated = true;
246                         }
247                         if (was_violated)
248                             nb_violated--;
249                         else
250                             nb_violated++;
251                     }
252                 }
253             }
254             ie[tidx] = ILevelCHeurStat.POS;
255         }
256     }

259     public void tryGivenLeaf(ClusNode leaf, boolean use_p_lab, int violated, double ss, int[] clusters) throws ClusException {
260         ILevelCStatistic stat = (ILevelCStatistic) leaf.getClusteringStat();
261         if (stat.getTotalWeight() <= m_MinLeafWeight) {
262             /* don't refine clusters that are too small */
263             return;
264         }
265         RowData leaf_data = (RowData) leaf.getVisitor();
266         double ss_leaf = stat.getSVarS(m_Scale);
267         double ss_offset = ss - ss_leaf;
268         int[] v_info = countViolatedConstaints(leaf_data, clusters);
269         int violated_offset = violated - v_info[0];
270         ClusLogger.info("Violated by leaf: " + v_info[0] + " internal ML: " + v_info[1] + " (of " + violated + " total)");
271         ClusSchema schema = getSchema();
272         ClusAttrType[] attrs = schema.getDescriptiveAttributes();
273         for (int i = 0; i < attrs.length; i++) {
274             ClusAttrType at = attrs[i];
275             if (at instanceof NumericAttrType)
276                 findNumericConstraints((NumericAttrType) at, leaf, use_p_lab, ss_offset, violated_offset, v_info[1], clusters);
277             else
278                 throw new ClusException("Unsupported descriptive attribute type: " + at.getName());
279         }
280     }

283     public void tryEachLeaf(ClusNode tree, ClusNode root, int violated, double ss, int[] clusters) throws ClusException {
284         int nb_c = tree.getNbChildren();
285         if (nb_c == 0) {
286             ILevelCStatistic ps = (ILevelCStatistic) tree.getClusteringStat();
287             int nbParLabel = countLabel(root, ps.getClusterID());
288             tryGivenLeaf(tree, nbParLabel <= 1, violated, ss, clusters);
289         }
290         else {
291             for (int i = 0; i < nb_c; i++) {
292                 ClusNode child = (ClusNode) tree.getChild(i);
293                 tryEachLeaf(child, root, violated, ss, clusters);
294             }
295         }
296     }

299     public ILevelCHeurStat computeCHeurStat(ClusNode leaf, ClusNode par, int[] ie, int[] clusters) {
300         RowData data = (RowData) leaf.getVisitor();
301         ILevelCStatistic cs = (ILevelCStatistic) leaf.getClusteringStat();
302         ILevelCHeurStat lstat = new ILevelCHeurStat(getSettings(), cs, m_NbClasses);
303         lstat.setIndices(m_ConstraintsIndex, m_Constraints, ie, clusters);
304         for (int i = 0; i < data.getNbRows(); i++) {
305             DataTuple tuple = data.getTuple(i);
306             lstat.updateWeighted(tuple, tuple.getWeight());
307         }
308         return lstat;
309     }

312     public int tryLabel(ILevelCHeurStat a, ILevelCHeurStat b, int parlabel, int nb) {
313         int v1 = a.computeMinimumExtViolated(parlabel, -1, nb < m_MaxNbClasses);
314         int label_a = a.getClusterID();
315         if (label_a == -1)
316             nb++;
317         int v2 = b.computeMinimumExtViolated(parlabel, label_a, nb < m_MaxNbClasses);
318         return (v1 == -1 || v2 == -1) ? -1 : v1 + v2;
319     }

322     public void assignLabels(ILevelCHeurStat a, ILevelCHeurStat b, ClusNode root, int parlabel, int nb) throws ClusException {
323         int v1 = a.computeMinimumExtViolated(parlabel, -1, nb < m_MaxNbClasses);
324         int label_a = a.getClusterID();
325         if (label_a == -1) {
326             nb++;
327             label_a = freeLabel(root, -1);
328             m_NbClasses = Math.max(label_a + 1, m_NbClasses);
329             a.setClusterID(label_a);
330         }
331         int v2 = b.computeMinimumExtViolated(parlabel, label_a, nb < m_MaxNbClasses);
332         if (b.getClusterID() == -1) {
333             int label_b = freeLabel(root, label_a);
334             m_NbClasses = Math.max(label_b + 1, m_NbClasses);
335             b.setClusterID(label_b);
336         }
337         if (v1 == -1 || v2 == -1) { throw new ClusException("Error: can't assign labels: v1 = " + v1 + " v2 = " + v2); }
338     }

349     public void enterBestTest(ClusNode tree, ClusNode root, int[] clusters) throws ClusException {
350         int nb_c = tree.getNbChildren();
351         if (nb_c == 0) {
352             if (tree == m_BestLeaf) {
353                 RowData data = (RowData) tree.getVisitor();
354                 tree.setTest(m_BestTest);
355                 int arity = tree.updateArity();
356                 NodeTest test = tree.getTest();
357                 for (int j = 0; j < arity; j++) {
358                     ClusNode child = new ClusNode();
359                     tree.setChild(child, j);
360                     RowData subset = data.applyWeighted(test, j);
361                     child.initClusteringStat(getStatManager(), subset);
362                     child.initTargetStat(getStatManager(), subset);
363                     child.getTargetStat().calcMean();
364                     child.setVisitor(subset);
365                 }
366                 int[] ie = createIE(data);
367                 ILevelCStatistic ps = (ILevelCStatistic) tree.getClusteringStat();
368                 ILevelCHeurStat left = computeCHeurStat((ClusNode) tree.getChild(0), tree, ie, clusters);
369                 ILevelCHeurStat right = computeCHeurStat((ClusNode) tree.getChild(1), tree, ie, clusters);
370                 int nbParLabel = countLabel(root, ps.getClusterID());
371                 int pLabel = -1;
372                 int nbLabels = m_NbClasses;
373                 if (nbParLabel <= 0) {
374                     /* if does not occur elsewhere then it becomes available for reuse */
375                     nbLabels = m_NbClasses - 1;
376                     pLabel = ps.getClusterID();
377                 }
378                 int v1 = tryLabel(left, right, pLabel, nbLabels);
379                 int v2 = tryLabel(right, left, pLabel, nbLabels);
380                 if (v1 == -1 || v2 == -1) {
381                     v1 = tryLabel(left, right, pLabel, nbLabels);
382                     v2 = tryLabel(right, left, pLabel, nbLabels);
383                 }
384                 if ((v1 <= v2 && v1 != -1) || v2 == -1) {
385                     assignLabels(left, right, root, pLabel, nbLabels);
386                 }
387                 else {
388                     assignLabels(right, left, root, pLabel, nbLabels);
389                 }
390                 storeLabels((ClusNode) tree.getChild(0), left);
391                 storeLabels((ClusNode) tree.getChild(1), right);
392                 tree.setVisitor(null);
393             }
394         }
395         else {
396             for (int i = 0; i < nb_c; i++) {
397                 ClusNode child = (ClusNode) tree.getChild(i);
398                 enterBestTest(child, root, clusters);
399             }
400         }
401     }

404     public int freeLabel(ClusNode tree, int ignore) {
405         boolean[] set = new boolean[m_NbClasses];
406         labelSet(tree, set);
407         for (int i = 0; i < set.length; i++) {
408             if (!set[i] && i != ignore)
409                 return i;
410         }
411         if (m_NbClasses < m_MaxNbClasses)
412             return m_NbClasses;
413         else
414             return -1;
415     }

418     public void labelSet(ClusNode tree, boolean[] set) {
419         int nb_c = tree.getNbChildren();
420         if (nb_c == 0) {
421             ILevelCStatistic cs = (ILevelCStatistic) tree.getClusteringStat();
422             if (cs.getClusterID() != -1)
423                 set[cs.getClusterID()] = true;
424         }
425         else {
426             for (int i = 0; i < nb_c; i++) {
427                 ClusNode child = (ClusNode) tree.getChild(i);
428                 labelSet(child, set);
429             }
430         }

434     public int countLabel(ClusNode tree, int label) {
435         int nb_c = tree.getNbChildren();
436         if (nb_c == 0) {
437             ILevelCStatistic cs = (ILevelCStatistic) tree.getClusteringStat();
438             return cs.getClusterID() == label ? 1 : 0;
439         }
440         else {
441             int count = 0;
442             for (int i = 0; i < nb_c; i++) {
443                 ClusNode child = (ClusNode) tree.getChild(i);
444                 count += countLabel(child, label);
445             }
446             return count;
447         }

451     public void iLevelCInduce(ClusNode root) throws ClusException {
452         double ss = root.estimateClusteringSS(m_Scale);
453         int[] clusters = assignAllInstances(root);
454         int violated = countViolatedConstaints(clusters);
455         computeHeuristic(violated, ss);
456         m_BestHeur = Double.POSITIVE_INFINITY;
457         while (true) {
458             /* reset current best values */
459             m_BestTest = null;
460             m_BestLeaf = null;
461             /* try to refine each leaf */
462             tryEachLeaf(root, root, violated, ss, clusters);
463             if (m_BestTest == null) {
464                 /* no improvement possible */
465                 return;
466             }
467             else {
468                 /* refine tree and continue */
469                 enterBestTest(root, root, clusters);
470                 ClusLogger.info("Tree:");
471                 root.printTree();
472                 ss = root.estimateClusteringSS(m_Scale);
473                 clusters = assignAllInstances(root);
474                 violated = countViolatedConstaints(clusters);
475                 double heur = computeHeuristic(violated, ss);
476                 if (Math.abs(heur - m_BestHeur) > 1e-6) { throw new ClusException("Error: heuristic " + heur + " <> " + m_BestHeur); }
477                 ClusLogger.info("CHECK heuristic " + heur + " == " + m_BestHeur + " [OK]");
478             }
479         }
480     }

503     public int[] countViolatedConstaints(RowData data, int[] clusters) {
504         int violated = 0;
505         int violated_internal = 0;
506         int[] ie = createIE(data);
507         ILevelConstraint[] constr = getSubsetConstraints(data);
508         for (int i = 0; i < constr.length; i++) {
509             ILevelConstraint ic = constr[i];
510             int type = ic.getType();
511             int t1 = ic.getT1().getIndex();
512             int t2 = ic.getT2().getIndex();
513             if (type == ILevelConstraint.ILevelCMustLink) {
514                 if (clusters[t1] != clusters[t2])
515                     violated++;
516             }
517             else {
518                 if (clusters[t1] == clusters[t2]) {
519                     violated++;
520                     if (ie[t1] != ILevelCHeurStat.EXT && ie[t2] != ILevelCHeurStat.EXT) {
521                         violated_internal++;
522                     }
523                 }
524             }
525         }
526         int[] result = new int[2];
527         result[0] = violated;
528         result[1] = violated_internal;
529         return result;
530     }

533     public int countViolatedConstaints(int[] clusters) {
534         int violated = 0;
535         for (int i = 0; i < m_Constraints.size(); i++) {
536             ILevelConstraint ic = (ILevelConstraint) m_Constraints.get(i);
537             int type = ic.getType();
538             int t1 = ic.getT1().getIndex();
539             int t2 = ic.getT2().getIndex();
540             if (type == ILevelConstraint.ILevelCMustLink) {
541                 if (clusters[t1] != clusters[t2])
542                     violated++;
543             }
544             else {
545                 if (clusters[t1] == clusters[t2])
546                     violated++;
547             }
548         }
549         return violated;
550     }

553     public ArrayList<ILevelConstraint> createConstraints(RowData data, int nbRows) {
554         ArrayList<ILevelConstraint> constr = new ArrayList<>();
555         ClusAttrType type = getSchema().getAttrType(getSchema().getNbAttributes() - 1);
556         if (type.getAttributeType().equals(AttributeType.Nominal)) {
557             NominalAttrType cls = (NominalAttrType) type;
558             m_MaxNbClasses = cls.getNbValues();
559             int nbConstraints = getSettings().getILevelC().getILevelCNbRandomConstraints();
560             for (int i = 0; i < nbConstraints; i++) {
561                 int t1i = ClusRandom.nextInt(ClusRandom.RANDOM_ALGO_INTERNAL, nbRows);
562                 int t2i = ClusRandom.nextInt(ClusRandom.RANDOM_ALGO_INTERNAL, nbRows);
563                 DataTuple t1 = data.getTuple(t1i);
564                 DataTuple t2 = data.getTuple(t2i);
565                 if (cls.getNominal(t1) == cls.getNominal(t2)) {
566                     constr.add(new ILevelConstraint(t1, t2, ILevelConstraint.ILevelCMustLink));
567                 }
568                 else {
569                     constr.add(new ILevelConstraint(t1, t2, ILevelConstraint.ILevelCCannotLink));
570                 }
571             }
572         }
573         return constr;
574     }

577     @Override
578     public ClusModel induceSingleUnpruned(ClusRun cr) throws ClusException, IOException, InterruptedException {
579         m_NbClasses = 1;
580         ClusRandom.reset(ClusRandom.RANDOM_ALGO_INTERNAL);
581         RowData data = (RowData) cr.getTrainingSet();
582         int nbTrain = data.getNbRows();
583         /* add in test data! */
584         RowData test = cr.getTestSet();
585         if (test != null) {
586             ArrayList<DataTuple> allData = new ArrayList<>();
587             data.addTo(allData);
588             test.addTo(allData);
589             data = new RowData(allData, data.getSchema());
590         }
591         ClusLogger.info("All data: " + data.getNbRows());
592         /* and process it ... */
593         data.addIndices();
594         m_NbTrain = data.getNbRows();
595         m_MinLeafWeight = getSettings().getModel().getMinimalWeight();
596         ArrayList<DataTuple> points = data.toArrayList();
597         /* load constraints from file */
598         if (getSettings().getILevelC().hasILevelCFile()) {
599             String fname = getSettings().getILevelC().getILevelCFile();
600             m_Constraints = ILevelConstraint.loadConstraints(fname, points);
601         }
602         else {
603             /* constraints are only on the training instances */
604             /* reason: nbTrain = number of training instances and training instances come first */
605             m_Constraints = createConstraints(data, nbTrain);
606         }
607         if (getSettings().getILevelC().isILevelCCOPKMeans()) {
608             COPKMeans km = new COPKMeans(m_MaxNbClasses, getStatManager());
609             COPKMeansModel model = null;
610             int sumIter = 0;
611             for (int i = 0; i < 100000; i++) {
612                 model = (COPKMeansModel) km.induce(data, m_Constraints);
613                 sumIter = Math.max(sumIter, model.getIterations());
614                 model.setCSets(i + 1);
615                 model.setAvgIter(sumIter);
616 
617                 if (!model.isIllegal())
618                     return model;
619                 m_Constraints = createConstraints(data, nbTrain);
620             }
621             return model;
622         }
623         else if (getSettings().getILevelC().isILevelCMPCKMeans()) {
624             MPCKMeansWrapper wrap = new MPCKMeansWrapper(getStatManager());
625             return wrap.induce(data, test, m_Constraints, m_MaxNbClasses);
626         }
627         else {
628             /* add derived constraints */
629             DerivedConstraintsComputer comp = new DerivedConstraintsComputer(points, m_Constraints);
630             comp.compute();
631             m_ConstraintsIndex = createConstraintsIndex();
632             ClusLogger.info("Number of instance level constraints: " + m_Constraints.size());
633             /* create initial node */
634             ClusNode root = new ClusNode();
635             root.initClusteringStat(m_StatManager, data);
636             root.initTargetStat(m_StatManager, data);
637             root.setVisitor(data);
638             /* assign root examples to initial cluster */
639             ILevelCStatistic ilevels = (ILevelCStatistic) root.getClusteringStat();
640             ilevels.setClusterID(0);
641             /* initialize scale */
642             m_Scale = (ClusNormalizedAttributeWeights) getStatManager().getClusteringWeights();
643             m_GlobalSS = ilevels.getSVarS(m_Scale);
644             ClusLogger.info("Global SS: " + m_GlobalSS);
645             /* induce tree now */
646             initSelectorAndSplit(root.getClusteringStat());
647             iLevelCInduce(root);
648             root.afterInduce(null);
649             cleanSplit();
650             return root;
651         }
652     }
