23  package si.ijs.kt.clus.main;
24  
25  import si.ijs.kt.clus.util.ClusLogger;
26  import si.ijs.kt.clus.util.ResourceInfo;
27  import si.ijs.kt.clus.util.jeans.io.MyFile;
28  
29  
30  public class ClusStat {
31  
32      public static long m_PrevTime;
33  
34      public static long m_InitialMemory;
35      public static long m_LoadedMemory;
36      public static long m_FinalMemory;
37  
38      public static long m_TimeTest;
39      public static long m_TimeSplit;
40      public static long m_TimeStat;
41      public static long m_TimeSort;
42      public static long m_TimeHeur;
43  
44      public static double m_TTimeTest;
45      public static double m_TTimeSplit;
46      public static double m_TTimeStat;
47      public static double m_TTimeSort;
48      public static double m_TTimeHeur;
49  
50      public static double m_TTimeTotal;
51  
52  
53      public static void initTime() {
54          m_PrevTime = ResourceInfo.getCPUTime();
55      }
56  
57  
58      public static void deltaStat() {
59          long now = ResourceInfo.getCPUTime();
60          m_TimeStat += now - m_PrevTime;
61          m_PrevTime = now;
62      }
63  
64  
65      public static void deltaTest() {
66          long now = ResourceInfo.getCPUTime();
67          m_TimeTest += now - m_PrevTime;
68          m_PrevTime = now;
69      }
70  
71  
72      public static void deltaSplit() {
73          long now = ResourceInfo.getCPUTime();
74          m_TimeSplit += now - m_PrevTime;
75          m_PrevTime = now;
76      }
77  
78  
79      public static void deltaSort() {
80          long now = ResourceInfo.getCPUTime();
81          m_TimeSort += now - m_PrevTime;
82          m_PrevTime = now;
83      }
84  
85  
86      public static void deltaHeur() {
87          long now = ResourceInfo.getCPUTime();
88          m_TimeHeur += now - m_PrevTime;
89          m_PrevTime = now;
90      }
91  
92  
93      public static void resetTimes() {
94          m_TimeTest = 0;
95          m_TimeSplit = 0;
96          m_TimeStat = 0;
97          m_TimeSort = 0;
98          m_TimeHeur = 0;
99      }
100 
101 
102     public static double addToTotal(long value, int div) {
103         double mean = (double) value / div;
104         m_TTimeTotal += mean;
105         return mean;
106     }
107 
108 
109     public static void addTimes(int div) {
110         m_TTimeTest += (double) m_TimeTest / div;
111         m_TTimeSplit += (double) m_TimeSplit / div;
112         m_TTimeStat += (double) m_TimeStat / div;
113         m_TTimeSort += (double) m_TimeSort / div;
114         m_TTimeHeur += (double) m_TimeHeur / div;
115     }
116 
117 
118     public static void updateMaxMemory() {
119         m_FinalMemory = Math.max(m_FinalMemory, ResourceInfo.getMemorySize());
120     }
121 
122 
123     public static void show() {
124         double sum = m_TTimeStat + m_TTimeTest + m_TTimeSplit + m_TTimeSort + m_TTimeHeur;
125 
126         ClusLogger.info("Mem usage (KB) [initial, loaded, max]: [" + m_InitialMemory + "," + m_LoadedMemory + "," + m_FinalMemory + "]");
127         ClusLogger.info("Total estimate: " + m_TTimeTotal);
128         ClusLogger.info("Total induction time: " + sum);
129         ClusLogger.info("Time for stats: " + m_TTimeStat);
130         ClusLogger.info("Time for evaluating: " + m_TTimeTest);
131         ClusLogger.info("Time for splitting: " + m_TTimeSplit);
132         ClusLogger.info("Time for sorting: " + m_TTimeSort);
133         ClusLogger.info("Time for heuristics: " + m_TTimeHeur);
134 
135         MyFile file = new MyFile("stats.txt");
136         file.log("Mem usage (KB) [initial, loaded, max]: [" + m_InitialMemory + "," + m_LoadedMemory + "," + m_FinalMemory + "]");
137         file.log("Total estimate: " + m_TTimeTotal);
138         file.log("Total induction time: " + sum);
139         file.log("Time for stats: " + m_TTimeStat);
140         file.log("Time for evaluating: " + m_TTimeTest);
141         file.log("Time for splitting: " + m_TTimeSplit);
142         file.log("Time for sorting: " + m_TTimeSort);
143         file.log("Time for heuristics: " + m_TTimeHeur);
144         file.close();
145     }
146 
147 }
