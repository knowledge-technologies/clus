4   import si.ijs.kt.clus.statistic.ClusStatistic;
5   
6   
7   public class ClusStopCriterionMinNbExamples implements ClusStopCriterion {
8   
9       protected int m_MinExamples;
10  
11  
12      public ClusStopCriterionMinNbExamples(int minExamples) {
13          m_MinExamples = minExamples;
14      }
15  
16  
17      @Override
18      public boolean stopCriterion(ClusStatistic tstat, ClusStatistic pstat, ClusStatistic missing) {
19          int n_pos = pstat.getNbExamples();
20          int n_neg = tstat.getNbExamples() - n_pos;
21          return n_pos < m_MinExamples || n_neg < m_MinExamples;
22      }
23  
24  
25      @Override
26      public boolean stopCriterion(ClusStatistic tstat, ClusStatistic[] pstat, int nbsplit) {
27          for (int i = 0; i < nbsplit; i++) {
28              if (pstat[i].getNbExamples() < m_MinExamples) { return true; }
29          }
30          return false;
31      }
32  }
