13  public abstract class SettingsBase implements Serializable {
14  
15      private static final long serialVersionUID = Settings.SERIAL_VERSION_ID;
16      /***********************************************************************
17       * Simple constants *
18       ***********************************************************************/
19      public final static String DEFAULT = "Default";
20      public final static String NONE = "None";
21      public final static String[] NONELIST = { "None" };
22      public final static String[] EMPTY = {};
23      public final static String[] INFINITY = { "Infinity" };
24      public final static String INFINITY_STRING = "Infinity";
25      public final static int INFINITY_VALUE = 0;
26      public final static double[] FOUR_ONES = { 1.0, 1.0, 1.0, 1.0 };
27      public final static Charset CHARSET = StandardCharsets.UTF_8;
28      
29      public enum YesNo {
30          Yes, No;
31      };
32  
33      protected int m_Position = 1; // position in the INI file
34  
35      protected INIFileSection m_Section;
36  
37  
38      public SettingsBase(int position, String name) {
39          m_Position = position;
40  
41          m_Section = new INIFileSection(name);
42      }
43  
44  
45      /**
46       * Method should handle the INIFileSection initialization.
47       */
48      public abstract void create();
49  
50  
51      /**
52       * Method should handle initialization of named values.
53       * E.g.: Use si.ijs.kt.clus.util.jeans.io.ini.INIFileInt.setNamedValue(int, String) and similar methods.
54       * This method is not used by all settings sections. It is not abstract, so that it does not have to be implemented
55       * by all those
56       * sections that do not need it.
57       */
58      public void initNamedValues() {
59      }
60  
61  
62      /**
63       * This method should be overriden to check for inconsistencies in the section settings.
64       * It is called by <code>si.ijs.kt.clus.main.settings.Settings.validateSettingsCompatibility()</code> method.
65       * If several sections are needed to determine if inconsistencies exist, implement the logic in
66       * <code>si.ijs.kt.clus.main.settings.Settings.validateSettingsCombined()</code>.
67       */
68      public List<String> validateSettingsInternal() {
69          return null;
70      }
71  
72  
73      /**
74       * Returns the position of this section in the INI file
75       * 
76       * @return Position
77       */
78      public int getPosition() {
79          return m_Position;
80      }
81  
82  
83      /** Returns INI section */
84      public INIFileSection getSection() {
85          return m_Section;
86      }
87  
88  
89      /** Convenience method for denoting illegal settings */
90      public static String formatInvalid(INIFileEntry settingEntry, Object settingValue) {
91          return formatInvalid(settingEntry, settingValue, null);
92      }
93  
94  
95      /** Convenience method for denoting illegal settings */
96      public static String formatInvalid(INIFileEntry settingEntry, Object settingValue, String additionalMessage) {
97          return String.format("%s = %s is invalid %s", settingEntry.getName(), settingValue, additionalMessage == null ? "" : "=> Additional info: " + additionalMessage);
98      }
99  }
