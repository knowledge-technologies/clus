46      public final static boolean CREATE_TRAIN_TUNE_TEST_SPLIT = true;
47  
48  
49      public void convert(String input, String output, double minfreq) throws Exception {
50          Clus clus = new Clus();
51          String appname = FileUtil.getName(input) + ".s";
52          clus.initializeAddOn(appname);
53          ClusStatManager mgr = clus.getStatManager();
54          Settings sett = clus.getSettings();
55          RowData data = clus.getData();
56          if (CREATE_TRAIN_TUNE_TEST_SPLIT) {
57              ClusRun run = clus.partitionData();
58              ClusStatistic[] stats = new ClusStatistic[1];
59              stats[0] = mgr.createClusteringStat();
60              data.calcTotalStats(stats);
61              if (!sett.getData().isNullTestFile()) {
62                  ClusLogger.info("Loading: " + sett.getData().getTestFile());
63                  if (minfreq != 0.0) {
64                      RowData test = run.getTestSet();
65                      test.calcTotalStats(stats);
66                  }
67                  else {
68                      clus.updateStatistic(sett.getData().getTestFile(), stats);
69                  }
70              }
71              if (!sett.getData().isNullPruneFile()) {
72                  ClusLogger.info("Loading: " + sett.getData().getPruneFile());
73                  if (minfreq != 0.0) {
74                      RowData tune = (RowData) run.getPruneSet();
75                      tune.calcTotalStats(stats);
76                  }
77                  else {
78                      clus.updateStatistic(sett.getData().getPruneFile(), stats);
79                  }
80              }
81              ClusStatistic.calcMeans(stats);
82              WHTDStatistic stat = (WHTDStatistic) stats[0];
83              stat.showRootInfo();
84              ClassHierarchy hier = mgr.getHier();
85              boolean[] removed = hier.removeInfrequentClasses(stat, minfreq);
86              if (minfreq != 0.0) {
87                  ClassesAttrType type = hier.getType();
88                  removeLabelsFromData((RowData) run.getTrainingSet(), type, removed);
89                  if (!sett.getData().isNullTestFile())
90                      removeLabelsFromData(run.getTestSet(), type, removed);
91                  if (!sett.getData().isNullPruneFile())
92                      removeLabelsFromData((RowData) run.getPruneSet(), type, removed);
93              }
94              hier.initialize();
95              if (minfreq != 0.0) {
96                  addIntermediateLabels((RowData) run.getTrainingSet(), hier);
97                  if (!sett.getData().isNullTestFile())
98                      addIntermediateLabels(run.getTestSet(), hier);
99                  if (!sett.getData().isNullPruneFile())
100                     addIntermediateLabels((RowData) run.getPruneSet(), hier);
101             }
102             hier.showSummary();
103             RowData train = (RowData) run.getTrainingSet();
104             ARFFFile.writeArff(output + ".train.arff", train);
105             if (!sett.getData().isNullTestFile()) {
106                 RowData test = run.getTestSet();
107                 ARFFFile.writeArff(output + ".test.arff", test);
108             }
109             if (!sett.getData().isNullPruneFile()) {
110                 RowData tune = (RowData) run.getPruneSet();
111                 ARFFFile.writeArff(output + ".valid.arff", tune);
112             }
113         }
114         else {
115             ARFFFile.writeArff(output, data);
116         }
117     }

119 
120     public void removeLabelsFromData(RowData data, ClassesAttrType type, boolean[] removed) {
121         for (int i = 0; i < data.getNbRows(); i++) {
122             ClassesTuple tuple = (ClassesTuple) data.getTuple(i).getObjVal(type.getArrayIndex());
123             tuple.removeLabels(removed);
124         }
125     }
126 
127 
128     public void addIntermediateLabels(RowData data, ClassHierarchy hier) {
129         ClassesAttrType type = hier.getType();
130         ArrayList scratch = new ArrayList();
131         boolean[] alllabels = new boolean[hier.getTotal()];
132         ArrayList leftdata = new ArrayList();
133         for (int i = 0; i < data.getNbRows(); i++) {
134             DataTuple tuple = data.getTuple(i);
135             scratch.clear();
136             Arrays.fill(alllabels, false);
137             ClassesTuple ct = (ClassesTuple) tuple.getObjVal(type.getArrayIndex());
138             ct.addIntermediateElems(hier, alllabels, scratch);
139             if (ct.getNbClasses() > 0) {
140                 leftdata.add(tuple);
141             }
142         }
143         data.setFromList(leftdata);
144     }

147     public static void main(String[] args) {
148         if (args.length != 2 && args.length != 4) {
149             ClusLogger.info("Usage: HMCConvertDAGData [-minfreq f] input.arff output.arff");
150             System.exit(0);
151         }
152         double minfreq = 0;
153         String input = args[0];
154         String output = args[1];
155         if (args[0].equals("-minfreq")) {
156             minfreq = Double.parseDouble(args[1]) / 100.0;
157             input = args[2];
158             output = args[3];
159         }
160         HMCConvertDAGData cnv = new HMCConvertDAGData();
161         try {
162             cnv.convert(input, output, minfreq);
163         }
164         catch (Exception e) {
165             System.err.println("Error: " + e);
166             e.printStackTrace();
167         }
168     }
