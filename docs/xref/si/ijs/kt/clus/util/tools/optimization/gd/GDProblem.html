275 	/**
276 	 * Estimate of expected value of covariance for given prediction. The covariance
277 	 * of this prediction with its true value is returned.
278 	 */
279 	private double computePredVsTrueValueCov(int iPred) {
280 		double[] covs = new double[getNbOfTargets()];
281 		int nbOfTargets = getNbOfTargets();
282 
283 		BitSet b = new BitSet(nbOfTargets);
284 
285 		for (int iTarget = 0; iTarget < nbOfTargets; iTarget++) {
286 			for (int iInstance = 0; iInstance < getNbOfInstances(); iInstance++) {
287 				double trueVal = getTrueValue(iInstance, iTarget);
288 				if (isValidValue(trueVal)) { // Not a valid true value, rare but happens. Can happen for linear terms.
289 					double pred = predictWithRule(iPred, iInstance, iTarget);
290 					if (!Double.isNaN(pred)) {
291 						covs[iTarget] += (trueVal * pred);
292 						b.set(iTarget);
293 					}
294 				}
295 			}
296 
297 			covs[iTarget] /= getNbOfInstances();
298 			if (getSettings().getRules().isOptNormalization()) {
299 				covs[iTarget] /= getNormFactor(iTarget);
300 			}
301 		}
302 
303 		double avgCov = 0;
304 
305 		for (int iTarget = 0; iTarget < nbOfTargets; iTarget++) {
306 			// avgCov += covs[iTarget] / nbOfTargets;
307 			avgCov += covs[iTarget] / b.cardinality();
308 		}
309 		return avgCov;
310 	}

316 	final protected double getWeightCov(int iFirst, int iSecond) {
317 		int min = Math.min(iFirst, iSecond);
318 		int max = Math.max(iFirst, iSecond);
319 		// if (Double.isNaN(m_covariances[min][max][0]))
320 		if (Double.isNaN(m_covariances[min][max])) {
321 			throw new Error("Asked covariance not yet computed. Something wrong in the covariances in GDProbl.");
322 		}
323 
324 		return m_covariances[min][max];
325 	}

327 	/**
328 	 * Compute the covariances for this weight/prediction dimension. This means that
329 	 * we compute all the pairs where this takes part. Because for covariance
330 	 * cov(a,b) = cov(b,a) compute only one of them.
331 	 * 
332 	 * @param dimension The dimension for which the covariances are computed
333 	 */
334 	private void computeWeightCov(int dimension) {
335 		// Because of symmetry cov(dimension, b) is already computed if for some earlier
336 		// phase b was dimension
337 		// Thus if covariances for b are computed, this does not have to be computed
338 		// anymore.
339 
340 		// The weights with index lower than this
341 		for (int iMin = 0; iMin < dimension; iMin++) {
342 			// If the covariances for the other part are already computed, this covariance
343 			// is already
344 			// computed also. Thus skip this covariance.
345 			if (!m_isCovComputed[iMin]) {
346 				// if (!Double.isNaN(m_covariances[iMin][dimension][0]))
347 				// System.err.println("WARNING: Covariances are recalculated, waste of
348 				// computation!");
349 				m_covariances[iMin][dimension] = computeCovFor2Preds(iMin, dimension);
350 			}
351 		}
352 
353 		m_covariances[dimension][dimension] = computeCovFor2Preds(dimension, dimension);
354 		for (int iMax = dimension + 1; iMax < getNumVar(); iMax++) {
355 			if (!m_isCovComputed[iMax]) {
356 				// if (!Double.isNaN(m_covariances[iMax][dimension][0]))
357 				// System.err.println("WARNING: Covariances are recalculated, waste of
358 				// computation!");
359 				m_covariances[dimension][iMax] = computeCovFor2Preds(dimension, iMax);
360 			}
361 		}
362 	}

365 	 * Compute covariance between two predictions (base learner predictions). This
366 	 * is a help function for computeWeightCov NOTE: Second parameter index HAS to
367 	 * be greater or equal to first
368 	 * 
369 	 * @param iPrevious Base learner index
370 	 * @param iLatter   Base learner index. NOTE: iLatter {@literal >=} iPrevious
371 	 * 
372 	 */
373 	// private double[] computeCovFor2Preds(int iFirstRule, int iSecondRule) {
374 	private double computeCovFor2Preds(int iPrevious, int iLatter) {
375 		// double eka = 0;
376 		// If one of them is linear term, it has to be the later one
377 		if (isRuleTerm(iLatter)) {
378 			// Covariance for 2 rule predictions
379 			return computeCovFor2Rules(iPrevious, iLatter);
380 		} else if (isRuleTerm(iPrevious)) {
381 			// Covariance for rule and linear term
382 			return computeCovForRuleAndLin(iPrevious, iLatter);
383 		} else {
384 			// Covariance for 2 linear terms
385 			return computeCovFor2Lin(iPrevious, iLatter);
386 		}
387 	}

389 	/**
390 	 * For two linear terms.
391 	 * 
392 	 * @param iLatter   Rule index. NOTE: iLatter {@literal >}= iPrevious
393 	 * 
394 	 */
395 	private double computeCovFor2Lin(int iPrevious, int iLatter) {
396 		int nbOfInstances = getNbOfInstances();
397 		int nbOfTargets = getNbOfTargets();
398 
399 		// Go through only those dimensions that linear term is not always zero for
400 		int iTarget = getLinTargetDim(iPrevious);
401 
402 		if (iTarget != getLinTargetDim(iLatter)) {
403 			return 0;
404 		}
405 
406 		// Compute value
407 		double avgCov = 0;
408 
409 		for (int iInstance = 0; iInstance < nbOfInstances; iInstance++) {
410 			avgCov += predictWithRule(iPrevious, iInstance, iTarget) * predictWithRule(iLatter, iInstance, iTarget);
411 		}
412 		avgCov /= (nbOfTargets * getNbOfInstances());
413 
414 		if (getSettings().getRules().isOptNormalization()) {
415 			avgCov /= getNormFactor(iTarget);
416 		}
417 
418 		// Store the value to all the places where this descriptive attribute is used.
419 		// TODO how to mark that these covariances are computed already!!!!
420 		// In practice takes so little time, that does not matter?
421 		return avgCov;
422 	}

424 	/**
425 	 * For rule and linear term we skip the instances rule does not cover.
426 	 * 
427 	 * @param iRule   Rule index.
428 	 * @param iLinear Linear term index. NOTE: iLinear {@literal >=} iRule
429 	 * 
430 	 */
431 	private double computeCovForRuleAndLin(int iRule, int iLinear) {
432 		// Go through only those dimensions that linear term is not always zero for
433 		int iTarget = getLinTargetDim(iLinear);
434 
435 		double predRule = getPredictionsWhenCovered(iRule, 0, iTarget);
436 
437 		// if rule does not cover target iTarget, then covariance is zero
438 		if (Double.isNaN(predRule)) {
439 			return 0d;
440 		}
441 
442 		int nbOfTargets = getRuleEnabledTargets(iRule);
443 
444 		double avgCov = 0;
445 
446 		// Go through the covered examples always jumping to next covered
447 		for (int iInstance = getRuleNextCovered(iRule, 0); iInstance >= 0; iInstance = getRuleNextCovered(iRule,
448 				iInstance + 1)) {
449 			avgCov += getPredictionsWhenCovered(iLinear, iInstance, iTarget);
450 		}
451 
452 		avgCov *= predRule;
453 
454 		avgCov /= (nbOfTargets * getNbOfInstances());
455 
456 		if (getSettings().getRules().isOptNormalization()) {
457 			avgCov /= getNormFactor(iTarget);
458 		}
459 
460 		if (Double.isNaN(avgCov)) {
461 			System.err.println("TEST");
462 		}
463 		return avgCov;
464 	}

467

478 	/**
479 	 * For rules the prediction is always the same. We first count: how many
480 	 * instances both rules cover (prediction != 0). Then the covariance is
481 	 * pred1*pred2*([nb both rules cover]/[nb of instances])
482 	 * 
483 	 * @param iPrevious
484 	 * @param iLatter   Rule index. NOTE: iLatter {@literal >=} iPrevious
485 	 * 
486 	 */
487 	private double computeCovFor2Rules(int iPrevious, int iLatter) {
488 		BitSet prev = (BitSet) (getRuleCovers(iPrevious).clone());
489 		BitSet latter = getRuleCovers(iLatter);
490 		prev.and(latter); // prev is now a AND bitset of both of these
491 
492 		int nbOfTargets = getNbOfTargets();
493 		double p1, p2, cov, avgCov = 0d, omitted = 0d;
494 
495 		for (int iTarget = 0; iTarget < nbOfTargets; iTarget++) {
496 			cov = 0d;
497 
498 			p1 = getPredictionsWhenCovered(iPrevious, 0, iTarget);
499 			p2 = getPredictionsWhenCovered(iLatter, 0, iTarget);
500 
501 			// if p1/p2=NaN then this is ROS SubsetAveraging
502 			if (!Double.isNaN(p1) && !Double.isNaN(p2)) {
503 				cov += p1 * p2;
504 
505 				if (getSettings().getRules().isOptNormalization()) {
506 					cov /= getNormFactor(iTarget);
507 				}
508 				avgCov += cov;
509 			} else {
510 				omitted++;
511 			}
512 		}
513 
514 		// if rules predict a disjunct set of targets, covariance is zero
515 		if (omitted == nbOfTargets) {
516 			return 0d;
517 		}
518 
519 		avgCov /= (nbOfTargets - omitted);
520 
521 		avgCov *= (double) prev.cardinality() / getNbOfInstances();
522 
523 		return avgCov;
524 	}

555 	/** Compute gradient for the given weight dimension */
556 	protected double getGradient(int iWeightDim, ArrayList<Double> weights) {
557 
558 		double gradient = 0;
559 		switch (getSettings().getRules().getOptGDLossFunction()) {
560 		case ZeroOneError:
561 			// gradient = loss01(trueValue, prediction);
562 			// break;
563 		case Huber:
564 		case RRMSE:
565 			// gradient = lossHuber(trueValue, prediction);
566 			// break;
567 			try {
568 				throw new Exception("0/1 or Huber loss function not yet implemented for Gradient descent.\n"
569 						+ "Using squared loss.\n");
570 			} catch (Exception s) {
571 				s.printStackTrace();
572 			} // TODO Huber and alpha computing
573 				// Default case
574 		case Squared:
575 		default:
576 			gradient = gradientSquared(iWeightDim, weights);
577 			break;
578 		}
579 
580 		return gradient;
581 	}

583 	/**
584 	 * Squared loss gradient. p. 18 in Friedman {@literal &} Popescu, 2004
585 	 * 
586 	 * @param iGradWeightDim Weight dimension for which the gradient is computed.
587 	 */
588 	private double gradientSquared(int iGradWeightDim, ArrayList<Double> weights) {
589 
590 		double gradient = getCovForPrediction(iGradWeightDim);
591 
592 		for (int iWeight = 0; iWeight < getNumVar(); iWeight++) {
593 			if (m_isWeightNonZero[iWeight]) {
594 				gradient -= weights.get(iWeight).doubleValue() * getWeightCov(iWeight, iGradWeightDim);
595 			}
596 		}
597 
598 		return gradient;
599 	}

624 	/** Recomputation of gradients for least squares loss function */
625 	public void modifyGradientSquared(int[] iChangedWeights) {
626 
627 		// New gradients are computed with the old gradients.
628 		// Only the changed gradients are stored here
629 		// However since we use affective gradients which are not YET changed,
630 		// we can use directly them.
631 
632 		// Could store only the changes to the gradients and do the actual change
633 		// afterwards in another for loop
634 		double[] oldGradsOfChanged = new double[iChangedWeights.length];
635 
636 		for (int iCopy = 0; iCopy < iChangedWeights.length; iCopy++) {
637 			oldGradsOfChanged[iCopy] = m_gradients[iChangedWeights[iCopy]];
638 		}
639 
640 		// The next few lines take about 80% of processor time.
641 
642 		/**
643 		 * If we reach linear terms with the first loop, we can skip most of the terms
644 		 * in the second loop (because cov is 0)
645 		 */
646 		boolean firstLinearTermReached = false;
647 		int nbOfTargs = getNbOfTargets();
648 		int nbOfChanged = iChangedWeights.length; // does this make more effective?
649 		int nbOfGrads = m_gradients.length;
650 
651 		// Index over the other gradients that are affecting (THE WEIGHTS THAT ALTERED)
652 		for (int iiAffecting = 0; iiAffecting < nbOfChanged; iiAffecting++) {
653 			if (!firstLinearTermReached && !isRuleTerm(iChangedWeights[iiAffecting]))
654 				firstLinearTermReached = true;
655 
656 			boolean secondLinearTermReached = false;
657 			double stepAmount = m_stepSize * oldGradsOfChanged[iiAffecting];
658 			// Index over the gradient we are changing (ALL GRADIENTS)
659 			for (int iWeightChange = 0; iWeightChange < nbOfGrads; iWeightChange++) {
660 				m_gradients[iWeightChange] -= getWeightCov(iChangedWeights[iiAffecting], iWeightChange) * stepAmount;
661 
662 				if (firstLinearTermReached) {
663 					if (secondLinearTermReached) {
664 						// Skip to only those linear terms, for which
665 						// cov(iWeightChange,iChangedWeights[iiAffecting])!=
666 						// 0
667 						iWeightChange += nbOfTargs - 1;
668 					} else if (!isRuleTerm(iWeightChange)) {
669 						// Jump to the first linear term that has covariance nonzero with iiAffecting.
670 						// This is linear term number linTargetDim(). However this is already the first
671 						// linear term and
672 						// there is already
673 						// ++ because of for loop -> -2
674 						iWeightChange += (getLinTargetDim(iChangedWeights[iiAffecting]) + nbOfTargs - 1) % nbOfTargs;
675 
676 						secondLinearTermReached = true;
677 					}
678 				}
679 
680 			}
681 		}
682 	}
683 
684 	/**
685 	 * Return the gradients with maximum absolute value. For the weights we want to
686 	 * change The function is assuming (if max allowed rule nb is set) that index 0
687 	 * of gradients includes the "default rule" which is not counted.
688 	 * 
689 	 * @param nbOfIterations Used for detecting if weight is banned (not usually
690 	 *                       used)
691 	 */
692 	public int[] getMaxGradients(int nbOfIterations) {
693 		/** Maximum number of nonzero elements. Can we add more? */
694 		int maxElements = getSettings().getRules().getOptGDMaxNbWeights();
695 		boolean maxNbOfWeightReached = false;
696 		if (maxElements > 0 && m_nbOfNonZeroRules >= maxElements) {
697 			// If maximum number of nonzero elements is reached,
698 			// search for the biggest one among the nonzero weights
699 
700 			maxNbOfWeightReached = true;
701 		}
702 
703 		double maxGrad = 0; // Maximum gradient
704 		for (int iGrad = 0; iGrad < m_gradients.length; iGrad++) {
705 			if (m_bannedWeights != null && m_bannedWeights[iGrad] > nbOfIterations) {
706 				// The weight is banned
707 				continue;
708 			}
709 			// We choose the gradient if max nb of weights is reached only if it is
710 			// already nonzero
711 			if (Math.abs(m_gradients[iGrad]) > maxGrad
712 					&& (!maxNbOfWeightReached || m_isWeightNonZero[iGrad] || iGrad == 0))
713 				maxGrad = Math.abs(m_gradients[iGrad]);
714 		}
715 
716 		ArrayList<Integer> iMaxGradients = new ArrayList<Integer>();
717 
718 		// The least allowed item.
719 		double minAllowed = getSettings().getRules().getOptGDGradTreshold() * maxGrad;
720 
721 		// Copy all the items that are greater to a returned array
722 		for (int iCopy = 0; iCopy < m_gradients.length; iCopy++) {
723 			if (m_bannedWeights != null && m_bannedWeights[iCopy] > nbOfIterations) {
724 				// The weight is banned
725 				continue;
726 			}
727 			if (Math.abs(m_gradients[iCopy]) >= minAllowed && (!maxNbOfWeightReached || m_isWeightNonZero[iCopy])
728 					|| iCopy == 0) {
729 				iMaxGradients.add(iCopy);
730 				// If the treshold is 1, we only want to change one dimension at time
731 				// Default rule is not counted
732 				if (getSettings().getRules().getOptGDGradTreshold() == 1.0 && iCopy != 0)
733 					break;
734 			}
735 		}
736 
737 		// If we have maximum amount of rules and treshold value is not 1, we may be
738 		// returning too many gradients and thus having too many rules. Thus select
739 		// only the ones needed.
740 		// However if maximum number of nonzero weights is already reached, we can't
741 		// take too much of these.
742 		if (maxElements > 0 && !maxNbOfWeightReached && getSettings().getRules().getOptGDGradTreshold() < 1.0) {
743 
744 			// Gradients that are already nonzero. (count also default rule, because it is
745 			// not counted in maxgradients)
746 			int nbOfOldGrads = 0;
747 
748 			// Count how many old ones we have in the current gradients
749 			for (int iGrad = 0; iGrad < iMaxGradients.size(); iGrad++) {
750 				if (m_isWeightNonZero[iMaxGradients.get(iGrad)] || iMaxGradients.get(iGrad) == 0) {
751 					nbOfOldGrads++;
752 				}
753 			}
754 
755 			// if number of new rules in gradients is greater than maximum number of
756 			// allowed elements, we have to get rid of some of them.
757 			int nbOfAllowedNewGradients = maxElements - m_nbOfNonZeroRules;
758 			if (nbOfAllowedNewGradients < iMaxGradients.size() - nbOfOldGrads) {
759 
760 				// ArrayList is slow for inserts in the middle.
761 				// LinkedList should be fast for iterating thru and inserting.
762 				LinkedList<Integer> iAllowedNewMaxGradients = new LinkedList<Integer>();
763 
764 				for (int iGrad = 0; iGrad < iMaxGradients.size(); iGrad++) {
765 
766 					// If the gradient is new.
767 					if (!m_isWeightNonZero[iMaxGradients.get(iGrad)] && iMaxGradients.get(iGrad) != 0) {
768 
769 						// Insertion sort to iAllowed by the greatness of gradient
770 						ListIterator<Integer> iAllowed = iAllowedNewMaxGradients.listIterator();
771 						while (iAllowed.hasNext()) {
772 							// If we already went too far
773 							if (Math.abs(m_gradients[iAllowed.next()]) < Math
774 									.abs(m_gradients[iMaxGradients.get(iGrad)])) {
775 								iAllowed.previous();
776 								break;
777 							}
778 						}
779 
780 						// We should now be on the insertion place
781 						iAllowed.add(iMaxGradients.get(iGrad));
782 						iMaxGradients.remove(iGrad);
783 						iGrad--;
784 
785 						// To keep the list smaller
786 						if (iAllowedNewMaxGradients.size() > nbOfAllowedNewGradients) {
787 							iAllowedNewMaxGradients.removeLast();
788 						}
789 					}
790 				}
791 
792 				// now we have in sorted order gradients in iAllowedNewMaxGradients. Let us add
793 				// only so much that is allowed
794 				ListIterator<Integer> iList = iAllowedNewMaxGradients.listIterator();
795 
796 				for (int addedElements = 0; addedElements < nbOfAllowedNewGradients; addedElements++) {
797 					iMaxGradients.add(iList.next());
798 				}
799 
800 			}
801 
802 		}
803 
804 		// Efficient enough
805 		int[] iMaxGradientsArray = new int[iMaxGradients.size()];
806 		for (int iCopy = 0; iCopy < iMaxGradients.size(); iCopy++) {
807 			iMaxGradientsArray[iCopy] = iMaxGradients.get(iCopy);
808 		}
809 		return iMaxGradientsArray;
810 	}

875 		double newFitness = m_earlyStopProbl.calcFitness(weights, m_ruleSet);
876 
877 		if (newFitness < m_minFitness) {
878 			m_minFitness = newFitness;
879 			// Copy the weights
880 			for (int iWeight = 0; iWeight < weights.size(); iWeight++) {
881 				m_minFitWeights.set(iWeight, weights.get(iWeight).doubleValue());

883 		}
884

886

888 			stop = true;
889 			ClusLogger.fine("GD: Independent test set error increase detected - overfitting.");
890 		}
891 
892 		return stop;
893 	}

903 	/**
904 	 * Return a maximum tree depth based on RulefFit (Friedman, Popescu. 2005)
905 	 * function used. Function is Pr(nbOfLeaves) =
906 	 * exp(-nbOfLeaves/(avgDepth/2))/(avgDepth-2).
907 	 * 
908 	 * @param unifRand Random uniform number on which the number is based on.
909 	 * @param avgDepth Average depth (L in RuleFit).
910 	 * @return Random depth
911 	 */
912 	public static int randDepthWighExponentialDistribution(double unifRand, int avgDepth) {
913 		// Friedman is computing the number of terminal leaves with exponential
914 		// distribution
915 		// To get the limit depth for this we need to compute
916 		// depth = ceil(log(2+floor(-(L-2)/(lg(L-2))lg(unifRand)))) +1
917 		// Here L is the average and 2+floor(-(L-2)/(lg(L-2))lg(unifRand)) is the number
918 		// of terminal leaves
919 		// However in Clus depth seems to be so that only root is on depth 0, and depth
920 		// 1 has two leaves. Thus
921 		// we leave the last +1 out and compute
922 		// ceil(log(2+floor(-(L-2)/(lg(L-2))lg(unifRand))))
923 		// Also we do not take the floor for computing amount of terminal nodes because
924 		// this reduces the average
925 		// depth too much. Without it the average depth for value 3 is in reality 2.7
926 		int maxDepths = 0;
927 
928 		if (unifRand == 0.0) {
929 			// This means that the limit depth should be infinite. Can't take logarithm of
930 			// this
931 			maxDepths = -1;
932 		} else {
933 			int avgNbLeaves = (int) (Math.pow(2, avgDepth)); // The root depth is 0, thus not avgDepth-1.
934 			// int terminalNodes = (int)
935 			// (2+Math.floor((double)(2-avgNbLeaves)/Math.log(avgNbLeaves-2)*Math.log(unifRand)));
936 			double terminalNodes = 2 + (2 - avgNbLeaves) / Math.log(avgNbLeaves - 2) * Math.log(unifRand);
937 			maxDepths = (int) Math.ceil(Math.log(terminalNodes) / Math.log(2.0)); // Binary logarithm. Root
938 		}
939 		// turha.add(new Integer(maxDepths));
940 		return maxDepths;
941 	}
