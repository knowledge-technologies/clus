56

64          return m_ILevelCMPCKMeans.getValue();
65      }
66  
67  
68      @Override
69      public void create() {
70          m_Section.addNode(m_ILevelCAlpha = new INIFileDouble("Alpha", 0.5));
71          m_Section.addNode(m_ILevelCFile = new INIFileString("File", NONE));
72          m_Section.addNode(m_ILevelNbRandomConstr = new INIFileInt("NbRandomConstraints", 0));
73          m_Section.addNode(m_ILevelCCOPKMeans = new INIFileBool("RunCOPKMeans", false));
74          m_Section.addNode(m_ILevelCMPCKMeans = new INIFileBool("RunMPCKMeans", false));
75          m_Section.setEnabled(false);
76      }
77  }
