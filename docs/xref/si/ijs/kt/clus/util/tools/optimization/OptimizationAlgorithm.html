36  /**
37   * Abstract super class for optimization of weights of base learners.
38   * 
39   * @author Timo Aho
40   */
41  public abstract class OptimizationAlgorithm {
42  
43      private ClusStatManager m_StatMgr;
44  
45  
46      /**
47       * Constructor for classification and regression optimization.
48       * 
49       * @param stat_mgr
50       *        Statistics
51       *        The true values and predictions for the instances. These are used by OptimProbl.
52       *        The optimization procedure is based on this data information
53       */
54      public OptimizationAlgorithm(ClusStatManager stat_mgr) {
55          m_StatMgr = stat_mgr;
56          // m_Probl = new DeProbl(stat_mgr, dataInformation);
57      }
58  
59  
60      /**
61       * Start the actual optimization.
62       * 
63       * @return The weights for the base learners.
64       */
65      abstract public ArrayList<Double> optimize();
66  
67  
68      final protected Settings getSettings() {
69          return m_StatMgr.getSettings();
70      }
71  
72  
73      public void postProcess(ClusRuleSet rset) {
74      }
75  
76  }
