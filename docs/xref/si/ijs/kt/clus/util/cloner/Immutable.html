11  /**
12   * marks the specific class as immutable and the cloner avoids cloning it
13   * 
14   * @author kostantinos.kougios
15   *
16   *         24 Mar 2011
17   */
18  @Target(TYPE)
19  @Retention(RUNTIME)
20  public @interface Immutable {
21  
22      /**
23       * by default all subclasses of the @Immutable class are not immutable. This can override it.
24       * 
25       * @return true for subclasses of @Immutable class to be regarded as immutable from the cloner
26       */
27      boolean subClass() default false;
28  }
