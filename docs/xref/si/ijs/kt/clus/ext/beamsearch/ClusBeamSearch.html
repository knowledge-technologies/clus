118     public void initializeHeuristic() {
119         ClusStatManager smanager = m_BeamInduce.getStatManager();
120         Settings sett = smanager.getSettings();
121         m_MaxTreeSize = sett.getBeamSearch().getBeamTreeMaxSize();
122         ClusLogger.info("BeamSearch : the maximal size of the trees is " + m_MaxTreeSize);
123         m_BeamPostPruning = sett.getBeamSearch().isBeamPostPrune();
124         m_Heuristic = (ClusBeamHeuristic) smanager.getHeuristic();
125         Heuristic attr_heur = sett.getBeamSearch().getBeamAttrHeuristic();
126         if (!attr_heur.equals(Heuristic.Default)) {
127             m_AttrHeuristic = smanager.createHeuristic(attr_heur);
128             m_Heuristic.setAttrHeuristic(m_AttrHeuristic);
129         }

160     public ClusBeam initializeBeam(ClusRun run) throws Exception {
161         ClusStatManager smanager = m_BeamInduce.getStatManager();
162         Settings sett = smanager.getSettings();
163         ClusBeam beam = new ClusBeam(sett.getBeamSearch().getBeamWidth(), sett.getBeamSearch().getBeamRemoveEqualHeur());
164         /* Create single leaf node */
165         RowData train = (RowData) run.getTrainingSet();
166         train.addIndices();// add the indices of the tuples
167         ClusStatistic stat = m_Induce.createTotalClusteringStat(train);
168         stat.calcMean();
169         m_Induce.initSelectorAndSplit(stat);
170         initSelector(m_Induce.getBestTest());
171         ClusLogger.info("Root statistic: " + stat);
172         /* Has syntactic constraints? */
173         ClusNode root = null;
174         String constr_file = sett.getConstraints().getConstraintFile();
175         if (StringUtils.unCaseCompare(constr_file, SettingsBase.NONE)) {
176             root = new ClusNode();
177             root.setClusteringStat(stat);
178         }
179         else {
180             ClusConstraintFile file = ClusConstraintFile.getInstance();
181             root = file.getClone(constr_file);
182             root.setClusteringStat(stat);
183             m_Induce.fillInStatsAndTests(root, train);
184         }
185         /* Make sure root also has target statistics computed */
186         root.initTargetStat(getStatManager(), train);
187         root.getTargetStat().calcMean();
188         root.getClusteringStat().setBeam(beam);// connection to the similarity measure...
189         root.getTargetStat().setBeam(beam);
190         /* Compute total weight */
191         double weight = root.getClusteringStat().getTotalWeight();
192         setTotalWeight(weight);
193         /* Evaluate the quality estimate */
194         double value = estimateBeamMeasure(root);
195         /* Add tree to beam */
196         beam.addModel(new ClusBeamModel(value, root));
197         /* Initialize Tree Distance */
198         m_BeamModelDistance = new ClusBeamModelDistance(run, beam);
199         /* Initialize the Syntactic Distance Constraint */
200         if (SettingsBeamSearch.BEAM_SYNT_DIST_CONSTR)
201             m_BeamSyntConstr = new ClusBeamSyntacticConstraint(run); // this can throw IOException
202         return beam;
203     }

206     public void refineGivenLeaf(ClusNode leaf, ClusBeamModel root, ClusBeam beam, ClusAttrType[] attrs) throws Exception {
207         MyArray arr = (MyArray) leaf.getVisitor();
208         RowData data = new RowData(arr.getObjects(), arr.size());
209         if (m_Induce.initSelectorAndStopCrit(leaf, data)) {
210             // stopping criterion is met (save this for further reference?)
211             return;
212         }
213         // init base value for heuristic
214         CurrentBestTestAndHeuristic sel = m_Induce.getBestTest();
215         FindBestTest find = m_Induce.getFindBestTest();
216         double base_value = root.getValue();
217         double leaf_add = m_Heuristic.computeLeafAdd(leaf);
218         m_Heuristic.setTreeOffset(base_value - leaf_add);
219         // find good splits
220         for (int i = 0; i < attrs.length; i++) {
221             // reset selector for each attribute
222             sel.resetBestTest();
223             double beam_min_value = beam.getMinValue();
224             sel.setBestHeur(beam_min_value);
225             // process attribute
226             ClusAttrType at = attrs[i];
227             // ClusLogger.info("Attribute: "+at.getName());
228             if (at instanceof NominalAttrType)
229                 find.findNominal((NominalAttrType) at, data, null);
230             else
231                 find.findNumeric((NumericAttrType) at, data, null);
232             // found good test for attribute ?
233             if (sel.hasBestTest()) {
234                 ClusNode ref_leaf = (ClusNode) leaf.cloneNode();
235                 ref_leaf.testToNode(sel);
236                 // output best test
237                 if (getSettings().getGeneral().getVerbose() > 0)
238                     ClusLogger.info("Test: " + ref_leaf.getTestString() + " -> " + sel.m_BestHeur + " (" + ref_leaf.getTest().getPosFreq() + ")");
239                 // create child nodes
240                 ClusStatManager mgr = m_Induce.getStatManager();
241                 int arity = ref_leaf.updateArity();
242                 NodeTest test = ref_leaf.getTest();
243                 for (int j = 0; j < arity; j++) {
244                     ClusNode child = new ClusNode();
245                     ref_leaf.setChild(child, j);
246                     RowData subset = data.applyWeighted(test, j);
247                     child.initClusteringStat(mgr, subset);
248                     // the following two calls could be removed, but are useful for printing the trees
249                     child.initTargetStat(mgr, subset);
250                     child.getTargetStat().calcMean();
251                 }
252                 // create new model
253                 ClusNode root_model = (ClusNode) root.getModel();
254                 ClusNode ref_tree = (ClusNode) root_model.cloneTree(leaf, ref_leaf);
255                 double new_heur = sanityCheck(sel.m_BestHeur, ref_tree);
256                 // Check for sure if _strictly_ better!
257 
258                 ClusBeamModel new_model = new ClusBeamModel(new_heur, ref_tree);
259                 new_model.setParentModelIndex(getCurrentModel());
260 
261                 if ((SettingsBeamSearch.BEAM_SIMILARITY != 0) && !SettingsBeamSearch.BEAM_SYNT_DIST_CONSTR) {
262                     new_model.setModelPredictions(m_BeamModelDistance.getPredictions(new_model.getModel()));
263                     if (!beam.modelAlreadyIn(new_model)) {
264                         // This version is linear wrt the beam-width
265                         m_BeamModelDistance.addDistToCandOpt(beam, new_model);
266                         if (beam.removeMinUpdatedOpt(new_model, m_BeamModelDistance) == 1)
267                             setBeamChanged(true);
268 
269                         // These were used for KDID 2006 paper
270                         /*
271                          * m_BeamModelDistance.calculatePredictionDistances(beam, new_model);
272                          * if (beam.removeMinUpdated(new_model) == 1) setBeamChanged(true);
273                          */
274                         /*
275                          * m_BeamModelDistance.calculatePredictionDistancesOpt(beam, new_model);
276                          * if (beam.removeMinUpdated(new_model) == 1) setBeamChanged(true);
277                          */
278                     }
279                 }
280                 else {
281                     if (SettingsBeamSearch.BEAM_SYNT_DIST_CONSTR) {
282                         ClusLogger.info("OLD HEUR = " + new_heur);
283                         new_model.setModelPredictions(m_BeamModelDistance.getPredictions(new_model.getModel()));
284                         new_heur -= SettingsBeamSearch.BEAM_SIMILARITY * m_BeamModelDistance.getDistToConstraint(new_model, m_BeamSyntConstr);
285                         ClusLogger.info("UPDT HEUR = " + new_heur);
286                     }
287                     // Check for sure if _strictly_ better!
288                     if (new_heur > beam_min_value) {
289 
290                         beam.addModel(new_model);
291                         setBeamChanged(true);
292                         // Uncomment the following to print each model that is added to the beam
293                         // ((ClusNode)new_model.getModel()).printTree();
294                     }
295                 }
296             }
297         }
298     }

301     public void refineEachLeaf(ClusNode tree, ClusBeamModel root, ClusBeam beam, ClusAttrType[] attrs) throws Exception {
302         int nb_c = tree.getNbChildren();
303         if (nb_c == 0) {
304             refineGivenLeaf(tree, root, beam, attrs);
305         }
306         else {
307             for (int i = 0; i < nb_c; i++) {
308                 ClusNode child = (ClusNode) tree.getChild(i);
309                 refineEachLeaf(child, root, beam, attrs);
310             }
311         }
312     }

315     public void refineModel(ClusBeamModel model, ClusBeam beam, ClusRun run) throws Exception {
316         ClusNode tree = (ClusNode) model.getModel();
317         /* Compute size */
318         if (m_MaxTreeSize >= 0) {
319             int size = tree.getNbNodes();
320             if (size + 2 > m_MaxTreeSize) { return; }
321         }
322         /* Sort the data into tree */
323         RowData train = (RowData) run.getTrainingSet();
324         m_Coll.initialize(tree, null);
325         int nb_rows = train.getNbRows();
326         for (int i = 0; i < nb_rows; i++) {
327             DataTuple tuple = train.getTuple(i);
328             tree.applyModelProcessor(tuple, m_Coll);
329         }
330         /* Data is inside tree, try to refine each leaf */
331         ClusAttrType[] attrs = train.getSchema().getDescriptiveAttributes();
332         refineEachLeaf(tree, model, beam, attrs);
333         /* Remove data from tree */
334         tree.clearVisitors();
335     }

338     public void refineBeam(ClusBeam beam, ClusRun run) throws Exception {
339         setBeamChanged(false);
340         ArrayList models = beam.toArray();
341         for (int i = 0; i < models.size(); i++) {
342             // ClusLogger.info("Refining model: "+i);
343             setCurrentModel(i);
344             ClusBeamModel model = (ClusBeamModel) models.get(i);
345             if (!(model.isRefined() || model.isFinished())) {
346                 if (m_Verbose)
347                     System.out.print("[*]");
348                 refineModel(model, beam, run);
349                 model.setRefined(true);
350                 model.setParentModelIndex(-1);
351             }
352             if (m_Verbose) {
353                 if (model.isRefined()) {
354                     System.out.print("[R]");
355                 }
356                 if (model.isFinished()) {
357                     System.out.print("[F]");
358                 }
359             }
360         }
361     }

370     public void estimateBeamStats(ClusBeam beam) {
371         SingleStat stat_heuristic = new SingleStat();
372         SingleStat stat_size = new SingleStat();
373         SingleStat stat_same_heur = new SingleStat();
374         ArrayList lst = beam.toArray();
375         HashSet tops = new HashSet();
376         for (int i = 0; i < lst.size(); i++) {
377             ClusBeamModel model = (ClusBeamModel) lst.get(i);
378             stat_heuristic.addFloat(model.getValue());
379             stat_size.addFloat(model.getModel().getModelSize());
380             NodeTest top = ((ClusNode) model.getModel()).getTest();
381             if (top != null) {
382                 if (!tops.contains(top)) {
383                     tops.add(top);
384                 }
385             }
386         }
387         Iterator iter = beam.getIterator();
388         while (iter.hasNext()) {
389             ClusBeamTreeElem elem = (ClusBeamTreeElem) iter.next();
390             stat_same_heur.addFloat(elem.getCount());
391         }
392         ArrayList stat = new ArrayList();
393         stat.add(stat_heuristic);
394         stat.add(stat_same_heur);
395         stat.add(stat_size);
396         stat.add(new Integer(tops.size()));
397         m_BeamStats.add(stat);
398     }

401     public String getLevelStat(int i) {
402         ArrayList stat = (ArrayList) m_BeamStats.get(i);
403         StringBuffer buf = new StringBuffer();
404         buf.append("Level: " + i);
405         for (int j = 0; j < stat.size(); j++) {
406             Object elem = stat.get(j);
407             buf.append(", ");
408             if (elem instanceof SingleStat) {
409                 SingleStat st = (SingleStat) elem;
410                 buf.append(st.getMean() + "," + st.getRange());
411             }
412             else {
413                 buf.append(elem.toString());
414             }
415         }
416         return buf.toString();
417     }

434     public void writeModel(ClusModelCollectionIO strm) throws IOException, ClusException {
435         saveBeamStats();
436         ArrayList beam = getBeam().toArray();
437         for (int i = 0; i < beam.size(); i++) {
438             ClusBeamModel m = (ClusBeamModel) beam.get(i);
439             ClusNode node = (ClusNode) m.getModel();
440             node.updateTree();
441             node.clearVisitors();
442         }
443         int pos = 1;
444         for (int i = beam.size() - 1; i >= 0; i--) {
445             ClusBeamModel m = (ClusBeamModel) beam.get(i);
446             ClusModelInfo info = new ClusModelInfo("B" + pos + ": " + m.getValue());
447             info.setScore(m.getValue());
448             info.setModel(m.getModel());
449             strm.addModel(info);
450             pos++;
451         }
452     }

460     public ClusNode beamSearch(ClusRun run) throws Exception {
461         reset();
462         ClusLogger.info("Starting beam search");
463         m_Induce = new ConstraintDFInduce(m_BeamInduce);
464         ClusBeam beam = initializeBeam(run);
465         // MyFile beamlog = new MyFile("beam.log", true);
466         // tryLogBeam(beamlog, beam, "Initial beam:");
467         int i = 0;
468         while (true) {
469             ClusLogger.info("Step: " + i);
470             refineBeam(beam, run);
471             if (isBeamChanged()) {
472                 // tryLogBeam(beamlog, beam, "Step:"+i);
473                 estimateBeamStats(beam);
474             }
475             else {
476                 break;
477             }
478             i++;
479         }
480         setBeam(beam);
481         double best = beam.getBestModel().getValue();
482         double worst = beam.getWorstModel().getValue();
483         ClusLogger.info("Worst = " + worst + " Best = " + best);
484         printBeamStats(i - 1);
485         ClusNode result = (ClusNode) beam.getBestAndSmallestModel().getModel();
486         // beamlog.close();
487         return result;
488     }

526     public double sanityCheck(double value, ClusNode tree) throws ClusException {
527         /*
528          * int size = tree.getModelSize();
529          * ClusLogger.info("Size = "+size);
530          * tree.printTree();
531          * ClusLogger.info("Evaluating measure...");
532          */
533         double expected = estimateBeamMeasure(tree);
534         if (Math.abs(value - expected) > 1e-6) {
535             ClusLogger.info("Bug in heurisitc: " + value + " <> " + expected);
536             PrintWriter wrt = new PrintWriter(System.out);
537             tree.printModel(wrt);
538             wrt.close();
539             System.out.flush();
540             throw new ClusException("Bug in heuristic: " + value + " <> " + expected);
541         }
542         return expected;
543     }
