28      private INIFileStringOrDouble m_TestSet;
29      private INIFileStringOrDouble m_PruneSet;
30      private INIFileStringOrInt m_PruneSetMax;
31      /** How many folds are we having in xval OR gives a file that defines the used folds (in the data set) */

53      public String getDataFile() {
54          return m_DataFile.getValue();
55      }
56  
57  
58      public boolean isNullFile() {
59          return StringUtils.unCaseCompare(m_DataFile.getValue(), NONE);
60      }
61  
62  
63      public void updateDataFile(String fname) {
64          if (isNullFile())
65              m_DataFile.setValue(fname);
66      }
67  
68  
69      public String getTestFile() {
70          return m_TestSet.getValue();
71      }
72  
73  
74      public boolean isNullTestFile() {
75          return m_TestSet.isDoubleOrNull(NONE);
76      }
77  
78  
79      public String getPruneFile() {
80          return m_PruneSet.getValue();
81      }
82  
83  
84      public boolean isNullPruneFile() {
85          return m_PruneSet.isDoubleOrNull(NONE);
86      }
87  
88  
89      public double getTestProportion() {
90          if (!m_TestSet.isDouble())
91              return 0.0;
92          return m_TestSet.getDoubleValue();
93      }
94  
95  
96      public double getPruneProportion() {
97          if (!m_PruneSet.isDouble())
98              return 0.0;
99          return m_PruneSet.getDoubleValue();
100     }
101 
102 
103     public int getPruneSetMax() {
104         if (m_PruneSetMax.isString(INFINITY_STRING))
105             return Integer.MAX_VALUE;
106         else
107             return m_PruneSetMax.getIntValue();
108     }
109 
110 
111     public boolean isNullXValFile() {
112         return m_XValFolds.isIntOrNull(NONE);
113     }
114 
115 
116     public boolean isLOOXVal() {
117         return m_XValFolds.isString("LOO");
118     }
119 
120 
121     public String getXValFile() {
122         return m_XValFolds.getValue();
123     }
124 
125 
126     public int getXValFolds() {
127         return m_XValFolds.getIntValue();
128     }
129 
130 
131     public void setXValFolds(int folds) {
132         m_XValFolds.setIntValue(folds);
133     }
134 
135 
136     public boolean isRemoveMissingTarget() {
137         return m_RemoveMissingTarget.getValue();
138     }
139 
140 
141     /** Do we want to normalize the data */
142     public NormalizeDataValues getNormalizeData() {
143         return m_NormalizeData.getValue();
144     }
145 
146 
147     public void setPruneSetMaxEnabled(boolean value) {
148         m_PruneSetMax.setEnabled(value);
149     }
150 
151 
152     public boolean isPruneSetString(String str) {
153         return m_PruneSet.isString(str);
154     }

156 
157     @Override
158     public void create() {
159         m_Section.addNode(m_DataFile = new INIFileString("File", NONE));
160         m_Section.addNode(m_TestSet = new INIFileStringOrDouble("TestSet", NONE));
161         m_Section.addNode(m_PruneSet = new INIFileStringOrDouble("PruneSet", NONE));
162         m_Section.addNode(m_PruneSetMax = new INIFileStringOrInt("PruneSetMax", INFINITY_STRING));
163         
164         m_Section.addNode(m_XValFolds = new INIFileStringOrInt("XVal"));
165         m_XValFolds.setIntValue(10);
166         
167         m_Section.addNode(m_RemoveMissingTarget = new INIFileBool("RemoveMissingTarget", false));
168         m_Section.addNode(m_NormalizeData = new INIFileEnum<>("NormalizeData", NormalizeDataValues.None));
169     }
