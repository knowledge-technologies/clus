149 	public void initializeAttributes(ClusAttrType[] descriptive, int nbRankings) {
150 		int num = -1;
151 		int nom = -1;
152 		// ClusLogger.info("NB = "+descriptive.length);
153 		int[] attributeDatasetIndices = descriptive[0].getSchema().getDescriptive().toIndices();
154 		for (int i = 0; i < descriptive.length; i++) {
155 			ClusAttrType type = descriptive[i];
156 			// type.setDatasetIndex(m_AttributeDatasetIndices[i]);
157 			m_AttributeDatasetIndices.put(type.getName(), (double) attributeDatasetIndices[i]);
158 			if (!type.isDisabled()) {
159 				// double[] info = new double[3];
160 				double[] info = new double[2 + nbRankings];
161 				if (type.getAttributeType().equals(AttributeType.Nominal)) {
162 					nom++;
163 					info[0] = 0; // type
164 					info[1] = nom; // order in nominal attributes
165 				}
166 				if (type.getAttributeType().equals(AttributeType.Numeric)) {
167 					num++;
168 					info[0] = 1; // type
169 					info[1] = num; // order in numeric attributes
170 				}
171 				for (int j = 0; j < nbRankings; j++) {
172 					info[2 + j] = 0; // current importance
173 				}
174 				// System.out.print(type.getName()+": "+info[1]+"\t");
175 				m_AllAttributes.put(type.getName(), info);
176 			}
177 		}
178 	}

181 	 * Writes fimp with where the attributes appear in the order that was chosen (as
182 	 * in the dataset or sorted by relevance).
183 	 */
184 	public void writeRanking(String fname) throws IOException {
185 		File franking = new File(fname + ".fimp");
186 		FileWriter wrtr = new FileWriter(franking);
187 
188 		wrtr.write(m_RankingDescription + "\n");
189 		wrtr.write(m_FimpTableHeader + "\n");
190 		wrtr.write(StringUtils.makeString('-', m_FimpTableHeader.length()) + "\n");
191 
192 		Set<Double> orderedKeys;
193 		if (m_Order == FimpOrdering.AS_IN_DATASET) {
194 			orderedKeys = m_OutputRanking.keySet();
195 		} else {
196 			orderedKeys = m_OutputRanking.descendingKeySet();
197 		}
198 		for (Double key : orderedKeys) {
199 			ArrayList<Triple<String, int[], double[]>> triples = m_OutputRanking.get(key);
200 			for (Triple<String, int[], double[]> triple : triples) {
201 				wrtr.write(fimpTableRow(triple.getFirst(), triple.getSecond(), triple.getThird()) + "\n");
202 			}
203 		}
204 		wrtr.flush();
205 		wrtr.close();
206 		if (getSettings().getGeneral().getVerbose() >= 1) {
207 			ClusLogger.info(String.format("Feature importances written to: %s.fimp", fname));
208 		}
209 	}

216 		int datasetIndex = (int) Math.round(m_AttributeDatasetIndices.get(attributeName).doubleValue());

339 	/**
340 	 * Implements the Fisher-Yates algorithm for uniform shuffling.
341 	 * 
342 	 * @param type
343 	 *            0 nominal, 1 numeric
344 	 * @param position
345 	 *            position at which the attribute whose values are being shuffled,
346 	 *            is
347 	 */
348 	public RowData createRandomizedOOBdata(OOBSelection selection, RowData data, int type, int position, int seed)
349 	        throws ClusException {
350 		RowData result = data;
351 		Random rndm = new Random(seed);
352 		for (int i = 0; i < result.getNbRows() - 1; i++) {
353 			// int rnd = i + ClusRandom.nextInt(ClusRandom.RANDOM_ALGO_INTERNAL,
354 			// result.getNbRows()- i);
355 			int rnd = i + rndm.nextInt(result.getNbRows() - i);
356 			DataTuple first = result.getTuple(i);
357 			DataTuple second = result.getTuple(rnd);
358 			boolean successfullySwapped = false;
359 			if (first instanceof SparseDataTuple) {
360 				if (type == 1) {
361 					double swap = ((SparseDataTuple) first).getDoubleValueSparse(position);
362 					((SparseDataTuple) first)
363 					        .setDoubleValueSparse(((SparseDataTuple) second).getDoubleValueSparse(position), position);
364 					((SparseDataTuple) second).setDoubleValueSparse(swap, position);
365 					successfullySwapped = true;
366 				} else {
367 					System.err.println(
368 					        "WARNING: type is not 1 (numeric). We will try to swap the values like in non-sparse case, but some things might go wrong, e.g.,\n"
369 					                + "java.lang.NullPointerException might occur.");
370 				}
371 			}
372 			if (!successfullySwapped) {
373 				if (type == 0) {// nominal
374 					int swap = first.getIntVal(position);
375 					first.setIntVal(second.getIntVal(position), position);
376 					second.setIntVal(swap, position);
377 				} else if (type == 1) {// numeric
378 					double swap = first.getDoubleVal(position);
379 					first.setDoubleVal(second.getDoubleVal(position), position);
380 					second.setDoubleVal(swap, position);
381 				} else {
382 					throw new ClusException("Error while making the random permutations for feature ranking!");
383 				}
384 			}
385 		}
386 		return result;
387 	}

389 	/**
390 	 * Finds all attributes that appear as (part of) the test in a node in the given
391 	 * tree. Depth first search is used to traverse the tree.
392 	 * 
393 	 * @param root
394 	 *            The root of the tree
395 	 * @return List of attributes' names
396 	 */
397 	public ArrayList<String> getInternalAttributesNames(ClusNode root) {
398 		ArrayList<String> attributes = new ArrayList<String>(); // list of attribute names in the tree
399 		ArrayList<ClusNode> stack = new ArrayList<ClusNode>(); // stack of nodes to be processed
400 		HashSet<String> discovered = new HashSet<String>(); // names that are currently in attributes, used for faster
401 		                                                    // look-up.
402 		if (!root.atBottomLevel()) {
403 			stack.add(root);
404 		}
405 		while (stack.size() > 0) {
406 			ClusNode top = stack.remove(stack.size() - 1);
407 			String name = top.getTest().getType().getName();
408 			if (!(discovered.contains(name) || top.atBottomLevel())) { // top not discovered yet and is internal node
409 				discovered.add(name);
410 				attributes.add(name);
411 			}
412 			for (int i = 0; i < top.getNbChildren(); i++) {
413 				if (!top.getChild(i).atBottomLevel()) {
414 					stack.add((ClusNode) top.getChild(i));
415 				}
416 			}
417 		}
418 		return attributes;
419 	}

421 	/**
422 	 * Calculates values of all error measures.
423 	 * 
424 	 * @return {@code [[listOfResultsForErr1, [sign1]], [listOfResultsForErr2, [sign2]], ...]},<br>
425 	 *         where {@code signI = errorI.shouldBeLow() ? -1.0 : 1.0}, and
426 	 *         {@code listOfResultsForErr} always contains the overall {@code Err}
427 	 *         error in the position 0, and possibly also per target calculations
428 	 *         for {@code error} in the positions i {@literal >} 0.
429 	 */
430 	public double[][][] calcAverageErrors(RowData data, ClusModel model, ClusStatManager mgr)
431 	        throws ClusException, InterruptedException {
432 		ClusSchema schema = data.getSchema();
433 		ClusErrorList error = computeErrorList(schema, mgr);
434 		/* attach model to given schema */
435 		schema.attachModel(model);
436 		/* iterate over tuples and compute error */
437 		for (int i = 0; i < data.getNbRows(); i++) {
438 			DataTuple tuple = data.getTuple(i);
439 			ClusStatistic pred = model.predictWeighted(tuple);
440 			error.addExample(tuple, pred);
441 		}
442 		// if (m_FimpTableHeader == null) {
443 		// setRForestDescription(error);
444 		// }
445 		/* return the average errors */
446 		double[][][] errors = new double[error.getNbErrors()][2][];
447 		for (int i = 0; i < errors.length; i++) {
448 			ClusError currentError = error.getError(i);
449 			int nbResultsPerError = 1;
450 			if (mgr.getSettings().getEnsemble().shouldPerformRankingPerTarget()
451 			        && (currentError instanceof ComponentError)) {
452 				nbResultsPerError += currentError.getDimension();
453 			}
454 			errors[i][0] = new double[nbResultsPerError];
455 			// compute overall error
456 			errors[i][0][0] = currentError.getModelError();
457 			// compute per target errors if necessary
458 			for (int dim = 1; dim < errors[i][0].length; dim++) {
459 				errors[i][0][dim] = currentError.getModelErrorComponent(dim - 1);
460 			}
461 			// should be low?
462 			errors[i][1] = new double[] { currentError.shouldBeLow() ? -1.0 : 1.0 };
463 		}
464 		return errors;
465 	}

467 	public ClusErrorList computeErrorList(ClusSchema schema, ClusStatManager mgr) {
468 		Settings sett = mgr.getSettings();
469 		ClusErrorList error = new ClusErrorList();
470 		NumericAttrType[] num = schema.getNumericAttrUse(AttributeUseType.Target);
471 		NominalAttrType[] nom = schema.getNominalAttrUse(AttributeUseType.Target);
472 		boolean hasNumeric = num.length > 0;
473 		boolean hasNominal = nom.length > 0;
474 		// Sometimes, instead of operating with modes (which depend on clustering attributes),
475 		// we operate with (numbers of) targets directly)
476 		if (mgr.getTargetMode() == ClusStatManager.Mode.HIERARCHICAL) {
477 			error.addError(
478 			        new HierErrorMeasures(error, mgr.getHier(), sett.getHMLC().getRecallValues().getDoubleVector(),
479 			                HierarchyMeasures.PooledAUPRC, false, getSettings().getOutput().isGzipOutput()));
480 		} else if (hasNominal && hasNumeric) {
481 			throw new UnsupportedOperationException("Unsupported! No error measure can handle numeric and nominal targets.");
482 		} else if (hasNominal) {
483 			if (sett.getMLC().getSectionMultiLabel().isEnabled()) {
484 				List<MultiLabelMeasures> measures = sett.getMLC().getMultiLabelRankingMeasures();
485 				for (MultiLabelMeasures measure : measures) {
486 					switch (measure) {
487 					case HammingLoss:
488 						error.addError(new HammingLoss(error, nom));
489 						break;
490 					case MLAccuracy:
491 						error.addError(new MLAccuracy(error, nom));
492 						break;
493 					case MLPrecision:
494 						error.addError(new MLPrecision(error, nom));
495 						break;
496 					case MLRecall:
497 						error.addError(new MLRecall(error, nom));
498 						break;
499 					case MLFOne:
500 						error.addError(new MLFOneMeasure(error, nom));
501 						break;
502 					case SubsetAccuracy:
503 						error.addError(new SubsetAccuracy(error, nom));
504 						break;
505 					case MacroPrecision:
506 						error.addError(new MacroPrecision(error, nom));
507 						break;
508 					case MacroRecall:
509 						error.addError(new MacroRecall(error, nom));
510 						break;
511 					case MacroFOne:
512 						error.addError(new MacroFOne(error, nom));
513 						break;
514 					case MicroPrecision:
515 						error.addError(new MicroPrecision(error, nom));
516 						break;
517 					case MicroRecall:
518 						error.addError(new MicroRecall(error, nom));
519 						break;
520 					case MicroFOne:
521 						error.addError(new MisclassificationError(error, nom));
522 						break;
523 					case OneError:
524 						error.addError(new OneError(error, nom));
525 						break;
526 					case Coverage:
527 						error.addError(new Coverage(error, nom));
528 						break;
529 					case RankingLoss:
530 						error.addError(new RankingLoss(error, nom));
531 						break;
532 					case AveragePrecision:
533 						error.addError(new AveragePrecision(error, nom));
534 						break;
535 					case AverageAUROC:
536 						error.addError(new MLaverageAUROC(error, nom));
537 						break;
538 					case AverageAUPRC:
539 						error.addError(new MLaverageAUPRC(error, nom));
540 						break;
541 					case WeightedAverageAUPRC:
542 						error.addError(new MLweightedAUPRC(error, nom));
543 						break;
544 					case PooledAUPRC:
545 						error.addError(new MLpooledAUPRC(error, nom));
546 						break;
547 					default:
548 						break;
549 					}
550 				}
551 			} else {
552 				error.addError(new Accuracy(error, nom));
553 			}
554 		} else if (hasNumeric) {
555 			error.addError(new RMSError(error, num));
556 		} else {
557 			throw new UnsupportedOperationException("Unsupported!");
558 		}
559 		return error;
560 	}

634 	public void createFimp(ClusRun cr, String appendixToFimpName, int expectedNumberTrees, int realNumberOfTrees)
635 	        throws IOException {
636 		if (expectedNumberTrees != realNumberOfTrees) {
637 			// copy the file from resources
638 			String fileName = "fimp_manipulation.py";
639 			try {
640 				InputStream is = si.ijs.kt.clus.Clus.class.getResourceAsStream("/" + fileName);
641 				String fullFileName = getSettings().getGeneric().getFileAbsolute(fileName);
642 				Files.copy(is, Paths.get(fullFileName), StandardCopyOption.REPLACE_EXISTING);
643 			} catch (IOException ex) {
644 				System.err.println("Error while copying " + fileName + " to the output folder.");
645 				ex.printStackTrace();
646 			}
647 		}
648 
649 		if (cr.getStatManager().getSettings().getEnsemble().shouldSortRankingByRelevance()) {
650 			m_Order = FimpOrdering.BY_RELEVANCE;
651 		} else {
652 			m_Order = FimpOrdering.AS_IN_DATASET;
653 		}
654 		if (m_Order == FimpOrdering.BY_RELEVANCE && getNbFeatureRankings() > 1) {
655 			if (getSettings().getGeneral().getVerbose() >= 1) {
656 				System.err.println(
657 				        "More than one feature ranking will be output. The attributes will be sorted with respect to the first ranking.");
658 			}
659 			// m_Order = FimpOrdering.AS_IN_DATASET;
660 		}
661 		computeFinalScores(realNumberOfTrees);
662 		computeRanks();
663 		prepareFeatureRankingForOutput();
664 
665 		String appName = cr.getStatManager().getSettings().getGeneric()
666 		        .getFileAbsolute(cr.getStatManager().getSettings().getGeneric().getAppName()) + appendixToFimpName;
667 		writeRanking(appName);
668 
669 		if (cr.getStatManager().getSettings().getOutput().isOutputJSONModel()) {
670 			writeJSON(cr);
671 		}
672 	}

674 	/**
675 	 * Computes the final scores (and ranks) and puts the attributes into a sorting
676 	 * tree, according to the chosen order.
677 	 */
678 	private void prepareFeatureRankingForOutput(FimpOrdering ordering) {
679 		for (String attributeName : m_AllAttributes.keySet()) {
680 			Double key;
681 			double[] scores = Arrays.copyOfRange(m_AllAttributes.get(attributeName), 2, 2 + m_NbFeatureRankings);
682 			if (ordering == FimpOrdering.AS_IN_DATASET) {
683 				key = m_AttributeDatasetIndices.get(attributeName);
684 			} else {
685 				key = scores[0]; // the only relevance or the first ranking if more than one is computed
686 			}
687 			if (!m_OutputRanking.containsKey(key)) {
688 				m_OutputRanking.put(key, new ArrayList<Triple<String, int[], double[]>>());
689 			}
690 			m_OutputRanking.get(key).add(
691 			        new Triple<String, int[], double[]>(attributeName, m_AttributeRanks.get(attributeName), scores));
692 		}
693 	}

721 	/**
722 	 * Comptues ranks of the features. To avoid the floating-point arithmetics
723 	 * issues we use approximately equal when computing the ranks ...
724 	 */
725 	private void computeRanks() {
726 		m_AttributeRanks = new HashMap<String, int[]>();
727 		ArrayList<String> attributeNames = new ArrayList<String>();
728 		for (String attributeName : m_AllAttributes.keySet()) {
729 			m_AttributeRanks.put(attributeName, new int[m_NbFeatureRankings]);
730 			attributeNames.add(attributeName);
731 		}
732 		for (int rankingInd = 0; rankingInd < m_NbFeatureRankings; rankingInd++) {
733 			final int ind = 2 + rankingInd;
734 			Collections.sort(attributeNames, new Comparator<String>() {
735 
736 				@Override
737 				public int compare(String attr1, String attr2) {
738 					return Double.compare(m_AllAttributes.get(attr2)[ind], m_AllAttributes.get(attr1)[ind]); // decreasing
739 				}
740 			});
741 			int lastRank = -200;
742 			double prevRelevance = Double.NaN;
743 			double thisRelevance;
744 			for (int i = 0; i < attributeNames.size(); i++) {
745 				String attributeName = attributeNames.get(i);
746 				thisRelevance = m_AllAttributes.get(attributeName)[ind];
747 				if (!ClusUtil.eq(thisRelevance, prevRelevance, ClusUtil.PICO)) { // (thisRelevance != prevRelevance){
748 					lastRank = i + 1;
749 				}
750 				m_AttributeRanks.get(attributeName)[rankingInd] = lastRank;
751 				prevRelevance = thisRelevance;
752 			}
753 		}
754 	}
