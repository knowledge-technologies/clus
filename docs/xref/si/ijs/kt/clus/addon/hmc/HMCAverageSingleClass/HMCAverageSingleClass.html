362     public String getClassStr(String file) {
363         StringBuffer result = new StringBuffer();
364         String value = FileUtil.getName(FileUtil.removePath(file));
365         String[] cmps = value.split("_");
366         String[] elems = cmps[cmps.length - 1].split("-");
367         for (int i = 0; i < elems.length; i++) {
368             if (i != 0)
369                 result.append("/");
370             result.append(elems[i]);
371         }
372         return result.toString();
373     }

384     public ClusModel loadModel(String file) throws IOException, ClusException, ClassNotFoundException {
385         String class_str = getClassStr(file);
386         ClusLogger.info("Loading: " + file + " class: " + class_str);
387         ClusModelCollectionIO io = ClusModelCollectionIO.load(file);
388         ClusModel sub_model = io.getModel("Original");
389         if (sub_model == null) { throw new ClusException("Error: .model file does not contain model named 'Original'"); }
390         m_NbModels++;
391         m_TotSize += sub_model.getModelSize();
392         return sub_model;
393     }

396     public void loadModelPerModel(String dir, ClusRun cr) throws IOException, ClusException, ClassNotFoundException, InterruptedException {
397         String[] files = FileUtil.dirList(dir, "model");
398         for (int i = 0; i < files.length; i++) {
399             ClusModel model = loadModel(FileUtil.cmbPath(dir, files[i]));
400             int class_idx = getClassIndex(files[i]);
401             for (int j = ClusModelInfoList.TRAINSET; j <= ClusModelInfoList.TESTSET; j++) {
402                 evaluateModelAndUpdateErrors(j, class_idx, model, cr);
403             }
404         }
405         INIFileNominalOrDoubleOrVector class_thr = getSettings().getHMLC().getClassificationThresholds();
406         if (class_thr.isVector()) {
407             HierClassTresholdPruner pruner = (HierClassTresholdPruner) getStatManager().getTreePruner(null);
408             for (int j = 0; j < pruner.getNbResults(); j++) {
409                 for (int traintest = ClusModelInfoList.TRAINSET; traintest <= ClusModelInfoList.TESTSET; traintest++) {
410                     RowData data = cr.getDataSet(traintest);
411                     ClusErrorList error = getEvalArray(traintest, j);
412                     error.setNbExamples(data.getNbRows(), data.getNbRows());
413                 }
414             }
415         }
416     }

419     // evaluate tree for one class on all examples and update errors
420     public void evaluateModelAndUpdateErrors(int train_or_test, int class_idx, ClusModel model, ClusRun cr) throws ClusException, IOException, InterruptedException {
421         RowData data = cr.getDataSet(train_or_test);
422         m_Clus.getSchema().attachModel(model);
423         INIFileNominalOrDoubleOrVector class_thr = getSettings().getHMLC().getClassificationThresholds();
424         if (class_thr.isVector()) {
425             HierClassTresholdPruner pruner = (HierClassTresholdPruner) getStatManager().getTreePruner(null);
426             for (int i = 0; i < data.getNbRows(); i++) {
427                 DataTuple tuple = data.getTuple(i);
428                 ClusStatistic prediction = model.predictWeighted(tuple);
429                 double[] predicted_distr = prediction.getNumericPred();
430                 ClassesTuple tp = (ClassesTuple) tuple.getObjVal(0);
431                 boolean actually_has_class = tp.hasClass(class_idx);
432                 for (int j = 0; j < pruner.getNbResults(); j++) {
433                     // update corresponding hierclasswiseacc
434                     boolean predicted_class = predicted_distr[0] >= pruner.getThreshold(j) / 100.0;
435                     HierClassWiseAccuracy acc = (HierClassWiseAccuracy) m_EvalArray[train_or_test][j].getError(0);
436                     acc.nextPrediction(class_idx, predicted_class, actually_has_class);
437                 }
438             }
439         }
440         for (int i = 0; i < data.getNbRows(); i++) {
441             DataTuple tuple = data.getTuple(i);
442             ClusStatistic prediction = model.predictWeighted(tuple);
443             double[] predicted_distr = prediction.getNumericPred();
444             m_PredProb[train_or_test][i][class_idx] = predicted_distr[0];
445         }
446     }

472     public void countClasses(RowData data, double[] counts) {
473         ClassHierarchy hier = getStatManager().getHier();
474         int sidx = hier.getType().getArrayIndex();
475         boolean[] arr = new boolean[hier.getTotal()];
476         for (int i = 0; i < data.getNbRows(); i++) {
477             DataTuple tuple = data.getTuple(i);
478             ClassesTuple tp = (ClassesTuple) tuple.getObjVal(sidx);
479             // count with parents
480             Arrays.fill(arr, false);
481             tp.fillBoolArrayNodeAndAncestors(arr);
482             for (int j = 0; j < arr.length; j++) {
483                 if (arr[j])
484                     counts[0] += 1.0;
485             }
486             // count without parents
487             hier.removeParentNodes(arr);
488             for (int j = 0; j < arr.length; j++) {
489                 if (arr[j])
490                     counts[1] += 1.0;
491             }
492         }
493     }

496     public void computeStats() throws ClusException, IOException, InterruptedException {
497         ClusRun cr = m_Clus.partitionData();
498         RegressionStat stat = (RegressionStat) getStatManager().createStatistic(AttributeUseType.Target);
499         RowData train = (RowData) cr.getTrainingSet();
500         RowData valid = (RowData) cr.getPruneSet();
501         RowData test = cr.getTestSet();
502         train.calcTotalStat(stat);
503         if (valid != null)
504             valid.calcTotalStat(stat);
505         if (test != null)
506             test.calcTotalStat(stat);
507         stat.calcMean();
508         ClassHierarchy hier = getStatManager().getHier();
509         PrintWriter wrt = getSettings().getGeneric().getFileAbsoluteWriter(getSettings().getGeneric().getAppName() + "-hmcstat.arff");
510         ClusSchema schema = new ClusSchema("HMC-Statistics");
511         schema.addAttrType(new StringAttrType("Class"));
512         schema.addAttrType(new NumericAttrType("Weight"));
513         schema.addAttrType(new NumericAttrType("MinDepth"));
514         schema.addAttrType(new NumericAttrType("MaxDepth"));
515         schema.addAttrType(new NumericAttrType("Frequency"));
516         double total = stat.getTotalWeight();
517         double[] classCounts = new double[2];
518         countClasses(train, classCounts);
519         if (valid != null)
