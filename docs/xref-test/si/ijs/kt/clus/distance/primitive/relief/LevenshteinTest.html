12  public class LevenshteinTest extends BaseTestCase {
13  
14    @Test
15    public void computeDistTest() {
16      Object[][] listOfExamples =
17          new Object[][] {
18            new Object[] {"asd", "1asd", 1},
19            new Object[] {"grk", "garac", 3},
20            // When the values are the same, the distance is zero.
21            new Object[] {"", "", 0},
22            new Object[] {"1", "1", 0},
23            new Object[] {"12", "12", 0},
24            new Object[] {"123", "123", 0},
25            new Object[] {"1234", "1234", 0},
26            new Object[] {"12345", "12345", 0},
27            new Object[] {"password", "password", 0},
28            // When one of the values is empty, the distance is the length of the
29            // other value.
30            new Object[] {"", "1", 1},
31            new Object[] {"", "12", 2},
32            new Object[] {"", "123", 3},
33            new Object[] {"", "1234", 4},
34            new Object[] {"", "12345", 5},
35            new Object[] {"", "password", 8},
36            new Object[] {"1", "", 1},
37            new Object[] {"12", "", 2},
38            new Object[] {"123", "", 3},
39            new Object[] {"1234", "", 4},
40            new Object[] {"12345", "", 5},
41            new Object[] {"password", "", 8},
42            // Whenever a single character is inserted or removed, the distance is
43            // one.
44            new Object[] {"password", "1password", 1},
45            new Object[] {"password", "p1assword", 1},
46            new Object[] {"password", "pa1ssword", 1},
47            new Object[] {"password", "pas1sword", 1},
48            new Object[] {"password", "pass1word", 1},
49            new Object[] {"password", "passw1ord", 1},
50            new Object[] {"password", "passwo1rd", 1},
51            new Object[] {"password", "passwor1d", 1},
52            new Object[] {"password", "password1", 1},
53            new Object[] {"password", "assword", 1},
54            new Object[] {"password", "pssword", 1},
55            new Object[] {"password", "pasword", 1},
56            new Object[] {"password", "pasword", 1},
57            new Object[] {"password", "passord", 1},
58            new Object[] {"password", "passwrd", 1},
59            new Object[] {"password", "passwod", 1},
60            new Object[] {"password", "passwor", 1},
61            // Whenever a single character is replaced, the distance is one.
62            new Object[] {"password", "Xassword", 1},
63            new Object[] {"password", "pXssword", 1},
64            new Object[] {"password", "paXsword", 1},
65            new Object[] {"password", "pasXword", 1},
66            new Object[] {"password", "passXord", 1},
67            new Object[] {"password", "passwXrd", 1},
68            new Object[] {"password", "passwoXd", 1},
69            new Object[] {"password", "passworX", 1},
70            // If characters are taken off the front and added to the back and all of
71            // the characters are unique, then the distance is two times the number of
72            // characters shifted, until you get halfway (and then it becomes easier
73            // to shift from the other direction).
74            new Object[] {"12345678", "23456781", 2},
75            new Object[] {"12345678", "34567812", 4},
76            new Object[] {"12345678", "45678123", 6},
77            new Object[] {"12345678", "56781234", 8},
78            new Object[] {"12345678", "67812345", 6},
79            new Object[] {"12345678", "78123456", 4},
80            new Object[] {"12345678", "81234567", 2},
81            // If all the characters are unique and the values are reversed, then the
82            // distance is the number of characters for an even number of characters,
83            // and one less for an odd number of characters (since the middle
84            // character will stay the same).
85            new Object[] {"12", "21", 2},
86            new Object[] {"123", "321", 2},
87            new Object[] {"1234", "4321", 4},
88            new Object[] {"12345", "54321", 4},
89            new Object[] {"123456", "654321", 6},
90            new Object[] {"1234567", "7654321", 6},
91            new Object[] {"12345678", "87654321", 8},
92            // The rest of these are miscellaneous interesting examples. They will
93            // be illustrated using the following key:
94            // = (the characters are equal)
95            // + (the character is inserted)
96            // - (the character is removed)
97            // # (the character is replaced)
98  
99            // Mississippi
100           // ippississiM
101           // -=##====##=+ --> 6
102           new Object[] {"Mississippi", "ippississiM", 6},
103           // eieio
104           // oieie
105           // #===# --> 2
106           new Object[] {"eieio", "oieie", 2},
107           // brad+angelina
108           // bra ngelina
109           // ===+++======= --> 3
110           new Object[] {"brad+angelina", "brangelina", 3},
111           // test international chars
112           // ?e?uli?ka
113           // e?uli?ka
114           // -======== --> 1
115           new Object[] {"?e?uli?ka", "e?uli?ka", 1},
116           new Object[] {"Mama", "Raca", 2}
117         };
118     for (int i = 0; i < listOfExamples.length; i++) {
119       String a = (String) listOfExamples[i][0], b = (String) listOfExamples[i][1];
120       Levenshtein lev = new Levenshtein(a, b);
121       double razd = Math.max(a.length(), Math.max(1, b.length()));
122 
123       assertEquals(
124           Double.parseDouble(listOfExamples[i][2].toString()) / razd,
125           lev.getDist(),
126           0.0001,
127           "test: " + i + ": " + Arrays.toString(listOfExamples[i]));
128     }
129   }
