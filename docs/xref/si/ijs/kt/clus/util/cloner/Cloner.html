131         @Override
132         public <T> T deepClone(T o, Map<Object, Object> clones) {
133             try {
134                 return cloneInternal(o, clones);

144     protected Object fastClone(final Object o, final Map<Object, Object> clones) throws IllegalAccessException {
145         final Class<? extends Object> c = o.getClass();
146         final IFastCloner fastCloner = fastCloners.get(c);
147         if (fastCloner != null)
148             return fastCloner.clone(o, deepCloner, clones);
149         return null;
150     }

158     public void registerConstant(final Class<?> c, final String privateFieldName) {
159         try {
160             final Field field = c.getDeclaredField(privateFieldName);
161             field.setAccessible(true);
162             final Object v = field.get(null);
163             ignoredInstances.put(v, true);
164         }
165         catch (final SecurityException e) {
166             throw new RuntimeException(e);
167         }
168         catch (final NoSuchFieldException e) {
169             throw new RuntimeException(e);
170         }
171         catch (final IllegalArgumentException e) {
172             throw new RuntimeException(e);
173         }
174         catch (final IllegalAccessException e) {
175             throw new RuntimeException(e);
176         }
177     }

212     /**
213      * registers all static fields of these classes. Those static fields won't be cloned when an instance
214      * of the class is cloned.
215      *
216      * This is useful i.e. when a static field object is added into maps or sets. At that point, there is no
217      * way for the cloner to know that it was static except if it is registered.
218      *
219      * @param classes
220      *        array of classes
221      */
222     public void registerStaticFields(final Class<?>... classes) {
223         for (final Class<?> c : classes) {
224             final List<Field> fields = allFields(c);
225             for (final Field field : fields) {
226                 final int mods = field.getModifiers();
227                 if (Modifier.isStatic(mods) && !field.getType().isPrimitive()) {
228                     registerConstant(c, field.getName());
229                 }
230             }
231         }
232     }

352     /**
353      * deep clones "o".
354      *
355      * @param <T>
356      *        the type of "o"
357      * @param o
358      *        the object to be deep-cloned
359      * @return a deep-clone of "o".
360      */
361     public <T> T deepClone(final T o) {
362         if (o == null)
363             return null;
364         if (!cloningEnabled)
365             return o;
366         if (dumpCloned != null) {
367             dumpCloned.startCloning(o.getClass());
368         }
369         final Map<Object, Object> clones = new IdentityHashMap<Object, Object>(16);
370         try {
371             return cloneInternal(o, clones);
372         }
373         catch (final IllegalAccessException e) {
374             throw new CloningException("error during cloning of " + o, e);
375         }
376     }

379     public <T> T deepCloneDontCloneInstances(final T o, final Object... dontCloneThese) {
380         if (o == null)
381             return null;
382         if (!cloningEnabled)
383             return o;
384         if (dumpCloned != null) {
385             dumpCloned.startCloning(o.getClass());
386         }
387         final Map<Object, Object> clones = new IdentityHashMap<Object, Object>(16);
388         for (final Object dc : dontCloneThese) {
389             clones.put(dc, dc);
390         }
391         try {
392             return cloneInternal(o, clones);
393         }
394         catch (final IllegalAccessException e) {
395             throw new CloningException("error during cloning of " + o, e);
396         }
397     }

401      * shallow clones "o". This means that if c=shallowClone(o) then
402      * c!=o. Any change to c won't affect o.
403      *
404      * @param <T>
405      *        the type of o
406      * @param o
407      *        the object to be shallow-cloned
408      * @return a shallow clone of "o"
409      */
410     public <T> T shallowClone(final T o) {
411         if (o == null)
412             return null;
413         if (!cloningEnabled)
414             return o;
415         try {
416             return cloneInternal(o, null);
417         }
418         catch (final IllegalAccessException e) {
419             throw new CloningException("error during cloning of " + o, e);
420         }
421     }

445     /**
446      * decides if a class is to be considered immutable or not
447      *
448      * @param clz
449      *        the class under check
450      * @return true if the clz is considered immutable
451      */
452     private boolean isImmutable(final Class<?> clz) {
453         final Boolean isIm = immutables.get(clz);
454         if (isIm != null)
455             return isIm;
456         if (considerImmutable(clz))
457             return true;
458 
459         final Class<?> immutableAnnotation = getImmutableAnnotation();
460         for (final Annotation annotation : clz.getDeclaredAnnotations()) {
461             if (annotation.annotationType() == immutableAnnotation) {
462                 immutables.put(clz, Boolean.TRUE);
463                 return true;
464             }
465         }
466         Class<?> c = clz.getSuperclass();
467         while (c != null && c != Object.class) {
468             for (final Annotation annotation : c.getDeclaredAnnotations()) {
469                 if (annotation.annotationType() == Immutable.class) {
470                     final Immutable im = (Immutable) annotation;
471                     if (im.subClass()) {
472                         immutables.put(clz, Boolean.TRUE);
473                         return true;
474                     }
475                 }
476             }
477             c = c.getSuperclass();
478         }
479         immutables.put(clz, Boolean.FALSE);
480         return false;
481     }

484     @SuppressWarnings("unchecked")
485     protected <T> T cloneInternal(final T o, final Map<Object, Object> clones) throws IllegalAccessException {
486         if (o == null)
487             return null;
488         if (o == this)
489             return null; // don't clone the cloner!
490         if (ignoredInstances.containsKey(o))
491             return o;
492         if (o instanceof Enum)
493             return o;
494         final Class<T> clz = (Class<T>) o.getClass();
495         // skip cloning ignored classes
496         if (nullInstead.contains(clz))
497             return null;
498         if (ignored.contains(clz))
499             return o;
500         for (final Class<?> iClz : ignoredInstanceOf) {
501             if (iClz.isAssignableFrom(clz))
502                 return o;
503         }
504         if (isImmutable(clz))
505             return o;
506         if (o instanceof IFreezable) {
507             final IFreezable f = (IFreezable) o;
508             if (f.isFrozen())
509                 return o;
510         }
511         final Object clonedPreviously = clones != null ? clones.get(o) : null;
512         if (clonedPreviously != null)
513             return (T) clonedPreviously;
514 
515         final Object fastClone = fastClone(o, clones);
516         if (fastClone != null) {
517             if (clones != null) {
518                 clones.put(o, fastClone);
519             }
520             return (T) fastClone;
521         }
522 
523         if (dumpCloned != null) {
524             dumpCloned.startCloning(o.getClass());
525         }
526         if (clz.isArray()) { return cloneArray(o, clones); }
527 
528         return cloneObject(o, clones, clz);
529     }

532     // clones o, no questions asked!
533     private <T> T cloneObject(T o, Map<Object, Object> clones, Class<T> clz) throws IllegalAccessException {
534         final T newInstance = newInstance(clz);
535         if (clones != null) {
536             clones.put(o, newInstance);
537         }
538         final List<Field> fields = allFields(clz);
539         for (final Field field : fields) {
540             final int modifiers = field.getModifiers();
541             if (!Modifier.isStatic(modifiers)) {
542                 if (!(nullTransient && Modifier.isTransient(modifiers))) {
543                     // request by Jonathan : transient fields can be null-ed
544                     final Object fieldObject = field.get(o);
545                     final boolean shouldClone = (cloneSynthetics || !field.isSynthetic()) && (cloneAnonymousParent || !isAnonymousParent(field));
546                     final Object fieldObjectClone = clones != null ? (shouldClone ? cloneInternal(fieldObject, clones) : fieldObject) : fieldObject;
547                     field.set(newInstance, fieldObjectClone);
548                     if (dumpCloned != null && fieldObjectClone != fieldObject) {
549                         dumpCloned.cloning(field, o.getClass());
550                     }
551                 }
552             }
553         }
554         return newInstance;
555     }

558     @SuppressWarnings("unchecked")
559     private <T> T cloneArray(T o, Map<Object, Object> clones) throws IllegalAccessException {
560         final Class<T> clz = (Class<T>) o.getClass();
561         final int length = Array.getLength(o);
562         final T newInstance = (T) Array.newInstance(clz.getComponentType(), length);
563         if (clones != null) {
564             clones.put(o, newInstance);
565         }
566         if (clz.getComponentType().isPrimitive() || isImmutable(clz.getComponentType())) {
567             System.arraycopy(o, 0, newInstance, 0, length);
568         }
569         else {
570             for (int i = 0; i < length; i++) {
571                 final Object v = Array.get(o, i);
572                 final Object clone = clones != null ? cloneInternal(v, clones) : v;
573                 Array.set(newInstance, i, clone);
574             }
575         }
576         return newInstance;
577     }

585     /**
586      * copies all properties from src to dest. Src and dest can be of different class, provided they contain same field
587      * names/types
588      *
589      * @param src
590      *        the source object
591      * @param dest
592      *        the destination object which must contain as minimum all the fields of src
593      */
594     public <T, E extends T> void copyPropertiesOfInheritedClass(final T src, final E dest) {
595         if (src == null)
596             throw new IllegalArgumentException("src can't be null");
597         if (dest == null)
598             throw new IllegalArgumentException("dest can't be null");
599         final Class<? extends Object> srcClz = src.getClass();
600         final Class<? extends Object> destClz = dest.getClass();
601         if (srcClz.isArray()) {
602             if (!destClz.isArray())
603                 throw new IllegalArgumentException("can't copy from array to non-array class " + destClz);
604             final int length = Array.getLength(src);
605             for (int i = 0; i < length; i++) {
606                 final Object v = Array.get(src, i);
607                 Array.set(dest, i, v);
608             }
609             return;
610         }
611         final List<Field> fields = allFields(srcClz);
612         final List<Field> destFields = allFields(dest.getClass());
613         for (final Field field : fields) {
614             if (!Modifier.isStatic(field.getModifiers())) {
615                 try {
616                     final Object fieldObject = field.get(src);
617                     field.setAccessible(true);
618                     if (destFields.contains(field)) {
619                         field.set(dest, fieldObject);
620                     }
621                 }
622                 catch (final IllegalArgumentException e) {
623                     throw new RuntimeException(e);
624                 }
625                 catch (final IllegalAccessException e) {
626                     throw new RuntimeException(e);
627                 }
628             }
629         }
630     }

634      * reflection utils
635      */
636     private void addAll(final List<Field> l, final Field[] fields) {
637         for (final Field field : fields) {
638             if (!field.isAccessible()) {
639                 field.setAccessible(true);
640             }
641             l.add(field);
642         }
643     }

646     /**
647      * reflection utils, override this to choose which fields to clone
648      */
649     protected List<Field> allFields(final Class<?> c) {
650         List<Field> l = fieldsCache.get(c);
651         if (l == null) {
652             l = new LinkedList<Field>();
653             final Field[] fields = c.getDeclaredFields();
654             addAll(l, fields);
655             Class<?> sc = c;
656             while ((sc = sc.getSuperclass()) != Object.class && sc != null) {
657                 addAll(l, sc.getDeclaredFields());
658             }
659             fieldsCache.putIfAbsent(c, l);
660         }
661         return l;
662     }
