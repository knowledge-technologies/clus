27  package si.ijs.kt.clus.util.tools;
28  
29  import java.io.PrintWriter;
30  
31  import si.ijs.kt.clus.error.common.ClusError;
32  import si.ijs.kt.clus.error.common.ClusErrorList;
33  import si.ijs.kt.clus.model.ClusModel;
34  import si.ijs.kt.clus.model.ClusModelInfo;
35  import si.ijs.kt.clus.model.io.ClusModelCollectionIO;
36  import si.ijs.kt.clus.util.ClusLogger;
37  
38  
39  public class ClusShowModelInfo {
40  
41      public static void main(String[] args) {
42          if (args.length != 1) {
43              ClusLogger.info("Usage: java clus.tools.ClusShowModelInfo somefile.model");
44              System.exit(1);
45          }
46          try {
47              PrintWriter output = new PrintWriter(System.out);
48              ClusModelCollectionIO dot_model_file = ClusModelCollectionIO.load(args[0]);
49              int nb_models = dot_model_file.getNbModels();
50              output.println("This .model file contains: " + nb_models + " models.");
51              // for (int i = 0; i < nb_models; i++) {
52              for (int i = 0; i < 1; i++) {
53                  ClusModelInfo model_info = dot_model_file.getModelInfo(i);
54                  ClusModel model = model_info.getModel();
55                  output.println("Model: " + i + ", Name: " + model_info.getName());
56                  output.println("Size: " + model.getModelSize());
57                  output.println();
58                  model.printModel(output);
59                  ClusErrorList train_error = model_info.getTrainingError();
60                  output.println();
61                  if (train_error != null) {
62                      output.println("Training Error:");
63                      train_error.showError(output);
64                  }
65                  else {
66                      output.println("No Training Error Available");
67                  }
68                  ClusErrorList test_error = model_info.getTestError();
69                  output.println();
70                  if (test_error != null) {
71                      output.println("Testing Error:");
72                      test_error.showError(output);
73                      /* One can also get an individual error, or iterate over all error measures */
74                      ClusError err = test_error.getErrorByName("Classification Error");
75                      if (err != null) {
76                          for (int j = 0; j < err.getDimension(); j++) {
77                              output.println("Target: " + j + " error: " + err.getModelErrorComponent(j));
78                          }
79                      }
80                  }
81                  else {
82                      output.println("No Testing Error Available");
83                  }
84              }
85              output.flush();
86          }
87          catch (Exception e) {
88              System.err.println(e.getMessage());
89          }
90      }
91  }
