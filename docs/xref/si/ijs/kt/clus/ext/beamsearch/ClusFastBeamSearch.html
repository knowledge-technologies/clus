47  /*
48   * The optimizations over ClusBeamSearch.java the following:
49   * (1) It computes the heuristic value of the refinements of a given leaf
50   * only once (refinement = new test node with two child leaves that can be
51   * placed instead of the leaf). This can be done because these values do
52   * not depend on the rest of the tree. You just add the value of the entire
53   * tree before the refinement and subtract the value of the leaf that is
54   * being refined. This is done in the line:
55   * double heuristic = test.getHeuristicValue() + offset;
56   * The possible refinements for each leaf are stored in an object of the
57   * class ClusBeamAttrSelector in each leaf. It has an array storing the best
58   * split for each attribute. The "non-fast" version, on the other hand,
59   * repeats the compuation of all heuristics each time the leaf is considered
60   * for refinement.
61   * (2) If a new test is introduced in a tree, then the data of the node where
62   * the test is added must be split between the two new leaves based on the
63   * outcome of the test. The "fast" version postpones this step, i.e., it does
64   * not do this each time a new tree is added to the beam. The reason is that
65   * other trees might kick it out of the beam later and then this partitioning
66   * step (which takes some time) would be done in vain. In the "fast" version,
67   * these splits are therefore only done for the remaining trees after the entire
68   * beam of trees is refined. This is done in the updateModelRefinement()
69   * method.
70   */
71  
72  public class ClusFastBeamSearch extends ClusBeamSearch {

74      ClusBeamSizeConstraints m_Constr;

76  
77      public ClusFastBeamSearch(Clus clus) throws IOException, ClusException {
78          super(clus);
79          m_Constr = new ClusBeamSizeConstraints();
80      }

83      @Override
84      public ClusBeam initializeBeam(ClusRun run) throws Exception {
85          ClusBeam beam = super.initializeBeam(run);
86          ClusBeamModel model = beam.getBestModel();
87          initModelRecursive((ClusNode) model.getModel(), (RowData) run.getTrainingSet());
88          return beam;
89      }

91  
92      public void initModelRecursive(ClusNode node, RowData data) {
93          if (node.atBottomLevel()) {
94              ClusBeamAttrSelector attrsel = new ClusBeamAttrSelector();
95              attrsel.setData(data);
96              node.setVisitor(attrsel);
97          }
98          else {
99              NodeTest test = node.getTest();
100             for (int j = 0; j < node.getNbChildren(); j++) {
101                 ClusNode child = (ClusNode) node.getChild(j);
102                 RowData subset = data.applyWeighted(test, j);
103                 initModelRecursive(child, subset);
104             }
105         }
106     }

109     public void computeGlobalHeuristic(NodeTest test, RowData data, CurrentBestTestAndHeuristic sel) throws ClusException {
110         sel.reset(2);
111         data.calcPosAndMissStat(test, ClusNode.YES, sel.getPosStat(), sel.getMissStat());
112         double global_heur = m_Heuristic.calcHeuristic(sel.getTotStat(), sel.getPosStat(), sel.getMissStat());
113         test.setHeuristicValue(global_heur);
114     }

117     @Override
118     public void refineGivenLeaf(ClusNode leaf, ClusBeamModel root, ClusBeam beam, ClusAttrType[] attrs) throws Exception {
119         ClusBeamAttrSelector attrsel = (ClusBeamAttrSelector) leaf.getVisitor();
120         if (attrsel.isStopCrit()) {
121             /* stopping criterion already succeeded for this node */
122             if (m_Verbose)
123                 System.out.print("[S:" + leaf.getClusteringStat() + "]");
124             return;
125         }
126         RowData data = attrsel.getData();
127         if (!attrsel.hasEvaluations()) {
128             if (m_Induce.initSelectorAndStopCrit(leaf, data)) {
129                 /* stopping criterion succeeds */
130                 attrsel.setStopCrit(true);
131                 return;
132             }
133             CurrentBestTestAndHeuristic sel = m_Induce.getBestTest();
134             FindBestTest find = m_Induce.getFindBestTest();
135             m_Heuristic.setTreeOffset(0.0);
136             attrsel.newEvaluations(attrs.length);
137             for (int i = 0; i < attrs.length; i++) {
138                 sel.resetBestTest();
139                 ClusAttrType at = attrs[i];
140                 if (at instanceof NominalAttrType)
141                     find.findNominal((NominalAttrType) at, data, null);
142                 else
143                     find.findNumeric((NumericAttrType) at, data, null);
144                 // found good test for attribute ?
145                 if (sel.hasBestTest()) {
146                     NodeTest test = sel.updateTest();
147                     if (hasAttrHeuristic()) {
148                         // has attribute heuristic -> recompute global heuristic
149                         computeGlobalHeuristic(test, data, sel);
150                     }
151                     attrsel.setBestTest(i, test);
152                 }
153             }
154         }
155         double offset = root.getValue() - m_Heuristic.computeLeafAdd(leaf);
156         NodeTest[] besttests = attrsel.getBestTests();
157         if (m_Verbose)
158             ClusLogger.info("[M:" + beam.getMinValue() + "]");
159         for (int i = 0; i < besttests.length; i++) {
160             NodeTest test = besttests[i];
161             if (test != null) {
162                 double beam_min_value = beam.getMinValue();
163                 double heuristic = test.getHeuristicValue() + offset;
164                 if (heuristic >= beam_min_value) {
165                     if (m_Verbose)
166                         System.out.print("[+]");
167                     ClusNode ref_leaf = (ClusNode) leaf.cloneNode();
168                     ref_leaf.setTest(test);
169                     // visitor is removed in updateModelRefinement() !
170                     ref_leaf.setVisitor(leaf.getVisitor());
171                     if (getSettings().getGeneral().getVerbose() > 0)
172                         ClusLogger.info("Test: " + ref_leaf.getTestString() + " -> " + ref_leaf.getTest().getHeuristicValue() + " (" + ref_leaf.getTest().getPosFreq() + ")");
173                     int arity = ref_leaf.updateArity();
174                     for (int j = 0; j < arity; j++) {
175                         ClusNode child = new ClusNode();
176                         ref_leaf.setChild(child, j);
177                     }
178                     ClusNode root_model = (ClusNode) root.getModel();
179                     ClusNode ref_tree = root_model.cloneTreeWithVisitors(leaf, ref_leaf);
180                     ClusBeamModel new_model = new ClusBeamModel(heuristic, ref_tree);
181                     new_model.setRefinement(ref_leaf);
182                     new_model.setParentModelIndex(getCurrentModel());
183                     beam.addModel(new_model);
184                     setBeamChanged(true);
185                 }
186                 else {
187                     if (m_Verbose)
188                         System.out.print("[-:" + heuristic + "]");
189                 }
190             }
191         }
192     }

195     @Override
196     public void refineModel(ClusBeamModel model, ClusBeam beam, ClusRun run) throws Exception {
197         ClusNode tree = (ClusNode) model.getModel();
198         ClusBeamModel new_model = model.cloneModel();
199         /* Create new model because value can be different */
200         new_model.setValue(sanityCheck(model.getValue(), tree));
201         if (isBeamPostPrune()) {
202             ClusNode clone = tree.cloneTreeWithVisitors();
203             m_Constr.enforce(clone, m_MaxTreeSize);
204            
205             if (m_Constr.isFinished()) {
206                 model.setFinished(true);
207                 return;
208             }
209             if (m_Constr.isModified()) {
210                 new_model.setModel(clone);
211                 new_model.setValue(estimateBeamMeasure(clone));
212             }
213         }
214         else {
215             if (m_MaxTreeSize >= 0) {
216                 int size = tree.getNbNodes();
217                 if (size + 2 > m_MaxTreeSize) {
218                     model.setFinished(true);
219                     return;
220                 }
221             }
222         }
223         RowData train = (RowData) run.getTrainingSet();
224         ClusAttrType[] attrs = train.getSchema().getDescriptiveAttributes();
225         refineEachLeaf((ClusNode) new_model.getModel(), new_model, beam, attrs);
226     }

229     public void updateModelRefinement(ClusBeamModel model) throws ClusException {
230         /* Get data into children of last refinement */
231         ClusNode leaf = (ClusNode) model.getRefinement();
232         if (leaf == null)
233             return;
234         ClusBeamAttrSelector attrsel = (ClusBeamAttrSelector) leaf.getVisitor();
235         RowData data = attrsel.getData();
236         ClusStatManager mgr = m_Induce.getStatManager();
237         for (int j = 0; j < leaf.getNbChildren(); j++) {
238             ClusNode child = (ClusNode) leaf.getChild(j);
239             ClusBeamAttrSelector casel = new ClusBeamAttrSelector();
240             RowData subset = data.applyWeighted(leaf.getTest(), j);
241             child.initTargetStat(mgr, subset);
242             child.initClusteringStat(mgr, subset);
243             casel.setData(subset);
244             child.setVisitor(casel);
245         }
246         leaf.setVisitor(null);
247         model.setRefinement(null);
248     }

251     @Override
252     public void refineBeam(ClusBeam beam, ClusRun run) throws Exception {
253         super.refineBeam(beam, run);
254         ArrayList models = beam.toArray();
255         for (int i = 0; i < models.size(); i++) {
256             ClusBeamModel model = (ClusBeamModel) models.get(i);
257             updateModelRefinement(model);
258         }
259     }
