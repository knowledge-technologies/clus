75      public void pruneInitialize(ClusNode node, int size) throws ClusException {
76          recursiveInitialize(node, size);
77          // compute error for each internal node and leaf
78          if (isUsingAdditiveError()) {
79              recursiveInitializeError(node, m_Data);
80          }
81          else {
82              recursiveInitializeErrorFromStatistic(node);
83          }
84      }

87      public void pruneExecute(ClusNode node, int size) {
88          computeCosts(node, size);
89          pruneToSizeK(node, size);
90      }

111     @Override
112     public void prune(int result, ClusNode node) throws ClusException {
113         if (m_MaxError == null) {
114             int size = m_MaxSize[result];
115             int orig = node.getNbNodes();
116             ClusLogger.info("Pruning to size (" + orig + "): " + size);
117             pruneInitialize(node, size);
118             pruneExecute(node, size);
119         }
120         else {
121             if (m_MaxSize.length == 0) {
122                 pruneMaxError(node, node.getNbNodes());
123             }
124             else {
125                 pruneMaxError(node, m_MaxSize[result]);
126             }
127         }
128         node.clearVisitors();
129     }

132     @Override
133     public void sequenceInitialize(ClusNode node) {
134         int max_size = node.getNbNodes();
135         int abs_max = getMaxSize();
136         if (abs_max != -1 && max_size > abs_max)
137             max_size = abs_max;
138         if ((max_size % 2) == 0)
139             max_size--;
140         m_MaxIndex = max_size;
141         m_CrIndex = m_MaxIndex;
142         recursiveInitialize(node, max_size);
143         setOriginalTree(node);
144     }

147     @Override
148     public void sequenceReset() {
149         m_CrIndex = m_MaxIndex;
150     }

153     @Override
154     public ClusNode sequenceNext() throws ClusException {
155         if (m_CrIndex > 0) {
156             ClusNode cloned = getOriginalTree().cloneTreeWithVisitors();
157             pruneExecute(cloned, m_CrIndex);
158             m_CrIndex -= 2;
159             return cloned;
160         }
161         else {
162             return null;
163         }
164     }

168     public void sequenceToElemK(ClusNode node, int k) {
169         pruneExecute(node, m_MaxIndex - 2 * k);

173     public void pruneMaxError(ClusNode node, int maxsize) throws ClusException {
174         pruneInitialize(node, maxsize);
175         int constr_ok_size = maxsize;
176         for (int crsize = 1; crsize <= maxsize; crsize += 2) {
177             ClusNode copy = node.cloneTreeWithVisitors();
178             pruneExecute(copy, crsize);
179             ClusErrorList cr_err = m_ErrorMeasure.getErrorClone();
180             ClusError err = cr_err.getFirstError();
181             // Can be made more efficient :-)
182             TreeErrorComputer.computeErrorStandard(copy, m_Data, err);
183             cr_err.setNbExamples(m_Data.getNbRows());
184             if (m_MaxError.length == 1) {
185                 double max_err = m_MaxError[0];
186                 // ClusLogger.info("dans SizeConstraintPruning, maxError = "+max_err);
187                 if (err.getModelError() <= max_err) {
188                     constr_ok_size = crsize;
189                     break;
190                 }
191             }
192             else {
193                 boolean isOK = true;
194                 for (int i = 0; i < m_MaxError.length; i++) {
195                     double err_i = m_MaxError[i];
196                     if (!Double.isNaN(err_i)) {
197                         if (err.getModelErrorComponent(i) > err_i) {
198                             isOK = false;
199                         }
200                     }
201                 }
202                 if (isOK) {
203                     constr_ok_size = crsize;
204                     break;
205                 }
206             }
207         }
208         pruneExecute(node, constr_ok_size);
209     }

212     private static void recursiveInitialize(ClusNode node, int size) {
213         /* Create array for each node */
214         SizeConstraintVisitor visitor = new SizeConstraintVisitor(size);
215         node.setVisitor(visitor);
216         /* Recursively visit children */
217         for (int i = 0; i < node.getNbChildren(); i++) {
218             ClusNode child = (ClusNode) node.getChild(i);
219             recursiveInitialize(child, size);
220         }
221     }
222 
223 
224     private void recursiveInitializeError(ClusNode node, RowData data) throws ClusException {
225         // get visitor and error measure
226         SizeConstraintVisitor visitor = (SizeConstraintVisitor) node.getVisitor();
227         ClusErrorList parent = getAdditiveError();
228         ClusError err = parent.getFirstError();
229         // compute error for given node
230         parent.reset();
231         TreeErrorComputer.computeErrorNode(node, data, err);
232         parent.setNbExamples(data.getNbRows());
233         visitor.error = err.getModelErrorAdditive();
234         // do recursive call for each child
235         NodeTest tst = node.getTest();
236         for (int i = 0; i < node.getNbChildren(); i++) {
237             ClusNode child = (ClusNode) node.getChild(i);
238             RowData subset = data.applyWeighted(tst, i);
239             recursiveInitializeError(child, subset);
240         }
241     }

244     private void recursiveInitializeErrorFromStatistic(ClusNode node) {
245         // compute error for given node
246         SizeConstraintVisitor visitor = (SizeConstraintVisitor) node.getVisitor();
247         visitor.error = node.getTargetStat().getError(m_TargetWeights);
248         // do recursive call for each child
249         for (int i = 0; i < node.getNbChildren(); i++) {
250             ClusNode child = (ClusNode) node.getChild(i);
251             recursiveInitializeErrorFromStatistic(child);
252         }
253     }

262     public double computeCosts(ClusNode node, int l) {
263         SizeConstraintVisitor visitor = (SizeConstraintVisitor) node.getVisitor();
264         if (visitor.computed[l])
265             return visitor.cost[l];
266         if (l < 3 || node.atBottomLevel()) {
267             visitor.cost[l] = computeNodeCost(node);
268         }
269         else {
270             visitor.cost[l] = computeNodeCost(node);
271             ClusNode ch1 = (ClusNode) node.getChild(0);
272             ClusNode ch2 = (ClusNode) node.getChild(1);
273             for (int k1 = 1; k1 <= l - 2; k1++) {
274                 int k2 = l - k1 - 1;
275                 double cost = computeCosts(ch1, k1) + computeCosts(ch2, k2);
276                 // ClusLogger.info("cost "+cost+" "+arr[l].cost);
277                 if (cost < visitor.cost[l]) {
278                     visitor.cost[l] = cost;
279                     visitor.left[l] = k1;
280                 }
281             }
282         }
283         // ClusLogger.info("Node: "+node+" cost "+l+" = "+arr[l]);
284         visitor.computed[l] = true;
285         return visitor.cost[l];
286     }

289     public static void pruneToSizeK(ClusNode node, int l) {
290         if (node.atBottomLevel())
291             return;
292         SizeConstraintVisitor visitor = (SizeConstraintVisitor) node.getVisitor();
293         if (l < 3 || visitor.left[l] == 0) {
294             node.makeLeaf();
295         }
296         else {
297             int k1 = visitor.left[l];
298             int k2 = l - k1 - 1;
299             pruneToSizeK((ClusNode) node.getChild(0), k1);
300             pruneToSizeK((ClusNode) node.getChild(1), k2);
301         }
302     }
