183     public void initialize(ClusSchema schema, Settings settMain, Clus clus) throws ClusException, IOException {
184         m_BagClus = clus;
185         m_Timer = new StopWatch();
186         
187         // optimize if not XVAL and HMC
188 
189         SettingsEnsemble sett = settMain.getEnsemble();
190         
191         m_WriteOOB = sett.shouldEstimateOOB();
192         
193         m_MaxTime = sett.getTimeBudget().getValue();
194 
195         m_OptMode = sett.shouldOptimizeEnsemble() && (
196                 m_StatManager.getTargetMode() == ClusStatManager.Mode.HIERARCHICAL ||
197                 m_StatManager.getTargetMode() == ClusStatManager.Mode.REGRESSION ||
198                 m_StatManager.getTargetMode() == ClusStatManager.Mode.CLASSIFY);
199 
200         m_EnsembleROSSubspaceSize = sett.calculateNbRandomAttrSelected(schema, RandomAttributeTypeSelection.Clustering);
201 
202         m_OutEnsembleAt = sett.getNbBaggingSets().getIntVectorSorted();
203         m_NbMaxBags = getNbTrees(m_OutEnsembleAt.length - 1);
204         settMain.getExperimental();
205         m_FeatRank = sett.shouldPerformRanking() && !SettingsExperimental.IS_XVAL;
206         if (m_FeatRank && !sett.shouldEstimateOOB() && sett.getRankingMethods().contains(EnsembleRanking.RForest)) {
207             System.err.println("For Feature Ranking RForest, OOB estimate of error should also be performed.");
208             System.err.println("OOB Error Estimate is set to true.");
209 
210             sett.setOOBestimate(true);
211         }
212         if (sett.shouldEstimateOOB())
213             m_OOBEstimation = new ClusOOBErrorEstimate(getStatManager().getTargetMode(), settMain);
214         settMain.getOutput().setWriteOOBFile(true);
215         if (m_FeatRank) {
216             if (m_BagClus.getSettings().getMLC().getSectionMultiLabel().isEnabled()) {
217                 List<MultiLabelMeasures> rankingMeasures = m_BagClus.getSettings().getMLC().getMultiLabelRankingMeasures();
218 
219                 if (rankingMeasures.contains(MultiLabelMeasures.All)) {
220                     m_BagClus.getSettings().getMLC().setToAllMultiLabelRankingMeasures();
221                 }
222             }
223         }
224         setNumberOfThreads(sett.getNumberOfThreads());
225 
226         sett.updateNbRandomAttrSelected(schema);
227         sett.determineBoostrapping();
228     }

248     public HashMap<EnsembleRanking, Integer> setNbFeatureRankings(ClusSchema schema, ClusStatManager mgr) {
249         HashMap<EnsembleRanking, Integer> nbRankings = new HashMap<>();
250         for (int forest = 0; forest < m_FeatureRankings.length; forest++) {
251             int nbTrees = getNbTrees(forest);
252             HashMap<EnsembleRanking, ClusEnsembleFeatureRanking> rankings = m_FeatureRankings[forest].getRankings();
253             for (EnsembleRanking r : rankings.keySet()){
254                 int nb = setNbFeatureRankings(r, rankings.get(r), schema, mgr, nbTrees);
255                 if (nbRankings.containsKey(r)){
256                     if (nbRankings.get(r).intValue() != nb){
257                         throw new RuntimeException("Wrong number of rankings.");
258                     } 
259                 } else {
260                     nbRankings.put(r, nb);
261                 }
262             }
263         }
264         return nbRankings;
265     }

268     /**
269      * Sets the number of feature rankings and the ranking description.
270      * 
271      * @param franking
272      * @param schema
273      * @param mgr
274      */
275     public int setNbFeatureRankings(EnsembleRanking rankingType, ClusEnsembleFeatureRanking franking, ClusSchema schema, ClusStatManager mgr, int nbTrees) {
276         ClusStatistic clusteringStat = mgr.getStatistic(AttributeUseType.Clustering);
277         // Settings sett = schema.getSettings();
278         SettingsEnsemble sett = schema.getSettings().getEnsemble();
279 
280         int nbRankings = 0;
281         ClusAttrType[] clusteringAttrs = schema.getAllAttrUse(AttributeUseType.Clustering);
282 
283         if (clusteringAttrs.length < 2 && sett.shouldPerformRankingPerTarget()) {
284             System.err.println("Situation:");
285             System.err.println("- there is only one clustering attribute");
286             System.err.println("- per target ranking is set to Yes");
287             System.err.println("Consequences:");
288             System.err.println("- per target ranking == overAll ranking");
289             System.err.println("- per target ranking will be set to No");
290             sett.setPerformRankingPerTarget(false);
291         }
292         ArrayList<String> rankingNames = new ArrayList<String>();
293         switch (rankingType) {
294             case RForest:
295                 ClusErrorList errLst = franking.computeErrorList(schema, mgr);
296                 int nbErrors = errLst.getNbErrors();
297                 for (int i = 0; i < nbErrors; i++) {
298                     nbRankings++; // overall
299                     rankingNames.add(String.format("%s:%s", errLst.getError(i).getName(), "overall"));
300 
301                     if (sett.shouldPerformRankingPerTarget() && (errLst.getError(i) instanceof ComponentError)) {
302                         int nbDim = errLst.getError(i).getDimension();
303                         nbRankings += nbDim;
304                         for (int j = 0; j < nbDim; j++) {
305                             rankingNames.add(String.format("%s:%s", errLst.getError(i).getName(), clusteringAttrs[j].getName()));
306                         }
307                     }
308                 }
309                 franking.setRForestFimpHeader(rankingNames);
310                 break;
311             case Genie3:
312                 nbRankings++; // overall
313                 rankingNames.add(String.format("overAll"));
314                 if (sett.shouldPerformRankingPerTarget()) {
315                     if (!(clusteringStat instanceof ComponentStatistic)) {
316                         System.err.println("Cannot perform per-target ranking for the given type(s) of targets!");
317                         System.err.println("This option is now set to false.");
318                         sett.setPerformRankingPerTarget(false);
319                     }
320                     else {
321                         int nbComponents = ((ComponentStatistic) clusteringStat).getNbStatisticComponents();
322                         nbRankings += nbComponents;
323                         for (int j = 0; j < nbComponents; j++) {
324                             rankingNames.add(String.format("%s", clusteringAttrs[j].getName()));
325                         }
326                     }
327                 }
328                 franking.setGenie3FimpHeader(rankingNames);
329                 break;
330             case Symbolic:
331                 String[] weights = sett.getSymbolicWeights();
332                 nbRankings = weights.length;
333                 franking.setSymbolicFimpHeader(weights);
334                 break;
335             default:
336                 throw new RuntimeException("Wrong feature ranking method: " + rankingType.toString());
337         }
338         franking.setNbFeatureRankings(nbRankings);
339         return nbRankings;
340     }

488             }
489         }

492             for (int i = 0; i < m_Optimizations.length; i++) {
493                 m_Optimizations[i].roundPredictions();
494             }
495         }
496 
497         postProcessForest(cr);

501          * output hierarchy.txt file (for HMC)

503          * this, potentially huge, file is re-written for every tree
504          */

510         // This section is for calculation of the similarity in the ensemble

537     public void induceRForestNoBootstrap(ClusRun cr) throws Exception {
538         // long summ_time = 0; // = ResourceInfo.getTime();
539 
540         TupleIterator train_iterator = m_OptMode ? cr.getTrainIter() : null; // = train set iterator
541         TupleIterator test_iterator = m_OptMode ? cr.getTestIter() : null; // = test set iterator
542

544         if (cr.getIsInternalXValRun()) {
545         	// SSL internal X-val
546         	max_number_of_bags = m_Schema.getSettings().getSSL().getNumberOfTreesSupervisionOptimisation(m_NbMaxBags);
547         }
548

676     public void induceBagging(ClusRun cr) throws ClusException, IOException, InterruptedException, ExecutionException {
677         int nbrows = cr.getTrainingSet().getNbRows();
678         ((RowData) cr.getTrainingSet()).addIndices(); // necessary to be able to print paths
679         if (cr.getTestSet() != null) {
680             cr.getTestSet().addIndices();
681         }
682 
683         TupleIterator train_iterator = m_OptMode ? cr.getTrainIter() : null; // = train set iterator
684         TupleIterator test_iterator = m_OptMode ? cr.getTestIter() : null; // = test set iterator
685 
686         OOBSelection oob_total = null; // = total OOB selection
687         OOBSelection oob_sel = null; // = current OOB selection
688 
689         SettingsEnsemble sett = getSettings().getEnsemble();
690 
691         // - Added by Jurica L. (IJS)
692         // - check if dataset contains unlabeled examples, then use different sampling, labeled and unlabled examples
693         // are sampled separately
694         int nbUnlabeled = 0;
695         nbUnlabeled = ((RowData) cr.getTrainingSet()).getNbUnlabeled();
696         if (nbUnlabeled > 0) {
697             if (nbUnlabeled > 0) { // sort datasets, so that labeled examples come first and unlabeled second, needed
698                                    // for semi-supervised bagging selection
699                 ((RowData) cr.getTrainingSet()).sortLabeledFirst();
700             }
701         }
702         // - end added by Jurica
703 
704         // We store the old maxDepth to this if needed. Thus we get the right depth to .out files etc.
705         int origMaxDepth = -1;
706         if (sett.isEnsembleRandomDepth()) {
707             // Random depth for the ensembles
708             // The original Max depth is used as the average
709             origMaxDepth = getSettings().getConstraints().getTreeMaxDepth();
710         }
711         BagSelection msel = null;
712         int[] bagSelections = sett.getBagSelection().getIntVectorSorted();
713         // bagSelections is either -1, 0, a value in [1,Iterations], or 2 values in [1,Iterations]
714 
715         Random bagSeedGenerator = new Random(getSettings().getGeneral().getRandomSeed());
716         
717         int max_number_of_bags = m_NbMaxBags;
718         if (cr.getIsInternalXValRun()) {
719         	// SSL internal X-val
720         	max_number_of_bags = m_Schema.getSettings().getSSL().getNumberOfTreesSupervisionOptimisation(m_NbMaxBags);
721         }
722         
723         int[] seeds = new int[max_number_of_bags];
724         for (int i = 0; i < max_number_of_bags; i++) {
725             seeds[i] = bagSeedGenerator.nextInt();
726         }
727 
728         ClusStatManager clonedStatManager = cloneStatManager(cr.getStatManager());
729 
730         ExecutorService executor = Executors.newFixedThreadPool(m_NbThreads);
731         ArrayList<Future<OneBagResults>> bagResults = new ArrayList<Future<OneBagResults>>();
732 
733         if (bagSelections[0] == -1) {
734             // normal bagging procedure
735             m_Timer.reset();
736             m_Timer.start();
737             for (int i = 1; i <= max_number_of_bags; i++) {
738                 ClusRandomNonstatic rnd = new ClusRandomNonstatic(seeds[i - 1]);
739 
740                 if (nbUnlabeled > 0) {
741                     msel = new BagSelectionSemiSupervised(nbrows, cr.getTrainingSet().getNbRows() - nbUnlabeled, nbUnlabeled, rnd);
742                 }
743                 else {
744                     msel = new BagSelection(nbrows, sett.getEnsembleBagSize(), rnd);
745                 }
746 
747                 if (sett.shouldEstimateOOB()) { // OOB estimate is on
748                     oob_sel = new OOBSelection(msel);
749                     if (i == 1) { // initialization
750                         oob_total = new OOBSelection(msel);
751                     }
752                     else {
753                         oob_total.addToThis(oob_sel);
754                     }
755                 }
756 
757                 if (sett.isVotingOOBWeighted() && !sett.shouldEstimateOOB()) {
758                     // find oob selection only if OOB is not turned on already
759                     oob_sel = new OOBSelection(msel);
760                 }
761 
762                 // OOBSelection current_oob_total = cloner.deepClone(oob_total);
763                 // induceOneBag(cr, i, origMaxDepth, oob_sel, oob_total, train_iterator, test_iterator, msel, rnd);
764                 // InduceOneBagCallable worker = new InduceOneBagCallable(this, cr, i, origMaxDepth, oob_sel,
765                 // current_oob_total, train_iterator, test_iterator, msel, rnd, clonedStatManager);
766                 InduceOneBagCallable worker = new InduceOneBagCallable(this, cr, i, origMaxDepth, oob_sel, train_iterator, test_iterator, msel, rnd, clonedStatManager); // statManagerClones[i
767                                                                                                                                                                          // -
768                                                                                                                                                                          // 1]
769                 Future<OneBagResults> submit = executor.submit(worker);
770                 bagResults.add(submit);
771             }
772         }
773         else if (bagSelections[0] == 0) {
774             // we assume that files _bagI.model exist, for I=1..m_NbMaxBags and build the forest from them
775             makeForestFromBags(cr, train_iterator, test_iterator);
776         }
777         else {
778             // only one or a range of bags needs to be computed (and the model output) e.g. because we want to run the
779             // forest in parallel,
780             // or because we want to add a number of trees to an existing forest.
781             for (int i = 1; i < bagSelections[0]; i++) {
782                 // we eventually want the same bags as when computing them sequentially.
783                 msel = new BagSelection(nbrows, sett.getEnsembleBagSize(), null); // not really necessary any more
784             }
785             for (int i = bagSelections[0]; i <= bagSelections[1]; i++) {
786                 ClusRandomNonstatic rnd = new ClusRandomNonstatic(seeds[i - 1]);
787 
788                 if (nbUnlabeled > 0) {
789                     msel = new BagSelectionSemiSupervised(nbrows, cr.getTrainingSet().getNbRows() - nbUnlabeled, nbUnlabeled, rnd);
790                 }
791                 else {
792                     msel = new BagSelection(nbrows, sett.getEnsembleBagSize(), rnd);
793                 }
794 
795                 if (sett.shouldEstimateOOB()) { // OOB estimate is on
796                     oob_sel = new OOBSelection(msel);
797                 }
798                 // induceOneBag(cr, i, origMaxDepth, oob_sel, oob_total, train_iterator, test_iterator, msel, rnd);
799                 // InduceOneBagCallable worker = new InduceOneBagCallable(this, cr, i, origMaxDepth, oob_sel, oob_total,
800                 // train_iterator, test_iterator, msel, rnd, clonedStatManager); //
801                 // cloner.deepClone(cr.getStatManager())
802                 InduceOneBagCallable worker = new InduceOneBagCallable(this, cr, i, origMaxDepth, oob_sel, train_iterator, test_iterator, msel, rnd, clonedStatManager); // cloner.deepClone(cr.getStatManager())
803                 Future<OneBagResults> submit = executor.submit(worker);
804                 bagResults.add(submit);
805             }
806         }
807 
808         executor.shutdown();
809         executor.awaitTermination(Integer.MAX_VALUE, TimeUnit.SECONDS);
810 
811         for (int i = 1; i <= bagResults.size(); i++) { // must start with 1 and end with the size
812             Future<OneBagResults> future = bagResults.get(i - 1);
813             try {
814                 OneBagResults results = future.get();
815                 if (results.getInductionTime() < 0)
816                     continue;
817                 if (!m_OptMode) {
818                     addModelToForests(results.getModel(), i);
819                 }
820                 if (sett.shouldPerformRanking()) {
821                     updateFeatureRankings(results.getModelIndex(), results.getFimportances());
822                 }
823 
824                 ClusRun crSingle = results.getSingleRun();
825                 // Lines that are commented out in this if-section are already executed in induceOneBag
826                 if (checkToOutEnsemble(i) && (sett.getBagSelection().getIntVectorSorted()[0] == -1)) {
827                     // crSingle.setInductionTime(m_SummTime);
828 
829                     postProcessForest(crSingle);
830                     if (m_WriteOOB) {  // sett.shouldEstimateOOB()
831                         if (i == max_number_of_bags) {
832                             m_OOBEstimation.postProcessForestForOOBEstimate(crSingle, oob_total, (RowData) cr.getTrainingSet(), m_BagClus, "");
833                         }
834                         else {
835                             m_OOBEstimation.postProcessForestForOOBEstimate(crSingle, oob_total, (RowData) cr.getTrainingSet(), m_BagClus, "_" + i + "_");
836                         }
837                     }
838                 }
839 
840             }
841             catch (InterruptedException e) {
842                 throw new ClusException(e.toString());
843             }
844         }
845 
846         // Restore the old maxDepth
847         if (origMaxDepth != -1) {
848             getSettings().getConstraints().setTreeMaxDepth(origMaxDepth);
849         }
850     }

878     public OneBagResults induceOneBag(ClusRun cr, int i, int origMaxDepth, OOBSelection oob_sel, TupleIterator train_iterator, TupleIterator test_iterator, BagSelection msel, ClusRandomNonstatic rnd, ClusStatManager unmodifiedManager) throws Exception {
879         if (m_MaxTime > 0 && m_Timer.getTime() / 1000 >= m_MaxTime) {
880             return new OneBagResults(null, null, null, -1, 0);
881         }
882         long one_bag_time = ResourceInfo.getTime();
883         SettingsEnsemble sett = cr.getStatManager().getSettings().getEnsemble();
884         SettingsOutput seto = cr.getStatManager().getSettings().getOutput();
885         boolean canForgetTheRun = true;
886 
887         ClusRun crSingle = null;
888         DepthFirstInduce ind = null;
889         ClusStatManager mgr = cloneStatManager(unmodifiedManager);
890 
891         ClusLogger.info("Bag: " + i);
892 
893         if (sett.isEnsembleRandomDepth()) {
894             // Set random tree max depth
895             getSettings().getConstraints().setTreeMaxDepth(GDProblem.randDepthWighExponentialDistribution(rnd.nextDouble(ClusRandomNonstatic.RANDOM_INT_RANFOR_TREE_DEPTH), origMaxDepth));
896         }
897 
898         if (sett.getEnsembleBootstrapping().equals(EnsembleBootstrapping.No)) {
899             crSingle = new ClusRun(cr.getTrainingSet(), cr.getSummary());
900         }
901         else {
902             crSingle = m_BagClus.partitionDataBasic(cr.getTrainingSet(), msel, cr.getSummary(), i);
903         }
904 
905         if (getSchema().isSparse()) {
906             ind = new DepthFirstInduceSparse(this, mgr, true);
907         }
908         else {
909             ind = new DepthFirstInduce(this, mgr, true);
910         }
911 
912         ind.initialize();
913 
914         crSingle.getStatManager().initClusteringWeights();
915 
916         ind.getStatManager().initClusteringWeights();
917 
918         initializeROSModel(ind, i);
919 
920         ClusModel model = ind.induceSingleUnpruned(crSingle, rnd);
921         one_bag_time = ResourceInfo.getTime() - one_bag_time;
922 
923         updateCounts((ClusNode) model, i);
924 
925         // OOB estimate for the parallel implementation is done in makeForestFromBags method <--- matejp: This is some
926         // old parallelisation
927         if (sett.shouldEstimateOOB() && (sett.getBagSelection().getIntVectorSorted()[0] == -1)) {
928             m_OOBEstimation.updateOOBTuples(oob_sel, (RowData) cr.getTrainingSet(), model, i);
929         }
930 
931         // Calculation of OOB for EnsembleVotingType.OOBModelWeighted/OOBTargetWeighted predictions; we do it
932         // independently of original OOB calculations above
933         if (sett.isVotingOOBWeighted()) {
934             saveOOBEstimates(model, oob_sel, (RowData) cr.getTrainingSet(), i);
935         }
936 
937         HashMap<EnsembleRanking, HashMap<String, double[][]>> fimportances = new HashMap<>();
938         if (m_FeatRank) {
939             for (EnsembleRanking r : sett.getRankingMethods()){
940                 ClusEnsembleFeatureRanking aRanking = m_FeatureRankings[0].getRankings().get(r);
941                 switch (r) {
942                     case RForest:
943                         fimportances.put(r, aRanking.calculateRFimportance(model, cr, oob_sel, rnd, ind.getStatManager()));
944                         break;
945     
946                     case Symbolic:
947                         String[] weights = sett.getSymbolicWeights();
948                         fimportances.put(r, aRanking.calculateSYMBOLICimportanceIteratively((ClusNode) model, weights));
949                         break;
950     
951                     case Genie3:
952                         fimportances.put(r, aRanking.calculateGENIE3importanceIteratively((ClusNode) model, ind.getStatManager()));
953                         break;
954     
955                     default:
956                         throw new RuntimeException("Unknown ranking method: " + sett.getRankingMethodName());
957                 }
958             }
959         }
960         
961         if (seto.shouldWritePerBagModelFiles() && (sett.getBagSelection().getIntVectorSorted()[0] != -1 || sett.isPrintEnsembleModelFiles())) {
962             ClusModelCollectionIO io = new ClusModelCollectionIO();
963             ClusModelInfo orig_info = crSingle.addModelInfo("Original");
964             orig_info.setModel(model);
965             // cut the trees
966             ArrayList<ClusNode> submodels = ClusNode.cutTrees((ClusNode) model);
967             for(int s = 0; s < submodels.size(); s++) {
968             	ClusModelInfo subInfo = crSingle.addModelInfo(ClusNode.getSubmodelName(s));
969             	subInfo.setModel(submodels.get(s));
970             }            
971             m_BagClus.saveModels(crSingle, io);
972             io.save(m_BagClus.getSettings().getGeneric().getFileAbsolute(cr.getStatManager().getSettings().getGeneric().getAppName() + "_bag" + i + ".model"));
973             // join them back so that the other stuff will still work
974             ((ClusNode) model).joinWithSubmodels(submodels);
975         }
976 
977         if (m_OptMode) {
978             // m_Optimization.updatePredictionsForTuples(model, train_iterator, test_iterator);
979             updatePredictionsForTuples(model, train_iterator, test_iterator, i);
980 
981             if (cr.getStatManager().getSettings().getOutput().isOutputPythonModel()) {
982                 writeTreeToTempPythonFile(cr, model, i);
983             }
984 
985             model = null;
986             unmodifiedManager = null;
987             ind = null;
988         }
989 
990         // printing paths taken by each example in each tree (used in ICDM'11 paper on "Random forest based feature
991         // induction")
992         if (sett.isPrintEnsemblePaths()) {
993             PrintWriter pw = new PrintWriter(new OutputStreamWriter(new FileOutputStream("tree_" + i + ".path")));
994             ((ClusNode) model).numberCompleteTree();
995             ((ClusNode) model).printPaths(pw, "", "", (RowData) cr.getTrainingSet(), oob_sel, false);
996             if (cr.getTestSet() != null) {
997                 ((ClusNode) model).printPaths(pw, "", "", cr.getTestSet(), null, true);
998             }
999         }
1000 
1001         // Valid only when test set is supplied
1002         if (checkToOutEnsemble(i) && sett.getBagSelection().getIntVectorSorted()[0] == -1) {
1003             // crSingle.setInductionTime(m_SummTime);
1004             crSingle.setInductionTime(one_bag_time);
1005             canForgetTheRun = false;
1006 
1007             if (m_OptMode && (i != m_NbMaxBags)) {
1008                 if (m_NbThreads > 1) {
1009                     giveParallelisationWarning(ParallelTrap.Optimization);
1010                 }
1011             }
1012         }
1013 
1014         if (canForgetTheRun) {
1015             crSingle.deleteData();
1016             crSingle.setModels(new ArrayList<ClusModelInfo>());
1017         }
1018 
1019 
1020         if (m_MaxTime > 0 && m_Timer.getTime() / 1000 >= m_MaxTime) {
1021             return new OneBagResults(null, null, null, -1, 0);
1022         }
1023         return new OneBagResults(model, fimportances, crSingle, one_bag_time, i);
1024     }

1043     private ClusError calculateOOBWeights(ClusModel model, OOBSelection oob_sel, RowData data) throws ClusException, InterruptedException {
1044         ClusError error = null;
1045         ClusErrorList OOBErrorList = new ClusErrorList();
1046 
1047         switch (getStatManager().getTargetMode()) {
1048             case REGRESSION:
1049                 // RMSE (less is better)
1050                 error = new RMSError(OOBErrorList, this.getSchema().getNumericAttrUse(AttributeUseType.Target));
1051                 break;
1052 
1053             case CLASSIFY:
1054                 // Accuracy (more is better)
1055                 error = new Accuracy(OOBErrorList, this.getSchema().getNominalAttrUse(AttributeUseType.Target));
1056                 break;
1057 
1058             case HIERARCHICAL:
1059                 // Pooled AUPRC (more is better)
1060                 // error = new HierErrorMeasures(OOBErrorList, m_StatManager.getHier(),
1061                 // m_StatManager.getSettings().getHMLC().getRecallValues().getDoubleVector(),
1062                 // HierarchyMeasures.PooledAUPRC, m_StatManager.getSettings().getOutput().isWriteCurves(),
1063                 // getSettings().getOutput().isGzipOutput());
1064                 break;
1065 
1066             default:
1067                 String msg = "Unable to calculate OOB estimates!";
1068                 ClusLogger.severe(msg);
1069                 throw new RuntimeException(msg);
1070         }
1071 
1072         OOBErrorList.addError(error);
1073         DataTuple tuple;
1074 
1075         for (int example = 0; example < data.getNbRows(); example++) {
1076             if (oob_sel.isSelected(example)) {

1395     /**
1396      * Selects random subspaces.
1397      * 
1398      * @param attrs
1399      *        -- For which attributes
1400      * @param select
1401      *        -- How many
1402      */
1403     public static ClusAttrType[] selectRandomSubspaces(ClusAttrType[] attrs, int select, int randomizerVersion, ClusRandomNonstatic rand) {
1404         int origsize = attrs.length;
1405         int[] samples = new int[origsize];
1406         int rnd;
1407         boolean randomize = true;
1408         int i = 0;
1409         if (rand == null) {
1410             while (randomize) {
1411                 rnd = ClusRandom.nextInt(randomizerVersion, origsize);
1412                 if (samples[rnd] == 0) {
1413                     samples[rnd]++;
1414                     i++;
1415                 }
1416                 randomize = i != select;
1417             }
1418         }
1419         else {
1420             while (randomize) {
1421                 rnd = rand.nextInt(randomizerVersion, origsize);
1422                 if (samples[rnd] == 0) {
1423                     samples[rnd]++;
1424                     i++;
1425                 }
1426                 randomize = i != select;
1427             }
1428         }
1429 
1430         ClusAttrType[] result = new ClusAttrType[select];
1431         int res = 0;
1432         for (int k = 0; k < origsize; k++) {
1433             if (samples[k] != 0) {
1434                 result[res] = attrs[k];
1435                 res++;
1436             }
1437         }
1438         return result;
1439     }
