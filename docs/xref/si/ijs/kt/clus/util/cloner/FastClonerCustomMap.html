13  @SuppressWarnings({ "unchecked", "rawtypes" })
14  public abstract class FastClonerCustomMap<T extends Map> implements IFastCloner {
15  
16      @Override
17      public Object clone(final Object t, final IDeepCloner cloner, final Map<Object, Object> clones) {
18          final T m = (T) t;
19          final T result = getInstance((T) t);
20          final Set<Map.Entry<Object, Object>> entrySet = m.entrySet();
21          for (final Map.Entry e : entrySet) {
22              final Object key = cloner.deepClone(e.getKey(), clones);
23              final Object value = cloner.deepClone(e.getValue(), clones);
24              result.put(key, value);
25          }
26          return result;
27      }
28  
29  
30      protected abstract T getInstance(T t);
31  }
