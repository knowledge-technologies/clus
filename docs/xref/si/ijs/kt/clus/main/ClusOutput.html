103 	public void writeHeader() throws IOException {
104 		String relname = m_Schema.getRelationName();
105 		m_Writer.println("Clus run " + relname);
106 		m_Writer.println(StringUtils.makeString('*', 9 + relname.length()));
107 		m_Writer.println();
108 		Date date = m_Schema.getSettings().getGeneric().getDate();
109 		m_Writer.println("Date: " + DateFormat.getInstance().format(date));
110 		m_Writer.println("File: " + m_Fname);
111 		int a_tot = m_Schema.getNbAttributes();
112 		int a_in = m_Schema.getNbDescriptiveAttributes();
113 		int a_out = m_Schema.getNbTargetAttributes();
114 		m_Writer.println("Attributes: " + a_tot + " (input: " + a_in + ", output: " + a_out + ")");
115 		m_Writer.println("Missing values: " + (m_Schema.hasMissing() ? "Yes" : "No"));
116 		if (ResourceInfo.isLibLoaded()) {
117 			m_Writer.println("Memory usage: " + ClusStat.m_InitialMemory + " kB (initial), " + ClusStat.m_LoadedMemory
118 			        + " kB (data loaded)");
119 		}
120 		m_Writer.println();
121 		m_Sett.show(m_Writer);
122 		m_Writer.flush();
123 	}

125 	public void writeBrief(ClusRun cr) throws IOException {
126 		String ridx = cr.getIndexString();
127 		m_Writer.println("Run: " + ridx);
128 		ClusErrorList te_err = cr.getTestError();
129 		if (te_err != null) {
130 			te_err.showErrorBrief(cr, ClusModelInfo.TEST_ERR, m_Writer);
131 		}
132 		ClusErrorList tr_err = cr.getTrainError();
133 		if (m_Sett.getOutput().isOutTrainError() && tr_err != null) {
134 			tr_err.showErrorBrief(cr, ClusModelInfo.TRAIN_ERR, m_Writer);
135 		}
136 		m_Writer.println();
137 	}

143 	public void writeOutput(ClusRun cr, boolean detail, boolean outputtrain)
144 	        throws IOException, ClusException, InterruptedException {
145 		ArrayList<ClusModel> models = new ArrayList<ClusModel>();
146 		String ridx = cr.getIndexString();
147 
148 		if (getSettings().getHMTR().isSectionHMTREnabled()) {
149 			m_Writer.println("HMTR Tree");
150 			m_Writer.println("---------");
151 			m_Writer.println();
152 			m_Writer.println(m_Schema.getHMTRHierarchy().printHierarchyTree());
153 		}
154 
155 		m_Writer.println("Run: " + ridx);
156 		m_Writer.println(StringUtils.makeString('*', 5 + ridx.length()));
157 		m_Writer.println();
158 		m_Writer.println("Statistics");
159 		m_Writer.println("----------");
160 		m_Writer.println();
161 		m_Writer.println("FTValue (FTest): " + m_Sett.getTree().getFTest());
162 
163 		if (!ClusOOBErrorEstimate.isOOBCalculation()) {
164 			double tsec = cr.getInductionTime() / 1000.0;
165 			double tpru = cr.getPruneTime() / 1000.0;
166 			long tpred = cr.getPredictionTime();
167 			long tpredavg = cr.getPredictionTimeAverage();
168 
169 			String parallelTime = "";
170 			if (getSettings().getEnsemble().isEnsembleWithParallelExecution()) {
171 				parallelTime = " (sequential "
172 				        + ClusFormat.FOUR_AFTER_DOT.format((cr.getInductionTimeSequential() / 1000.0)) + " sec)";
173 			}
174 
175 			// Compute statistics
176 			String cpu = ResourceInfo.isLibLoaded() ? " (CPU)" : "";
177 			m_Writer.println("Induction Time: " + ClusFormat.FOUR_AFTER_DOT.format(tsec) + " sec" + parallelTime + cpu);
178 			m_Writer.println("Pruning Time: " + ClusFormat.FOUR_AFTER_DOT.format(tpru) + " sec" + cpu);
179 			m_Writer.println("Prediction Time (total for ClusModel.Original): " + System.lineSeparator() + "\t"
180 			        + ClusFormat.FOUR_AFTER_DOT.format(tpred / Math.pow(10, 3)) + " microsecs" + System.lineSeparator()
181 			        + "\t" + ClusFormat.FOUR_AFTER_DOT.format(tpred / Math.pow(10, 6)) + " millisecs"
182 			        + System.lineSeparator() + "\t" + ClusFormat.FOUR_AFTER_DOT.format(tpred / Math.pow(10, 9))
183 			        + " secs");
184 			m_Writer.println("Prediction Time (average for ClusModel.Original): "
185 			        + ClusFormat.FOUR_AFTER_DOT.format(tpredavg / Math.pow(10, 3)) + " microsecs");
186 		}
187 
188 		// Prepare models for printing if required
189 		for (int i = 0; i < cr.getNbModels(); i++) {
190 			ClusModelInfo mi = cr.getModelInfo(i);
191 			if (mi != null) {
192 				ClusModel root = mi.getModel();
193 				if (mi.shouldPruneInvalid()) {
194 					root = root.prune(ClusModel.PRUNE_INVALID);
195 				}
196 				models.add(root);
197 			} else {
198 				models.add(null);
199 			}
200 		}
201 
202 		m_Writer.println("Model information:");
203 		for (int i = 0; i < cr.getNbModels(); i++) {
204 			ClusModelInfo mi = cr.getModelInfo(i);
205 			if (mi != null) {
206 				ClusModel model = models.get(i);
207 				// A model info without an actual model is possible
208 				// E.g., to report error measures in HMCAverageSingleClass8
209 				if (model != null) {
210 					m_Writer.print("     " + mi.getName() + ": ");
211 					String info_str = model.getModelInfo();
212 					String[] info = info_str.split("\\s*\\,\\s*");
213 					for (int j = 0; j < info.length; j++) {
214 						if (j > 0)
215 							m_Writer.print(StringUtils.makeString(' ', mi.getName().length() + 7));
216 						m_Writer.println(info[j]);
217 					}
218 				}
219 			}
220 		}
221 
222 		// print multi-label thresholds
223 		if (cr.getStatManager().getSettings().getMLC().shouldShowThresholds()) {
224 			String mlThresholdsTitle = "MultiLabelThresholds:";
225 			m_Writer.println(System.lineSeparator() + mlThresholdsTitle);
226 			m_Writer.println(StringUtils.makeString('-', mlThresholdsTitle.length()));
227 			for (int i = 0; i < cr.getNbModels(); i++) {
228 				ClusModel root = models.get(i);
229 				String modelName = cr.getModelInfo(i).getName();
230 				if (cr.getStatManager().getSettings().getMLC().shouldShowThresholds(modelName)) {
231 					if (root instanceof ClusForest) {
232 						m_Writer.println(modelName);
233 						ClusForest forest = (ClusForest) root;
234 						int forestSize = forest.getModelSize();
235 						for (int tree = 0; tree < forestSize; tree++) {
236 							((ClusNode) forest.getModel(tree)).printMultiLabelThresholds(m_Writer, tree);
237 						}
238 						m_Writer.println();
239 					} else if (root instanceof ClusNode) {
240 						m_Writer.println(modelName);
241 						((ClusNode) root).printMultiLabelThresholds(m_Writer, -1);
242 						m_Writer.println();
243 					}
244 				}
245 			}
246 		}
247 
248 		// Compute basename - not needed
249 		String bName = FileUtil.getName(m_Fname);
250 		m_Writer.println();
251 		ClusErrorList te_err = cr.getTestError();
252 		if (m_Sett.getOutput().isOutFoldError() || detail) {
253 			if (outputtrain) {
254 				ClusErrorList tr_err = cr.getTrainError();
255 				if (tr_err != null) {
256 					if (ClusOOBErrorEstimate.isOOBCalculation()) {
257 						m_Writer.println("Out-Of-Bag Estimate of Error");
258 						m_Writer.println("----------------------------");
259 					} else {
260 						m_Writer.println("Training error");
261 						m_Writer.println("--------------");
262 					}
263 					m_Writer.println();
264 					tr_err.showError(cr, ClusModelInfo.TRAIN_ERR, bName + ".train", m_Writer, m_Sett);
265 					// tr_err.showError(cr, ClusModelInfo.TRAIN_ERR, m_Writer);
266 					m_Writer.println();
267 				}
268 				ClusErrorList.printExtraError(cr, ClusModelInfo.TRAIN_ERR, m_Writer);
269 			}
270 			ClusErrorList va_err = cr.getValidationError();
271 			if (!ClusOOBErrorEstimate.isOOBCalculation() && va_err != null && m_Sett.getOutput().isOutValidError()) {
272 				m_Writer.println("Validation error");
273 				m_Writer.println("----------------");
274 				m_Writer.println();
275 				va_err.showError(cr, ClusModelInfo.VALID_ERR, bName + ".valid", m_Writer, m_Sett);
276 				// va_err.showError(cr, ClusModelInfo.VALID_ERR, m_Writer);
277 				m_Writer.println();
278 			}
279 			if (!ClusOOBErrorEstimate.isOOBCalculation() && te_err != null && m_Sett.getOutput().isOutTestError()) {
280 				m_Writer.println("Testing error");
281 				m_Writer.println("-------------");
282 				m_Writer.println();
283 				te_err.showError(cr, ClusModelInfo.TEST_ERR, bName + ".test", m_Writer, m_Sett);
284 				// te_err.showError(cr, ClusModelInfo.TEST_ERR, m_Writer);
285 				m_Writer.println();
286 			}
287 		}
288 		StatisticPrintInfo info = m_Sett.getOutput().getStatisticPrintInfo();
289 		if (!ClusOOBErrorEstimate.isOOBCalculation()) {
290 
291 			for (int i = 0; i < cr.getNbModels(); i++) {
292 				if (cr.getModelInfo(i) != null && models.get(i) != null && m_Sett.getOutput().shouldShowModel(i)) {
293 					ClusModelInfo mi = cr.getModelInfo(i);
294 					ClusModel root = models.get(i);
295 					String modelname = mi.getName() + " Model";
296 					m_Writer.println(modelname);
297 					m_Writer.println(StringUtils.makeString('*', modelname.length()));
298 					m_Writer.println();
299 					if (m_Sett.getOutput().isPrintModelAndExamples()) {
300 						RowData pex = (RowData) cr.getTrainingSet();
301 						// ClusLogger.info(te_err);
302 						if (te_err != null)
303 							pex = cr.getTestSet();
304 						root.printModelAndExamples(m_Writer, info, pex);
305 					} else {
306 						root.printModel(m_Writer, info);
307 					}
308 					m_Writer.println();
309 				}
310 			}
311 		}
312 
313 		if (!ClusOOBErrorEstimate.isOOBCalculation() && getSettings().getOutput().isOutputPythonModel()) {
314 			String appName = m_Fname.substring(0, m_Fname.lastIndexOf(".out"));
315 			PythonModelType pyModelType = cr.getStatManager().getSettings().getOutput().getPythonModelType();
316 			if (pyModelType == PythonModelType.Object) {
317 				// copy the files from resources
318 				String[] fileNames = new String[] { "tree_as_object.py", "prediction_aggregators.py" };
319 
320 				for (String fileName : fileNames) {
321 					try {
322 						InputStream is = si.ijs.kt.clus.Clus.class.getResourceAsStream("/" + fileName);
323 						String fullFileName = getSettings().getGeneric().getFileAbsolute(fileName);
324 						Files.copy(is, Paths.get(fullFileName), StandardCopyOption.REPLACE_EXISTING);
325 					} catch (IOException ex) {
326 						System.err.println("Error while copying " + fileName + " to the output folder.");
327 						ex.printStackTrace();
328 					}
329 
330 				}
331 			}
332 			String pyName = appName + "_models.py";
333 			File pyscript = new File(getSettings().getGeneric().getFileAbsolute(pyName));
334 			PrintWriter wrtr = new PrintWriter(new FileOutputStream(pyscript));
335 			if (getSettings().getEnsemble().isEnsembleMode()) {
336 				// If in ensemble mode, we create two files:
337 				// - <app name>_models.py with ensemble methods which use the imported methods
338 				// from
339 				// - <app name>_trees.py where the trees are defined
340 				String treeFile = ClusForest.getTreeFile(appName);
341 				// write ensemble file
342 				ClusForest.writePythonAggregation(wrtr, treeFile, cr.getStatManager().getTargetType());
343 				int maxSize = -1;
344 				int maxSizedForest = -1;
345 				for (int i = 0; i < cr.getNbModels(); i++) {
346 					ClusModel root = models.get(i);
347 					if (root instanceof ClusForest) {
348 						int trees = ((ClusForest) root).getNbModels();
349 						if (trees > maxSize) {
350 							maxSize = trees;
351 							maxSizedForest = i;
352 						}
353 						((ClusForest) root).writePythonEnsembleFile(wrtr, treeFile, pyModelType);
354 					}
355 				}
356 				ClusLogger.info(String.format("Python ensemble aggregation code written to: %s", pyName));
357 				// print only max sized forest trees, since it contains the other forests
358 				ClusForest root = (ClusForest) models.get(maxSizedForest);
359 				if (getSettings().getEnsemble().shouldOptimizeEnsemble()) {
360 					root.joinPythonForestInOneFile(cr);
361 				} else {
362 					root.printForestToPython(pyModelType);
363 				}
364 			} else {
365 				// Otherwise, we create only the <app name>_models.py file
366 				// Tested for clus.jar something.s case
367 				if (pyModelType == PythonModelType.Object) {
368 					wrtr.println("from tree_as_object import *\n\n");
369 				}
370 				HashMap<Integer, String> pythonNames = new HashMap<Integer, String>();
371 				pythonNames.put(ClusModel.DEFAULT, "default");
372 				pythonNames.put(ClusModel.ORIGINAL, "original");
373 				pythonNames.put(ClusModel.PRUNED, "pruned");
374 				String defPattern = "def %s(xs):";
375 				HashMap<String, Integer> descrIndices = ClusUtil.getDescriptiveAttributesIndices(cr.getStatManager());
376 				for (int i = 0; i < cr.getNbModels(); i++) {
377 					ClusModel root = models.get(i);
378 					if (pythonNames.containsKey(i)) {
379 						switch (pyModelType) {
380 						case Function:
381 							wrtr.println(String.format(defPattern, pythonNames.get(i)));
382 							root.printModelToPythonScript(wrtr, descrIndices);
383 							break;
384 						case Object:
385 							root.printModelToPythonScript(wrtr, descrIndices, pythonNames.get(i));
386 							break;
387 						default:
388 							throw new RuntimeException("Wrong PythonModelType: "
389 							        + cr.getStatManager().getSettings().getOutput().getPythonModelType());
390 						}
391 					} else {
392 						System.err.println(
393 						        "Warning: this is not DEFAULT/ORIGINAL/PRUNED model and will not be printed. Extend the pythonNames hash map!");
394 					}
395 				}
396 			}
397 			wrtr.close();
398 		}
399 
400 		if (!ClusOOBErrorEstimate.isOOBCalculation() && getSettings().getOutput().isOutputDatabaseQueries()) {
401 			int starttree = getSettings().getExhaustiveSearch().getStartTreeCpt();
402 			int startitem = getSettings().getExhaustiveSearch().getStartItemCpt();
403 			ClusModel root = models.get(cr.getNbModels() - 1);
404 			// use the following lines for creating a SQL file that will put the tree into a
405 			// database
406 			String out_database_name = m_Sett2.getGeneric().getAppName() + ".txt";
407 			PrintWriter database_writer = m_Sett2.getGeneric().getFileAbsoluteWriter(out_database_name);
408 			root.printModelToQuery(database_writer, cr, starttree, startitem,
409 			        getSettings().getExhaustiveSearch().isExhaustiveSearch());
410 			database_writer.close();
411 			ClusLogger.info("The queries are in " + out_database_name);
412 		}
413 
414 		if (!ClusOOBErrorEstimate.isOOBCalculation() && getSettings().getOutput().isOutputClowdFlowsJSON()) {
415 			JsonObject output = new JsonObject();
416 			StringWriter settingsStringWriter = new StringWriter();
417 			PrintWriter settingsWriter = new PrintWriter(settingsStringWriter);
418 			m_Sett.show(settingsWriter);
419 			output.addProperty("settings", settingsStringWriter.toString());
420 			JsonArray outputModels = new JsonArray();
421 			output.add("models", outputModels);
422 			for (int i = 0; i < cr.getNbModels(); i++) {
423 				if (cr.getModelInfo(i) != null && models.get(i) != null && m_Sett.getOutput().shouldShowModel(i)) {
424 					ClusModelInfo mi = cr.getModelInfo(i);
425 					ClusModel root = models.get(i);
426 					String modelname = mi.getName();
427 					JsonObject currentModel = new JsonObject();
428 					currentModel.addProperty("name", modelname);
429 					RowData pex = (RowData) cr.getTrainingSet();
430 					// ClusLogger.info(te_err);
431 					if (te_err != null)
432 						pex = cr.getTestSet();
433 					currentModel.add("representation", root.getModelJSON(info, pex));
434 					outputModels.add(currentModel);
435 				}
436 			}
437 
438 			String jsonFileName = m_Sett2.getGeneric().getAppName() + ".json";
439 			PrintWriter jsonWriter = m_Sett2.getGeneric().getFileAbsoluteWriter(jsonFileName);
440 			jsonWriter.write(output.toString());
441 			// System.out.print(output.toString());
442 			jsonWriter.close();
443 		}
444 		m_Writer.flush();
445 	}

447 	public String getQuotient(int a, int b) {
448 		double val = b == 0 ? 0.0 : (double) a / b;
449 		return ClusFormat.ONE_AFTER_DOT.format(val);
450 	}

452 	public void writeSummary(ClusSummary summary) throws IOException {
453 		m_Writer.println("Summary");
454 		m_Writer.println("*******");
455 		m_Writer.println();
456 
457 		int runs = summary.getNbRuns();
458 		m_Writer.println("Runs: " + runs);
459 
460 		double tsec = summary.getInductionTime() / 1000.0;
461 		m_Writer.println("Induction time: " + ClusFormat.FOUR_AFTER_DOT.format(tsec) + " sec");
462 
463 		double psec = summary.getPrepareTime() / 1000.0;
464 		m_Writer.println("Preprocessing time: " + ClusFormat.ONE_AFTER_DOT.format(psec) + " sec");
465 
466 		long tpred = summary.getPredictionTime();
467 		long tpredavg = summary.getPredictionTimeAverage();
468 		m_Writer.println("Prediction Time (total for ClusModel.Original, " + runs + " runs, "
469 		        + summary.getPredictionTimeNbExamples() + " examples): " + System.lineSeparator() + "\t"
470 		        + ClusFormat.FOUR_AFTER_DOT.format(tpred / Math.pow(10, 3)) + " microsecs" + System.lineSeparator()
471 		        + "\t" + ClusFormat.FOUR_AFTER_DOT.format(tpred / Math.pow(10, 6)) + " millisecs"
472 		        + System.lineSeparator() + "\t" + ClusFormat.FOUR_AFTER_DOT.format(tpred / Math.pow(10, 9)) + " secs");
473 		m_Writer.println("Prediction Time (average for ClusModel.Original): "
474 		        + ClusFormat.FOUR_AFTER_DOT.format(tpredavg / Math.pow(10, 3)) + " microsecs");
475 
476 		m_Writer.println("Mean number of tests");
477 		for (int i = ClusModel.ORIGINAL; i <= ClusModel.PRUNED; i++) {
478 			ClusModelInfo mi = summary.getModelInfo(i);
479 			if (mi != null)
480 				m_Writer.println("     " + mi.getName() + ": " + getQuotient(mi.getModelSize(), runs));
481 		}
482 		m_Writer.println();
483 		String bName = FileUtil.getName(m_Fname);
484 		ClusErrorList tr_err = summary.getTrainError();
485 		if (m_Sett.getOutput().isOutTrainError() && tr_err != null) {
486 			m_Writer.println("Training error");
487 			m_Writer.println("--------------");
488 			m_Writer.println();
489 			tr_err.showError(summary, ClusModelInfo.TRAIN_ERR, bName + ".train", m_Writer, m_Sett);
490 			// tr_err.showError(summary, ClusModelInfo.TRAIN_ERR, m_Writer);
491 			m_Writer.println();
492 		}
493 		ClusErrorList va_err = summary.getValidationError();
494 		if (va_err != null) {
495 			m_Writer.println("Validation error");
496 			m_Writer.println("----------------");
497 			m_Writer.println();
498 			va_err.showError(summary, ClusModelInfo.VALID_ERR, bName + ".valid", m_Writer, m_Sett);
499 			// va_err.showError(summary, ClusModelInfo.VALID_ERR, m_Writer);
500 			m_Writer.println();
501 		}
502 		ClusErrorList te_err = summary.getTestError();
503 		if (te_err != null) {
504 			m_Writer.println("Testing error");
505 			m_Writer.println("-------------");
506 			m_Writer.println();
507 			te_err.showError(summary, ClusModelInfo.TEST_ERR, bName + ".test", m_Writer, m_Sett);
508 			// te_err.showError(summary, ClusModelInfo.TEST_ERR, m_Writer);
509 		}
510 		m_Writer.println();
511 		m_Writer.flush();
512 	}
