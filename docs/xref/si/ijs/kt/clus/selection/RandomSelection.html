23  package si.ijs.kt.clus.selection;
24  
25  import java.util.Random;
26  
27  
28  public class RandomSelection extends ClusSelection {
29  
30      protected int m_NbSelected;
31      protected boolean[] m_Selection;
32      int m_Seed;
33  
34  
35      public RandomSelection(int nbrows, double sel) {
36          super(nbrows);
37          m_Seed = 0; // default seed is 0
38          makeSelection(nbrows, (int) Math.round(sel * nbrows));
39      }
40  
41  
42      public RandomSelection(int nbrows, int nbsel) {
43          super(nbrows);
44          m_Seed = 0; // default seed is 0
45          makeSelection(nbrows, nbsel);
46      }
47  
48      public RandomSelection(int nbrows, double sel, int randomSeed) {
49  		super(nbrows);
50  		m_Seed = randomSeed;
51  		makeSelection(nbrows, (int)Math.round(sel*nbrows));
52  	}
53          
54          public RandomSelection(int nbrows, int nbsel, int randomSeed) {
55  		super(nbrows);
56  		m_Seed = randomSeed;
57  		makeSelection(nbrows, nbsel);
58  	}
59  
60      public void setRandomSeed(int seed){
61      	m_Seed = seed;
62      }
63          
64      @Override
65      public int getNbSelected() {
66          return m_NbSelected;
67      }
68  
69  
70      @Override
71      public boolean isSelected(int row) {
72          return m_Selection[row];
73      }
74  
75  
76      private final void makeSelection(int nbrows, int nbsel) {
77          m_NbSelected = nbsel;
78          m_Selection = new boolean[nbrows];
79          //Random rnd = new Random(0);
80          Random rnd = new Random(m_Seed);
81          for (int i = 0; i < m_NbSelected; i++) {
82              int j = 0;
83              int p = rnd.nextInt(nbrows - i) + 1; // Select one of the remaining positions
84  
85              while (p > 0 && j < nbrows) {
86                  if (!m_Selection[j]) {
87                      p--;
88                      if (p == 0)
89                          m_Selection[j] = true;
90                  }
91                  j++;
92              }
93          }
94      }
95  }
