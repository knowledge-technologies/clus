31       * Returns weights
32       */
33      public static ArrayList<Double> main(ClusStatManager clusStatManager, OptimizationProblem.OptimizationParameter optInfo, ClusRuleSet rset) {
34          int nbOfWeights = optInfo.m_rulePredictions.length;
35          int nbOfRules = nbOfWeights; // Only rules
36          // We are ignoring any other base functions than rules here
37  
38          SettingsRules set = clusStatManager.getSettings().getRules();
39  
40          int nbTargs = (clusStatManager.getStatistic(AttributeUseType.Target)).getNbAttributes();
41          // Parameter data matrix dimension. Do we give descriptive dims also?
42          int nbDescrForDataMatrix = 0;
43          int nbRows = optInfo.m_trueValues.length;
44  
45          // if (rules.m_implicitLinearTerms != null) {
46          if (set.getOptAddLinearTerms().equals(OptimizationGDAddLinearTerms.YesSaveMemory)) {
47              nbDescrForDataMatrix = clusStatManager.getSchema().getNumericAttrUse(AttributeUseType.Descriptive).length;
48              nbOfWeights += nbDescrForDataMatrix * nbTargs;
49          }
50  
51          // Optimization normalization
52          double[] normFactors = OptimizationProblem.initNormFactors(nbTargs, clusStatManager.getSettings());
53          double[] targetAvg = OptimizationProblem.initMeans(nbTargs);
54  
55          // Change the default prediction - it will otherwise be set to zero
56          for (int iTarg = 0; iTarg < nbTargs; iTarg++) {
57              if (optInfo.m_rulePredictions[0].m_prediction[iTarg][0] != targetAvg[iTarg]) {
58                  System.err.println("Error: Difference in main for target nb " + iTarg + ". The values are " + optInfo.m_rulePredictions[0].m_prediction[iTarg][0] + " and " + targetAvg[iTarg]);
59                  System.exit(1);
60              }
61              optInfo.m_rulePredictions[0].m_prediction[iTarg][0] = Math.sqrt(normFactors[iTarg]); // e.g. 2* std dev
62          }
63  
64          OptimizationParameter trainingSet = optInfo;
65          OptimizationParameter validationSet = null;
66          // If early stopping criteria is chosen, reserve part of the training set for early stop testing.
67          if (set.getOptGDEarlyStopAmount() > 0) {
68  
69              int nbDataTest = (int) Math.ceil(nbRows * set.getOptGDEarlyStopAmount());
70  
71              // Create the early stopping data variables.
72              validationSet = new OptimizationParameter(optInfo.m_rulePredictions.length, optInfo.m_baseFuncPredictions.length, nbDataTest, nbTargs, optInfo.m_implicitLinearTerms);
73              trainingSet = new OptimizationParameter(optInfo.m_rulePredictions.length, optInfo.m_baseFuncPredictions.length, nbRows - nbDataTest, nbTargs, optInfo.m_implicitLinearTerms);
74              OptimizationProblem.splitDataIntoValAndTrainSet(clusStatManager, optInfo, validationSet, trainingSet);
75          }
76  
77          double[] weights = new double[nbOfWeights];
78  
79          // Rule predictions
80          double[] rulePreds = new double[nbOfRules * nbTargs];
81          for (int iRule = 0; iRule < nbOfRules; iRule++) {
82              for (int iTarg = 0; iTarg < nbTargs; iTarg++) {
83                  // These are same for both training and validation set
84                  rulePreds[iRule * nbTargs + iTarg] = trainingSet.m_rulePredictions[iRule].m_prediction[iTarg][0] / Math.sqrt(normFactors[iTarg]);
85              }
86          }
87          // Covers
88          // nbOfWeights, first training set, then validation set
89          boolean[] ruleCovers = new boolean[nbOfRules * nbRows];
90  
91          int nbInstTrain = trainingSet.m_rulePredictions[0].m_cover.length();
92          int nbInstVal = validationSet.m_rulePredictions[0].m_cover.length();
93          for (int iRule = 0; iRule < nbOfRules; iRule++) {
94              // Index for ruleCovers array
95              int iIndex = 0;
96              // index for TrainingSet and validationSet
97              int iInst = 0;
98              int iMaxInst = nbInstTrain;
99              OptimizationParameter targetData = trainingSet;
100 
101             // We first go trhough training set, then validation set
102             for (; iInst < iMaxInst; iInst++, iIndex++) {
103                 ruleCovers[iRule * nbRows + iIndex] = targetData.m_rulePredictions[iRule].m_cover.get(iInst);
104                 if (iInst == nbInstTrain - 1) {
105                     iInst = -1;
106                     targetData = validationSet;
107                     iMaxInst = nbInstVal;
108                 }
109             }
110         }
111 
112         // Data for linear terms and true values
113         double[] binData = new double[nbRows * (nbTargs + nbDescrForDataMatrix)];
114 
115         // Index for bindata array
116         int iIndex = 0;
117         // index for TrainingSet and validationSet
118         int iInst = 0;
119         int iMaxInst = nbInstTrain;
120         OptimizationParameter targetData = trainingSet;
121 
122         for (; iInst < iMaxInst; iInst++, iIndex++) {
123 
124             for (int iDescrDim = 0; iDescrDim < nbDescrForDataMatrix; iDescrDim++) {
125                 binData[iIndex * (nbTargs + nbDescrForDataMatrix) + iDescrDim] =
126                         // Get the first (index [0])linear term for each descr dim
127                         targetData.m_implicitLinearTerms.predict(iDescrDim * nbTargs, targetData.m_trueValues[iInst].m_dataExample, 0, nbTargs) / Math.sqrt(normFactors[0]);
128             }
129             // For true values the shifting has not yet been done, we have to do it here
130             for (int iTarDim = 0; iTarDim < nbTargs; iTarDim++) {
131                 binData[iIndex * (nbTargs + nbDescrForDataMatrix) + nbDescrForDataMatrix + iTarDim] = (targetData.m_trueValues[iInst].m_targets[iTarDim] - targetAvg[iTarDim]) / Math.sqrt(normFactors[iTarDim]);
132             }
133 
134             // We first go trhough training set, then validation set
135             if (iInst == nbInstTrain - 1) {
136                 iInst = -1;
137                 targetData = validationSet;
138                 iMaxInst = nbInstVal;
139             }
140 
141         }
142 
143         // Create class instance
144         CallExternGD mappedFile = new CallExternGD();
145         // String settings = "nbOfIterations 10000\nminTVal 0.0\nlinTermsUsed 0\nnbOfTargs 2\nnbTrainData 3\nnbValData
146         // 2\nnbOfRules 3";
147         String settings = "";
148         if (set.getOptAddLinearTerms().equals(OptimizationGDAddLinearTerms.YesSaveMemory))
149             settings += "linTermsUsed 1\n";
150         settings += "nbOfTargs " + nbTargs + "\nnbTrainData " + nbInstTrain + "\nnbValData " + nbInstVal + "\nnbOfRules " + nbOfRules + "\nnbOfDescrAttr " + nbDescrForDataMatrix;
151         settings += "\nnbOfIterations " + set.getOptGDMaxIter() + "\nminTVal " + set.getOptGDGradTreshold();
152         settings += "\nnbOfDiffTVal " + set.getOptGDNbOfTParameterTry() + "\nnbNonZeroWeights " + set.getOptGDMaxNbWeights() + "\n";
153 
154         // System.out.print("data, rulepreds, covers");
155         // for (int i = 0; i < 10; i++)
156         // System.out.print(" " + binData[i] + "\t" + rulePreds[i] + "\t" + ruleCovers[i] + "\n");
157         weights = mappedFile.externalOptim(settings, binData, rulePreds, ruleCovers);
158 
159         weights[0] = undoNormalization(weights[0], rulePreds, nbTargs, targetAvg, rset, normFactors); // Undo the effect
160                                                                                                       // on default rule
161                                                                                                       // System.out.print("Toimii
162                                                                                                       // Javakin\n");
163         ArrayList<Double> result = new ArrayList<Double>(nbOfWeights);
164         for (int i = 0; i < nbOfWeights; i++) {
165             result.add(weights[i]);
166         }
167         return result;
168     }

191     // Returns new weight
192     static final private double undoNormalization(double defaultRuleWeights, double[] defaultRulePreds, int nbTargs, double[] targMeans, ClusRuleSet rset, double[] normFactors) {
193         double[] newDefault = new double[nbTargs];
194         for (int iTarg = 0; iTarg < nbTargs; iTarg++) {
195             // Let's add the prediction got in the optimization to the average
196             newDefault[iTarg] = defaultRulePreds[iTarg] * Math.sqrt(normFactors[iTarg]) // Undo normalization;
197                     * defaultRuleWeights // Include weight into the prediction
198                     + targMeans[iTarg];
199         }
200         rset.getRule(0).setNumericPrediction(newDefault);
201         return 1.0;
202     }
