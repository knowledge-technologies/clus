31  public class OptXValBinTree extends MyNode {
32  
33      public final static long serialVersionUID = Settings.SERIAL_VERSION_ID;
34  
35      protected int m_NbTests;
36      protected int m_NbFoldTests;
37      protected long m_Time;
38  
39  
40      public double[] getFIs() {
41          double[] arr = new double[getMaxLeafDepth()];
42          getFIs(arr, 0);
43          return arr;
44      }
45  
46  
47      private void getFIs(double[] arr, int depth) {
48          if (m_NbTests > 0)
49              arr[depth] += (double) m_NbFoldTests / m_NbTests;
50          for (int j = 0; j < getNbChildren(); j++) {
51              OptXValBinTree tree = (OptXValBinTree) getChild(j);
52              tree.getFIs(arr, depth + 1);
53          }
54      }
55  
56  
57      public double[] getNodes() {
58          double[] arr = new double[getMaxLeafDepth()];
59          getNodes(arr, 0);
60          return arr;
61      }
62  
63  
64      private void getNodes(double[] arr, int depth) {
65          if (m_NbTests > 0)
66              arr[depth]++;
67          for (int j = 0; j < getNbChildren(); j++) {
68              OptXValBinTree tree = (OptXValBinTree) getChild(j);
69              tree.getNodes(arr, depth + 1);
70          }
71      }
72  
73  
74      public double[] getTimes() {
75          double[] arr = new double[getMaxLeafDepth()];
76          getTimes(arr, 0);
77          return arr;
78      }
79  
80  
81      private void getTimes(double[] arr, int depth) {
82          arr[depth] += m_Time;
83          for (int j = 0; j < getNbChildren(); j++) {
84              OptXValBinTree tree = (OptXValBinTree) getChild(j);
85              tree.getTimes(arr, depth + 1);
86          }
87      }
88  
89  
90      public static void processs(MyArray nodes, MyArray left, MyArray right, OptXValBinTree tree) {
91          for (int i = 0; i < nodes.size(); i++) {
92              Object obj = nodes.elementAt(i);
93              if (obj instanceof ClusNode) {
94                  ClusNode node = (ClusNode) obj;
95                  tree.m_Time += node.m_Time;
96                  if (node.getNbChildren() > 0) {
97                      left.addElement(node.getChild(0));
98                      right.addElement(node.getChild(1));
99  
100                     tree.m_NbTests++;
101                     tree.m_NbFoldTests++;
102                 }
103             }
104             else {
105                 OptXValNode node = (OptXValNode) obj;
106                 tree.m_Time += node.m_Time;
107                 ClusNode[] cnodes = node.getNodes();
108                 for (int j = 0; j < cnodes.length; j++) {
109                     tree.m_Time += cnodes[j].m_Time;
110                     if (cnodes[j].getNbChildren() > 0) {
111                         Object n1 = cnodes[j].getChild(0);
112                         Object n2 = cnodes[j].getChild(1);
113                         if (n1 != null)
114                             left.addElement(n1);
115                         if (n2 != null)
116                             right.addElement(n2);
117                     }
118                 }
119                 for (int j = 0; j < node.getNbChildren(); j++) {
120                     OptXValSplit split = (OptXValSplit) node.getChild(j);
121                     if (split.getNbChildren() > 0) {
122                         left.addElement(split.getChild(0));
123                         right.addElement(split.getChild(1));
124 
125                         tree.m_NbTests++;
126                         tree.m_NbFoldTests += split.getFolds().length;
127                     }
128                 }
129             }
130         }
131     }
132 
133 
134     public static OptXValBinTree convertTree(MyArray crnodes) {
135         OptXValBinTree tree = new OptXValBinTree();
136         if (crnodes.size() != 0) {
137             tree.setNbChildren(2);
138             MyArray left = new MyArray();
139             MyArray right = new MyArray();
140             processs(crnodes, left, right, tree);
141             tree.setChild(convertTree(left), 0);
142             tree.setChild(convertTree(right), 1);
143         }
144         return tree;
145     }
146 
147 
148     public static OptXValBinTree convertTree(Object object) {
149         MyArray arr = new MyArray();
150         arr.addElement(object);
151         return convertTree(arr);
152     }
153 }
