72      public HammingLossError(ClusErrorList par, SetAttrType[] set, ClusAttributeWeights weights, boolean printall, ClusDistance innerDistance, int numberOfPossibleValues) {
73          super(par, set);
74          m_nbEx = new int[m_Dim];

76          m_SumSqErr = new double[m_Dim];
77          m_Weights = weights;

79          this.innerDistance = innerDistance;
80          this.numberOfTotalPossibleValues = numberOfPossibleValues;
81      }

86          for (int i = 0; i < m_Dim; i++) {
87              m_SumErr[i] = 0.0;
88              m_SumSqErr[i] = 0.0;
89              m_nbEx[i] = 0;
90          }
91      }

100     @Override
101     public double getModelErrorComponent(int i) {
102 
103         //int nb = getNbExamples();
104         int nb = m_nbEx[i];
105 
106         //ClusLogger.info(m_SumErr[i]);
107         double err = nb != 0.0 ? m_SumErr[i] / nb : 0.0;
108         if (m_Weights != null)
109             err *= m_Weights.getWeight(getAttr(i));
110 
111         return err;
112     }

115     @Override
116     public double getModelError() {
117         double ss_tree = 0.0;
118         int nb = 0;
119         for (int i = 0; i < m_Dim; i++) {
120             nb += m_nbEx[i];
121         }
122         if (m_Weights != null) {
123             for (int i = 0; i < m_Dim; i++) {
124                 ss_tree += m_SumErr[i] * m_Weights.getWeight(getAttr(i));
125             }
126             return nb != 0.0 ? ss_tree / nb : 0.0;
127         }
128         else {
129             for (int i = 0; i < m_Dim; i++) {
130                 ss_tree += m_SumErr[i];
131             }
132             return nb != 0.0 ? ss_tree / nb : 0.0;
133         }
134     }

137     @Override
138     public double getModelErrorStandardError() {
139         double sum_err = 0.0;
140         double sum_sq_err = 0.0;
141         for (int i = 0; i < m_Dim; i++) {
142             if (m_Weights != null) {
143                 sum_err += m_SumErr[i];
144                 sum_sq_err += m_SumSqErr[i];
145             }
146             else {
147                 sum_err += m_SumErr[i] * m_Weights.getWeight(getAttr(i));
148                 sum_sq_err += m_SumSqErr[i] * sqr(m_Weights.getWeight(getAttr(i)));
149             }
150         }
151         //double n = getNbExamples() * m_Dim;
152         double n = 0;
153         for (int i = 0; i < m_Dim; i++) {
154             n += m_nbEx[i];
155         }
156 
157         if (n <= 1) {
158             return Double.POSITIVE_INFINITY;
159         }
160         else {
161             double ss_x = (n * sum_sq_err - sqr(sum_err)) / (n * (n - 1));
162             return Math.sqrt(ss_x / n);
163         }
164     }

166 
167     public final static double sqr(double value) {
168         return value * value;
169     }
170 
171 
172     @Override
173     public void addExample(double[] real, double[] predicted) {
174         for (int i = 0; i < m_Dim; i++) {
175             double err = sqr(real[i] - predicted[i]);
176             ClusLogger.info(err);
177             if (!Double.isInfinite(err) && !Double.isNaN(err)) {
178                 m_SumErr[i] += err;
179                 m_SumSqErr[i] += sqr(err);
180                 m_nbEx[i]++;
181             }
182         }
183     }

186     @Override
187     public void addExample(DataTuple tuple, ClusStatistic pred) throws ClusException {
188         Set predicted = ((SetStatistic) pred).getSetPred();
189         Set realSet = getAttr(0).getSet(tuple);
190         double error = new HammingLossDistance(innerDistance, numberOfTotalPossibleValues).calcDistance(realSet, predicted);
191         m_SumErr[0] += error;
192         m_nbEx[0]++;
193     }

196     @Override
197     public void addExample(DataTuple real, DataTuple pred) throws ClusException {
198         Set realSet = getAttr(0).getSet(real);
199         Set predicted = getAttr(0).getSet(pred);
200         double error = new HammingLossDistance(innerDistance, numberOfTotalPossibleValues).calcDistance(realSet, predicted);
201         m_SumErr[0] += error;
202         m_nbEx[0]++;
203     }

211     @Override
212     public void add(ClusError other) {
213         HammingLossError oe = (HammingLossError) other;
214         for (int i = 0; i < m_Dim; i++) {
215             m_SumErr[i] += oe.m_SumErr[i];
216             m_SumSqErr[i] += oe.m_SumSqErr[i];
217             m_nbEx[i] += oe.m_nbEx[i];
218         }
219     }

222     @Override
223     public void showModelError(PrintWriter out, int detail) {
224     	ClusNumberFormat fr = getFormat();
225         StringBuffer buf = new StringBuffer();
226         if (m_PrintAllComps) {
227             buf.append("[");
228             for (int i = 0; i < m_Dim; i++) {
229                 if (i != 0)
230                     buf.append(",");
231                 buf.append(fr.format(getModelErrorComponent(i)));
232             }
233             if (m_Dim > 1)
234                 buf.append("]: ");
235             else
236                 buf.append("]");
237         }
238         if (m_Dim > 1 || !m_PrintAllComps) {
239             buf.append(fr.format(getModelError()));
240         }
241         out.println(buf.toString());
242     }

266     @Override
267     public double computeLeafError(ClusStatistic stat) {
268         RegressionStat rstat = (RegressionStat) stat;
269         return rstat.getSVarS(m_Weights) * rstat.getNbAttributes();
