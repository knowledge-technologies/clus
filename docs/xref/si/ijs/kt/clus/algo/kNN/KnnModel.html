99      public KnnModel(ClusRun cr, int k, DistanceWeights weighting, KnnModel master) {
100         this.m_ClusRun = cr;
101         this.m_K = k;
102         this.m_MaxK = Math.max(this.m_K, master.m_MaxK);
103         master.m_MaxK = this.m_MaxK;
104         this.m_WeightingOption = weighting;
105         this.m_Search = master.m_Search;
106         this.m_StatTemplate = master.m_StatTemplate;
107         this.m_Master = master;
108     }

117     public KnnModel(ClusRun cr, int k, DistanceWeights weighting, int maxK, boolean isSparse, ClusAttrType[] necessaryDescriptiveAttributes, int[] trainingExamplesWithMissing) throws ClusException, IOException, InterruptedException {
118         this.m_ClusRun = cr;
119         this.m_K = k;
120         this.m_MaxK = Math.max(Math.max(this.m_K, this.m_MaxK), maxK);
121         this.m_WeightingOption = weighting;
122         // settings file name; use name for .weight file
123         Settings sett = this.m_ClusRun.getStatManager().getSettings();
124         String fName = sett.getGeneric().getAppName();
125         // Initialize attribute weighting according to settings file
126         AttributeWeighting attrWe = new NoWeighting();
127 
128         String attrWeighting = sett.getKNN().getKNNAttrWeight();
129         boolean loadedWeighting = false;
130         if (attrWeighting.toLowerCase().compareTo("none") == 0) {
131 
132         }
133         else if (attrWeighting.startsWith("RF")) {
134             try {
135                 String[] wS = attrWeighting.split(",");
136                 int nbBags = 100; // Default value
137                 if (wS.length == 2)
138                     nbBags = Integer.parseInt(wS[1]);
139                 else
140                 	sett.getKNN().setKNNAttrWeight(attrWeighting + "," + nbBags);
141                 attrWe = new RandomForestWeighting(this.m_ClusRun, nbBags);
142             }
143             catch (Exception e) {
144                 throw new ClusException("Error at reading attributeWeighting value. RF value detected, but error accured while reading number of bags.");
145             }
146         }
147         else if (attrWeighting.startsWith("[") && attrWeighting.endsWith("]")) {
148             try {
149                 String[] wS = attrWeighting.substring(1, attrWeighting.length() - 1).split(",");
150                 double[] we = new double[wS.length];
151                 for (int i = 0; i < we.length; i++)
152                     we[i] = Double.parseDouble(wS[i]);
153                 attrWe = new UserDefinedWeighting(we);
154             }
155             catch (Exception e) {
156                 throw new ClusException("Error at reading attributeWeighting value. User defined entry detected, but value cannot be read.");
157             }
158 
159         }
160         else {
161             // Probably file name.. try to load weighting.
162             attrWe = AttributeWeighting.loadFromFile(fName + ".weight");
163             ClusLogger.info(attrWe.toString());
164             if (attrWe != null)
165                 loadedWeighting = true;
166             else
167                 throw new ClusException("Unrecognized attributeWeighting value (" + attrWeighting + ")");
168         }
169 
170         if (!(attrWe instanceof NoWeighting) && !loadedWeighting) {
171             AttributeWeighting.saveToFile(attrWe, fName + ".weight");
172         }
173 
174         // Initialize distance according to settings file
175         // int dist = sett.getKNN().getKNNDistance();
176         Distance dist = sett.getKNN().getDistance();
177 
178         SearchDistance searchDistance = createSearchDistance(cr, sett.getKNN(), attrWe);
179 
180         // Select search method according to settings file.
181         SearchMethod searchMethod = sett.getKNN().getSearchMethod();
182         switch (searchMethod) {
183             case VPTree:
184                 this.m_Search = new VPTree(this.m_ClusRun, searchDistance);
185                 break;
186             case KDTree:
187                 this.m_Search = new KDTree(this.m_ClusRun, searchDistance);
188                 break;
189             case BruteForce:
190                 this.m_Search = new BruteForce(this.m_ClusRun, searchDistance);
191                 break;
192             case Oracle:
193                 this.m_Search = new OracleBruteForce(this.m_ClusRun, searchDistance);
194                 break;
195             default:
196                 throw new RuntimeException("Wrong search method: " + searchMethod.toString());
197         }
198 
199         // debug info
200         if (sett.getGeneral().getVerbose() >= 1) {
201             ClusLogger.info("Search method: " + this.m_Search.getClass());
202             ClusLogger.info("Search distance: " + searchDistance.getBasicDistance().getClass());
203             ClusLogger.info("Number of neighbours: " + this.m_K);
204             ClusLogger.info("Distance weights: " + sett.getKNN().getKNNDistanceWeights());
205             
206         }
207 
208         // build tree, preprocessing
209         if (trainingExamplesWithMissing != null) {
210         	this.m_Search.buildForMissingTargetImputation(maxK, trainingExamplesWithMissing, sett.getKNN());
211         } else {
212         	this.m_Search.build(m_MaxK);
213         }
214         RowData train = this.m_ClusRun.getDataSet(ClusModelInfoList.TRAINSET);
215         RowData test = this.m_ClusRun.getDataSet(ClusModelInfoList.TESTSET);
216         if (searchMethod == SearchMethod.Oracle && trainingExamplesWithMissing == null) {
217             if (sett.getKNN().mustNotComputeTrainingError(train.getNbRows())) {
218                 sett.getOutput().setOutTrainError(false);
219                 System.err.println("Training error will not be computed, since we do not know the neighbours for each training instance.");
220             }
221             if (test != null && sett.getKNN().mustNotComputeTestError(test.getNbRows())) {
222                 sett.getOutput().setOutTestError(false);
223                 System.err.println("Testing error will not be computed, since we do not know the neighbours for each testing instance.");
224             }
225         }
226         
227         // special MLC version
228         m_IsMlcKnn = sett.getKNN().isMlcKnn();
229         
230         // save prediction template
231         // @todo : should all this be repalced with:
232         if (sett.getKNN().isMlcKnn()) {
233         	m_StatTemplate = new KnnMlcStat(sett, cr.getStatManager().getSchema().getNominalAttrUse(AttributeUseType.Target));
234         } else {
235         	m_StatTemplate = cr.getStatManager().getStatistic(AttributeUseType.Target);
236         }
237         ClusStatistic trainingStat = cr.getStatManager().getTrainSetStat(AttributeUseType.Target);
238         m_StatTemplate.setTrainingStat(trainingStat);
239 
240         // if( m_ClusRun.getStatManager().getMode() == ClusStatManager.MODE_CLASSIFY ){
241         // if(sett.getSectionMultiLabel().isEnabled()){
242         // m_StatTemplate = new
243         // ClassificationStat(this.cr.getDataSet(ClusRun.TRAINSET).m_Schema.getNominalAttrUse(AttributeUseType.Target),
244         // sett.getMultiLabelTrheshold());
245         // } else{
246         // m_StatTemplate = new
247         // ClassificationStat(this.cr.getDataSet(ClusRun.TRAINSET).m_Schema.getNominalAttrUse(AttributeUseType.Target));
248         // }
249         // }
250         // else if( m_ClusRun.getStatManager().getMode() == ClusStatManager.MODE_REGRESSION )
251         // m_StatTemplate = new
252         // RegressionStat(this.cr.getDataSet(ClusRun.TRAINSET).m_Schema.getNumericAttrUse(AttributeUseType.Target));
253         // else if( m_ClusRun.getStatManager().getMode() == ClusStatManager.MODE_TIME_SERIES ){
254         // // TimeSeriesAttrType attr =
255         // this.cr.getDataSet(ClusRun.TRAINSET).m_Schema.getTimeSeriesAttrUse(AttributeUseType.Target)[0];
256         // // m_StatTemplate = new TimeSeriesStat(attxr, new DTWTimeSeriesDist(attr), 0 );
257         // ClusLogger.info("-------------");
258         // m_StatTemplate = m_ClusRun.getStatManager().getStatistic(AttributeUseType.Target);
259         // ClusLogger.info(m_StatTemplate.getDistanceName());
260         // ClusLogger.info("----------------");
261         // }else if( m_ClusRun.getStatManager().getMode() == ClusStatManager.MODE_HIERARCHICAL ){
262         // m_StatTemplate = m_ClusRun.getStatManager().getStatistic(AttributeUseType.Target);
263         // ClusLogger.info("----------------------");
264         // ClusLogger.info(m_StatTemplate.getDistanceName());
265         // ClusLogger.info(m_StatTemplate.getClass());
266         // ClusLogger.info("----------------------");

270    	public static SearchDistance createSearchDistance(ClusRun cr, SettingsKNN sett, AttributeWeighting attrWe) {
271    		Distance dist = sett.getDistance();
272 
273         SearchDistance searchDistance = new SearchDistance();
274         ClusDistance distance;
275         RowData data = null;
276 		try {
277 			data = cr.getDataSet(ClusModelInfoList.TRAINSET);
278 		} catch (ClusException | IOException | InterruptedException e) {
279 			e.printStackTrace();
280 		}
281         ClusAttrType[] necessaryDescriptiveAttributes = KnnClassifier.getNecessaryDescriptiveAttributes(data);
282         
283         // descriptive attributes with weight 0 can be safely omitted from distance computation
284         necessaryDescriptiveAttributes = ClusDistance.attributesWithNonZeroWeight(necessaryDescriptiveAttributes, attrWe);
285         boolean isSparse = data.isSparse();
286         switch(dist) {
287 	        case Euclidean:
288 	        	distance = new EuclideanDistance(searchDistance, isSparse, necessaryDescriptiveAttributes);
289 	        	break;
290 	        case Chebyshev:
291 	        	distance = new ChebyshevDistance(searchDistance, isSparse, necessaryDescriptiveAttributes);
292 	        	break;
293 	        case Manhattan:
294 	        	distance = new ManhattanDistance(searchDistance, isSparse, necessaryDescriptiveAttributes);
295 	        	break;
296         	default:
297         		throw new RuntimeException("Wrong distance.");
298         }
299         
300      // initialize min values of numeric attributes: needed for normalization in the distance computation
301         int[] data_types = new int[] { ClusModelInfoList.TRAINSET, ClusModelInfoList.TESTSET, ClusModelInfoList.VALIDATIONSET };
302         double[] mins = null;
303         double[] maxs = null;
304         int nb_attrs = -1;
305         for (int type = 0; type < data_types.length; type++) {
306             try {
307 				data = cr.getDataSet(type);
308 			} catch (ClusException | IOException | InterruptedException e) {
309 				e.printStackTrace();
310 			}
311             ClusSchema schema = cr.getStatManager().getSchema();
312             if (data != null) {
313                 if (mins == null) {
314                     nb_attrs = schema.getNbAttributes();
315                     mins = new double[nb_attrs];
316                     Arrays.fill(mins, Double.POSITIVE_INFINITY);
317                     maxs = new double[nb_attrs];
318                     Arrays.fill(maxs, Double.NEGATIVE_INFINITY);
319                 }
320                 // compute max and min for every numeric attribute
321                 for (int tuple_ind = 0; tuple_ind < data.getNbRows(); tuple_ind++) {
322                     for (int i = 0; i < nb_attrs; i++) {
323                         ClusAttrType attr_type = schema.getAttrType(i);
324                         if (!attr_type.isDisabled() && attr_type instanceof NumericAttrType) {
325                             double t = attr_type.getNumeric(data.getTuple(tuple_ind));
326                             if (t < mins[i] && t != Double.POSITIVE_INFINITY) {
327                                 mins[i] = t;
328                             }
329                             if (t > maxs[i] && t != Double.POSITIVE_INFINITY) {
330                                 maxs[i] = t;
331                             }
332                         }
333                     }
334                 }
335             }
336         }
337 
338         searchDistance.setDistance(distance);
339         distance.setWeighting(attrWe);
340         searchDistance.setNormalizationWeights(mins, maxs);
341         return searchDistance;
342    	}

349     public ClusStatistic predictWeighted(DataTuple tuple, ArrayList<Integer> targetsNeeded) throws ClusException {
350         LinkedList<DataTuple> nearest = new LinkedList<DataTuple>(); // the first m_K neigbhours of the m_MaxK
351                                                                      // neighbours: OK, because the neighbours are
352                                                                      // sorted from the nearest to the farthest
353 
354         if (this.m_Master == null) { // this is the master
355             this.m_CurrentNeighbours = this.m_Search.returnNNs(tuple, this.m_MaxK);
356             this.m_CurrentTuple = tuple;
357             for (int neighbour = 0; neighbour < this.m_K; neighbour++) {
358                 nearest.add(this.m_CurrentNeighbours.get(neighbour));
359             }
360         }
361         else { // this is a slave
362             if (this.m_Master.m_CurrentTuple != tuple) { throw new RuntimeException("The neighbours were computed for tuple\n" + this.m_Master.m_CurrentTuple.toString() + "\nbut now, we are dealing with tuple\n" + tuple.toString()); }
363             for (int neighbour = 0; neighbour < this.m_K; neighbour++) {
364                 nearest.add(this.m_Master.m_CurrentNeighbours.get(neighbour));
365             }
366         }
367 
368         // Initialize distance weighting according to setting file
369         DistanceWeighting weighting;
370         switch (this.m_WeightingOption) {
371             case OneOverD:
372                 weighting = new WeightOver(nearest, this.m_Search, tuple);
373                 break;
374 
375             case OneMinusD:
376                 weighting = new WeightMinus(nearest, this.m_Search, tuple);
377                 break;
378 
379             case Constant:
380                 weighting = new WeightConstant(nearest, this.m_Search, tuple);
381                 break;
382 
383             default:
384                 throw new RuntimeException("DistanceWeights unknown!");
385         }
386 
387         // Vote
388         ClusStatistic stat = m_StatTemplate.cloneStat();
389         if (stat instanceof TimeSeriesStat) {
390             for (DataTuple dt : nearest) {
391                 ClusStatistic dtStat = m_StatTemplate.cloneStat();
392                 dtStat.setSDataSize(1);
393                 dtStat.updateWeighted(dt, 0);
394                 dtStat.computePrediction();
395                 stat.addPrediction(dtStat, weighting.weight(dt));
396             }
397             stat.computePrediction();
398             return stat;
399         }
400         else {
401             for (DataTuple dt : nearest)
402                 stat.updateWeighted(dt, weighting.weight(dt));
403             if (targetsNeeded != null){
404             	for (int target : targetsNeeded) {
405             		if (!stat.isAnyLabeled(target)) {
406             			return null;
407             		}
408             	}
409             }
410             stat.calcMean();
411         }
412         return stat;
413     }
