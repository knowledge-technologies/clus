4   import si.ijs.kt.clus.util.jeans.util.compound.DoubleObject;
5   import si.ijs.kt.clus.util.jeans.util.sort.MSortable;
6   
7   
8   public class RowDataSortHelper implements MSortable {
9   
10      protected int m_NbRows = 0;
11      public DataTuple[] missing;
12      public DataTuple[] zero;
13      public DoubleObject[] other;
14  
15  
16      public void resize(int nbrows) {
17          if (nbrows > m_NbRows) {
18              missing = new DataTuple[nbrows];
19              zero = new DataTuple[nbrows];
20              DoubleObject[] new_other = new DoubleObject[nbrows];
21              if (m_NbRows > 0) {
22                  System.arraycopy(other, 0, new_other, 0, m_NbRows);
23              }
24              for (int i = m_NbRows; i < nbrows; i++) {
25                  new_other[i] = new DoubleObject();
26              }
27              other = new_other;
28              m_NbRows = nbrows;
29          }
30      }
31  
32  
33      @Override
34      public double getDouble(int i) {
35          return other[i].getValue();
36      }
37  
38  
39      @Override
40      public void swap(int i, int j) {
41          DoubleObject obj_i = other[i];
42          other[i] = other[j];
43          other[j] = obj_i;
44      }
45  }
