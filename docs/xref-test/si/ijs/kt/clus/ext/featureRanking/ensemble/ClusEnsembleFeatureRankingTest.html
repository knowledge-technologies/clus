10  public class ClusEnsembleFeatureRankingTest {
11  
12  	public static Pair<ArrayList<Triple<String, String, String>>, String[]> generateNames() throws IOException {
13  		String formatSetting = ClusFeatureRankingTest.getSubfolder() + "/test%s%s.s";
14  		String formatFimpTruth = ClusFeatureRankingTest.getSubfolder() + "/test%s%sTrees%s.fimpTruth";
15  		String formatFimpNew = ClusFeatureRankingTest.getSubfolder() + "/test%s%sTrees%s%s.fimp";
16  		String[] tasks = new String[] { "DagHMLC-RealWorld", "TreeHMLC-RealWorld", "MLC-RealWorld", "MTR-RealWorld",
17  		        "STC-RealWorld" };
18  		String[][] rankingSubsets = new String[][] { { "Genie3" }, { "RForest" }, { "Symbolic" },
19  		        { "RForest", "Genie3" }, { "Symbolic", "Genie3" }, { "Symbolic", "RForest" },
20  		        { "Symbolic", "RForest", "Genie3" } };
21  		ArrayList<Triple<String, String, String>> settingsFimps = new ArrayList<>();
22  		for (String task : tasks) {
23  			for (String[] subset : rankingSubsets) {
24  				String sFile = String.format(formatSetting, task, String.join("", subset));
25  				for (String size : new String[] { "2", "4" }) {
26  					for (String ranking : subset) {
27  						String fimpTruth = String.format(formatFimpTruth, task, ranking, size);
28  						String fimpNew = String.format(formatFimpNew, task, String.join("", subset), size, ranking);
29  						settingsFimps.add(new Triple<String, String, String>(sFile, fimpTruth, fimpNew));
30  					}
31  				}
32  			}
33  		}
34  		String[] args = new String[] { "-silent", "-forest", "" };
35  		return new Pair<ArrayList<Triple<String, String, String>>, String[]>(settingsFimps, args);
36  	}
37  }
