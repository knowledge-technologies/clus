152     public final RowData getDataSet(int whichone) throws ClusException, IOException, InterruptedException {
153         switch (whichone) {
154             case TRAINSET:
155                 return (RowData) getTrainingSet();
156             case TESTSET:
157                 return getTestSet();
158             case VALIDATIONSET:
159                 return (RowData) getPruneSet();
160         }

214     // If the test set is specified as a separate file, this method first reads the entire
215     // file into memory, while the above method provides an interator that reads tuples one by one
216     public final RowData getTestSet() throws IOException, ClusException {
217         if (m_Test == null)
218             return null;
219         RowData data = (RowData) m_Test.getData();
220         if (data == null) {
221             data = (RowData) m_Test.createInMemoryData();
222             m_Test = data.getIterator();
223         }
224         return data;
225     }

253     public void combineTrainAndValidSets() throws InterruptedException, ClusException {
254         RowData valid = (RowData) getPruneSet();
255         if (valid != null) {
256             RowData train = (RowData) getTrainingSet();
257             ArrayList lst = train.toArrayList();
258             lst.addAll(valid.toArrayList());
259             setTrainingSet(new RowData(lst, train.getSchema()));
260             setPruneSet(null, null);
261             changePruneError(null);
262             copyTrainingData();
263         }
264     }

281     		if (m_UnlabeledData == null) return null;
282     		RowData data = (RowData)m_UnlabeledData.getData();
283     		if (data == null) {
284     			data = (RowData)m_UnlabeledData.createInMemoryData();
285     			m_UnlabeledData = data.getIterator();

287     		return data;

296     public void changeTestError(ClusErrorList par) throws ClusException {
297         m_Summary.setTestError(par);
298         int nb_models = getNbModels();
299         for (int i = 0; i < nb_models; i++) {
300             ClusModelInfo my = getModelInfo(i);
301             my.setTestError(par.getErrorClone());
302         }
303     }
304 
305 
306     public void changePruneError(ClusErrorList par) throws ClusException {
307         m_Summary.setValidationError(par);
308         int nb_models = getNbModels();
309         for (int i = 0; i < nb_models; i++) {
310             ClusModelInfo my = getModelInfo(i);
311             my.setValidationError(par != null ? par.getErrorClone() : null);
312         }
313     }
