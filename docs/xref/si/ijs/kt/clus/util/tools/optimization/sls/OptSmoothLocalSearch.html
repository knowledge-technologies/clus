133     private boolean removeBadRules() {
134         for (int rule : A) {
135             if (estimates[rule] < -2 * errorMargin) {
136                 // remove rule from biased set
137                 A.remove(rule);
138 
139                 // go calculate estimates again
140                 return true;
141             }
142         }
143         return false; // no new estimate calculation is needed
144     }
145

147     private boolean findBiasedRules() {
148         for (int rule = 0; rule < initialRuleSet.getModelSize(); rule++) {
149             if (!A.contains(rule) && estimates[rule] > 2 * errorMargin) {
150                 // add rule to biased set
151                 A.add(rule);
152 
153                 // go calculate estimates again
154                 return true;
155             }
156         }
157         return false; // no new estimate calculation is needed
158     }

230      * 
231      * @param probabilityBiased
232      * @return A randomly sampled rule set with bias towards some of the elements
233      */
234     ClusRuleSet randomSampleWithBias(double probabilityBiased) {
235         ClusRuleSet returnSet = new ClusRuleSet(statManager);
236         double prob, probA;
237         probA = 1 - probabilityBiased;
238 
239         // shuffle indices because we are truncating the iteration
240         // Collections.shuffle(m_indices, m_randGen); // important to use the same rnd generator as with everything
241         // else, in order to get repeatable results
242 
243         int rule;
244         for (int index = 0; index < m_maxRulesNb; index++) {
245             rule = m_indices.get(index);
246             // ask God
247             prob = m_randGen.nextDouble();
248 
249             if ((A.contains(rule) && prob >= probA) || (!A.contains(rule) && prob >= probabilityBiased)) {
250                 // returnSet.add(initialRuleSet.getRule(rule));
251                 // we will not be using .add() because it checks for uniquiness, which we already have from the
252                 // initialruleset
253                 returnSet.getRules().add(initialRuleSet.getRule(rule));

255         }

292     /**
293      * Calculate expected value for objectiveFunction given error margin and a rule set
294      */
295     double getEstimate(ClusRule ruleToCheck, double probabilityBiased) {
296         ArrayList<Double> differenceValuesArray = new ArrayList<Double>(100);
297 
298         double sum = 0, tmp = 0, mean = 0;
299         double error = errorMargin;
300         double counter = 0, tmpSetEstimate1, tmpSetEstimate2;
301 
302         int numberOfIterationsBeforeCalculation = 1;
303 
304         ClusRuleSet rndSet;
305         ClusRuleSet setWithRuleToCheck;
306         ClusRuleSet setWithoutRuleToCheck;
307 
308         // estimate mean within errorMargin
309         while (error >= errorMargin) {
310             counter++;
311 
312             // get randomly sampled rule set with bias
313             rndSet = randomSampleWithBias(probabilityBiased);
314             setWithRuleToCheck = rndSet.cloneRuleSetNonUnique();
315             setWithoutRuleToCheck = rndSet.cloneRuleSetNonUnique();
316 
317             // add / remove ruleToCheck from two sets that we are estimating the quality of (add it uniquely by using
318             // ClusRule.add() method)
319             setWithRuleToCheck.add(ruleToCheck);
320             setWithoutRuleToCheck.remove(ruleToCheck);
321 
322             // should this really be here?? the statistic is not correct for the rules that are in the set
323             // setWithRuleToCheck.setTargetStat(initialRuleSet.getTargetStat());
324             // setWithoutRuleToCheck.setTargetStat(initialRuleSet.getTargetStat());
325 
326             // sample random from rules
327             // tmpSetEstimate1 = objectiveFunction.apply(randomSampleWithoutBias(rulesWith)); // randomly sample without
328             // bias and estimate quality
329             // tmpSetEstimate2 = objectiveFunction.apply(randomSampleWithoutBias(rulesWithout));
330             tmpSetEstimate1 = objectiveFunction.apply(setWithRuleToCheck); // estimate quality
331             tmpSetEstimate2 = objectiveFunction.apply(setWithoutRuleToCheck);
332 
333             // get difference in objective function values
334             tmp = tmpSetEstimate1 - tmpSetEstimate2;
335             sum += tmp;
336             differenceValuesArray.add(tmp);
337 
338             // calculate mean and SD every K iterations
339             if (counter % numberOfIterationsBeforeCalculation == 0) {
340                 mean = sum / counter;
341                 tmp = 0;
342                 for (double d : differenceValuesArray) {
343                     tmp += Math.pow((mean - d), 2);
344                 }
345 
346                 tmp = Math.sqrt((1 / (counter - 1)) * tmp); // this is SD
347 
348                 error = tmp / Math.sqrt(counter); // this is Std Error
349 
350                 // ClusLogger.info(Math.abs(error-errorMargin));
351 
352                 // // try to reset recalculation iteration interval so that we dont iterate forever
353                 // numberOfIterationsBeforeCalculation = Math.max(
354                 // 1,
355                 // (int) ((numberOfIterationsBeforeCalculation / counter) * numberOfIterationsBeforeCalculation)
356                 // );
357                 // if (counter >numberOfIterationsBeforeCalculation)
358                 // System.err.println(counter);
359             }
360         }
361 
362         return mean;
363     }
