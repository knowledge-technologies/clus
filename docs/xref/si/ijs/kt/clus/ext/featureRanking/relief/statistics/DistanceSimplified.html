12  public class DistanceSimplified extends Statistics {
13  	/**
14  	 * m_SumDistAttr[target index][number of neighbours][attribute]: current sum of
15  	 * products (1 - distAttr)(1 - distClustering),
16  	 * for the given number of neighbours and attribute.
17  	 * If {@code target index} is 0, this is for overall ranking(s).
18  	 * Otherwise, this is for the ranking(s) for the target with the index
19  	 * {@code target index - 1}.
20  	 * 
21  	 * Problem of this method is that everything seems to be the same ...
22  	 */
23  	private double[][][] m_SumDistAttrTarget;
24  	
25  	private double[] tempSumDistAttrTarget;
26  	
27  	private static double EPS = 0.01;
28  //	private double m_NumberOfBorderline;
29  
30  	public DistanceSimplified(ClusReliefFeatureRanking relief, int nbTargets, int nbDiffNbNeighbours,
31  	        int nbDescriptiveAttributes) {
32  		initializeSuperFields(relief, nbDescriptiveAttributes);
33  		m_SumDistAttrTarget = new double[nbTargets][nbDiffNbNeighbours][nbDescriptiveAttributes];
34  		
35  		tempSumDistAttrTarget = new double[nbDescriptiveAttributes];
36  	}
37  
38  	@Override
39  	public void updateTempStatistics(int targetIndex, boolean isStdClassification, DataTuple tuple, RowData data,
40  	        NearestNeighbour neigh, double neighWeightNonnormalized, int trueIndex, int targetValue)
41  	        throws ClusException {
42  		double targetDistance = 0.0;
43  		if (targetIndex >= 0 && !isStdClassification) {
44  			targetDistance = mRelief.computeDistance1D(tuple, data.getTuple(neigh.getIndexInDataset()), mRelief.getTargetAttribute(trueIndex));
45  		} else {
46  			targetDistance = mRelief.computeDistance(tuple, data.getTuple(neigh.getIndexInDataset()),
47  			        ClusReliefFeatureRanking.TARGET_SPACE);
48  		}
49  //		m_NumberOfBorderline += 1.0;
50  		for (int attrInd = 0; attrInd < m_NbDescriptiveAttrs; attrInd++) {
51  			ClusAttrType attr = mRelief.getDescriptiveAttribute(attrInd);
52  			double distAttr = mRelief.computeDistance1D(tuple, data.getTuple(neigh.getIndexInDataset()), attr);
53  			tempSumDistAttrTarget[attrInd] += (1.0 / (targetDistance + EPS)) * (-distAttr) * neighWeightNonnormalized;
54  		}
55  	}
56  
57  	@Override
58  	public void updateStatistics(int targetIndex, int numNeighInd, double sumNeighbourWeights) {
59          for (int attrInd = 0; attrInd < m_NbDescriptiveAttrs; attrInd++) {
60              m_SumDistAttrTarget[targetIndex + 1][numNeighInd][attrInd] += tempSumDistAttrTarget[attrInd] / sumNeighbourWeights;
61          }
62  	}
63  
64  	@Override
65  	public void resetTempFields() {
66  		Arrays.fill(tempSumDistAttrTarget, 0.0);
67  	}
68  
69  	@Override
70  	public double computeImportances(int targetIndex, int nbNeighInd, int attrInd, boolean isStdClassification,
71  	        double[] successfulItearions) {
72  		return m_SumDistAttrTarget[targetIndex + 1][nbNeighInd][attrInd] / successfulItearions[targetIndex + 1];
73  	}
74  
75  }
