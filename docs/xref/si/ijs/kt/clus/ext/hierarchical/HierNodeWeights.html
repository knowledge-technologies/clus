51      /*
52       * The key issue is assigning weights. I suggest the following procedure.
53       * For the root, a weight of 1 is assigned.
54       * For all other nodes Y, the weight should be
55       * w * Sum( weight (X) ) where X is a parent of Y.
56       * This is equivalent to flattening the lattice into a tree
57       * (by copying the subtrees that have multiple parents).
58       */
59  
60      public boolean allParentsOk(ClassTerm term, boolean[] computed) {
61          for (int j = 0; j < term.getNbParents(); j++) {
62              ClassTerm parent = term.getParent(j);
63              if (parent.getIndex() != -1 && !computed[parent.getIndex()])
64                  return false;
65          }
66          return true;
67      }

70      public void initExponentialDepthWeightsDAG(ClassHierarchy hier, HierarchyWeight wtype, double w0) {
71      	// 1 ---> 1/w0, otherwise: the weights in this and DAG case differ by a factor w0
72      	double rootWeight = 1.0 / w0;
73          boolean[] weight_computed = new boolean[hier.getTotal()];
74          ArrayList<ClassTerm> todo = new ArrayList<ClassTerm>();
75          for (int i = 0; i < hier.getTotal(); i++) {
76              ClassTerm term = hier.getTermAt(i);
77              todo.add(term);
78          }
79          int nb_done = 0;
80          while (nb_done < hier.getTotal()) {
81              for (int i = todo.size() - 1; i >= 0; i--) {
82                  ClassTerm term = todo.get(i);
83                  if (allParentsOk(term, weight_computed)) {
84                      int maxDepth = 0;
85                      int minDepth = Integer.MAX_VALUE;
86                      for (int j = 0; j < term.getNbParents(); j++) {
87                          ClassTerm parent = term.getParent(j);
88                          maxDepth = Math.max(maxDepth, parent.getMaxDepth() + 1);
89                          minDepth = Math.min(minDepth, parent.getMinDepth() + 1);
90                      }
91                      // 1 + min/max depth over the parents' min/max depths
92                      term.setMinDepth(minDepth);
93                      term.setMaxDepth(maxDepth);
94                      double agg_wi;
95                      if (wtype.equals(HierarchyWeight.ExpMinParentWeight)) {
96                          agg_wi = Double.MAX_VALUE;
97                          for (int j = 0; j < term.getNbParents(); j++) {
98                              ClassTerm parent = term.getParent(j);
99                              if (parent.getIndex() == -1)
100                                 agg_wi = Math.min(agg_wi, rootWeight);
101                             else
102                                 agg_wi = Math.min(agg_wi, m_Weights[parent.getIndex()]);
103                         }
104                     } else if (wtype.equals(HierarchyWeight.ExpMaxParentWeight)) {
105                             agg_wi = Double.MIN_VALUE;
106                             for (int j = 0; j < term.getNbParents(); j++) {
107                                 ClassTerm parent = term.getParent(j);
108                                 if (parent.getIndex() == -1)
109                                     agg_wi = Math.max(agg_wi, rootWeight);
110                                 else
111                                     agg_wi = Math.max(agg_wi, m_Weights[parent.getIndex()]);
112                             }
113                     } else { // max and sum aggregation
114                         agg_wi = 0.0;
115                         for (int j = 0; j < term.getNbParents(); j++) {
116                             ClassTerm parent = term.getParent(j);
117                             if (parent.getIndex() == -1)
118                                 agg_wi += rootWeight;
119                             else
120                                 agg_wi += m_Weights[parent.getIndex()];
121                         }
122                         if (wtype.equals(HierarchyWeight.ExpAvgParentWeight)) {
123                             agg_wi = agg_wi / term.getNbParents();
124                         }
125                     }
126 
127                     m_Weights[term.getIndex()] = w0 * agg_wi;
128                     weight_computed[term.getIndex()] = true;
129                     todo.remove(i);
130                     nb_done++;
131                 }
132             }
133         }
134     }

137     public void initExponentialDepthWeightsRec(ClassTerm node, int depth, double w0) {
138         for (int i = 0; i < node.getNbChildren(); i++) {
139             ClassTerm child = (ClassTerm) node.getChild(i);
140             child.setMinDepth(depth);
141             child.setMaxDepth(depth);
142             m_Weights[child.getIndex()] = calcExponentialDepthWeight(depth, w0);
143             initExponentialDepthWeightsRec(child, depth + 1, w0);
144         }
145     }

148     public void initNoWeights(ClassHierarchy hier) {
149         boolean[] weight_computed = new boolean[hier.getTotal()];
150         ArrayList<ClassTerm> todo = new ArrayList<>();
151         for (int i = 0; i < hier.getTotal(); i++) {
152             ClassTerm term = hier.getTermAt(i);
153             todo.add(term);
154         }
155         int nb_done = 0;
156         while (nb_done < hier.getTotal()) {
157             for (int i = todo.size() - 1; i >= 0; i--) {
158                 ClassTerm term = (ClassTerm) todo.get(i);
159                 if (allParentsOk(term, weight_computed)) {
160                     int maxDepth = 0;
161                     int minDepth = Integer.MAX_VALUE;
162                     for (int j = 0; j < term.getNbParents(); j++) {
163                         ClassTerm parent = term.getParent(j);
164                         maxDepth = Math.max(maxDepth, parent.getMaxDepth() + 1);
165                         minDepth = Math.min(minDepth, parent.getMinDepth() + 1);
166                     }
167                     term.setMinDepth(minDepth);
168                     term.setMaxDepth(maxDepth);
169                     m_Weights[term.getIndex()] = 1.0;
170                     weight_computed[term.getIndex()] = true;
171                     todo.remove(i);
172                     nb_done++;
173                 }
174             }
175         }
176     }

179     public void initExponentialDepthWeights(ClassHierarchy hier, HierarchyWeight wtype, double w0) {
180         m_Weights = new double[hier.getTotal()];
181         if (wtype.equals(HierarchyWeight.NoWeight)) {
182             initNoWeights(hier);
183         }
184         else {
185             ClassTerm root = hier.getRoot();
186             if (hier.isTree()) {
187                 initExponentialDepthWeightsRec(root, 0, w0);
188                 m_Name = "Exponential depth weights (tree) " + w0;
189             }
190             else {
191                 root.setMinDepth(-1);
192                 root.setMaxDepth(-1);
193                 initExponentialDepthWeightsDAG(hier, wtype, w0);
194                 m_Name = "Exponential depth weights (DAG) " + w0;
195             }
196         }
197     }
198 
199 
200     private final static double calcExponentialDepthWeight(int depth, double w0) {
201         return Math.pow(w0, depth);
202     }
