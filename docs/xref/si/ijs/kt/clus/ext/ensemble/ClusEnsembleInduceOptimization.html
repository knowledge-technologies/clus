42      public ClusEnsembleInduceOptimization(TupleIterator train, TupleIterator test, Settings sett) throws IOException, ClusException {
43  
44          m_Settings = sett;
45  
46          // m_HashCodeTuple = new int[nb_tuples];
47          m_TuplePositions = new HashMap<DataTuple, Integer>();
48          int count = 0;
49          if (train != null) {
50              train.init();
51              DataTuple train_tuple = train.readTuple();
52              while (train_tuple != null) {
53                  // m_HashCodeTuple[count] = train_tuple.hashCode();
54                  m_TuplePositions.put(train_tuple, count);
55                  count++;
56                  train_tuple = train.readTuple();
57              }
58          }
59          if (test != null) {
60              test.init();
61              DataTuple test_tuple = test.readTuple();
62              while (test_tuple != null) {
63                  // m_HashCodeTuple[count] = test_tuple.hashCode();
64                  m_TuplePositions.put(test_tuple, count);
65                  count++;
66                  test_tuple = test.readTuple();
67              }
68          }
69      }

134     /**
135      * The average predictions {@code avg_predictions} of {@code nb_models - 1}, and the predictions {@code predictions}
136      * of one additional model
137      * are combined into the average predictions of all {@code nb_models}.
138      * 
139      * @param avg_predictions
140      * @param predictions
141      * @param nb_models
142 
143      */
144     public static double[] incrementPredictions(double[] avg_predictions, double[] predictions, double nb_models) {
145         // the current averages are stored in the avg_predictions
146         int plength = avg_predictions.length;
147         double[] result = new double[plength];
148         for (int i = 0; i < plength; i++) {
149             // result[i] = avg_predictions[i] + (predictions[i] - avg_predictions[i]) / nb_models;
150             result[i] = computeNextAverage(avg_predictions[i], predictions[i], nb_models);
151         }
152         return result;
153     }

156     public double[][] incrementPredictions(double[][] sum_predictions, double[][] predictions, int nb_models) {
157         // the current sums are stored in sum_predictions
158         double[][] result = new double[sum_predictions.length][];
159 
160         if (getSettings().getEnsemble().isEnsembleROSEnabled() && getSettings().getEnsemble().getEnsembleROSVotingType().equals(EnsembleROSVotingType.SubspaceAveraging)) {
161             /* get enabled targets for the model */
162             int[] enabled = m_EnsembleROSInfo.getOnlyTargets(m_EnsembleROSInfo.getModelSubspace(nb_models - 1));
163 
164             for (int i = 0; i < sum_predictions.length; i++) {
165                 if (enabled[i] == 1) {
166                     result[i] = new double[sum_predictions[i].length];
167                     // if target is enables, then update results[i][j]
168                     for (int j = 0; j < sum_predictions[i].length; j++) {
169                         // result[i][j] = sum_predictions[i][j] + (predictions[i][j] - sum_predictions[i][j]) /
170                         // m_EnsembleROSInfo.getCoverageOpt(i);
171                         result[i][j] = computeNextAverage(sum_predictions[i][j], predictions[i][j], m_EnsembleROSInfo.getCoverageOpt(i));
172                     }
173                 }
174                 else { // target not enabled for this model; retain what we have until now
175                     result[i] = sum_predictions[i];
176                 }
177             }
178         }
179         else {
180             for (int i = 0; i < sum_predictions.length; i++) {
181                 result[i] = new double[sum_predictions[i].length];
182                 for (int j = 0; j < sum_predictions[i].length; j++) {
183                     // result[i][j] = sum_predictions[i][j] + (predictions[i][j] - sum_predictions[i][j]) / nb_models;
184                     result[i][j] = computeNextAverage(sum_predictions[i][j], predictions[i][j], nb_models);
185                 }
186             }
187         }
188 
189         return result;
190     }

193     /**
194      * Computes the average of {@code nbValues} values, given the average {@code currentAverage} of {@code nbValues - 1}
195      * values
196      * and the next value.
197      * 
198      * @param currentAverege
199      * @param nextValue
200      * @param nbValues
201 
202      */
203     private static double computeNextAverage(double currentAverege, double nextValue, double nbValues) {
204         return nextValue / nbValues + currentAverege * (nbValues - 1) / nbValues;
205     }

208     /**
209      * Returns the componentwise sum of the current sums and another predictions
210      * 
211      * @param sum_predictions
212      *        The current sums
213      * @param predictions
214 
215      */
216     public static double[][] incrementPredictions(double[][] sum_predictions, double[][] predictions) {
217         double[][] result = new double[sum_predictions.length][];
218         for (int i = 0; i < sum_predictions.length; i++) {
219             result[i] = new double[sum_predictions[i].length];
220             for (int j = 0; j < sum_predictions[i].length; j++) {
221                 result[i][j] = sum_predictions[i][j] + predictions[i][j];
222             }
223         }
224         return result;
225     }

228     /**
229      * Transform the class counts to majority vote (the one with max votes gets 1)
230      * 
231      * @param counts
232 
233      */
234     public static double[][] transformToMajority(double[][] counts) {
235         int[] maxPerTarget = new int[counts.length];
236         for (int i = 0; i < counts.length; i++) {
237             maxPerTarget[i] = -1;
238             double m_max = Double.NEGATIVE_INFINITY;
239             for (int j = 0; j < counts[i].length; j++) {
240                 if (counts[i][j] > m_max) {
241                     maxPerTarget[i] = j;
242                     m_max = counts[i][j];
243                 }
244             }
245         }
246         double[][] result = new double[counts.length][];// all values set to zero
247         for (int m = 0; m < counts.length; m++) {
248             result[m] = new double[counts[m].length];
249             result[m][maxPerTarget[m]]++; // the positions of max class will be 1
250         }
251         return result;
252     }

255     /**
256      * Transform the class counts to probability distributions.
257      * 
258      * @param counts
259 
260      */
261     public static double[][] transformToProbabilityDistribution(double[][] counts) {
262         double[] sumPerTarget = new double[counts.length];
263         for (int i = 0; i < counts.length; i++)
264             for (int j = 0; j < counts[i].length; j++)
265                 sumPerTarget[i] += counts[i][j];
266         double[][] result = new double[counts.length][];
267 
268         for (int m = 0; m < counts.length; m++) {
269             result[m] = new double[counts[m].length];
270             for (int n = 0; n < counts[m].length; n++) {
271                 result[m][n] = counts[m][n] / sumPerTarget[m];
272             }
273         }
274         return result;
275     }
