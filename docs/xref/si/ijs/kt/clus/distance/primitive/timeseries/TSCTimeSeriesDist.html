41      @Override
42      public double calcDistance(TimeSeries t1, TimeSeries t2) throws ClusException {
43          // this calculates the Correlation coefficient of two TimeSeries
44          // the two TimeSeries have same length
45          double[] ts1 = t1.getValues();
46          double[] ts2 = t2.getValues();
47          double mean_ts1 = calculateMean(ts1);
48          double mean_ts2 = calculateMean(ts2);
49          double cc = 0;
50          double sum_ts1_sqr = 0;
51          double sum_ts2_sqr = 0;
52          double sum_ts1_ts2 = 0;
53  
54          if (ts1.length != ts2.length)
55              throw new ClusException("TimeSeriesCorrelation applies only to Time Series with equal length");
56  
57          for (int k = 0; k < ts1.length; k++) {
58              sum_ts1_ts2 += (ts1[k] - mean_ts1) * (ts2[k] - mean_ts2);
59              sum_ts1_sqr += Math.pow((ts1[k] - mean_ts1), 2);
60              sum_ts2_sqr += Math.pow((ts2[k] - mean_ts2), 2);
61          }
62          cc = 1 - Math.abs(sum_ts1_ts2 / Math.sqrt(sum_ts1_sqr * sum_ts2_sqr));
63          return cc;
64      }

93      public double calculateMean(double[] ts) {
94          double sum = 0;
95          for (int k = 0; k < ts.length; k++)
96              sum += ts[k];
97          sum = sum / ts.length;
98          return sum;
99      }
