103 	public void rememberMinMax(NominalAttrType at, RowData data) throws ClusException {
104 		findNominal(at, data, null);
105 		ClusStatistic m_CStat = m_StatManager.createClusteringStat();
106 		m_CStat.add(m_BestTest.m_TestStat[0]);
107 		double heur = m_BestTest.calcHeuristic(m_BestTest.m_TotStat, m_CStat);
108 		if (heur != Double.POSITIVE_INFINITY && heur != Double.NEGATIVE_INFINITY) {
109 			if (heur > hMaxB) {
110 				hMaxB = heur;
111 			}
112 			if (heur < hMinB) {
113 				hMinB = heur;
114 			}
115 		}
116 		// ClusLogger.info(heur+" "+hMinB+" -- "+hMaxB);
117 	}

120 	public void findNominal(NominalAttrType at, RowData data, ClusRandomNonstatic rnd) throws ClusException {
121 		// long start_time = System.currentTimeMillis();
122 		RowData sample = createSample(data, rnd);
123 		int nbvalues = at.getNbValues();
124 		m_BestTest.reset(nbvalues + 1); // Reset positive statistic
125 		int nb_rows = sample.getNbRows();
126 
127 		if (!data.getSchema().getSettings().getAttribute().isNullGIS()) { // handle Daniela
128 			// daniela
129 			// generateMatrix for each attribute GISHeuristic
130 			Integer[] indicesSorted = new Integer[nb_rows]; // matejp: no sorting here, but needed for technical reasons
131 			for (int i = 0; i < nb_rows; i++) {
132 				indicesSorted[i] = i;
133 			}
134 			try {
135 				ClusHeuristic m_Heuristic = m_StatManager.getHeuristic();
136 				if (m_Heuristic instanceof VarianceReductionHeuristicCompatibility) {
137 					VarianceReductionHeuristicCompatibility gisHeuristic = (VarianceReductionHeuristicCompatibility) m_Heuristic;
138 					ClusAttrType[] arr = at.getSchema().getAllAttrUse(AttributeUseType.GIS); // numeric and string
139 																									// GIS
140 					if (arr.length == 1) {
141 						gisHeuristic.readMatrixFromFile(sample); // 1 coordinate and a distance.csv file associated
142 					} else
143 						gisHeuristic.generateMatrix(sample); // 2 coordinates
144 				} else if (m_Heuristic instanceof GISHeuristic) {
145 					GISHeuristic gisHeuristic = (GISHeuristic) m_Heuristic;
146 					ClusAttrType[] arr = at.getSchema().getAllAttrUse(AttributeUseType.GIS); // numeric and string
147 																									// GIS
148 					if (arr.length == 1) {
149 						gisHeuristic.readMatrixFromFile(sample); // 1 coordinate and a distance.csv file associated
150 					} else
151 						gisHeuristic.generateMatrix(sample, indicesSorted); // 2 coordinates
152 				}
153 			} catch (Exception e) {
154 				e.printStackTrace();
155 				throw new ClusException("Error calculating GISHEuristics!"); // this try/catch block is weird
156 			}
157 
158 			m_BestTest.m_PosStat.setData(sample);
159 			m_BestTest.m_TotStat.setData(sample);
160 			m_BestTest.m_TotStat.setSplitIndex(sample.getNbRows());
161 			m_BestTest.m_PosStat.setPrevIndex(0);
162 			m_BestTest.m_PosStat.initializeSum();
163 
164 			for (int i = 0; i < nb_rows; i++) {
165 				DataTuple tuple = sample.getTuple(i);
166 				int value = at.getNominal(tuple);
167 				m_BestTest.m_TestStat[value].setData(sample);
168 				m_BestTest.m_TestStat[value].initializeSum();
169 				m_BestTest.m_TestStat[value].setPrevIndex(0);
170 			}
171 			// daniela end
172 		}
173 
174 		if (nbvalues == 2 && !at.hasMissing()) {
175 			// Only count ones for binary attributes (optimization)
176 			for (int i = 0; i < nb_rows; i++) {
177 				DataTuple tuple = sample.getTuple(i);
178 				int value = at.getNominal(tuple);
179 				// The value "1" has index 0 in the list of attribute values
180 				if (value == 0) {
181 					m_BestTest.m_TestStat[0].updateWeighted(tuple, i);
182 					m_BestTest.m_TestStat[0].setSplitIndex(i + 1); // daniela
183 				}
184 			}
185 			// Also compute the statistic for the zeros
186 			m_BestTest.m_TestStat[1].copy(m_BestTest.m_TotStat);
187 			m_BestTest.m_TestStat[1].subtractFromThis(m_BestTest.m_TestStat[0]);
188 		} else {
189 			// Regular code for non-binary attributes
190 			for (int i = 0; i < nb_rows; i++) {
191 				DataTuple tuple = sample.getTuple(i);
192 				int value = at.getNominal(tuple);
193 				m_BestTest.m_TestStat[value].updateWeighted(tuple, i);
194 				m_BestTest.m_TestStat[value].setSplitIndex(i + 1); // daniela
195 			}
196 		}
197 
198 		/*
199 		 * long stop_time = System.currentTimeMillis(); long elapsed = stop_time -
200 		 * start_time; m_Timer += elapsed;
201 		 */
202 
203 		// ClusLogger.info("done");
204 
205 		// Find best split
206 		m_Split.findSplit(m_BestTest, at);
207 	}

234 	public void findNominalExtraTree(NominalAttrType at, RowData data, ClusRandomNonstatic rnd) throws ClusException {
235 		Random rn;
236 		if (rnd == null) {
237 			rn = ClusRandom.getRandom(ClusRandom.RANDOM_EXTRATREE);
238 			ClusEnsembleInduce.giveParallelisationWarning(ParallelTrap.StaticRandom);
239 		} else {
240 			rn = rnd.getRandom(ClusRandomNonstatic.RANDOM_EXTRATREE);
241 		}
242 		// Reset positive statistic
243 		RowData sample = createSample(data, rnd);
244 		int nbvalues = at.getNbValues();
245 		m_BestTest.reset(nbvalues + 1);
246 		// For each attribute value
247 		int nb_rows = sample.getNbRows();
248 		for (int i = 0; i < nb_rows; i++) {
249 			DataTuple tuple = sample.getTuple(i);
250 			int value = at.getNominal(tuple);
251 			m_BestTest.m_TestStat[value].updateWeighted(tuple, i);
252 		}
253 		// Find the split
254 		m_Split.findExtraTreeSplit(m_BestTest, at, rn);
255 	}

257 	public void findNumeric(NumericAttrType at, RowData data, ClusRandomNonstatic rnd) throws Exception {
258 		RowData sample = createSample(data, rnd);
259 		DataTuple tuple;
260 		// if (at.isSparse()) {
261 		// sample.sortSparse(at, m_SortHelper);
262 		// }
263 		// else {
264 		// sample.sort(at);
265 		// }
266 		Integer[] indicesSorted = sample.smartSort(at);
267 
268 		m_BestTest.reset(2);
269 		// Missing values
270 		int pos = 0;
271 		int nb_rows = sample.getNbRows();
272 		// Copy total statistic into corrected total
273 		m_BestTest.copyTotal();
274 
275 		if (at.isMissing(sample.getTuple(indicesSorted[pos]))) { // at.hasMissing()
276 			// matejp: changed if-condition since sparse attributes do not detect missing
277 			// values in the reading phase
278 			// The conditions are equivalent since all missing values are in the front
279 			// because of sorting
280 			while (pos < nb_rows && at.isMissing(tuple = sample.getTuple(indicesSorted[pos]))) {
281 				m_BestTest.m_MissingStat.updateWeighted(tuple, indicesSorted[pos]);
282 				pos++;
283 			}
284 			m_BestTest.subtractMissing();
285 		}
286 
287 		double prev = Double.NaN;
288 		boolean isGIS = !data.getSchema().getSettings().getAttribute().isNullGIS();
289 		if (isGIS) {
290 			gisMatrixForFindNumberic(sample, nb_rows, pos, indicesSorted, at); // matejp: moved to a separate method for
291 																				// readability
292 		}
293 
294 		double minValue = (pos < nb_rows) ? at.getNumeric(sample.getTuple(indicesSorted[nb_rows - 1])) : Double.NaN;
295 		boolean isSparseAtr = at.isSparse();
296 
297 		double tot_corr_SVarS = 0.0; // does not matter which value we choose;
298 		// boolean isEfficient = m_BestTest.m_Heuristic.isEfficient();
299 		// if(isEfficient){
300 		// tot_corr_SVarS =
301 		// m_BestTest.m_TotCorrStat.getSVarS(m_BestTest.m_Heuristic.getClusteringAttributeWeights());
302 		//// m_BestTest.m_Heuristic.setSplitStatSVarS(tot_corr_SVarS);
303 		// }
304 		boolean middleSplits = data.getSchema().getSettings().getTree().getSplitPosition()
305 				.equals(SplitPositions.Middle);
306 		if (isGIS) {
307 			for (int i = pos; i < nb_rows; i++) {
308 				tuple = sample.getTuple(indicesSorted[i]);
309 				double value = at.getNumeric(tuple);
310 				if (value != prev) {
311 					m_BestTest.updateNumericGIS(value, at, indicesSorted);
312 					prev = value;
313 					m_BestTest.m_PosStat.setPrevIndex(i); // daniela
314 				}
315 				m_BestTest.m_PosStat.updateWeighted(tuple, i);
316 				m_BestTest.m_PosStat.setSplitIndex(i + 1); // daniela
317 				if (isSparseAtr && value == minValue) {
318 					break;
319 				}
320 			}
321 		} else {
322 			for (int i = pos; i < nb_rows; i++) {
323 				tuple = sample.getTuple(indicesSorted[i]);
324 				double value = at.getNumeric(tuple);
325 				if (value != prev) {
326 					double middle = prev + (value - prev) / 2; // numerically more stable version for (value + prev) / 2
327 					if (middleSplits && isFinite(middle)) { // choose the splitting point: in the middle or the exact
328 															// attribute value
329 						m_BestTest.updateNumeric(middle, at, tot_corr_SVarS, false); // isEfficient
330 					} else {
331 						m_BestTest.updateNumeric(value, at, tot_corr_SVarS, false);
332 					}
333 					prev = value;
334 				}
335 				m_BestTest.m_PosStat.updateWeighted(tuple, indicesSorted[i]);
336 				if (isSparseAtr && value == minValue) {
337 					break;
338 				}
339 			}
340 		}
341 	}

343 	private void gisMatrixForFindNumberic(RowData sample, int nb_rows, int pos, Integer[] indicesSorted,
344 			NumericAttrType at) throws ClusException {
345 		DataTuple tuple;
346 		double prev = Double.NaN;
347 		// daniela
348 		m_BestTest.m_PosStat.setData(sample);
349 		m_BestTest.m_TotCorrStat.setData(sample);
350 		m_BestTest.m_TotCorrStat.setSplitIndex(sample.getNbRows());
351 		m_BestTest.m_PosStat.setPrevIndex(0);
352 		m_BestTest.m_PosStat.initializeSum();
353 		m_BestTest.m_TotCorrStat.initializeSum();
354 		// for scaling of h -regression (daniela)
355 		if ((m_BestTest.m_Heuristic instanceof GISHeuristic
356 				|| m_BestTest.m_Heuristic instanceof VarianceReductionHeuristicCompatibility)
357 				&& (SettingsTree.ALPHA != 1.0)) {
358 			for (int i = pos; i < nb_rows; i++) {
359 				tuple = sample.getTuple(indicesSorted[i]); // every such tuple is positive or, as some people say, sekoj
360 															// vakov tuple pripaga vo positive
361 				double value = at.getNumeric(tuple);
362 				if (value != prev) {
363 					if (!Double.isNaN(value)) {
364 						m_BestTest.calculateHMinMax(value, at);
365 						// ClusLogger.info(at +" za: "+m_BestTest.hMax+"-->"+m_BestTest.hMin);
366 					}
367 					prev = value;
368 					m_BestTest.m_PosStat.setPrevIndex(i);
369 				}
370 				m_BestTest.m_PosStat.updateWeighted(tuple, i);
371 				m_BestTest.m_PosStat.setSplitIndex(i + 1); // add Daniela
372 			}
373 			prev = Double.NaN;
374 			m_BestTest.m_PosStat.setData(sample);
375 			m_BestTest.m_TotCorrStat.setData(sample);
376 			m_BestTest.m_TotCorrStat.setSplitIndex(sample.getNbRows());
377 			m_BestTest.m_PosStat.reset();
378 			m_BestTest.m_PosStat.setPrevIndex(0);
379 			m_BestTest.m_PosStat.initializeSum();
380 		}
381 		// end for scaling of h -regression
382 
383 		// daniela generateMatrix for each attribute gisheuristic
384 		try {
385 			ClusHeuristic m_Heuristic = m_StatManager.getHeuristic();
386 			// ClusLogger.info(m_Heuristic);
387 			if (SettingsTree.ALPHA != 1.0 && m_Heuristic instanceof VarianceReductionHeuristicCompatibility) {
388 				VarianceReductionHeuristicCompatibility gisHeuristic = (VarianceReductionHeuristicCompatibility) m_Heuristic;
389 				ClusAttrType[] arr = at.getSchema().getAllAttrUse(AttributeUseType.GIS); // numeric and string GIS
390 				if (arr.length == 1) {
391 					gisHeuristic.readMatrixFromFile(sample); // 1 coordinate and a distance.csv file associated
392 				} else {
393 					gisHeuristic.generateMatrix(sample); // 2 coordinates
394 				}
395 			}
396 
397 			if (m_Heuristic instanceof GISHeuristic && SettingsTree.ALPHA != 1.0) {
398 				GISHeuristic gisHeuristic = (GISHeuristic) m_Heuristic;
399 				ClusAttrType[] arr = at.getSchema().getAllAttrUse(AttributeUseType.GIS); // numeric and string GIS
400 				if (arr.length == 1) {
401 					gisHeuristic.readMatrixFromFile(sample); // 1 coordinate and a distance.csv file associated
402 				} else {
403 					gisHeuristic.generateMatrix(sample, indicesSorted); // 2 coordinates
404 				}
405 			}
406 		} catch (Exception e) {
407 			e.printStackTrace();
408 			throw new ClusException("Error calculating GISHEuristics!"); // this try/catch block is weird
409 		}
410 	}

412 	public void findNumericExtraTree(NumericAttrType at, RowData orig_data, ClusRandomNonstatic rnd)
413 			throws ClusException {
414 		// TODO: if this method gets completed, sampling of the RowDatas must be
415 		// included as well
416 
417 		Random rn;
418 		if (rnd == null) {
419 			rn = ClusRandom.getRandom(ClusRandom.RANDOM_EXTRATREE);
420 			ClusEnsembleInduce.giveParallelisationWarning(ParallelTrap.StaticRandom);
421 		} else {
422 			rn = rnd.getRandom(ClusRandomNonstatic.RANDOM_EXTRATREE);
423 		}
424 
425 		RowData data = createSample(orig_data, rnd);
426 		DataTuple tuple;
427 		int idx = at.getArrayIndex();
428 
429 		// Sort values from large to small
430 		// if (at.isSparse()) {
431 		// data.sortSparse(at, m_SortHelper);
432 		// } else {
433 		// data.sort(at);
434 		// }
435 
436 		Integer[] indicesSorted = data.smartSort(at);
437 
438 		m_BestTest.reset(2);
439 		// Missing values
440 		int pos = 0;
441 		int nb_rows = data.getNbRows();
442 		// Copy total statistic into corrected total
443 		m_BestTest.copyTotal();
444 		if (at.hasMissing()) {
445 			// Because of sorting, all missing values are in the front :-)
446 			while (pos < nb_rows && (tuple = data.getTuple(indicesSorted[pos])).hasNumMissing(idx)) {
447 				m_BestTest.m_MissingStat.updateWeighted(tuple, indicesSorted[pos]);
448 				pos++;
449 			}
450 			m_BestTest.subtractMissing();
451 		}
452 		// now indicesSorted[pos] is the index of the tuple that has the highest (and
453 		// non-missing) value of the attribute
454 
455 		// Generate the random split value based on the original data
456 		if (pos == nb_rows) {// this can prevent illegal tests;
457 			m_BestTest.m_BestHeur = Double.NEGATIVE_INFINITY;
458 		} else {
459 			// double value = at.getNumeric(tuple);
460 			double min_value = at.getNumeric(data.getTuple(indicesSorted[nb_rows - 1]));  // data.getTuple(indicesSorted[nb_rows - 1]).getDoubleVal(idx);
461 			double max_value = at.getNumeric(data.getTuple(indicesSorted[pos]));  // data.getTuple(indicesSorted[pos]).getDoubleVal(idx);
462 			double split_value = (max_value - min_value) * rn.nextDouble() + min_value;
463 			for (int i = pos; i < nb_rows; i++) {
464 				tuple = data.getTuple(indicesSorted[i]);
465 				if (at.getNumeric(tuple) <= split_value) {  // (tuple.getDoubleVal(idx) <= split_value)
466 					break;
467 				}
468 				m_BestTest.m_PosStat.updateWeighted(tuple, indicesSorted[i]);
469 			}
470 
471 			m_BestTest.updateNumeric(split_value, at); // we test only one split per attribute --> no need for
472 														// precomputing tot_corr_SVarS
473 			// m_BestTest.updateNumeric((value + prev)/2, at); //FIXME: splitting point is
474 			// in the middle of two values -> this is maybe better?
475 		}
476 
477 		// System.err.println("Inverse splits not yet included!");
478 		// TODO: m_Selector.updateInverseNumeric(split_value, at);
479 	}

624 	private RowData createSample(RowData original, ClusRandomNonstatic rnd) {
625 		int N = getSettings().getTree().getTreeSplitSampling();
626 		if (N == 0) {
627 			return original.sample(N, rnd);
628 		} else {
629 			String message = String.format("The value of SplitSampling = %d will result in wrong results.\n"
630 					+ "Use SplitSampling = 0 or correct the code.", N);
631 			throw new RuntimeException(message);
632 		}
633 	}
