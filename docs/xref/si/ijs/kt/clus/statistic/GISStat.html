85      /**
86       * Clone this statistic by taking the given weight into account.
87       * This is used for example to get the weighted prediction of default rule.
88       */
89      @Override
90      public ClusStatistic copyNormalizedWeighted(double weight) {
91          //    GISStat newStat = (GISStat) cloneSimple();
92          GISStat newStat = normalizedCopy();
93          for (int iTarget = 0; iTarget < newStat.getNbAttributes(); iTarget++) {
94              newStat.m_Means[iTarget] = weight * newStat.m_Means[iTarget];
95          }
96          return newStat;
97      }

100     @Override
101     public void copy(ClusStatistic other) {
102         GISStat or = (GISStat) other;
103         m_SumWeight = or.m_SumWeight;
104         m_NbExamples = or.m_NbExamples;
105         m_NbTarget = or.m_NbTarget;
106         System.arraycopy(or.m_SumWeights, 0, m_SumWeights, 0, m_NbAttrs);
107         System.arraycopy(or.m_SumValues, 0, m_SumValues, 0, m_NbAttrs);
108         System.arraycopy(or.m_SumSqValues, 0, m_SumSqValues, 0, m_NbAttrs);
109     }

116     public GISStat normalizedCopy() {
117         GISStat copy = (GISStat) cloneSimple();
118         copy.m_NbExamples = 0;

121         return copy;

125     @Override
126     public void add(ClusStatistic other) {
127         GISStat or = (GISStat) other;
128         m_SumWeight += or.m_SumWeight;
129         m_NbExamples += or.m_NbExamples;
130         for (int i = 0; i < m_NbAttrs; i++) {
131             m_SumWeights[i] += or.m_SumWeights[i];
132             m_SumValues[i] += or.m_SumValues[i];
133             m_SumSqValues[i] += or.m_SumSqValues[i];
134         }
135     }

177     public double getVarLisa(int i) {
178         double n_tot = m_SumWeight;
179         double k_tot = m_SumWeights[i];
180         double sv_tot = m_SumValues[i];
181         double ss_tot = m_SumSqValues[i];
182         if (k_tot <= MathUtil.C1E_9 && m_Training != null) {
183             return m_Training.getSVarS(i);
184         }
185         else {
186             return (k_tot > 1.0) ? ss_tot * (n_tot - 1) / (k_tot - 1) - n_tot * sv_tot / k_tot * sv_tot / k_tot : 0.0;
187         }
188     }

191     public double getVarLisa(ClusAttributeWeights scale) {
192         double result = 0.0;
193         for (int i = 0; i < m_NbAttrs; i++) {
194             double n_tot = m_SumWeight;
195             double k_tot = m_SumWeights[i];
196             double sv_tot = m_SumValues[i];
197             double ss_tot = m_SumSqValues[i];
198 
199             if (k_tot == n_tot) {
200                 result += (ss_tot - sv_tot * sv_tot / n_tot) * scale.getWeight(m_Attrs[i]);
201             }
202             else {
203                 if (k_tot <= MathUtil.C1E_9 && m_Training != null) {
204                     result += m_Training.getSVarS(i) * scale.getWeight(m_Attrs[i]);
205                 }
206                 else {
207                     result += (ss_tot * (n_tot - 1) / (k_tot - 1) - n_tot * sv_tot / k_tot * sv_tot / k_tot) * scale.getWeight(m_Attrs[i]);
208                 }
209             }
210         }
211         return result / m_NbAttrs;
212     }

215     public double getVarLisaDiff(ClusAttributeWeights scale, ClusStatistic other) {
216         double result = 0.0;
217         RegressionStat or = (RegressionStat) other;
218         for (int i = 0; i < m_NbAttrs; i++) {
219             double n_tot = m_SumWeight - or.m_SumWeight;
220             double k_tot = m_SumWeights[i] - or.getSumWeights(i); //or.m_SumWeights[i];   matejp was here
221             double sv_tot = m_SumValues[i] - or.getSumValues(i); //or.m_SumValues[i];
222             double ss_tot = m_SumSqValues[i] - or.getSumSqValues(i); //or.m_SumSqValues[i];
223             if (k_tot == n_tot) {
224                 result += (ss_tot - sv_tot * sv_tot / n_tot) * scale.getWeight(m_Attrs[i]);
225             }
226             else {
227                 if (k_tot <= MathUtil.C1E_9 && m_Training != null) {
228                     result += m_Training.getSVarS(i) * scale.getWeight(m_Attrs[i]);
229                 }
230                 else {
231                     result += (ss_tot * (n_tot - 1) / (k_tot - 1) - n_tot * sv_tot / k_tot * sv_tot / k_tot) * scale.getWeight(m_Attrs[i]);
232                 }
233             }
234         }
235         return result / m_NbAttrs;
236     }
