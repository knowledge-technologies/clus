61      public ClusInductionAlgorithm createInduce(ClusSchema schema, Settings sett, CMDLineArgs cargs) throws ClusException, IOException {
62          schema.addIndices(ClusSchema.ROWS);
63          int nb_num = schema.getNbNumericDescriptiveAttributes();
64          sett.getExperimental();
65          if (SettingsExperimental.XVAL_OVERLAP && nb_num > 0)
66              return new OptXValIndOV(schema, sett);
67          else
68              return new OptXValIndNO(schema, sett);
69      }

72      public final void addFoldNrs(RowData set, XValMainSelection sel) {
73          int nb = set.getNbRows();
74          for (int i = 0; i < nb; i++) {
75              int fold = sel.getFold(i);
76              DataTuple tuple = set.getTuple(i);
77              tuple.setIndex(fold + 1);
78          }
79      }
80  
81  
82      public final static void showFoldsInfo(PrintWriter writer, Object root) {
83          OptXValBinTree bintree = OptXValBinTree.convertTree(root);
84  
85          double[] fis = bintree.getFIs();
86          double[] nodes = bintree.getNodes();
87          double[] times = bintree.getTimes();
88          MDoubleArray.divide(fis, nodes);
89  
90          writer.println("FoldsInfo");
91          writer.println("Nodes:  " + MDoubleArray.toString(nodes));
92          writer.println("f(i-1): " + MDoubleArray.toString(fis));
93          writer.println("Time:   " + MDoubleArray.toString(times));
94      }
95  
96  
97      public final static void showForest(PrintWriter writer, OptXValNode root) {
98          writer.println("XVal Forest");
99          writer.println("***********");
100         writer.println();
101         showFoldsInfo(writer, root);
102         writer.println();
103         root.printTree(writer, "");
104         writer.println();
105     }

108     public final void xvalRun(String appname, Date date) throws Exception {
109         Settings sett = m_Clus.getSettings();
110         ClusSchema schema = m_Clus.getSchema();
111         RowData set = m_Clus.getRowDataClone();
112         XValMainSelection sel = schema.getXValSelection(set);
113         addFoldNrs(set, sel);
114         OptXValInduce induce = (OptXValInduce) m_Clus.getInduce();
115         induce.initialize(sel.getNbFolds());
116         Long time = null;
117         if (Clus.isDebug()) {
118             time = ResourceInfo.getCPUTime();
119         }
120 
121         OptXValNode root = null;
122         int nbr = 0;
123         while (true) {
124             root = induce.optXVal(set);
125             nbr++;
126             if (Clus.isDebug()) {
127                 if ((ResourceInfo.getCPUTime() - time) > 5000.0)
128                     break;
129             }
130 
131         }
132         ClusSummary summary = m_Clus.getSummary();
133         if (Clus.isDebug()) {
134             if (Clus.isDebug()) {
135                 summary.setInductionTime((long) ClusStat.addToTotal(ResourceInfo.getCPUTime() - time, nbr));
136             }
137 
138         }
139 
140         if (Clus.isDebug()) {
141             ClusStat.addTimes(nbr);
142         }
143 
144         // Output whole tree
145         // MultiScore score = m_Clus.getMultiScore();
146         ClusOutput output = new ClusOutput(appname + ".out", schema, sett);
147         output.writeHeader();
148         ClusNode tree = root.getTree(0);
149         ClusRun cr = m_Clus.partitionData();
150         tree.afterInduce(null);
151         // m_Clus.storeAndPruneModel(cr, tree);
152         // m_Clus.calcError(cr, null);
153         output.writeOutput(cr, true);
154         output.close();
155         // Output xval trees
156         output = new ClusOutput(appname + ".xval", schema, sett);
157         output.writeHeader();
158 
159         if (SettingsExperimental.SHOW_XVAL_FOREST)
160             showForest(output.getWriter(), root);
161         for (int i = 0; i < sel.getNbFolds(); i++) {
162             XValSelection msel = new XValSelection(sel, i);
163             cr = m_Clus.partitionData(msel, i + 1);
164             tree = root.getTree(i + 1);
165             tree.afterInduce(null);
166             // m_Clus.storeAndPruneModel(cr, tree);
167             // m_Clus.calcError(cr, summary);
168             if (sett.getOutput().isOutputFoldModels())
169                 output.writeOutput(cr, false);
170         }
171         output.writeSummary(summary);
172         output.close();
173     }
