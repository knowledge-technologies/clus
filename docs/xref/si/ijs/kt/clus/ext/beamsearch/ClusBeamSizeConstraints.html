36  public class ClusBeamSizeConstraints {
37  
38      protected int m_LeafCount, m_StopCount;
39      protected boolean m_IsModified;
40      protected boolean m_Debug;

42  
43      public void enforce(ClusNode root, int size) {
44          reset();
45          initVisitors(root, size);
46          computeCostUsingConstraints(root, size);
47          ClusBeamSizeConstraintInfo info = (ClusBeamSizeConstraintInfo) root.getVisitor();
48          pruneUsingConstraints(root, size, info.realcost[size]);
49          pruneNonMarkedNodes(root);
50          removeVisitors(root);
51      }
52  
53  
54      public boolean isModified() {
55          return m_IsModified;
56      }
57  
58  
59      public boolean isFinished() {
60          return m_StopCount >= m_LeafCount;
61      }
62  
63  
64      public void reset() {
65          m_LeafCount = 0;
66          m_StopCount = 0;
67          m_IsModified = false;
68      }

71      public static double computeLowerBound(ClusStatistic stat, int l) {
72          ClassificationStat cs = (ClassificationStat) stat;
73          double[] clcnts = cs.getClassCounts(0);
74          double[] c2 = new double[clcnts.length];
75          System.arraycopy(clcnts, 0, c2, 0, c2.length);
76          Arrays.sort(c2);
77          int s = (l - 1) / 2;
78          double result = 0.0;
79          for (int i = s + 2; i <= c2.length; i++) {
80              result += c2[c2.length - i];
81          }
82          return result;
83      }

86      public static void computeCostUsingConstraints(ClusNode node, int l) {
87          ClusBeamSizeConstraintInfo info = (ClusBeamSizeConstraintInfo) node.getVisitor();
88          if (info.computed[l]) { return; }
89          boolean is_leaf = node.atBottomLevel();
90          if (l < 3 || (is_leaf && ((ClusBeamAttrSelector) info.visitor).isStopCrit())) {
91              info.realcost[l] = info.lowcost[l] = node.getClusteringStat().getError();
92          }
93          else if (is_leaf) {
94              info.realcost[l] = node.getClusteringStat().getError();
95              info.lowcost[l] = computeLowerBound(node.getClusteringStat(), l);
96          }
97          else {
98              info.realcost[l] = info.lowcost[l] = node.getClusteringStat().getError();
99              ClusNode ch1 = (ClusNode) node.getChild(0);
100             ClusNode ch2 = (ClusNode) node.getChild(1);
101             ClusBeamSizeConstraintInfo i1 = (ClusBeamSizeConstraintInfo) ch1.getVisitor();
102             ClusBeamSizeConstraintInfo i2 = (ClusBeamSizeConstraintInfo) ch2.getVisitor();
103             for (int k1 = 1; k1 <= l - 2; k1++) {
104                 int k2 = l - k1 - 1;
105                 computeCostUsingConstraints(ch1, k1);
106                 computeCostUsingConstraints(ch2, k2);
107                 double realcost1 = i1.realcost[k1];
108                 double realcost2 = i2.realcost[k2];
109                 if (realcost1 + realcost2 < info.realcost[l]) {
110                     info.realcost[l] = realcost1 + realcost2;
111                 }
112                 double lowcost1 = i1.lowcost[k1];
113                 double lowcost2 = i2.lowcost[k2];
114                 if (lowcost1 + lowcost2 < info.lowcost[l]) {
115                     info.lowcost[l] = lowcost1 + lowcost2;
116                 }
117             }
118         }
119         info.computed[l] = true;
120     }

123     public void pruneUsingConstraints(ClusNode node, int l, double b) {
124         ClusBeamSizeConstraintInfo info = (ClusBeamSizeConstraintInfo) node.getVisitor();
125         info.marked = true;
126         // if (m_Debug) ClusLogger.info("Bound: "+b);
127         // if (m_Debug) ClusLogger.info("Node: "+node+" Stat: "+node.getTotalStat());
128         if (b <= info.bound[l]) {
129             // if (m_Debug) ClusLogger.info("b <= info.bound[l]: "+b+" <= "+info.bound[l]);
130             return;
131         }
132         for (int i = 1; i <= l; i++) {
133             if (b > info.bound[i])
134                 info.bound[i] = b;
135         }
136         if (info.lowcost[l] > b || Math.abs(info.lowcost[l] - node.getClusteringStat().getError()) < 1e-12) {
137             // if (m_Debug) ClusLogger.info("info.lowcost[l] > b || info.lowcost[l] ==
138             // node.getTotalStat().getError()"+info.lowcost[l]+" > b = "+b+" err = "+node.getTotalStat().getError());
139             return;
140         }
141         if (l >= 3 && !node.atBottomLevel()) {
142             for (int k1 = 1; k1 <= l - 2; k1++) {
143                 int k2 = l - k1 - 1;
144                 ClusNode ch1 = (ClusNode) node.getChild(0);
145                 ClusNode ch2 = (ClusNode) node.getChild(1);
146                 ClusBeamSizeConstraintInfo i1 = (ClusBeamSizeConstraintInfo) ch1.getVisitor();
147                 ClusBeamSizeConstraintInfo i2 = (ClusBeamSizeConstraintInfo) ch2.getVisitor();
148                 if (i1.lowcost[k1] + i2.lowcost[k2] <= b) {
149                     double b1 = b - i2.lowcost[k2];
150                     double b2 = b - i1.lowcost[k1];
151                     pruneUsingConstraints(ch1, k1, b1);
152                     pruneUsingConstraints(ch2, k2, b2);
153                 }
154             }
155         }
156     }

159     public void pruneNonMarkedNodes(ClusNode node) {
160         ClusBeamSizeConstraintInfo info = (ClusBeamSizeConstraintInfo) node.getVisitor();
161         if (node.atBottomLevel()) {
162             m_LeafCount++;
163             ClusBeamAttrSelector attrsel = (ClusBeamAttrSelector) info.visitor;
164             if (attrsel.isStopCrit())
165                 m_StopCount++;
166         }
167         else {
168             if (!info.marked) {
169                 ClusBeamAttrSelector attrsel = new ClusBeamAttrSelector();
170                 info.visitor = attrsel;
171                 node.makeLeaf();
172                 attrsel.setStopCrit(true);
173                 m_IsModified = true;
174             }
175             else {
176                 for (int i = 0; i < node.getNbChildren(); i++) {
177                     ClusNode child = (ClusNode) node.getChild(i);
178                     pruneNonMarkedNodes(child);
179                 }
180             }
181         }
182     }

185     public static void initVisitors(ClusNode node, int size) {
186         ClusBeamSizeConstraintInfo info = new ClusBeamSizeConstraintInfo(size);
187         info.visitor = node.getVisitor();
188         node.setVisitor(info);
189         for (int i = 0; i < node.getNbChildren(); i++) {
190             ClusNode child = (ClusNode) node.getChild(i);
191             initVisitors(child, size);
192         }
193     }
194 
195 
196     public static void removeVisitors(ClusNode node) {
197         /* Restore old visitors */
198         ClusBeamSizeConstraintInfo info = (ClusBeamSizeConstraintInfo) node.getVisitor();
199         node.setVisitor(info.visitor);
200         for (int i = 0; i < node.getNbChildren(); i++) {
201             ClusNode child = (ClusNode) node.getChild(i);
202             removeVisitors(child);
203         }
204     }
