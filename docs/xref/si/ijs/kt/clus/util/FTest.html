55      public static int getLevelAndComputeArray(double significance, int verbosityLevel) {
56          int maxlevel = FTEST_SIG.length - 1;
57          for (int level = 0; level < maxlevel; level++) {
58              if (Math.abs(significance - FTEST_SIG[level]) / FTEST_SIG[level] < 0.01) { return level; }
59          }
60          FTEST_SIG[maxlevel] = significance;
61          initializeFTable(significance, verbosityLevel);
62          return maxlevel;
63      }

143     // Calling getCriticalFCommonsMath() is slow, so build a table
144     public static void initializeFTable(double sig, int verbosityLevel) {
145         int df = 3;
146         double value = 0.0;
147         double limit = getCriticalFCommonsMath(sig, 1e5);
148         ArrayList<Double> values = new ArrayList<Double>();
149         do {
150             value = getCriticalFCommonsMath(sig, df);
151             values.add(new Double(value));
152             df++;
153         }
154         while ((value - limit) / limit > 0.05);
155         if (verbosityLevel > 0) {
156             ClusLogger.info("F-Test = " + sig + " limit = " + ClusFormat.TWO_AFTER_DOT.format(limit) + " values = " + values.size());
157         }
158         FTEST_LIMIT = limit;
159         FTEST_VALUE = new double[values.size() + 3];
160         for (int i = 0; i < values.size(); i++) {
161             FTEST_VALUE[i + 3] = values.get(i).doubleValue();
162         }
163     }

166     // ftest: Signif, total SS, residual SS, 2nd DF (1st is 1)
167     // (this implementation only works for F tests with 1 and n d.f., sorry)
168     // ftest predicate succeeds iff H0 is rejected at Signif
169     // only works correctly for signif 1.0, 0.1, 0.05, 0.01, 0.005, 0.001
170 
171     public static boolean ftest(int level, double sst, double ssr, int df) {
172         if (level == 0)
173             return true;
174         if (sst <= 0.0)
175             return false; // added to avoid 0/0; SST = 0 => no improvement possible
176         if (ssr == 0.0)
177             return true; // avoid x/0; SSR = 0 => F is infinite
178         double f = df * (sst - ssr) / ssr;
179         double cf = getCriticalF(level, df);
180         return f > cf;
181     }

189     public static double calcVarianceReductionHeuristic(double n_tot, double ss_tot, double ss_sum) {
190         double value = ss_tot - ss_sum;
191         if (value < MathUtil.C1E_9) {
192             // Gain too small
193             return Double.NEGATIVE_INFINITY;
194         }
195         if (SettingsTree.FTEST_LEVEL == 0) {
196             // No F-test -> just return value
197             return value;
198         }
199         int n_2 = (int) Math.floor(n_tot - 2.0 + 0.5);
200         if (n_2 <= 0) {
201             return Double.NEGATIVE_INFINITY;
202         }
203         else {
204             if (FTest.ftest(SettingsTree.FTEST_LEVEL, ss_tot, ss_sum, n_2)) {
205                 return value;
206             }
207             else {
208                 return Double.NEGATIVE_INFINITY;
209             }
210         }
