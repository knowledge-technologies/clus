85      @Override
86      public void findSplit(CurrentBestTestAndHeuristic node, NominalAttrType type) throws ClusException {
87          // ClusLogger.info("find split for attr: " + type);
88          double unk_freq = 0.0;
89          int nbvalues = type.getNbValues();
90          boolean isin[] = new boolean[nbvalues];
91          //boolean acceptable = true; // can only be changed for phylogenetic trees
92          // If has missing values?
93          if (type.hasMissing()) {
94              ClusStatistic unknown = node.m_TestStat[nbvalues];
95              m_MStat.copy(node.m_TotStat);
96              m_MStat.subtractFromThis(unknown);
97              unk_freq = unknown.getTotalWeight() / node.m_TotStat.getTotalWeight();
98          }
99          else {
100             m_MStat.copy(node.m_TotStat);
101         }
102         int card = 0;
103         double pos_freq = 0.0;
104         double bheur = Double.NEGATIVE_INFINITY;
105         // Not working for rules except if constraint of tests to '1' is desired!
106         if (nbvalues == 2 && (!getStatManager().isRuleInduceOnly() || getStatManager().getSettings().getRules().isConstrainedToFirstAttVal())) {
107             // Handle binary splits efficiently
108             card = 1;
109             isin[0] = true;
110             ClusStatistic CStat = node.m_TestStat[0];
111             bheur = node.calcHeuristic(m_MStat, CStat);
112             // showTest(type, isin, -1, bheur, m_MStat, m_CStat);
113             pos_freq = CStat.getTotalWeight() / m_MStat.getTotalWeight();
114         }
115         else if ((getStatManager().getClusterMode() == ClusStatManager.Mode.PHYLO) && (getStatManager().getSettings().getPhylogeny().getPhylogenySequence().equals(PhylogenySequence.DNA))) {
116             // for phylogenetic trees with DNA sequences, we use an optimization method: tests like pos10={A,t} are
117             // based on the results for tests pos10=A and pos10=T
118             // we do not do this for protein sequences, since there the alphabet is much larger, which would complicate
119             // things
120             // else if (false) {
121             boolean[] valid = new boolean[nbvalues];
122             // we try all subsets of size 1 (e.g. pos10=A) and 2 (e.g. pos10={AT}) (nbvalues = 5)
123             // first we try all subsets of size 1
124             for (int j = 0; j < nbvalues; j++) {
125                 m_PStat.reset();
126                 m_PStat.add(node.m_TestStat[j]);
127                 double mheur = node.calcHeuristic(m_MStat, m_PStat);
128                 if (mheur > bheur) {
129                     bheur = mheur;
130                     isin = new boolean[nbvalues];
131                     isin[j] = true;
132                     card = 1;
133                     // Calculate pos freq (of current best one)
134                     pos_freq = m_PStat.getTotalWeight() / m_MStat.getTotalWeight();
135                 }
136                 if (mheur > Double.NEGATIVE_INFINITY) {
137                     valid[j] = true;
138                 }
139             }
140             // then we try all subsets of size 2 (if both of the singleton subsets returned a valid heuristic value)
141             for (int j = 0; j < nbvalues; j++) {
142                 if (valid[j])
143                     for (int k = j + 1; k < nbvalues; k++) {
144                         if (valid[k]) {
145                             m_PStat.reset();
146                             m_CStat.copy(m_PStat);
147                             m_CStat.add(node.m_TestStat[j]);
148                             m_PStat.add(node.m_TestStat[j]);
149                             m_PStat.add(node.m_TestStat[k]);
150                             // double mheur = node.calcHeuristic(m_MStat, m_PStat); // use this if you want to compute
151                             // the heuristic only in the case both singletons were valid
152                             ClusStatistic[] csarray = new ClusStatistic[2];
153                             csarray[0] = m_PStat;
154                             csarray[1] = m_CStat;
155                             double mheur = node.calcHeuristic(m_MStat, csarray, 2); // use this if you want to compute
156                                                                                     // the heuristic only in the case
157                                                                                     // both singletons were valid AND
158                                                                                     // you want to reuse their
159                                                                                     // computations
160                                                                                     // ClusLogger.info("mheur: " +
161                                                                                     // mheur);
162                             if (mheur > bheur) {
163                                 bheur = mheur;
164                                 isin = new boolean[nbvalues];
165                                 isin[j] = true;
166                                 isin[k] = true;
167                                 card = 2;
168                                 // Calculate pos freq (of current best one)
169                                 pos_freq = m_PStat.getTotalWeight() / m_MStat.getTotalWeight();
170                             }
171                         }
172                     }
173             }
174         }
175         else {
176             // Try to add values to subsets
177             // Each iteration the cardinality increases by at most one
178             m_PStat.reset();
179             int bvalue = 0;
180             if ((m_PStat instanceof CombStat) && ((CombStat) m_PStat).getSettings().getRules().isHeurRuleDist()) {
181                 ((ClusRuleHeuristicDispersion) node.m_Heuristic).setDataIndexes(new int[0]);
182             }
183             boolean allowSubsetSplits = getStatManager().getSettings().getModel().isNominalSubsetTests();
184             while ((bvalue != -1) && ((card + 1) < nbvalues)) {
185                 bvalue = -1;
186                 for (int j = 0; j < nbvalues; j++) {
187                     if (!isin[j]) {
188                         // Try to add this one to the positive stat
189                         m_CStat.copy(m_PStat);
190                         m_CStat.add(node.m_TestStat[j]);
191                         if ((m_PStat instanceof CombStat) && ((CombStat) m_PStat).getSettings().getRules().isHeurRuleDist()) {
192                             boolean isin_current[] = new boolean[nbvalues];
193                             for (int k = 0; k < nbvalues; k++) {
194                                 isin_current[k] = isin[k];
195                             }
196                             isin_current[j] = true;
197                             ((ClusRuleHeuristicDispersion) node.m_Heuristic).setDataIndexes(isin_current);
198                         }
199                         // Calc heuristic
200 
201                         boolean isin_current[] = new boolean[nbvalues];
202                         for (int k = 0; k < nbvalues; k++) {
203                             isin_current[k] = isin[k];
204                         }
205                         isin_current[j] = true;
206                         double mheur = node.calcHeuristic(m_MStat, m_CStat);
207                         // showTest(type, isin_current, -1, mheur, m_MStat, m_CStat);
208                         // ClusLogger.info("mheur: " + mheur);
209                         if (mheur > bheur) {
210                             bheur = mheur;
211                             bvalue = j;
212                             // Calculate pos freq (of current best one)
213                             pos_freq = m_CStat.getTotalWeight() / m_MStat.getTotalWeight();
214                             node.checkAcceptable(m_MStat, m_CStat);
215                         }
216                     }
217                 }
218                 if (bvalue != -1) {
219                     card++;
220                     isin[bvalue] = true;
221                     m_PStat.add(node.m_TestStat[bvalue]);
222                 }
223                 if (!allowSubsetSplits) {
224                     // just generate equality tests for nominal attributes
225                     break;
226                 }
227             }
228         }
229         // ClusLogger.info("pos_freq_orig = " + pos_freq);
230         // ClusLogger.info("attr: " + type + " best heur: " + bheur);
231         if (bheur > node.m_BestHeur + ClusHeuristic.DELTA) {
232             node.m_UnknownFreq = unk_freq;
233             node.m_BestHeur = bheur;
234             node.m_TestType = CurrentBestTestAndHeuristic.TYPE_TEST;
235             node.m_BestTest = new SubsetTest(type, card, isin, pos_freq);
236             node.resetAlternativeBest();
237             // ClusLogger.info("attr: " + type + " best test: " + node.m_BestTest.getString());
238         }
239         else if (getStatManager().getSettings().getTree().showAlternativeSplits() && (((bheur > node.m_BestHeur - ClusHeuristic.DELTA) && (bheur < node.m_BestHeur + ClusHeuristic.DELTA)) || (bheur == Double.POSITIVE_INFINITY))) {
240             // if same heuristic: add to alternatives (list will later be pruned to remove those tests that do
241             // not yield exactly the same subsets)
242             node.addAlternativeBest(new SubsetTest(type, card, isin, pos_freq));
243         }
244     }

247     @Override
248     public void findRandomSplit(CurrentBestTestAndHeuristic node, NominalAttrType type, Random rn) throws ClusException {
249         double unk_freq = 0.0;
250         int nbvalues = type.getNbValues();
251         boolean isin[] = new boolean[nbvalues];
252         // If has missing values?
253         if (type.hasMissing()) {
254             ClusStatistic unknown = node.m_TestStat[nbvalues];
255             m_MStat.copy(node.m_TotStat);
256             m_MStat.subtractFromThis(unknown);
257             unk_freq = unknown.getTotalWeight() / node.m_TotStat.getTotalWeight();
258         }
259         else {
260             m_MStat.copy(node.m_TotStat);
261         }
262         int card = 0;
263         double pos_freq = 0.0;
264         // Generate non-empty and non-full subset
265         while (true) {
266             for (int i = 0; i < isin.length; i++) {
267                 isin[i] = rn.nextBoolean();
268             }
269             int sum = 0;
270             for (int i = 0; i < isin.length; i++) {
271                 if (isin[i]) {
272                     sum++;
273                 }
274             }
275             if (!((sum == 0) || (sum == nbvalues))) {
276                 card = sum;
277                 break;
278             }
279         }
280         // Calculate statistics ...
281         m_PStat.reset();
282         for (int j = 0; j < nbvalues; j++) {
283             if (isin[j]) {
284                 m_PStat.add(node.m_TestStat[j]);
285             }
286         }
287         pos_freq = m_PStat.getTotalWeight() / m_MStat.getTotalWeight();
288         node.m_UnknownFreq = unk_freq;
289         node.m_BestHeur = node.calcHeuristic(m_MStat, m_PStat);
290         node.m_TestType = CurrentBestTestAndHeuristic.TYPE_TEST;
291         node.m_BestTest = new SubsetTest(type, card, isin, pos_freq);
292     }

295     @Override
296     public void findExtraTreeSplit(CurrentBestTestAndHeuristic node, NominalAttrType type, Random rnd) throws ClusException {
297         // ClusLogger.info("find split for attr: " + type);
298         double minAllowedFrequency = 0.01;
299         double unk_freq = 0.0;
300         int nbvalues = type.getNbValues();
301         boolean isin[] = new boolean[nbvalues];
302         // If has missing values?
303         if (type.hasMissing()) {
304             ClusStatistic unknown = node.m_TestStat[nbvalues];
305             m_MStat.copy(node.m_TotStat);
306             m_MStat.subtractFromThis(unknown);
307             unk_freq = unknown.getTotalWeight() / node.m_TotStat.getTotalWeight();
308         }
309         else {
310             m_MStat.copy(node.m_TotStat);
311         }
312         int card = 0;
313         double pos_freq = 0.0;
314         double bheur = Double.NEGATIVE_INFINITY;
315         boolean found_test = false;
316 
317         if (nbvalues == 2 && (!getStatManager().isRuleInduceOnly() || getStatManager().getSettings().getRules().isConstrainedToFirstAttVal())) {
318             // Handle binary splits efficiently
319             card = 1;
320             isin[0] = true;
321             m_PStat.reset();
322             for (int j = 0; j < nbvalues; j++) { // for loop is not really necessary, we simply follow the style from
323                                                  // the else case
324                 if (isin[j]) {// if selected
325                     m_PStat.add(node.m_TestStat[j]);
326                 }
327             }
328             // <--- ClusStatistic CStat = node.m_TestStat[0];
329             bheur = node.calcHeuristic(m_MStat, m_PStat); // <--- bheur = node.calcHeuristic(m_MStat, CStat);
330 //            showTest(type, isin, -1, bheur, m_MStat, m_PStat); // <--- showTest(type, isin, -1, bheur, m_MStat, CStat);
331             pos_freq = m_PStat.getTotalWeight() / m_MStat.getTotalWeight(); // <--- CStat.m_SumWeight /
332                                                                             // m_MStat.m_SumWeight;
333 
334             boolean acc_test = node.m_IsAcceptable && (m_MStat.getTotalWeight() >= 4.0) && (m_PStat.getTotalWeight() >= 2.0) && ((m_MStat.getTotalWeight() - m_PStat.getTotalWeight()) >= 2.0) && (pos_freq > minAllowedFrequency) && (pos_freq < 1.0 - minAllowedFrequency);
335             // <---
336             // boolean acc_test = node.m_IsAcceptable && (m_MStat.m_SumWeight >= 4.0) && (CStat.m_SumWeight >= 2.0) &&
337             // ((m_MStat.m_SumWeight - CStat.m_SumWeight) >= 2.0) && (pos_freq > minAllowedFrequency) && (pos_freq < 1.0
338             // - minAllowedFrequency);
339 
340             if (acc_test) {// select only meaningful splits
341                 found_test = true;
342             }
343 
344         }
345         else if ((getStatManager().getClusterMode() == ClusStatManager.Mode.PHYLO) && (getStatManager().getSettings().getPhylogeny().getPhylogenySequence().equals(PhylogenySequence.DNA))) {
346             System.err.println("Extra-Tree split selection not implemented for Phylogentic trees.");
347             throw new ClusException("Error while searching for a random split in: " + getClass().getName());
348         }
349         else {
350             int count = 0; // allow for nbTries opportunities to select a valid random test... otherwise assign worst
351                            // split score
352             int nbTries = 10;
353             // boolean select = true;
354             while (!found_test && count < nbTries) { // && select) {
355                 count++;
356                 // random selection of a subset of classes
357                 int sum = 0;
358                 for (int i = 0; i < isin.length; i++) {
359                     isin[i] = rnd.nextBoolean();
360                     if (isin[i])
361                         sum++;
362                 }
363 
364                 card = sum;
365 
366                 // if (!((sum == 0) || (sum == nbvalues))) {
367                 if (sum == 0 || sum == nbvalues) {
368                     // card = sum;
369                     // select = false;
370                 }
371                 else {
372                     m_PStat.reset();
373                     for (int j = 0; j < nbvalues; j++) {
374                         if (isin[j]) {// if selected
375                             m_PStat.add(node.m_TestStat[j]);
376                         }
377                     }
378                     pos_freq = m_PStat.getTotalWeight() / m_MStat.getTotalWeight();
379 
380                     node.checkAcceptable(m_MStat, m_PStat);
381                     boolean acc_test = node.m_IsAcceptable && (m_MStat.getTotalWeight() >= 4.0) && (m_PStat.getTotalWeight() >= 2.0) && ((m_MStat.getTotalWeight() - m_PStat.getTotalWeight()) >= 2.0) && (pos_freq > minAllowedFrequency) && (pos_freq < 1.0 - minAllowedFrequency);
382                     // node.getStat(i);
383 
384                     // acc_test = acc_test && node.getHeuristic().stopCriterion(node.getTotStat(), m_PStat, m_MStat);
385 
386                     // if (((node.m_TotStat.m_NbExamples - m_PStat.m_SumWeight) > 0) && (m_PStat.m_SumWeight >
387                     // 0)){//select
388                     // only meaningful splits
389                     // found_test = true;
390                     // }
391                     if (acc_test) {// select only meaningful splits
392                         found_test = true;
393                     }
394                 }
395             }
396 
397             if (count == nbTries || card == nbvalues || card == 0) {
398             	if (getStatManager().getSettings().getGeneral().getVerbose() >= 2) {
399             		System.err.println("Due to the randomness in split search, a usefull split was not found in " + count + " tries:");
400             		System.err.println("   Cardinality = " + card);
401                     System.err.println("   nb values = " + nbvalues);
402             	}                
403             }
404             if (found_test) {
405                 bheur = node.calcHeuristic(m_MStat, node.m_TestStat[0]);
406             }
407         }
408 
409         boolean valid_test = node.m_IsAcceptable && (m_MStat.getTotalWeight() >= 4.0) && (m_PStat.getTotalWeight() >= 2.0) && ((m_MStat.getTotalWeight() - m_PStat.getTotalWeight()) >= 2.0) && (pos_freq > minAllowedFrequency) && (pos_freq < 1.0 - minAllowedFrequency);
410 
411 //        showTest(type, isin, -1, bheur, m_MStat, m_CStat);  // Why is this here? Maybe showTest(type, isin, -1, bheur, m_MStat, m_MStat); lol
412         if (found_test && valid_test) {
413             node.m_UnknownFreq = unk_freq;
414             node.m_BestHeur = bheur;
415             node.m_TestType = CurrentBestTestAndHeuristic.TYPE_TEST;
416             node.m_BestTest = new SubsetTest(type, card, isin, pos_freq);
417             node.resetAlternativeBest();
418         }
419         else {
420             node.m_BestHeur = Double.NEGATIVE_INFINITY;// this should ensure that this test is not considered at all
421         }
422         node.checkAcceptable(m_MStat, m_PStat);
423 
424     }

427     // makes a random subsetsplit for a given attribute, where the subset includes one value and excludes another value
428     // used in PERT trees
429     public void findRandomPertSplit(CurrentBestTestAndHeuristic node, NominalAttrType type, Random rn, int valueincl, int valueexcl) {
430         int nbvalues = type.getNbValues();
431         boolean isin[] = new boolean[nbvalues];
432 
433         // Generate non-empty and non-full subset
434         int card = 0;
435         double pos_freq = 0.0;
436         while (true) {
437             for (int i = 0; i < isin.length; i++) {
438                 isin[i] = rn.nextBoolean();
439             }
440             isin[valueincl] = true;
441             isin[valueexcl] = false;
442             int sum = 0;
443             for (int i = 0; i < isin.length; i++) {
444                 if (isin[i]) {
445                     sum++;
446                 }
447             }
448             if (!((sum == 0) || (sum == nbvalues))) {
449                 card = sum;
450                 break;
451             }
452         }
453         // Calculate statistics ...
454         m_PStat.reset();
455         for (int j = 0; j < nbvalues; j++) {
456             if (isin[j]) {
457                 m_PStat.add(node.m_TestStat[j]);
458             }
459         }
460         node.m_TestType = CurrentBestTestAndHeuristic.TYPE_TEST;
461         node.m_BestTest = new SubsetTest(type, card, isin, pos_freq);
462     }
