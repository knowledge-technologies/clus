27      public void processAttribute(ClassesAttrType attr, RowData data){
28      	String attrName = attr.getName();
29      	m_Hierarchies.put(attrName, attr.getHier());
30      	boolean[][] presentAndFinal = getPresentTerms(attr.getHier(), data);
31          double upperDistanceBound = upperBound(attr, presentAndFinal);
32          m_HierarUpperBounds.put(attrName, upperDistanceBound);
33      }

35      /**
36       * Calculates the weighted Euclidean distance between the two classes in the hierarchy, where the weight of
37       * a label is {@code m_HierarWeight^depth}. The distance is at the end normalised by the upper bound
38       * for this hierarchy.
39       * 
40       * @param t1
41       *        First tuple
42       * @param t2
43       *        Second tuple
44       * @param attr
45       *        The dimension in which the distance between {@code t1} and {@code t2} is computed.
46       * @return Normalised weighted Euclidean distance.
47       */
48      public double calculateDist(DataTuple t1, DataTuple t2, ClassesAttrType attr) {
49          String name = attr.getName();
50          ClassHierarchy hier = m_Hierarchies.get(name);
51          int sidx = hier.getType().getArrayIndex();
52          ClassesTuple tp1 = (ClassesTuple) t1.getObjVal(sidx);
53          ClassesTuple tp2 = (ClassesTuple) t2.getObjVal(sidx);
54          HashSet<Integer> symmetricDifferenceOfLabelSets = new HashSet<Integer>();
55  
56          // add labels for t1
57          for (int j = 0; j < tp1.getNbClasses(); j++) {
58              int labelIndex = tp1.getClass(j).getIndex();
59              symmetricDifferenceOfLabelSets.add(labelIndex);
60          }
61          // remove the intersection labels of t1 and t2, and add labels(t2) - labels(t1)
62          for (int j = 0; j < tp2.getNbClasses(); j++) {
63              int labelIndex = tp2.getClass(j).getIndex();
64              if (symmetricDifferenceOfLabelSets.contains(labelIndex)) {
65                  symmetricDifferenceOfLabelSets.remove(labelIndex);
66              }
67              else {
68                  symmetricDifferenceOfLabelSets.add(labelIndex);
69              }
70          }
71          double dist = weightedEuclideanInt(hier, symmetricDifferenceOfLabelSets);
72          return dist / m_HierarUpperBounds.get(name);
73      }

75      /**
76       * Computes the depth of each term in the hierarchy, which can be DAG or tree-shaped.
77       */
78      private static void computeDepthsOfTerms(ClassHierarchy hier) {
79          ArrayList<ClassTerm> toProcess = new ArrayList<ClassTerm>();
80          HashMap<ClassTerm, Integer> numberOfProcessedParents = new HashMap<ClassTerm, Integer>();
81          for (int i = 0; i < hier.getTotal(); i++) {
82              ClassTerm term = hier.getTermAt(i);
83              numberOfProcessedParents.put(term, term.getNbParents());
84              if (numberOfProcessedParents.get(term) == 0) {
85                  toProcess.add(term);
86              }
87          }
88          toProcess.add(hier.getRoot());
89          while (toProcess.size() > 0) {
90              ClassTerm term = toProcess.remove(toProcess.size() - 1);
91              // compute the depth
92              if (term.getNbParents() == 0) {
93                  term.setDepth(-1.0); // root
94              }
95              else {
96                  double avgDepth = 0.0;
97                  for (int i = 0; i < term.getNbParents(); i++) {
98                      avgDepth += term.getParent(i).getDepth();
99                  }
100                 avgDepth /= term.getNbParents();
101                 term.setDepth(avgDepth + 1.0);
102             }
103             // see the children
104             for (int i = 0; i < term.getNbChildren(); i++) {
105                 ClassTerm child = (ClassTerm) term.getChild(i);
106                 int nbParentsToProcess = numberOfProcessedParents.get(child);
107                 if (nbParentsToProcess == 1) {
108                     toProcess.add(child);
109                 }
110                 numberOfProcessedParents.put(child, nbParentsToProcess - 1);
111             }
112         }
113 
114     }

116     /**
117      * Finds the indices of the classes in the hierarchy that at least one example belongs to. 
118      * @param hier
119      * @param data
120      * @return answer[class index] = true iff at least one example belongs to class.
121      */
122     private static boolean[][] getPresentTerms(ClassHierarchy hier, RowData data){
123     	int hierarchySize = hier.getTotal();
124     	boolean[] present = new boolean[hierarchySize];
125     	boolean[] finalClasses = new boolean[hierarchySize];
126     	int nbFinal = 0;
127     	int sidx = hier.getType().getArrayIndex();
128     	boolean isTree = hier.isTree();
129     	// we can break the loop if all classes are present, since in that case,
130     	// the upper bound for d(x1, x2) is achieved in two leaves unless hierarchy is a line
131     	// root ---> a1 ---> a2 ---> ... ---> an ... Better safe than sorry ... We will not break the loop.
132     	for(int i = 0; i < data.getNbRows(); i++){ //  && nbPresent < hierarchySize
133     		ClassesTuple tp = (ClassesTuple) data.getTuple(i).getObjVal(sidx);
134     		int nbClasses = tp.getNbClasses();
135     		if (nbClasses == 0) {
136     			continue; // no classes
137     		}
138     		int lastClassIndex = -123;
139     		if(isTree){
140     			lastClassIndex = tp.getClass(nbClasses - 1).getIndex();
141     		} else{
142     			lastClassIndex = mostSpecificTermIndex(tp);
143     		}
144     		if(!finalClasses[lastClassIndex]){
145     			finalClasses[lastClassIndex] = true;
146     			nbFinal++;
147     		}
148     		if(present[lastClassIndex]){ // if DAG, the leaf may not be the last in the list
149     			continue;  // we have already processed an example that covers all classes of the current
150     		}
151     		for(int j = 0; j < nbClasses; j++){
152     			int ind = tp.getClass(j).getIndex();
153     			if (! present[ind]){
154     				present[ind] = true;
155     				
156     			}
157     		}
158     	}
159     	return new boolean[][]{present, finalClasses, new boolean[]{nbFinal > 1}};
160     }

162     /**
163      * If the hierarchy {@code hier} is tree-shaped, this efficiently (in linear time) computes the maximal distance {@code d(x, y)} over the vectors
164      * {@code x} and {@code y} that correspond to the labels in the hierarchy and are present in the data.<br>
165      * If the hierarchy is only DAG, then the time complexity is O(squared number of different labels in data).
166      * 
167      * @param attr
168      *        Hierarchical attribute under consideration
169      * @param presentAndFinalClasses An array of 3 arrays:
170      * <ul>
171      * <li>[..., is the i-th label present in the data, ...] </li>
172      * <li>[..., is the i-th label the most specific for some example in the data, ...] </li>
173      * <li>[is the number of different most specific labels in the data at least two]</li>
174      * </ul>
175      * @return {@code max d(x, y)} over the label sets x and y in the dataset.
176      */
177     private double upperBound(ClassesAttrType attr, boolean[][] presentAndFinalClasses) {
178     	boolean atLeastTwoClasses = presentAndFinalClasses[2][0];
179     	if(atLeastTwoClasses){
180 	        ClassHierarchy hier = m_Hierarchies.get(attr.getName());
181 	        if(hier.isTree()){
182 	        	return upperBoundTree(hier, presentAndFinalClasses[0], presentAndFinalClasses[1]);
183 	        } else {
184 	        	return upperBoundDAG(hier, presentAndFinalClasses[0], presentAndFinalClasses[1]);
185 	        } 
186     	} else{
187         	return 1.0; // does not matter what is the normalisation constant, the distance will be always 0
188         }
189 
190         
191     }

193     /**
194      * Finds the upper bound of the weighted Euclidean distance for a given data. Covers the following cases:<br>
195      * <ul>
196      * <li> any type of tree hierarchy (e.g. line) </li>
197      * <li> at least two terms of the hierarchy must be present in the data</li>
198      * </ul>
199      * @param hier The hierarchy
200      * @param presentClasses true at that components that correspond to terms that are present in the data
201      * @param finalClasses true at that components that correspond to the deepest term for some example in the data
202 
203      */
204     private double upperBoundTree(ClassHierarchy hier, boolean[] presentClasses, boolean[] finalClasses){
205     	computeDepthsOfTerms(hier);
206     	double bound = 0.0;
207         ClassTerm root = hier.getRoot();
208         
209         ArrayList<ClassTerm> extremeTerms = findDistantTerms(hier, root, presentClasses, finalClasses);  // must have length two!
210         bound = weightedEuclidean(hier, extremeTerms.get(0), extremeTerms.get(1));        
211         return bound;
212     }

214     /**
215      * We iterate over all pairs of different final classes and return the maximal distance achieved.
216      * @param hier
217      * @param presentClasses
218      * @param finalClasses
219 
220      */
221     private double upperBoundDAG(ClassHierarchy hier, boolean[] presentClasses, boolean[] finalClasses){
222     	ArrayList<Integer> finalClassesOnly = new ArrayList<Integer>();
223     	for(int termInd = 0; termInd < finalClasses.length; termInd++){
224     		if(finalClasses[termInd]){
225     			finalClassesOnly.add(termInd);
226     		}
227     	}
228     	double bound = 0.0;
229     	for(int i = 0; i < finalClassesOnly.size(); i++){
230     		ClassTerm term1 = hier.getTermAt((finalClassesOnly.get(i)));
231     		for(int j = i + 1; j < finalClassesOnly.size(); j++){
232     			ClassTerm term2 = hier.getTermAt((finalClassesOnly.get(j)));
233     			double dist = weightedEuclidean(hier, term1, term2);
234     			if(dist > bound + MathUtil.C1E_9){
235     				bound = dist;
236     			}
237     		}
238     	}	
239     	return bound;
240     }

242     /**
243      * Recursively finds the two most distant terms of the hierarchy that represent the final classes in the data, it two such terms exist. 
244      * @param root The current root of the hierarchy.
245      * @param presentClasses see upperBound
246      * @param finalClasses see upperBound
247 
248      */
249     private ArrayList<ClassTerm> findDistantTerms(ClassHierarchy hier, ClassTerm root, boolean[] presentClasses, boolean[] finalClasses){
250     	ArrayList<ClassTerm> extremeTerms = new ArrayList<ClassTerm>();
251     	int rootInd = root.getIndex();
252     	if(rootInd >= 0 && !presentClasses[rootInd]){
253     		// the part of the hierarchy rooted here, is not in the data --> empty list
254     		return extremeTerms;
255     	}
256     	ArrayList<ArrayList<ClassTerm>> childrenResults = new ArrayList<ArrayList<ClassTerm>>();
257     	int nbCh = root.getNbChildren();
258     	for(int i = 0; i < nbCh; i++){
259     		ArrayList<ClassTerm> resultCh = findDistantTerms(hier, (ClassTerm) root.getChild(i), presentClasses, finalClasses);
260     		if(resultCh.size() > 0){
261     			childrenResults.add(resultCh);
262     		}
263     	}
264     	
265     	boolean rootIsApproprite = rootInd >= 0 && finalClasses[rootInd];
266     	int nbChildrenCandidates = childrenResults.size();
267     	// find the indices of <= 2 children that have the deepest first term
268 		int notAnIndex = -1234;
269 		// double[] depths = new double[]{-10.0,-10.0};  // initial value does not matter
270 		int[] inds = new int[]{notAnIndex, notAnIndex}; // something that is not an index (-1 not good choice because of the root term)
271 		for(int i = 0; i < nbChildrenCandidates; i++){
272 			double thisDepth = childrenResults.get(i).get(0).getDepth();
273 			int whereTo = 0;			
274 			while(whereTo < inds.length){				
275 				if (inds[whereTo] == notAnIndex || thisDepth > childrenResults.get(inds[whereTo]).get(0).getDepth() + MathUtil.C1E_6){
276 					break;
277 				}
278 				whereTo++;
279 			}
280 			// move everything worse than the current one place to right: if whereTo = the last position or even worse, nothing happens
281 			for(int worseItemInd = inds.length - 2; worseItemInd >= whereTo ; worseItemInd--){
282 				inds[worseItemInd + 1] = inds[worseItemInd];
283 			}
284 			if (whereTo < inds.length){
285 				inds[whereTo] = i;
286 				}    		
287     	}
288     	
289     	if (nbChildrenCandidates == 0){
290     		// empty + possibly the root
291     		if(rootIsApproprite){
292     			extremeTerms.add(root);
293     		}
294     	} else if (nbChildrenCandidates == 1){
295     		extremeTerms.add(childrenResults.get(0).get(0)); // always in
296     		if(childrenResults.get(0).size() == 1 && rootIsApproprite){  // add root, if root is final
297     			extremeTerms.add(root); 
298     		} else if(childrenResults.get(0).size() == 2 && rootIsApproprite){ 	// add the other child term or root
299     			double distChildRootComb = weightedEuclidean(hier, childrenResults.get(0).get(0), root);
300     			double distChildComb = weightedEuclidean(hier, childrenResults.get(0).get(0), childrenResults.get(0).get(1));
301     			if(distChildRootComb > distChildComb){
302     				extremeTerms.add(root);
303     			} else{
304     				extremeTerms.add(childrenResults.get(0).get(1));
305     			}
306     		} else if(childrenResults.get(0).size() == 2){ // add the other child term
307     			extremeTerms.add(childrenResults.get(0).get(1));
308     		}
309     	} else{ // root cannot be an option, either two deepest 1st components or one of the children
310     		double maxDistDiffChildren = weightedEuclidean(hier, childrenResults.get(inds[0]).get(0),
311     															 childrenResults.get(inds[1]).get(0));
312     		double maxInnerDistChild = 0.0;
313     		int optimal = -1;
314     		for(int i = 0; i < nbChildrenCandidates; i++){
315     			if (childrenResults.get(i).size() > 1){
316 	    			double dist = weightedEuclidean(hier, childrenResults.get(i).get(0), childrenResults.get(i).get(1));
317 	    			if(dist > maxInnerDistChild + MathUtil.C1E_9){
318 	    				optimal = i;
319 	    				maxInnerDistChild = dist;
320 	    			}
321     			}
322     		}
323     		if(maxDistDiffChildren > maxInnerDistChild){
324     			extremeTerms.add(childrenResults.get(inds[0]).get(0));
325     			extremeTerms.add(childrenResults.get(inds[1]).get(0));
326     		} else{
327     			extremeTerms.addAll(childrenResults.get(optimal));
328     		}
329     	}
330     	return extremeTerms;
331     }

333     private double weightedEuclideanInt(ClassHierarchy hier, HashSet<Integer> symmetricDifferenceOfLabelSets){
334     	double dist = 0.0;
335         for (int labelIndex : symmetricDifferenceOfLabelSets) {
336             dist += hier.getWeight(labelIndex);
337         }
338         return dist;
339     }
340     
341     private double weightedEuclideanTerm(ClassHierarchy hier, HashSet<ClassTerm> symmetricDifferenceOfLabelSets){
342     	double dist = 0.0;
343         for (ClassTerm term : symmetricDifferenceOfLabelSets) {
344             dist += hier.getWeight(term.getIndex());
345         }
346         return dist;
347     }

349     private double weightedEuclidean(ClassHierarchy hier, ClassTerm finalTerm1, ClassTerm finalTerm2){
350     	return weightedEuclideanTerm(hier, MathUtil.symmetricDifference(finalTerm1.getAllAncestors(true), finalTerm2.getAllAncestors(true)));
351     }

353     /**
354      * Finds the most specific label among the ones given in ClassesTuple. Assumes that the argument contains at least one class.
355      * @param tp
356 
357      */
358     private static int mostSpecificTermIndex(ClassesTuple tp){
359     	int ind = 0;
360     	HashMap<Integer, Integer> termIndices = new HashMap<Integer, Integer>();
361     	int nbClasses = tp.getNbClasses();
362     	for(int i = 0; i < nbClasses; i++){
363     		termIndices.put(tp.getClass(i).getIndex(), i);
364     	}
365     	boolean continueSearch = true;
366     	while (continueSearch){
367     		continueSearch = false;
368     		ClassTerm term = tp.getClass(ind).getTerm();
369     		if (term.atBottomLevel()){
370     			return term.getIndex();
371     		}
372     		for(int i = 0; i < term.getNbChildren(); i++){
373     			int j = ((ClassTerm) term.getChild(i)).getIndex();
374     			if(termIndices.containsKey(j)){
375     				ind = termIndices.get(j);
376     				continueSearch = true;
377     				break;
378     			}
379     		}
380     	}
381     	return tp.getClass(ind).getIndex();
382     }

386      * The weight, used in Weighted Euclidean distance.
387      * 
388      * @param hier
389      * @return The the biggest among the weights of the labels that are smaller than 1.0.
390      */
391     private double getWeight(ClassHierarchy hier) {
392         double[] weights = hier.getWeights();
393         double maxW = 0.0;
394         for (double w : weights) {
395             if (w < 1.0 - MathUtil.C1E_6 && w > maxW) {
396                 maxW = w;
397             }
398         }
399         return maxW;
400     }
