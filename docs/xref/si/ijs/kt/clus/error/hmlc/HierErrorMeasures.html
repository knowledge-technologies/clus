53      public HierErrorMeasures(ClusErrorList par, ClassHierarchy hier, double[] recalls, HierarchyMeasures optimize, boolean wrCurves, boolean isGzipOutput) {
54          super(par, hier.getTotal());
55          m_Hier = hier;
56          m_OptimizeMeasure = optimize;
57          m_WriteCurves = wrCurves;
58          m_RecallValues = recalls;
59          m_EvalClass = hier.getEvalClassesVector();
60          // m_EvalClass = new boolean[hier.getTotal()];
61          // m_EvalClass[19] = true;
62          m_ClassWisePredictions = new BinaryPredictionList[hier.getTotal()];
63          m_ROCAndPRCurves = new ROCAndPRCurve[hier.getTotal()];
64          for (int i = 0; i < hier.getTotal(); i++) {
65              BinaryPredictionList predlist = new BinaryPredictionList();
66              m_ClassWisePredictions[i] = predlist;
67              m_ROCAndPRCurves[i] = new ROCAndPRCurve(predlist);
68          }
69  
70          m_IsGzipOutput = isGzipOutput;
71      }

74      @Override
75      public void addExample(DataTuple tuple, ClusStatistic pred) {
76          ClassesTuple tp = (ClassesTuple) tuple.getObjVal(m_Hier.getType().getArrayIndex());
77          double[] predarr = ((WHTDStatistic) pred).getNumericPred();
78          boolean[] actual = tp.getVectorBooleanNodeAndAncestors(m_Hier);
79          for (int i = 0; i < m_Dim; i++) {
80              m_ClassWisePredictions[i].addExample(actual[i], predarr[i]);
81          }
82      }

95      @Override
96      public boolean isComputeForModel(String name) {
97          if (name.equals("Original") || name.equals("Pruned") || name.equals("Default")) {
98              return true;
99          }
100         else if (name.startsWith("Original ") && name.contains("-nn model with ") || name.equals(KnnClassifier.DEFAULT_MODEL_NAME_WITH_CONSTANT_WEIGHTS)) {
101             // this is kNN hackish solution
102             return true;
103         }
104         else if (name.startsWith("Forest with ") && !name.contains("T = ")) { // added because of the option Iterations
105                                                                               // = [10, 20, 30, ...]
106             return true;
107         }
108         return false;
109     }

118     @Override
119     public double getModelError() {
120         computeAll();
121         switch (m_OptimizeMeasure) {
122             case AverageAUROC:
123                 return m_AverageAUROC;
124             case AverageAUPRC:
125                 return m_AverageAUPRC;
126             case WeightedAverageAUPRC:
127                 return m_WAvgAUPRC;
128             case PooledAUPRC:
129                 return m_PooledAUPRC;
130         }
131         return 0.0;
132     }

137         return m_EvalClass[idx] && includeZeroFreqClasses(idx); // && m_ClassWisePredictions[idx].hasBothPosAndNegEx();

170     // prints the evaluation results for each single predicted class
171     public void printResultsRec(ClusNumberFormat fr, PrintWriter out, ClassTerm node, boolean[] printed) {
172         int idx = node.getIndex();
173         // avoid printing a given node several times
174         if (printed[idx])
175             return;
176         printed[idx] = true;
177         if (isEvalClass(idx)) {
178             ClassesValue val = new ClassesValue(node);
179             out.print("      " + idx + ": " + val.toStringWithDepths(m_Hier));
180             out.print(", AUROC: " + fr.format(m_ROCAndPRCurves[idx].getAreaROC()));
181             out.print(", AUPRC: " + fr.format(m_ROCAndPRCurves[idx].getAreaPR()));
182             out.print(", Freq: " + fr.format(m_ClassWisePredictions[idx].getFrequency()));
183             if (m_RecallValues != null) {
184                 int nbRecalls = m_RecallValues.length;
185                 for (int i = 0; i < nbRecalls; i++) {
186                     int rec = (int) Math.floor(100.0 * m_RecallValues[i] + 0.5);
187                     out.print(", P" + rec + "R: " + fr.format(100.0 * m_ROCAndPRCurves[idx].getPrecisionAtRecall(i)));
188                 }
189             }
190             out.println();
191         }
192         for (int i = 0; i < node.getNbChildren(); i++) {
193             printResultsRec(fr, out, (ClassTerm) node.getChild(i), printed);
194         }
195     }

198     public void printResults(ClusNumberFormat fr, PrintWriter out, ClassHierarchy hier) {
199         ClassTerm node = hier.getRoot();

201         for (int i = 0; i < node.getNbChildren(); i++) {

203         }

229     public void computeAll() {
230         BinaryPredictionList pooled = new BinaryPredictionList();
231         ROCAndPRCurve pooledCurve = new ROCAndPRCurve(pooled);
232         compatibility(m_ROCAndPRCurves, pooledCurve);
233         for (int i = 0; i < m_Dim; i++) {
234             if (isEvalClass(i)) {
235                 m_ClassWisePredictions[i].sort();
236                 m_ROCAndPRCurves[i].computeCurves();
237                 m_ROCAndPRCurves[i].computePrecisions(m_RecallValues);
238                 outputPRCurve(i, m_ROCAndPRCurves[i]);
239                 outputROCCurve(i, m_ROCAndPRCurves[i]);
240                 m_ROCAndPRCurves[i].clear();
241                 pooled.add(m_ClassWisePredictions[i]);
242                 m_ClassWisePredictions[i].clearData();
243             }
244         }
245         pooled.sort();
246         pooledCurve.computeCurves();
247         outputPRCurve(-1, pooledCurve);
248         outputROCCurve(-1, pooledCurve);
249         pooledCurve.clear();
250         // Compute averages
251         int cnt = 0;
252         double sumAUROC = 0.0;
253         double sumAUPRC = 0.0;
254         double sumAUPRCw = 0.0;
255         double sumFrequency = 0.0;
256         for (int i = 0; i < m_Dim; i++) {
257             // In compatibility mode, averages never include classes with zero frequency in test set
258             if (isEvalClass(i)) {
259                 double freq = m_ClassWisePredictions[i].getFrequency();
260                 sumAUROC += m_ROCAndPRCurves[i].getAreaROC();
261                 sumAUPRC += m_ROCAndPRCurves[i].getAreaPR();
262                 sumAUPRCw += freq * m_ROCAndPRCurves[i].getAreaPR();
263                 sumFrequency += freq;
264                 cnt++;
265             }
266         }
267         m_AverageAUROC = sumAUROC / cnt;
268         m_AverageAUPRC = sumAUPRC / cnt;
269         m_WAvgAUPRC = sumAUPRCw / sumFrequency;
270         m_PooledAUPRC = pooledCurve.getAreaPR();
271         // Compute average precisions at recall values
272         if (m_RecallValues != null) {
273             int nbRecalls = m_RecallValues.length;
274             m_AvgPrecisionAtRecall = new double[nbRecalls];
275             for (int j = 0; j < nbRecalls; j++) {
276                 int nbClass = 0;
277                 for (int i = 0; i < m_Dim; i++) {
278                     if (isEvalClass(i)) {
279                         double prec = m_ROCAndPRCurves[i].getPrecisionAtRecall(j);
280                         m_AvgPrecisionAtRecall[j] += prec;
281                         nbClass++;
282                     }
283                 }
284                 m_AvgPrecisionAtRecall[j] /= nbClass;
285             }
286         }
287     }

290     public void ouputCurve(int ci, ArrayList points, PrintWriter curves) {
291         String clName = "ALL";
292         if (ci != -1) {
293             ClassTerm cl = m_Hier.getTermAt(ci);
294             clName = "\"" + cl.toStringHuman(m_Hier) + "\"";
295         }
296         for (int i = 0; i < points.size(); i++) {
297             double[] pt = (double[]) points.get(i);
298             curves.println(clName + "," + pt[0] + "," + pt[1]);
299         }
300     }

344     @Override
345     public void showModelError(PrintWriter out, String bName, int detail) throws IOException {
346         if (m_WriteCurves && bName != null) {
347             writeCSVFilesPR(bName + ".pr.csv");
348             writeCSVFilesROC(bName + ".roc.csv");
349         }
350         ClusNumberFormat fr1 = ClusFormat.SIX_AFTER_DOT;
351         computeAll();
352         out.println();
353         out.println("      Average AUROC:            " + m_AverageAUROC);
354         out.println("      Average AUPRC:            " + m_AverageAUPRC);
355         out.println("      Average AUPRC (weighted): " + m_WAvgAUPRC);
356         out.println("      Pooled AUPRC:             " + m_PooledAUPRC);
357         if (m_RecallValues != null) {
358             int nbRecalls = m_RecallValues.length;
359             for (int i = 0; i < nbRecalls; i++) {
360                 int rec = (int) Math.floor(100.0 * m_RecallValues[i] + 0.5);
361                 out.println("      P" + rec + "R: " + (100.0 * m_AvgPrecisionAtRecall[i]));
362             }
363         }
364         if (detail != ClusError.DETAIL_SMALL) {
365             printResults(fr1, out, m_Hier);
366         }
367         if (m_PRCurves != null) {
368             m_PRCurves.close();
369             m_PRCurves = null;
370         }
371         if (m_ROCCurves != null) {
372             m_ROCCurves.close();
373             m_ROCCurves = null;
374         }
375     }
