13  public class FastClonerTreeMap implements IFastCloner {
14  
15      @Override
16      @SuppressWarnings({ "unchecked", "rawtypes" })
17      public Object clone(final Object t, final IDeepCloner cloner, final Map<Object, Object> clones) {
18          final TreeMap<Object, Object> m = (TreeMap) t;
19          final TreeMap result = new TreeMap(m.comparator());
20          for (final Map.Entry e : m.entrySet()) {
21              final Object key = cloner.deepClone(e.getKey(), clones);
22              final Object value = cloner.deepClone(e.getValue(), clones);
23              result.put(key, value);
24          }
25          return result;
26      }
27  }
