31      /**
32       * unlabeled criteria is the criteria by which the unlabeled data will be added to the training set (used by the
33       * Self Training algorithm)
34       */
35      public enum SSLUnlabeledCriteria {
36          /**
37           * Threshold - all of the unlabeled instances with confidence of prediction greater than Threshold will be added
38           * to the training set
39           */
40          Threshold,
41  
42          /**
43           * KMostConfident - K unlabeled instances with the most confident predictions will be added to the training set,
44           * K is a parameter
45           */
46          KMostConfident,
47  
48          /**
49           * KPercentageMostConfident - K percentage of unlabeled instances with the most confident predictions will be
50           * added to the training set
51           */
52          KPercentageMostConfident,
53  
54          /** AutomaticOOB - the optimal threshold will be selected on the basis of OOB error of the labeled examples */
55          AutomaticOOB,
56  
57          /** ExhaustiveSearch - the optimal threshold will be selected from the specified list at each iteration */
58          ExhaustiveSearch,
59  
60          /**
61           * KPercentageMostAverage - the threshold is set to average confidence score of the K percentage most confident
62           * examples (set initially, and then this threshold is used throughout iterations)
63           */
64          KPercentageMostAverage,
65  
66          /**
67           * AutomaticOOBInitial - the threshold will be selected according to AutomaticOOB at initial iteration, this
68           * threshold is in next iterations
69           */
70          AutomaticOOBInitial
71      }
72

75      /** Stopping criteria for self training */
76      public enum SSLStoppingCriteria {
77          /** If no instance met the UnlabeledCriteria, i.e., no instance was added to the training set */
78          NoneAdded,
79  
80          /** Iterations - stop after specified number of iterations */
81          Iterations,
82  
83          /**
84           * (Leistner et al., 2009), based on Out of Bag Error estimate of Random Forest (applicable only if the base
85           * method for the Self Training is Random Forest)
86           */
87          Airbag
88      }

92      /** Confidence (i.e., reliability) score for Self-Training */
93      public enum SSLConfidenceMeasure {
94          /** Variance - standard deviation of votes of ensemble */
95          Variance,
96  
97          /** RandomUniform - uniformly distribuited random scores */
98          RandomUniform,
99  
100         /** RandomGaussian - normally distribuited random scores */
101         RandomGaussian,
102 
103         /**
104          * Oracle - real score, just for testing purposes, actual error on unlabeled examples is calculated to establish
105          * reliability scores
106          */
107         Oracle,
108 
109         /**
110          * RForestProximities - extrapolation of error of unlabeled examples via OOB error of labeled examples in their
111          * Random Forest proximity
112          */
113         RForestProximities,
114 
115         /** RForestProximitiesV2 - ?? */
116         RForestProximitiesV2,
117 
118         /**
119          * "Precision" - used for HMC, on the basis of OOB examples, classification threshold is found such that
120          * precision
121          * of precision is larger that the specified threshold
122          */
123         Precision,
124 
125         /** Reliability score for MLC and HMLC, based on empirical probabilities of classes */
126         ClassesProbabilities
127     }

252     public int[] getInternalFoldIndices() {
253     	int[] answer;
254     	int internalFold = m_SSL_InternalFold.getValue();
255         if (internalFold != DEFAULT_INTERNAL_FOLD){
256             if (internalFold < 0 || internalFold >= getSSLInternalFolds()) {
257                 throw new RuntimeException("We must have 0 <= internal fold < internal fold indices");
258             }
259         }
260     	if (internalFold == DEFAULT_INTERNAL_FOLD) {
261     		answer = new int[getSSLInternalFolds()];
262     		for (int i = 0; i < answer.length; i++) {
263     			answer[i] = i;
264     		}
265     	} else {
266     		answer = new int[] {internalFold};
267     	}
268     	return answer;
269     }
