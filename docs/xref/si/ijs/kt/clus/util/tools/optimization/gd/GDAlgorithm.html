112 	 * Initialize GD optimization for new run with same predictions and true values
113 	 * This can be used if some parameters change. Thus we e.g. do not compute
114 	 * covariances again
115 	 */
116 	public void initGDForNewRunWithSamePredictions() {
117 		m_GDProbl.initGDForNewRunWithSamePredictions();
118 
119 		m_weights = m_GDProbl.getInitialWeightVector();
120 
121 		// Oscillation detection
122 		m_prevChange = null;
123 		m_iPrevDimension = null;
124 		m_iNewDimension = null;
125 		m_newChange = null;
126 		m_minStepSizeReduction = 1;
127 
128 		m_earlyStopStepsizeReducedNb = 0;
129 
130 		// If weigths can be banned, oscillation is stored
131 		if (m_GDProbl.m_bannedWeights != null) {
132 			m_iOscillatingWeights = new ArrayList<Integer>();
133 		} else {
134 			m_iOscillatingWeights = null;
135 		}
136 	}

138 	/**
139 	 * Optimize the weights for the given data with gradient descent algorithm. This
140 	 * is the algorithm by Friedman, Popescu 2004
141 	 */
142 	@Override
143 	public ArrayList<Double> optimize() {
144 		ClusLogger.info(
145 				"Gradient descent: Optimizing rule weights (" + getSettings().getRules().getOptGDMaxIter() + ") ");
146 
147 		PrintWriter wrt_log = null;
148 
149 		if (GDProblem.m_printGDDebugInformation) {
150 			try {
151 				wrt_log = new PrintWriter(new OutputStreamWriter(new FileOutputStream("gradDesc.log")));
152 			} catch (Exception e) {
153 				e.printStackTrace();
154 				ClusLogger.severe("Log file could not be opened. Logging omitted.");
155 			}
156 		}
157 
158 		if (m_GDProbl.isClassifTask()) {
159 			try {
160 				throw new ClusException(
161 						"Classification not yeat implemented for gradient descent. Skipping the optimization.");
162 			} catch (Exception s) {
163 				s.printStackTrace();
164 			}
165 			return null;
166 		}
167 		// Compute initial gradients
168 		m_GDProbl.fullGradientComputation(m_weights);
169 
170 		int nbOfIterations = 0;
171 		while (nbOfIterations < getSettings().getRules().getOptGDMaxIter()) {
172 
173 			if (nbOfIterations % (Math.ceil(getSettings().getRules().getOptGDMaxIter() / 50.0)) == 0) {
174 				ClusLogger.finer(String.format("%s %%",
175 						nbOfIterations * 100 / Math.ceil(getSettings().getRules().getOptGDMaxIter())));
176 			}
177 			if (nbOfIterations % m_earlyStopStep == 0) {
178 
179 				if (getSettings().getRules().getOptGDEarlyStopAmount() > 0 && m_GDProbl.isEarlyStop(m_weights)) {
180 					if (GDProblem.m_printGDDebugInformation) {
181 						wrt_log.println("Increase in test fitness. Reducing step size or stopping.");
182 					}
183 
184 					ClusLogger.fine("Overfitting after " + nbOfIterations + " iterations.");
185 
186 					if (!getSettings().getRules().isOptGDIsDynStepsize()
187 							&& m_earlyStopStepsizeReducedNb < getSettings().getRules().getOptGDNbOfStepSizeReduce()) {
188 						m_earlyStopStepsizeReducedNb++;
189 						m_GDProbl.dropStepSize(0.1); // Drop stepsize to tenth.
190 						m_GDProbl.restoreBestWeight(m_weights); // restoring the weight with minimum fitness
191 						m_GDProbl.fullGradientComputation(m_weights);
192 
193 						ClusLogger.fine("Reducing step, continuing.");
194 					} else { // If dynamic step size, stop always
195 						ClusLogger.info("Stopping.");
196 
197 						if (GDProblem.m_printGDDebugInformation) {
198 							wrt_log.println("Early stopping detected after " + nbOfIterations + " iterations.");
199 						}
200 						break;
201 					}
202 				}
203 			}
204 
205 			// Print
206 			OutputLog(nbOfIterations, wrt_log); // Weights
207 
208 			// 3)
209 			// Search for maximum gradients
210 			int[] maxGradients = m_GDProbl.getMaxGradients(nbOfIterations);
211 
212 			boolean oscillation = false;
213 
214 			// For detecting oscillation
215 			storeGradientsForOscillation(maxGradients);
216 
217 			// b) // Take the gradient step on the peak dimension
218 
219 			// First compute the change values and detect possible oscillation
220 
221 			// Change of the weight according to this gradient.
222 			double[] valueChange = new double[maxGradients.length];
223 
224 			boolean debugPrint = getSettings().getGeneral().getVerbose() > 0 && false; // DEBUG
225 
226 			if (debugPrint) {
227 				System.out.println("\nDEBUG: Computing covariances, total " + maxGradients.length);
228 			}
229 
230 			for (int iiGradient = 0; iiGradient < maxGradients.length; iiGradient++) {
231 				if (debugPrint) {
232 					System.out.print(iiGradient % 10);
233 				}
234 				// c) We are changing the value of the weight. Let's compute the covariance if
235 				// it is not yet done
236 				m_GDProbl.computeCovariancesIfNeeded(maxGradients[iiGradient]);
237 
238 				// Change of the weight according to this gradient.
239 				valueChange[iiGradient] = m_GDProbl.howMuchWeightChanges(maxGradients[iiGradient]);
240 
241 				// For detecting oscillation. If oscillation is already true, do not change it
242 				// to false.
243 				if (nbOfIterations < 100) {
244 					oscillation = (detectOscillation(iiGradient, valueChange[iiGradient]) || oscillation);
245 				}
246 			}
247 			if (debugPrint) {
248 				System.out.println("\nDEBUG: Computing covariances ended");
249 			}
250 			// If oscillation was detected we do not take real steps.
251 			// we reduce the step size until the new step is smaller than the first one.
252 			// It should be smaller because otherwise we are going even further from the
253 			// optimal point.
254 			if (oscillation && !getSettings().getRules().isOptGDIsDynStepsize()) {
255 				if (GDProblem.m_printGDDebugInformation) {
256 					wrt_log.println("Detected oscillation, reducing step size of: " + m_GDProbl.m_stepSize);
257 				}
258 
259 				if (debugPrint) {
260 					ClusLogger.info("DEBUG: Detected oscillation on iteration " + nbOfIterations
261 							+ ", reducing step size of: " + m_GDProbl.m_stepSize);
262 				}
263 				// System.exit(1); //DEBUG
264 				// For MaxLoss combination the oscillation may be because we are finding some
265 				// local
266 				// optimum point for some weight and it is the most significant. Let us put this
267 				// weight
268 				// to a banned list for some iterations
269 				if (nbOfIterations > 10 && (getSettings().getRules().getOptGDMTGradientCombine()
270 						.equals(OptimizationGDMTCombineGradient.MaxLoss)
271 						|| getSettings().getRules().getOptGDMTGradientCombine()
272 								.equals(OptimizationGDMTCombineGradient.MaxLossFast))) {
273 					putOscillatingWeightsToBan(nbOfIterations);
274 				} else {
275 					// This is needed if we are changing more than one dimension at time.
276 					// The step size is made for the biggest oscillation and thus some of the other
277 					// oscillating ones
278 					// never come back to sensible values.
279 					reversePreviousStep(); // Reverse the previous step. We also skip this step.
280 					reduceStepSizeDueOscillation();
281 				}
282 				continue; // Start next iteration
283 			} else {
284 				// Store the current data for use in the next iterations.
285 				if (nbOfIterations < 100) {
286 					storeTheOscillationData();
287 				}
288 			}
289 
290 			// After it is sure no oscillation is detected, make the changes
291 			for (int iiGradient = 0; iiGradient < maxGradients.length; iiGradient++) {
292 				m_weights.set(maxGradients[iiGradient],
293 						m_weights.get(maxGradients[iiGradient]).doubleValue() + valueChange[iiGradient]);
294 			}
295 
296 			// d) compute the gradients again for the weights that changed
297 			// This must not affect the computation during this iteration
298 			m_GDProbl.modifyGradients(maxGradients, m_weights);
299 
300 			nbOfIterations++;
301 		} // While
302 
303 		ClusLogger.info("Done!");
304 
305 		if (getSettings().getRules().getOptGDEarlyStopAmount() > 0) {
306 			m_GDProbl.isEarlyStop(m_weights); // Check if current weights have better fitness than the best so far
307 			m_GDProbl.restoreBestWeight(m_weights); // Result are the weights with best fitness
308 		}
309 
310 		if (GDProblem.m_printGDDebugInformation) {
311 			wrt_log.println("The result of optimization");
312 		}
313 		OutputLog(nbOfIterations, wrt_log);
314 		if (GDProblem.m_printGDDebugInformation) {
315 			wrt_log.close();
316 		}
317 		// System.err.println("CHANGING THE WEIGHTS< REMOVE THISE");
318 		// for (int i=0; i< m_weights.size(); i++)
319 		// m_weights.set(i,1.0); // default rule
320 
321 		return m_weights;
322 	}

324 	/** If weight banning is in use, store the oscillation dimensions here */
325 	private ArrayList<Integer> m_iOscillatingWeights;

327 	/**
328 	 * Puts oscillating weights to ban, so that we can continue on the rest of
329 	 * weights. This is used only if for maxLoss combining something happens.
330 	 */
331 	private void putOscillatingWeightsToBan(int iterationNb) {
332 		for (int iWeight = 0; iWeight < m_iOscillatingWeights.size(); iWeight++) {
333 			m_GDProbl.m_bannedWeights[m_iOscillatingWeights.get(iWeight)] = iterationNb + 50;
334 		}
335 		m_iOscillatingWeights.clear();
336 	}

338 	/** Maximum step size reduction found on this iteration */
339 	private double m_minStepSizeReduction;
340 
341 	/** Reduce the step size. Take the biggest needed reduction and do that */
342 	private void reduceStepSizeDueOscillation() {
343 		// System.err.print("Dropping the step size.\n");
344 
345 		// We are lowering the step size just enough to prevent further oscillation
346 		// We make the new step size a little smaller than is limit (because of rounding
347 		// mistakes)
348 		m_GDProbl.dropStepSize(m_minStepSizeReduction * 0.99);
349 		m_minStepSizeReduction = 1; // Step done, reset this now
350 	}

366 	/**
367 	 * A iteration has been done and oscillation not detected - store the previous
368 	 * changes
369 	 */
370 	private void storeTheOscillationData() {
371 		if (getSettings().getRules().isOptGDIsDynStepsize())
372 			return;
373 		m_prevChange = m_newChange.clone();
374 		m_iPrevDimension = m_iNewDimension.clone();
375 	}
376 
377 	/**
378 	 * Reverse previous (not this) step if oscillation found. Thus the state should
379 	 * be like it was before. This has to be called before storeTheOscillationData.
380 	 */
381 	private void reversePreviousStep() {
382 		// First reverse the weights
383 		for (int iiGradient = 0; iiGradient < m_iPrevDimension.length; iiGradient++) {
384 			m_weights.set(m_iPrevDimension[iiGradient],
385 					m_weights.get(m_iPrevDimension[iiGradient]).doubleValue() - m_prevChange[iiGradient]);
386 		}
387 
388 		// Compute the gradients again. This should no be done so often that this costs
389 		// very much. Thus we
390 		// Can compute them again from scratch.
391 		m_GDProbl.fullGradientComputation(m_weights);
392 
393 		// Empty the oscillation control parameters (previous step was never done)
394 		m_prevChange = null;
395 		m_iPrevDimension = null;
396 
397 		// Also this has to be cleared.
398 		if (m_iOscillatingWeights != null)
399 			m_iOscillatingWeights.clear();
400 	}

402 	/**
403 	 * Stores the used gradients for oscillation detection later. Also
404 	 * initialization tasks for oscillation.
405 	 */
406 	private void storeGradientsForOscillation(int[] maxGradients) {
407 		if (getSettings().getRules().isOptGDIsDynStepsize())
408 			return;
409 		m_iNewDimension = maxGradients.clone();
410 		m_newChange = new double[maxGradients.length];
411 	}
412 
413 	/**
414 	 * Oscillation is detected if same dimension is changed such that the direction
415 	 * is different and the new step is bigger than previous.
416 	 * 
417 	 * @param iiNewChange The index of MaxGradients/m_iNewDimension we are going on.
418 	 * @param valueChange How much the weight is going to be changed
419 	 * @return If oscillation was detected.
420 	 */
421 	private boolean detectOscillation(int iiNewChange, double valueChange) {
422 		if (getSettings().getRules().isOptGDIsDynStepsize())
423 			return false;
424 		boolean detectOscillation = false;
425 
426 		// Store the value for the change, dimensions have already been stored
427 		m_newChange[iiNewChange] = valueChange;
428 
429 		// Go through the previous changes and check if some of them is for same
430 		// dimension
431 		for (int iiPrevChange = 0; m_prevChange != null && iiPrevChange < m_prevChange.length; iiPrevChange++) {
432 
433 			// Is the modified dimension same as before
434 			if (m_iPrevDimension[iiPrevChange] == m_iNewDimension[iiNewChange]) {
435 				// Does the change tell us about oscillation
436 
437 				if (valueChange * m_prevChange[iiPrevChange] < 0
438 						&& Math.abs(valueChange) > Math.abs(m_prevChange[iiPrevChange])) {
439 
440 					// System.err.println("DEBUG: Found out oscillation for weight " +
441 					// m_iPrevDimension[iiPrevChange] +
442 					// " that was nb " +iiPrevChange + " last iteration and "+ iiNewChange + "
443 					// now.");
444 					// OSCILLATION DETECTED
445 					if (m_iOscillatingWeights != null) {
446 						m_iOscillatingWeights.add(m_iPrevDimension[iiPrevChange]);
447 					}
448 
449 					// System.err.print("\nLikely oscillation detected in GDAlg.\n The dimension "
450 					// + m_iPrevDimension[iiPrevChange]
451 					// + " was changed consequently with change values: " +
452 					// m_prevChange[iiPrevChange]
453 					// + " and " + valueChange + ".\n");
454 
455 					// How much we need step size reduction
456 					double needReduction = Math.abs(m_prevChange[iiPrevChange]) / Math.abs(valueChange);
457 
458 					// If the reduction is greater than any of the previous on this iteration, do
459 					// it.
460 					if (needReduction < m_minStepSizeReduction)
461 						m_minStepSizeReduction = needReduction;
462 
463 					detectOscillation = true;
464 					// System.exit(1);
465 				}
466 
467 				break; // The dimension was found. do not check the latter ones.
468 			}
469 		}
470 
471 		return detectOscillation;
472 	}

474 	/** Print the current weights output file. */
475 	public void OutputLog(int iterNro, PrintWriter wrt) {
476 		if (!GDProblem.m_printGDDebugInformation)
477 			return;
478 
479 		ClusNumberFormat fr = ClusFormat.SIX_AFTER_DOT;
480 		double trainingFitness = m_GDProbl.calcFitness(m_weights, m_GDProbl.m_ruleSet);
481 		double testFitness = 0;
482 		if (getSettings().getRules().getOptGDEarlyStopAmount() > 0)
483 			testFitness = m_GDProbl.m_earlyStopProbl.calcFitness(m_weights, m_GDProbl.m_ruleSet);
484 
485 		wrt.print("Iteration " + iterNro + " ");
486 		if (getSettings().getRules().isOptGDIsDynStepsize())
487 			wrt.print("Step size: " + m_GDProbl.m_stepSize + " ");
488 		wrt.print("(" + fr.format(trainingFitness) + ", " + fr.format(testFitness) + "): ");
489 		// else
490 		// wrt.print("Iteration " + iterNro + ": ");
491 		for (int i = 0; i < m_weights.size(); i++) {
492 			wrt.print(fr.format((double) m_weights.get(i)) + "\t");
493 		}
494 		wrt.print("\n");
495 		// m_GDProbl.printGradientsToFile(nbOfIterations, wrt_log); //gradients //DEBUG
496 	}
