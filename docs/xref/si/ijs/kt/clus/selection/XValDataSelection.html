23  package si.ijs.kt.clus.selection;
24  
25  import java.io.FileInputStream;
26  import java.io.IOException;
27  import java.io.InputStreamReader;
28  import java.io.LineNumberReader;
29  
30  import si.ijs.kt.clus.data.type.primitive.IndexAttrType;
31  import si.ijs.kt.clus.util.exception.ClusException;
32  
33  
34  public class XValDataSelection extends XValMainSelection {
35  
36      protected IndexAttrType m_Attr;
37  
38  
39      public XValDataSelection(IndexAttrType type) {
40          super(type.getMaxValue(), type.getNbRows());
41          m_Attr = type;
42      }
43  
44  
45      @Override
46      public int getFold(int row) {
47          return m_Attr.getValue(row) - 1;
48      }
49  
50  
51      public static XValDataSelection readFoldsFile(String fname, int nbrows) throws IOException, ClusException {
52          IndexAttrType attr = new IndexAttrType("XVAL");
53          attr.setNbRows(nbrows);
54          for (int i = 0; i < nbrows; i++) {
55              attr.setValue(i, -1);
56          }
57          int fold = 0;
58  
59          try (LineNumberReader rdr = new LineNumberReader(new InputStreamReader(new FileInputStream(fname)))) {
60              String line = rdr.readLine();
61              while (line != null) {
62                  line = line.trim();
63                  if (!line.equals("")) {
64                      fold++;
65                      String[] tokens = line.split("[\\,\\s]+");
66                      for (int j = 0; j < tokens.length; j++) {
67                          try {
68                              int exid = Integer.parseInt(tokens[j]);
69                              if (attr.getValue(exid) != -1) {
70                                  throw new ClusException("Example id " + exid + " occurs twice in folds file: " + fname);
71                              }
72                              else {
73                                  attr.setValue(exid, fold);
74                              }
75                          }
76                          catch (NumberFormatException e) {
77                              throw new ClusException("Illegal number: " + tokens[j] + " in folds file: " + fname);
78                          }
79                      }
80                  }
81                  line = rdr.readLine();
82              }
83          }
84  
85          for (int i = 0; i < nbrows; i++) {
86              if (attr.getValue(i) == -1) { throw new ClusException("Folds file does not define fold for example " + (i + 1) + ": " + fname); }
87          }
88          return new XValDataSelection(attr);
89      }
90  }
