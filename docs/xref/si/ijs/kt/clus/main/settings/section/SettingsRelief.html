75      /**
76       * Returns a list, containing the numbers of the iterations.
77       * If this setting is given as a (list of) proportion(s) of the instances
78       * in the training set, we convert it into absolute values.
79       * <p>
80       * If the only value is 1, i.e., 1.0, we assume that this means
81       * all (100%) instances.
82       * 
83       * @param nbInstances
84       *        The number of instances in the training set.
85  
86       */
87      public int[] getReliefNbIterationsValue(int nbInstances) {
88          double[] values;
89          if (m_ReliefNbIterations.isVector()) {
90              values = m_ReliefNbIterations.getDoubleVector();
91          }
92          else {
93              values = new double[] { m_ReliefNbIterations.getDouble() };
94          }
95          boolean[] types = new boolean[] { true, true }; // {is double, is integer}
96          boolean containsOne = false;
97          for (double value : values) {
98              if (Math.abs(1.0 - value) >= MathUtil.C1E_9) { // value != 1
99                  int roundedValue = (int) Math.round(value);
100                 int wrong_type = Math.abs(roundedValue - value) >= MathUtil.C1E_9 ? 1 : 0;
101                 types[wrong_type] = false;
102             }
103             else {
104                 containsOne = true;
105             }
106         }
107         int[] iterations = new int[values.length];
108         if (values.length == 1 && containsOne) {
109             System.err.print(String.format("Warning! Ambiguous setting: %s = 1.0", m_ReliefNbIterations.getName()));
110             System.err.println(" Assuming 1.0 means 100%.");
111         }
112         if (!(types[0] || types[1])) {
113             throw new RuntimeException(String.format("The values for the setting %s must be of the same type!", m_ReliefNbIterations.getName()));
114         }
115         else if (types[0]) {
116             for (int i = 0; i < values.length; i++) {
117                 double value = values[i];
118                 if (0.0 <= value && value <= 1.0) {
119                     iterations[i] = (int) Math.round(value * nbInstances);
120                 }
121                 else {
122                     throw new RuntimeException("The value is not in the interval [0, 1]!");
123                 }
124             }
125         }
126         else {
127             for (int i = 0; i < values.length; i++) {
128                 iterations[i] = (int) Math.round(values[i]);
129             }
130         }
131         return iterations;
132     }

163     @Override
164     public void create() {
165         m_Section.addNode(m_ReliefNbNeighbours = new INIFileNominalOrIntOrVector("Neighbours", NONELIST));
166         m_ReliefNbNeighbours.setInt(RELIEF_NEIGHBOUR_DEFAULT);
167 
168         m_Section.addNode(m_ReliefNbIterations = new INIFileNominalOrDoubleOrVector("Iterations", NONELIST));
169 //        m_ReliefNbIterations.setNominal(RELIEF_ITERATIONS_DEFAULT);
170 		m_ReliefNbIterations.setDouble(RELIEF_ITERATIONS_DEFAULT);
171         m_Section.addNode(m_ReliefShouldHaveNeighbourWeighting = new INIFileBool("WeightNeighbours", false));
172         /* following Weka, the authors do not give any suggestions */
173         m_Section.addNode(m_ReliefWeightingSigma = new INIFileDouble("WeightingSigma", 0.5));
174         m_Section.addNode(m_ChosenInstances = new INIFileNominalOrIntOrVector("ChosenInstances", NONELIST));
175         m_ChosenInstances.setIntVector(DUMMY_INSTANCES);
176         m_Section.addNode(m_MultilabelDistance = new INIFileEnum<MultilabelDistance>("MultilabelDistance", MultilabelDistance.HammingLoss));
177         m_Section.addNode(mReliefStatistics = new INIFileEnum<ReliefStatisticsType>("ReliefStatisticsType", ReliefStatisticsType.DistanceClassic));
178         m_Section.addNode(m_ReliefIndicesInRankingNames = new INIFileBool("UseIndicesInRankingNames", false));
179         // sss: semi-supervsied scenario
180         m_Section.addNode(m_MissingTargetHandling = new INIFileEnum<SettingsRelief.MissingTargetHandling>("MissingTargetHandling", MissingTargetHandling.Impute));        
181         m_Section.setEnabled(false);
182     }
