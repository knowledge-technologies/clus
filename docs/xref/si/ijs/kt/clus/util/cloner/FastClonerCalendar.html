15  public class FastClonerCalendar implements IFastCloner {
16  
17      @Override
18      public Object clone(final Object t, final IDeepCloner cloner, final Map<Object, Object> clones) {
19          final GregorianCalendar gc = new GregorianCalendar();
20          Calendar c = (Calendar) t;
21          gc.setTimeInMillis(c.getTimeInMillis());
22          gc.setTimeZone((TimeZone) c.getTimeZone().clone());
23          return gc;
24      }
25  }
