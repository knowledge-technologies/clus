107     // generate matrix & write to file matejp: introduced permutation and everything that comes from this
108     public void generateMatrix(RowData data, Integer[] permutation) throws IOException {
109         // PrintWriter pw = new PrintWriter(new FileWriter("distances.txt"));
110         m_distances.clear();
111         ClusSchema schema = data.getSchema();
112         int N = data.getNbRows();
113         double maxdist = 0;
114         double d = 0;
115         // double W = 0;
116         double minDistLine = Double.POSITIVE_INFINITY;
117         double maxOnMinDistLine = 0;
118         double[][] w = new double[N][N];
119         // max distance in the file
120         for (int i = 0; i < N; i++) {
121             minDistLine = Double.POSITIVE_INFINITY;
122             for (int j = 0; j < N; j++) {
123                 int tupleI = permutation[i];
124                 int tupleJ = permutation[j];
125                 DataTuple exi = data.getTuple(tupleI); // i
126                 DataTuple exj = data.getTuple(tupleJ); // j
127                 ClusAttrType xt = schema.getNumericAttrUse(AttributeUseType.GIS)[0];
128                 ClusAttrType yt = schema.getNumericAttrUse(AttributeUseType.GIS)[1];
129                 double xi = xt.getNumeric(exi);
130                 double yi = yt.getNumeric(exi);
131                 double xj = xt.getNumeric(exj);
132                 double yj = yt.getNumeric(exj);
133                 Boolean longlat = schema.getSettings().getTree().isLonglat();
134                 if (longlat == true)
135                     d = LanLongdistance(xi, yi, xj, yj);
136                 else
137                     d = Math.sqrt((xi - xj) * (xi - xj) + (yi - yj) * (yi - yj));
138                 if (d > maxdist)
139                     maxdist = d;
140                 if (d != 0 && d < minDistLine)
141                     minDistLine = d;
142             }
143             if (minDistLine > maxOnMinDistLine)
144                 maxOnMinDistLine = minDistLine;
145         }
146         double bandwidth = schema.getSettings().getTree().getBandwidth();
147         double b = bandwidth * maxdist;
148         // ClusLogger.info(maxdist);
149         if (maxOnMinDistLine != Double.POSITIVE_INFINITY && b < maxOnMinDistLine)
150             b = maxOnMinDistLine;
151         // calculate the weights
152         for (int i = 0; i < N; i++) {
153             for (int j = 0; j < N; j++) {
154                 int tupleI = permutation[i]; // these two replace all
155                 int tupleJ = permutation[j]; // occurrences of i and j in this block
156 
157                 if (tupleI > tupleJ) {
158                     w[tupleI][tupleJ] = w[tupleJ][tupleI];
159                     // W += w[tupleI][tupleJ];
160                     continue;
161                 }
162                 DataTuple exi = data.getTuple(tupleI); // example i
163                 DataTuple exj = data.getTuple(tupleJ); // example j
164                 ClusAttrType xti = schema.getNumericAttrUse(AttributeUseType.GIS)[0];
165                 ClusAttrType yti = schema.getNumericAttrUse(AttributeUseType.GIS)[1];
166                 double xii = xti.getNumeric(exi);
167                 double yii = yti.getNumeric(exi);
168                 double xji = xti.getNumeric(exj);
169                 double yji = yti.getNumeric(exj);
170                 Boolean longlat = schema.getSettings().getTree().isLonglat();
171                 if (longlat == true)
172                     d = LanLongdistance(xii, yii, xji, yji);
173                 else
174                     d = Math.sqrt((xii - xji) * (xii - xji) + (yii - yji) * (yii - yji));
175                 /*
176                  * distance over descriptive and all atributes
177                  * NumericAttrType[] arr = schema.getNumericAttrUse(AttributeUseType.Descriptive);
178                  * int countA=arr.length; double dd=0;double d=0;
179                  * double xi[] = new double[countA];double xj[] = new double[countA];
180                  * for (int ii = 0; ii < countA; ii++) {
181                  * ClusAttrType xt = schema.getNumericAttrUse(AttributeUseType.Descriptive)[ii];
182                  * xi[ii] = xt.getNumeric(exi); xj[ii] = xt.getNumeric(exj);
183                  * dd += (xi[ii]-xj[ii])*(xi[ii]-xj[ii]); //euc over all atrb}
184                  */
185                 SpatialMatrixType spatialMatrix = schema.getSettings().getTree().getSpatialMatrix();
186                 if (d >= b)
187                     w[tupleI][tupleJ] = 0;
188                 else {
189                     if (d == 0)
190                         w[tupleI][tupleJ] = 1;
191                     else {
192                         switch (spatialMatrix) {
193                             case Binary:
194                                 w[tupleI][tupleJ] = 1; // 0; break; //binary
195                             case Euclidean:
196                                 w[tupleI][tupleJ] = 1 - d / b;
197                                 break; // euclidian
198                             case Modified:
199                                 w[tupleI][tupleJ] = (1 - (d * d) / (b * b)) * (1 - (d * d) / (b * b));
200                                 break; // modified
201                             case Gaussian:
202                                 w[tupleI][tupleJ] = Math.exp(-(d * d) / (b * b));
203                                 break; // gausian
204                             default:
205                                 w[tupleI][tupleJ] = 0;
206                                 break;
207                         }
208                         // W += w[tupleI][tupleJ];
209                         // //ClusLogger.info(i+"\t"+j+"\t"+exi+"\t"+exj+"\t"+(long)(i*N+j)+"\t"+w[i][j]);
210                     }
211                     m_distances.put((long) (tupleI * N + tupleJ), w[tupleI][tupleJ]); // write to hasp map only, not in
212                                                                                       // file
213                     // pw.println(i+"#"+j+" "+d);
214                     // pw.println(i+"\t"+W); //write to file
215                 }
216             }
217         }
218         // ClusLogger.info("spatialMatrix end: "+ System.currentTimeMillis());
219         // pw.close();
220     }

223     public void readMatrixFromFile(RowData data) throws IOException {
224         // PrintWriter pw = new PrintWriter(new FileWriter("distancesDistance.txt"));
225         m_distancesS.clear();
226         m_distancesN.clear();
227         ClusSchema schema = data.getSchema();
228         // int N = data.getNbRows();
229         double maxdist = 0;
230         double minDistLine = Double.POSITIVE_INFINITY;
231         double maxOnMinDistLine = 0;
232         double iEqual = Double.POSITIVE_INFINITY;
233         String filename = "distances.csv";
234         BufferedReader br = new BufferedReader(new FileReader(filename));
235         double bandwidth = schema.getSettings().getTree().getBandwidth();
236         double w;
237         if (bandwidth != 100.0) {
238             while (br.ready()) {
239                 String s = br.readLine();
240                 StringTokenizer st = new StringTokenizer(s, ",");
241                 Long ii = Long.parseLong(st.nextToken());
242                 Long jj = Long.parseLong(st.nextToken());
243                 double d = Double.parseDouble(st.nextToken());
244                 if (iEqual != ii) {
245                     m_distancesN.put((ii + "#" + ii), 0.0);
246                     iEqual = ii; // pw.println(ii+" "+ii+" "+0.0);
247                 }
248                 if (ii > jj) {
249                     m_distancesN.put((jj + "#" + ii), d); // pw.println(jj+" "+ii+" "+d);
250                 }
251                 else {
252                     m_distancesN.put((ii + "#" + jj), d); // pw.println(ii+" "+jj+" "+d);
253                 }
254             }
255 
256             for (double d : m_distancesN.values()) {
257                 if (d > maxdist)
258                     maxdist = d;
259                 if (d != 0 && d < minDistLine)
260                     minDistLine = d;
261             }
262             if (minDistLine > maxOnMinDistLine)
263                 maxOnMinDistLine = minDistLine;
264             double b = bandwidth * maxdist;
265             if (maxOnMinDistLine != Double.POSITIVE_INFINITY && b < maxOnMinDistLine)
266                 b = maxOnMinDistLine;
267             SpatialMatrixType spatialMatrix = schema.getSettings().getTree().getSpatialMatrix();
268             for (Map.Entry<String, Double> entry : m_distancesN.entrySet()) {
269                 String i = entry.getKey();
270                 double d = entry.getValue();
271                 if (d >= b)
272                     w = 0;
273                 else {
274                     if (d == 0)
275                         w = 1;
276                     else {
277                         switch (spatialMatrix) {
278                             case Binary:
279                                 w = 0;
280                                 break; // binary
281                             case Euclidean:
282                                 w = 1 - d / b;
283                                 break; // euclidian
284                             case Modified:
285                                 w = Math.pow(((1 - (d * d) / (b * b)) * (1 - (d * d) / (b * b))), 2);
286                                 break; // modified
287                             case Gaussian:
288                                 w = Math.exp(-(d * d) / (b * b));
289                                 break; // gausian
290                             default:
291                                 w = 0;
292                                 break;
293                         }
294                     }
295                     m_distancesS.put((i), w); // write to hasp map non-zero only, not in file
296                     // ClusLogger.info((i)+" "+w);
297                 }
298             }
299         }
300         else {
301             while (br.ready()) {
302                 String s = br.readLine();
303                 StringTokenizer st = new StringTokenizer(s, ",");
304                 // Long ii=Long.parseLong(st.nextToken());
305                 // Long jj=Long.parseLong(st.nextToken());
306                 String ii = st.nextToken().toLowerCase();
307                 String jj = st.nextToken().toLowerCase();
308                 double d = Double.parseDouble(st.nextToken());
309                 SpatialMatrixType spatialMatrix = schema.getSettings().getTree().getSpatialMatrix();
310                 switch (spatialMatrix) {
311                     case Binary:
312                         if (d == 1)
313                             w = 1;
314                         else
315                             w = 0;
316                         break; // binary
317                     case Euclidean:
318                         if (d == 1)
319                             w = 0.01;
320                         else
321                             w = 1 - 1 / d;
322                         break; // euclidean
323                     default:
324                         w = 0;
325                         break;
326                 }
327                 // if (ii>jj) m_distancesS.put((jj+"#"+ii), w); else m_distancesS.put((ii+"#"+jj), w);
328                 m_distancesS.put((ii + "#" + jj), w);
329                 m_distancesS.put((jj + "#" + ii), w);
330             }
331         }
332         br.close();
333     }

442     @Override
443     public double calcHeuristic(ClusStatistic c_tstat, ClusStatistic c_pstat, ClusStatistic missing) {
444         double value = 0.0;
445         if (c_tstat instanceof ClassificationStat) {
446             ClassificationStat tstat = (ClassificationStat) c_tstat;
447             ClassificationStat pstat = (ClassificationStat) c_pstat;
448             if (stopCriterion(c_tstat, c_pstat, missing))
449                 return Double.NEGATIVE_INFINITY;
450             double n_tot = tstat.getTotalWeight();// Equal for all target attributes
451             double n_pos = pstat.getTotalWeight();
452             double n_neg = n_tot - n_pos;
453             double tot_ent = tstat.entropy(); // Initialize entropy's
454             double pos_ent = pstat.entropy();
455             double neg_ent = tstat.entropyDifference(pstat);
456             value = tot_ent - (n_pos * pos_ent + n_neg * neg_ent) / n_tot; // gain
457             if (value < MathUtil.C1E_6)
458                 return Double.NEGATIVE_INFINITY;
459             if (m_GainRatio) {
460                 double si = ClassificationStat.computeSplitInfo(n_tot, n_pos, n_neg);
461                 if (si < MathUtil.C1E_6)
462                     return Double.NEGATIVE_INFINITY;
463                 value = value / si;
464             }
465         }
466         else {
467             if (c_tstat instanceof RegressionStat) {
468                 RegressionStat tstat = (RegressionStat) c_tstat;
469                 RegressionStat pstat = (RegressionStat) c_pstat;
470                 if (stopCriterion(c_tstat, c_pstat, missing))
471                     return Double.NEGATIVE_INFINITY;
472                 double ss_tot = tstat.getSVarS(m_ClusteringWeights);
473                 double ss_pos = pstat.getSVarS(m_ClusteringWeights);
474                 double ss_neg = tstat.getSVarSDiff(m_ClusteringWeights, pstat);
475                 value = FTest.calcVarianceReductionHeuristic(tstat.getTotalWeight(), ss_tot, ss_pos + ss_neg);
476             }
477             else {
478                 WHTDStatistic tstat = (WHTDStatistic) c_tstat;
479                 WHTDStatistic pstat = (WHTDStatistic) c_pstat;
480                 if (stopCriterion(c_tstat, c_pstat, missing))
481                     return Double.NEGATIVE_INFINITY;
482                 double ss_tot = tstat.getSVarS(m_ClusteringWeights);
483                 double ss_pos = pstat.getSVarS(m_ClusteringWeights);
484                 double ss_neg = tstat.getSVarSDiff(m_ClusteringWeights, pstat);
485                 value = FTest.calcVarianceReductionHeuristic(tstat.getTotalWeight(), ss_tot, ss_pos + ss_neg);
486                 if (getSettings().getGeneral().getVerbose() >= 10) {
487                     ClusLogger.info("TOT: " + tstat.getDebugString());
488                     ClusLogger.info("POS: " + pstat.getDebugString());
489                     ClusLogger.info("-> (" + ss_tot + ", " + ss_pos + ", " + ss_neg + ") " + value);
490                 }
491             }
492         }
493         return value;
494     }

497     @Override
498     public double calcHeuristic(ClusStatistic c_tstat, ClusStatistic[] c_pstat, int nbsplit) {
499         if (stopCriterion(c_tstat, c_pstat, nbsplit))
500             return Double.NEGATIVE_INFINITY;
501         double value = 0.0;
502         if (c_tstat instanceof ClassificationStat) {
503             ClassificationStat tstat = (ClassificationStat) c_tstat; // Total Entropy
504             double n_tot = tstat.getTotalWeight();
505             value = tstat.entropy();
506             for (int i = 0; i < nbsplit; i++) { // Subset entropy
507                 ClassificationStat pstat = (ClassificationStat) c_pstat[i];
508                 double n_set = pstat.getTotalWeight();
509                 value -= n_set / n_tot * pstat.entropy();
510             }
511             if (value < MathUtil.C1E_6)
512                 return Double.NEGATIVE_INFINITY;
513             if (m_GainRatio) {
514                 double si = 0;// Compute split information
515                 for (int i = 0; i < nbsplit; i++) {
516                     double n_set = c_pstat[i].getTotalWeight();
517                     if (n_set >= MathUtil.C1E_6) {
518                         double div = n_set / n_tot;
519                         si -= div * Math.log(div);
520                     }
521                 }
522                 si /= MathUtil.M_LN2;
523                 if (si < MathUtil.C1E_6)
524                     return Double.NEGATIVE_INFINITY; // Return calculated gainratio
525                 value = value / si;
526             }
527         }
528         else {
529             if (c_tstat instanceof RegressionStat) {
530                 double ss_sum = 0.0;
531                 RegressionStat tstat = (RegressionStat) c_tstat;
532                 RegressionStat[] pstat = (RegressionStat[]) c_pstat;
533                 for (int i = 0; i < nbsplit; i++)
534                     ss_sum += pstat[i].getSVarS(m_ClusteringWeights);
535                 double ss_tot = tstat.getSVarS(m_ClusteringWeights);
536                 value = FTest.calcVarianceReductionHeuristic(tstat.getTotalWeight(), ss_tot, ss_sum);
537             }
538             else {
539                // WHTDStatistic tstat = (WHTDStatistic) c_tstat;
540                 // WHTDStatistic[] pstat = (WHTDStatistic[]) c_pstat;
541                 // Compute |S|Var[S]
542                 // double ss_tot = tstat.getSVarS(m_ClusteringWeights);
543             }
544         }
545         return value;
546     }

549     public double LanLongdistance(double lat1, double lon1, double lat2, double lon2) {
550         double dist;
551         if (lat1 == lat2 && lon1 == lon2)
552             dist = 0;
553         else {
554             double theta = lon1 - lon2;
555             dist = Math.sin(deg2rad(lat1)) * Math.sin(deg2rad(lat2)) + Math.cos(deg2rad(lat1)) * Math.cos(deg2rad(lat2)) * Math.cos(deg2rad(theta));
556             dist = rad2deg(Math.acos(dist));
557             dist = dist * 60 * 1.1515 * 1.609344;
558         }
559         return (dist);
