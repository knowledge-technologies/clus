69      /**
70       * Initialize:
71       * -load settings
72       * -create schema
73       * -read in data
74       * 
75       * @throws IOException
76       * @throws ClusException
77       */
78      public void initialize() throws IOException, ClusException {
79          // Load settings file
80          ARFFFile arff = null;
81          ClusLogger.info("Loading '" + m_Sett.getGeneric().getAppName() + "'");
82          ClusRandom.initialize(m_Sett);
83          ClusReader reader = new ClusReader(m_Sett.getData().getDataFile(), m_Sett);
84          ClusLogger.info();
85          ClusLogger.info("Reading ARFF Header");
86          arff = new ARFFFile(reader);
87          m_Schema = arff.read(m_Sett);
88          // Count rows and move to data segment
89          ClusLogger.info("Reading CSV Data");
90          // Updata schema based on settings
91          m_Sett.updateTarget(m_Schema);
92          m_Schema.initializeSettings(m_Sett);
93          m_Sett.getAttribute().setTarget(m_Schema.getTarget().toString());
94          m_Sett.getAttribute().setDisabled(m_Schema.getDisabled().toString());
95          m_Sett.getAttribute().setClustering(m_Schema.getClustering().toString());
96          m_Sett.getAttribute().setDescriptive(m_Schema.getDescriptive().toString());
97          // Load data from file
98          if (ResourceInfo.isLibLoaded()) {
99              ClusStat.m_InitialMemory = ResourceInfo.getMemory();
100         }
101         ClusView view = m_Schema.createNormalView();
102         m_Data = view.readData(reader, m_Schema);
103         reader.close();
104         // Preprocess and initialize induce
105         m_Sett.update(m_Schema);
106         getSettings().getExperimental();
107         // Set XVal field in Settings
108         SettingsExperimental.IS_XVAL = true;
109         ClusLogger.info("Has missing values: " + m_Schema.hasMissing());
110     }

138     private void InitLearner() {
139 
140         if (this.m_Sett.getSIT().getLearnerName().equals("KNN")) {
141             ClusLogger.info("Using KNN Learner");
142             this.m_Learner = new KNNLearner();
143         }
144         else {
145             ClusLogger.info("Using Clus Learner");
146             this.m_Learner = new ClusLearner();
147 
148         }
149         // this.m_Learner = new AvgLearner();
150         this.m_Learner.init(this.m_Data, this.m_Sett);
151         int mt = new Integer(m_Sett.getSIT().getMainTarget()) - 1;
152         ClusAttrType mainTarget = m_Schema.getAttrType(mt);
153         this.m_Learner.setMainTarget(mainTarget);
154 
155     }

181     /**
182      * Initialize the SearchAlgorithm
183      */
184     private void InitSearchAlgorithm() {
185         String search = m_Sett.getSIT().getSearchName();
186         if (search.equals("OneTarget")) {
187             this.m_Search = new OneTarget();
188             ClusLogger.info("Search = single target");
189         }
190         else if (search.equals("AllTargets")) {
191             this.m_Search = new AllTargets();
192             ClusLogger.info("Search = full multi target");
193         }
194         else if (search.equals("GeneticSearch")) {
195             this.m_Search = new GeneticSearch();
196             ClusLogger.info("Search = Genetic search strategy");
197         }
198         else if (search.equals("SIT")) {
199             this.m_Search = new GreedySIT();
200             ClusLogger.info("Search = SIT, with stop criterion");
201         }
202         else if (search.equals("NoStop")) {
203             this.m_Search = new NoStopSearch();
204             ClusLogger.info("Search = SIT, no stop criterion");
205         }
206         else if (search.equals("TC")) {
207             this.m_Search = new TC();
208             ClusLogger.info("Search = TC");
209         }
210         else {
211             System.err.println("Search strategy unknown!");
212         }
213 
214         this.m_Search.setMTLearner(this.m_Learner);
215         this.m_Search.setSettings(this.m_Sett);
216 
217     }

220     /**
221      * Start the search for the optimal subset using the current learner and search algorithm
222      * 
223      * @return Targetset The found subset
224      * @throws ClusException 
225      */
226     public TargetSet search() throws ClusException {
227 
228         int mt = new Integer(m_Sett.getSIT().getMainTarget()) - 1;
229         ClusAttrType mainTarget = m_Schema.getAttrType(mt);
230         IntervalCollection candidates = new IntervalCollection(m_Sett.getAttribute().getTarget());
231         TargetSet candidateSet = new TargetSet(m_Schema, candidates);
232         return m_Search.search(mainTarget, candidateSet);
233 
234     }

266     public void singleRun() throws ClusException {
267         ClusLogger.info("Starting single run");
268         /* Init the Learner */
269         InitLearner();
270         /* Init the Search algorithm */
271         InitSearchAlgorithm();
272 
273         ErrorOutput errOut = new ErrorOutput(this.m_Sett);
274         try {
275             errOut.writeHeader();
276         }
277         catch (IOException e) {
278             
279             e.printStackTrace();
280         }
281 
282         TargetSet trgset = search();
283 
284         // compute the error of the final set
285         int mt = new Integer(m_Sett.getSIT().getMainTarget()) - 1;
286         ClusAttrType mainTarget = m_Schema.getAttrType(mt);
287         int errorIdx = mainTarget.getArrayIndex();
288         // predict a few folds
289         int nbFolds = 20;
290         this.m_Learner.initXVal(nbFolds);
291         // learn a model for each fold
292         ArrayList<RowData[]> folds = new ArrayList<RowData[]>();
293         for (int f = 0; f < nbFolds; f++) {
294             folds.add(m_Learner.LearnModel(trgset, f));
295         }
296 
297         //double finalerror = Evaluator.getPearsonCorrelation(folds, errorIdx);
298         Evaluator.getPearsonCorrelation(folds, errorIdx);
299 
300         // errOut.addFold(0,0,m_Learner.getName(),m_Search.getName(),Integer.toString(mt+1),finalerror,"["+trgset.toString()+"]");
301 
302     }

305     public void XValRun() throws Exception {
306         ErrorOutput errOut = new ErrorOutput(this.m_Sett);
307         errOut.writeHeader();
308         ClusLogger.info("Starting XVal run");
309 
310         XValRandomSelection m_XValSel = null;
311         int nrFolds = 26;
312         try {
313 
314             m_XValSel = new XValRandomSelection(m_Data.getNbRows(), nrFolds);
315         }
316         catch (ClusException e) {
317             e.printStackTrace();
318         }
319 
320         int mt = new Integer(m_Sett.getSIT().getMainTarget()) - 1;
321         ClusAttrType mainTarget = m_Schema.getAttrType(mt);
322         int errorIdx = mainTarget.getArrayIndex();
323 
324         for (int i = 0; i < nrFolds; i++) {
325             ClusLogger.info("Outer XVAL fold " + (i + 1));
326             XValSelection msel = new XValSelection(m_XValSel, i);
327             RowData train = (RowData) m_Data.cloneData();
328             RowData test = (RowData) train.select(msel);
329 
330             ClusLogger.info(test.getNbRows());
331 
332             /* Init the Learner */
333             InitLearner(train);
334             /* Init the Search algorithm */
335             InitSearchAlgorithm();
336 
337             Long d = (new Date()).getTime();
338             TargetSet searchResult = search();
339 
340             // find the error
341             m_Learner.setTestData(test);
342 
343             RowData[] predictions = m_Learner.LearnModel(searchResult);
344 
345             /*
346              * RowData t = predictions[0];
347              * DataTuple tt = t.getTuple(0);
348              * double dt = mainTarget.getNumeric(tt);
349              * RowData p = predictions[1];
350              * DataTuple tp = p.getTuple(0);
351              * double dp = mainTarget.getNumeric(tp);
352              */
353             Long new_d = (new Date()).getTime();
354             Long dif = new_d - d;
355 
356             double error = 0;
357             String errorName = m_Sett.getSIT().getError();
358             if (errorName.equals("MSE")) {
359                 error = Evaluator.getMSE(predictions, errorIdx);
360             }
361             else if (errorName.equals("MisclassificationError")) {
362                 error = Evaluator.getMisclassificationError(predictions, errorIdx);
363             }
364             else {
365 
366                 error = Evaluator.getPearsonCorrelation(predictions, errorIdx);
367             }
368 
369             // errOut.addFold(0,i,m_Learner.getName(),m_Search.getName(),Integer.toString(mt),error,"\""+searchResult.toString()+"
370             // \"",dt,dp);
371             errOut.addFold(0, i, m_Learner.getName(), m_Search.getName(), Integer.toString(mt + 1), error, "\"" + searchResult.toString() + " \"", dif);
372 
373         }
374

378     public void YATSXValRun() throws Exception {
379         ErrorOutput errOut = new ErrorOutput(this.m_Sett);
380         errOut.writeHeader();
381         ClusLogger.info("Starting XVal run");
382 
383         XValRandomSelection m_XValSel = null;
384         int nrFolds = 500;
385         try {
386 
387             m_XValSel = new XValRandomSelection(m_Data.getNbRows(), nrFolds);
388         }
389         catch (ClusException e) {
390             e.printStackTrace();
391         }
392 
393         int mt = new Integer(m_Sett.getSIT().getMainTarget()) - 1;
394         ClusAttrType mainTarget = m_Schema.getAttrType(mt);
395         int errorIdx = mainTarget.getArrayIndex();
396 
397         for (int i = 0; i < nrFolds; i++) {
398             ClusLogger.info("Outer XVAL fold " + (i + 1));
399             XValSelection msel = new XValSelection(m_XValSel, i);
400             RowData train = (RowData) m_Data.cloneData();
401             RowData test = (RowData) train.select(msel);
402 
403             // ClusLogger.info(test.getNbRows());
404 
405             /* Init the Learner */
406             InitLearner(train);
407             /* Init the Search algorithm */
408             InitSearchAlgorithm();
409 
410             Long d = (new Date()).getTime();
411             TargetSet searchResult = search();
412 
413             // find the error
414             m_Learner.setTestData(test);
415 
416             RowData[] predictions = m_Learner.LearnModel(searchResult);
417 
418             // add new data to training
419 
420             RowData pred = predictions[1];
421             RowData xtr_train = test.deepCloneData();
422             for (int t = 0; t < pred.getNbRows(); t++) {
423                 DataTuple tp = pred.getTuple(t);
424                 double dp = mainTarget.getNumeric(tp);
425                 DataTuple clone = xtr_train.getTuple(t);
426                 ((NumericAttrType) mainTarget).setNumeric(clone, dp);
427 
428             }
429 
430             RowData new_train = new RowData(train.getSchema(), train.getNbRows() + xtr_train.getNbRows());
431 
432             for (int j = 0; j < train.getNbRows(); j++) {
433                 new_train.setTuple(train.getTuple(j), j);
434             }
435             for (int j = train.getNbRows(); j < train.getNbRows() + xtr_train.getNbRows(); j++) {
436                 new_train.setTuple(xtr_train.getTuple(j - train.getNbRows()), j);
437             }
438 
439             InitLearner(xtr_train);
440             m_Learner.setTestData(test);
441 
442             // predictions = m_Learner.LearnModel(searchResult);
443 
444             Long new_d = (new Date()).getTime();
445             Long dif = new_d - d;
446 
447             double error = 0;
448             String errorName = m_Sett.getSIT().getError();
449 
450             if (errorName.equals("RME")) {
451                 error = Evaluator.getRelativeError(predictions, errorIdx);
452                 ClusLogger.info(error);
453             }
454             else if (errorName.equals("MSE")) {
455                 error = Evaluator.getMSE(predictions, errorIdx);
456             }
457             else if (errorName.equals("MisclassificationError")) {
458                 error = Evaluator.getMisclassificationError(predictions, errorIdx);
459             }
460             else {
461 
462                 error = Evaluator.getPearsonCorrelation(predictions, errorIdx);
463             }
464 
465             // errOut.addFold(0,i,m_Learner.getName(),m_Search.getName(),Integer.toString(mt),error,"\""+searchResult.toString()+"
466             // \"",dt,dp);
467             errOut.addFold(0, i, m_Learner.getName(), m_Search.getName(), Integer.toString(mt + 1), error, "\"" + searchResult.toString() + " \"", dif);
468 
469         }
470

478     /**
479      * @param args
480      * @throws Exception
481      */
482     public static void main(String[] args) throws Exception {
483         Sit sit = Sit.getInstance();
484         Settings sett = sit.getSettings();
485         CMDLineArgs cargs = new CMDLineArgs(sit);
486         cargs.process(args);
487         if (cargs.getNbMainArgs() == 0) {
488             sit.showHelp();
489             ClusLogger.info();
490             ClusLogger.info("Expected main argument");
491             System.exit(0);
492         }
493         if (cargs.allOK()) {
494             sett.getGeneric().setDate(new Date());
495             sett.getGeneric().setAppName(cargs.getMainArg(0));
496             sit.initSettings(cargs);
497 
498         }
499         else {
500             System.err.println("Arguments not ok?!");
501         }
502         sit.initialize();
503 
504         /* Search for the optimal subset */
505         sit.m_SearchSelection = 1;
506         sit.XValRun();
507 
508         // sit.YATSXValRun();
509 
510         ClusLogger.info("Finished");
511     }
