282 		// note: keep the order of the inicialization fixed, sometimes it is important
283 		printMessage("Preprocessing steps ...", 1, getSettings().getGeneral().getVerbose());
284 		m_NbExamples = m_Data.getNbRows();
285 		setReliefDescription(m_NbNeighbours, m_NbIterations);
286 
287 		initializeNeigbhourLoadSave();
288 		initializeNeighbourWeights();
289 		initializeInstanceOrder();
290 		initializeTimeSeriesDistance();
291 		initializeDescriptiveTargetAttributes();
292 		initializeNumericRanges();	
293 		// semi-supervised: this can modify the data, so keep it as high as possible
294 		prepareForSemiSupervised();
295 		
296 		
297 		initializeRankings();
298 		m_IsMLC = getSettings().getMLC().getSectionMultiLabel().isEnabled();
299 		initializeTargetProbabilities();
300 
301 		// depends on m_TargetProbabilities ...
302 		if (m_IsMLC) {
303 			m_MLCDistanceType = getSettings().getRelief().getMultilabelDistance();
304 			double[] labelProbabilities = new double[m_NbTargetAttrs];
305 			String labelPresent = "1";
306 			for (int i = 0; i < m_NbTargetAttrs; i++) {
307 				NominalAttrType attr = (NominalAttrType) m_DescriptiveTargetAttr[TARGET_SPACE][i];
308 				labelProbabilities[i] = m_TargetProbabilities[i + 1][attr.getValueIndex(labelPresent)];
309 			}
310 			m_MLCDist = new MultiLabelDistance(m_MLCDistanceType, m_DescriptiveTargetAttr[TARGET_SPACE],
311 			        labelProbabilities);
312 		}
313 
314 		// check for hierarchical attributes
315 		for (int space : SPACE_TYPES) {
316 			for (int attrInd = 0; attrInd < m_DescriptiveTargetAttr[space].length; attrInd++) {
317 				if (m_DescriptiveTargetAttr[space][attrInd].isClasses()) {
318 					ClassesAttrType attr = (ClassesAttrType) m_DescriptiveTargetAttr[space][attrInd];
319 					m_HierarMLCDist.processAttribute(attr, m_Data);
320 				}
321 			}
322 		}
323 
324 		// attribute relevance estimation: current statistics
325 		ReliefStatisticsType statType = getSettings().getRelief().getReliefStatisticsType();
326 		if (Arrays.stream(m_SSLWs).min().getAsDouble() < m_BASIC_SSLw && !statType.equals(ReliefStatisticsType.DistanceClassic)) {
327 			throw new RuntimeException("Using unsupervised data is supported only in DistanceClassic statisic type, but yours is " + statType);
328 		}
329 		switch (statType) {
330 		case DistanceClassic:
331 			mStats = new Classic(this, m_NbGeneralisedTargetAttrs, m_NbNeighbours.length, m_NbDescriptiveAttrs);
332 			break;
333 		case DistanceSimplified:
334 			mStats = new DistanceSimplified(this, m_NbGeneralisedTargetAttrs, m_NbNeighbours.length,
335 			        m_NbDescriptiveAttrs);
336 			break;
337 		case Steepness:
338 			mStats = new Steepness(this, m_NbGeneralisedTargetAttrs, m_NbNeighbours.length, m_NbDescriptiveAttrs,
339 			        m_MaxNbNeighbours, false, true); // TODO un-hardcode false
340 			break;
341 		case Variance:
342 			mStats = new Variance(this, m_NbGeneralisedTargetAttrs, m_NbNeighbours.length, m_NbDescriptiveAttrs);
343 			break;
344 		default:
345 			throw new RuntimeException("Wrong value for statistics type: " + statType);
346 		}
347 		// nearest neighbours
348 		m_NearestNeighbours = new HashMap<Integer, HashMap<Integer, NearestNeighbour[][]>>();
349 		// number of threads
350 		m_NbThreads = m_Data.getSchema().getSettings().getEnsemble().getNumberOfThreads();		
351 	}

394 			attrType = space == DESCRIPTIVE_SPACE ? AttributeUseType.Descriptive : AttributeUseType.Target;

397 		}

399 		m_NbTargetAttrs = m_DescriptiveTargetAttr[TARGET_SPACE].length;
400 		m_IsSparse = m_Data.isSparse();
401 		// efficient distance computation for sparse tuples
402 		int nbNonNumericDescriptive = m_NbDescriptiveAttrs; // if not sparse, the variables names are misleading:
403 		int nbNonNumericTarget = m_NbTargetAttrs; // these are simply the numbers of desc./targ. attributes ...
404 
405 		if (m_IsSparse) {
406 			nbNonNumericDescriptive -= m_Data.getSchema().getNbNumericAttrUse(AttributeUseType.Descriptive);
407 			nbNonNumericTarget -= m_Data.getSchema().getNbNumericAttrUse(AttributeUseType.Target);
408 		}
409 		m_DistanceDescriptiveTargetAttr[DESCRIPTIVE_SPACE] = new int[nbNonNumericDescriptive];
410 		m_DistanceDescriptiveTargetAttr[TARGET_SPACE] = new int[nbNonNumericTarget];
411 		for (int space : SPACE_TYPES) {
412 			int place = 0;
413 			for (int i = 0; i < m_DescriptiveTargetAttr[space].length; i++) {
414 				if (!m_IsSparse || !m_DescriptiveTargetAttr[space][i].isNumeric()) {
415 					m_DistanceDescriptiveTargetAttr[space][place] = i;
416 					place++;
417 				}
418 			}
419 			if (place != m_DistanceDescriptiveTargetAttr[space].length) {
420 				throw new RuntimeException("Something fishy with the number of nonumeric attributes.");
421 			}
422 		}

454 	private void initializeNumericRanges() {
455 		// compute min and max of numeric attributes
456 		m_numMins = new HashMap<String, Double>();
457 		m_numMaxs = new HashMap<String, Double>();
458 		double value;
459 		String attrName;
460 		for (int space : SPACE_TYPES) {
461 			AttributeUseType attrType = space == DESCRIPTIVE_SPACE ? AttributeUseType.Descriptive : AttributeUseType.Target;
462 			for (NumericAttrType numAttr : m_Data.m_Schema.getNumericAttrUse(attrType)) {
463 				attrName = numAttr.getName();
464 				m_numMins.put(attrName, Double.POSITIVE_INFINITY);
465 				m_numMaxs.put(attrName, Double.NEGATIVE_INFINITY);
466 				for (int example = 0; example < m_NbExamples; example++) {
467 					value = numAttr.getNumeric(m_Data.getTuple(example));
468 					if (value < m_numMins.get(attrName)) { // equivalent to ... && value != Double.POSITIVE_INFINITY
469 						m_numMins.put(attrName, value);
470 					}
471 					if (value > m_numMaxs.get(attrName) && value != Double.POSITIVE_INFINITY) {
472 						m_numMaxs.put(attrName, value);
473 					}
474 				}
475 			}
476 		}
477 	}

504 	private void computeAmountsOfSupervision() {
505 		// try load from file
506 		// TODO
507 		double[] distancesToNotUnlabeled = new double[m_NbExamples];
508 		// compute the nearest neighbours
509 		for (int i = 0; i < m_NbExamples; i++) {
510 			try {
511 				distancesToNotUnlabeled[i] = findDistanceToFirstNotUnlabeledNeighbour(i);
512 			} catch (ClusException e) {
513 				e.printStackTrace();
514 			}
515 		}
516 		double dMin = Arrays.stream(distancesToNotUnlabeled).min().getAsDouble();
517 		double dMax = Arrays.stream(distancesToNotUnlabeled).max().getAsDouble();
518 		double wMin = m_SSLW[0];
519 		double wMax = m_SSLW[1];
520 		// take care of some degenerate cases
521 		if (dMax == 0.0) {
522 			ClusLogger.info("The data seems to be supervised or all nearest neighbours are 0 away.");
523 		}
524 		else if ((dMax - dMin) / dMax < ClusUtil.MICRO) {
525 			Arrays.fill(m_SSLWs, (wMin + wMax) / 2.0); // something in between if everything is the same
526 		}
527 		else {
528 			// linear interpolation
529 			for (int i = 0; i < m_SSLWs.length; i++) {

731 	 * In each iteration, this method updates distance statistics after the
732 	 * neighbours of a chosen tuple are found.
733 	 * 
734 	 */
735 	private int updateDistanceStatistics(RowData data, DataTuple tuple, int tupleIndex, NearestNeighbour[][] nearestNeighbours,
736 	        int targetIndex) throws ClusException {
737 		int tempInsufficientNbNeighbours = 0;
738 		int nbTargetValues = m_NbTargetValues[targetIndex + 1];
739 		for (int targetValue = 0; targetValue < nbTargetValues; targetValue++) {
740 			// The sums sum_neigh w_neigh * d, where w is non-normalised weight and d is
741 			// d_class * d_attr or d_class
742 			// etc.
743 			mStats.resetTempFields();
744 			double sumNeighbourWeights = 0.0;
745 
746 			boolean isStdClassification = m_IsStandardClassification[targetIndex + 1];
747 			int trueIndex = getTrueTargetIndex(targetIndex);

751 				if (nearestNeighbours[targetValue].length < m_MaxNbNeighbours) {
752 					tempInsufficientNbNeighbours++;
753 				}
754 				sumNeighbourWeights += m_NeighbourWeights[neighbour];
755 				double neighWeightNonnormalized = m_NeighbourWeights[neighbour];				
756 				NearestNeighbour neigh = nearestNeighbours[targetValue][neighbour];
757 				// take care of semi-supervised: 
758 				// neighWeightNonnormalized will actually be the weight of the pair (tuple, neigh).

762 
763 				if (neighbour + 1 == m_NbNeighbours[numNeighInd]) {
764 					mStats.updateStatistics(targetIndex, numNeighInd, sumNeighbourWeights);
765 					numNeighInd++;
766 				}
767 			}
768 		}
769 		return tempInsufficientNbNeighbours;
770 	}

807 	 *            if -1, this is the overall ranking, hence STC is the task at hand.
808 	 *            Otherwise, we are computing statistics that are used in per-target
809 	 *            ranking.
810 	 * @return A list with number of examples with a particular value of the target.
811 	 *         In the last place, we have the number of examples whose value of the
812 	 *         target is missing. The counts for non-missing values are normalised

822 		}
823 		if (m_NbExamples > MathUtil.C1E_9 + targetProbabilities[nbValues]) { // otherwise: targetProbabilities = {0, 0,

826 			// ignored
827 			// The formula for standard classification class weighting still holds, i.e.,
828 			// sum over other classes of
829 			// the terms P(other class) / (1 - P(class)), equals 1
830 			for (int value = 0; value < nbValues; value++) {

832 			}
833 		}

1181 	 * Calculates distance between the numeric values of the component {@code attr}.
1182 	 * In the case of missing values, we follow Weka's solution and not the paper
1183 	 * Theoretical and Empirical Analysis of ReliefF and RReliefF, by Robnik Sikonja
1184 	 * and Kononenko (time complexity ...).
1185 	 * 
1186 	 * @param t1
1187 	 *            The first tuple
1188 	 * @param t2
1189 	 *            The second tuple
1190 	 * @param attr
1191 	 *            The numeric attribute/dimension in which the distance between
1192 	 *            {@code t1} and {@code t2} is computed.
1193 	 * @param normalizationFactor
1194 	 *            Typically,
1195 	 *            <p>
1196 	 *            {@code normalizationFactor = 1 / (m_numMaxs[attr.name()] - m_numMins[attr.name()])}.
1197 	 * @return If {@code v1} and {@code v2} are the numeric values of the attribute
1198 	 *         {@code attr} for the instances {@code t1} and {@code t2}, the value
1199 	 *         <p>
1200 	 *         {@code |v1 - v2| / normalizationFactor}
1201 	 *         <p>
1202 	 *         is returned.
1203 	 */
1204 	public double computeNumericDist1D(DataTuple t1, DataTuple t2, NumericAttrType attr, double normalizationFactor) {
1205 		double v1 = attr.getNumeric(t1);
1206 		double v2 = attr.getNumeric(t2);
1207 		double t;
1208 		if (isMissing(v1)) { // t1.hasNumMissing(attr.getArrayIndex())
1209 			if (isMissing(v2)) { // t2.hasNumMissing(attr.getArrayIndex())
1210 				t = BOTH_MISSING_DIST;
1211 			} else {
1212 				t = (v2 - m_numMins.get(attr.getName())) / normalizationFactor;
1213 				t = Math.max(t, 1.0 - t);
1214 			}
1215 		} else {
1216 			if (isMissing(v2)) {
1217 				t = (v1 - m_numMins.get(attr.getName())) / normalizationFactor;
1218 				t = Math.max(t, 1.0 - t);
1219 			} else {
1220 				t = Math.abs(v1 - v2) / normalizationFactor;
1221 			}
1222 		}
1223 		return t;
1224 	}

1293 	 * Computes a random permutation with Fisher-Yates algorithm.
1294 	 * 
1295 	 * @param examples
1296 	 *            The number of examples that we will place in a random order. *
1297 	 * @return A permutation given as a list whose i-th element is the index of the
1298 	 *         example that is processed in i-th iteration.
1299 	 */
1300 	private int[] randomPermutation(int examples) {
1301 		int[] permuted = new int[examples];
1302 		// fill
1303 		for (int i = 0; i < permuted.length; i++) {
1304 			permuted[i] = i;
1305 		}
1306 		// shuffle
1307 		for (int i = permuted.length - 1; i > 0; i--) {
1308 			int ind = m_rnd.nextInt(i + 1);
1309 			int temp = permuted[ind];
1310 			permuted[ind] = permuted[i];
1311 			permuted[i] = temp;
1312 		}
1313 		return permuted;
1314 	}
