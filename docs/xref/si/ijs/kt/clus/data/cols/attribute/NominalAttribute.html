55      @Override
56      public ClusAttribute select(ClusSelection sel, int nbsel) {
57          int s_data = 0;
58          int s_subset = 0;
59          int[] data = m_Data;
60          m_Data = new int[m_NbRows - nbsel];
61          int[] subset = new int[nbsel];
62          for (int i = 0; i < m_NbRows; i++) {
63              if (sel.isSelected(i))
64                  subset[s_subset++] = data[i];
65              else
66                  m_Data[s_data++] = data[i];
67          }
68          m_NbRows -= nbsel;
69          NominalAttribute s_attr = new NominalAttribute(m_Type);
70          s_attr.m_Data = subset;
71          s_attr.m_NbRows = nbsel;
72          return s_attr;
73      }

77      public void insert(ClusAttribute attr, ClusSelection sel, int nb_new) {
78          int s_data = 0;
79          int s_subset = 0;
80          int[] data = m_Data;
81          m_Data = new int[nb_new];
82          int[] subset = ((NominalAttribute) attr).m_Data;
83          for (int i = 0; i < nb_new; i++) {
84              if (sel.isSelected(i))
85                  m_Data[i] = subset[s_subset++];
86              else
87                  m_Data[i] = data[s_data++];
88          }
89          m_NbRows = nb_new;
90      }

154     public boolean read(ClusReader data, int row) throws IOException {
155         String value = data.readString();
156         if (value == null)
157             return false;
158         if (value.equals("?")) {
159             m_Type.incNbMissing();
160             m_Data[row] = m_Type.getNbValues();
161         }
162         else {
163             Integer i = m_Type.getValueIndex(value);
164             if (i != null) {
165                 m_Data[row] = i.intValue();
166             }
167             else {
168                 throw new IOException("Illegal value '" + value + "' for attribute " + getName() + " at row " + (row + 1));
169             }
170         }
171         return true;
172     }
