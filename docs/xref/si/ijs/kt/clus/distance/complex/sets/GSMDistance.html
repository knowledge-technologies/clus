14  public class GSMDistance extends SetDistance {
15  
16      private static final long serialVersionUID = Settings.SERIAL_VERSION_ID;
17  
18      private int maximal_possible_distance = 1;
19  
20  
21      /*
22       * public GSMDistance(SetAttrType attr, ClusDistance innerDistance) {
23       * super(attr,innerDistance);
24       * }
25       */
26  
27      public GSMDistance(SetAttrType attr, ClusDistance innerDistance, int maximal_possible_distance) {
28          super(attr, innerDistance);
29      }
30  
31  
32      @Override
33      public double calcDistance(Set set1, Set set2) throws ClusException {
34          ClusDistance clusDistance = m_ChildDistances[0];
35  
36          double distance = 0;
37          int m = set1.size();
38          int n = set2.size();
39          int maxCostFlow = Math.abs(m - n);
40          int flow = Math.min(m, n);
41  
42          double[][] distances = new double[m + 1][n + 1];
43          int i = 0, j = 0;
44  
45          if (flow > 0)
46              for (Object element1 : set1.getValues()) {
47                  j = 0;
48                  for (Object element2 : set2.getValues()) {
49                      if (clusDistance == null) {
50                          distances[i][j] = Math.abs((Double) element1 - (Double) element2);
51                      }
52                      else {
53                          distances[i][j] = clusDistance.calcDistance(element1, element2);
54                      }
55                      j++;
56                  }
57                  i++;
58              }
59  
60          int mini = -1;
61          int minj = -1;
62          while (flow > 0) {
63              double minCost = Double.MAX_VALUE;
64              for (i = 0; i < m; i++) {
65                  for (j = 0; j < n; j++) {
66                      if (distances[i][j] < minCost) {
67                          minCost = distances[i][j];
68                          mini = i;
69                          minj = j;
70                      }
71                  }
72              }
73              flow--;
74              distance += minCost;
75              //there is no more flow capacity there so put cost to max
76              Arrays.fill(distances[mini], Double.MAX_VALUE);
77              for (i = 0; i < m; i++)
78                  distances[i][minj] = Double.MAX_VALUE;
79          }
80  
81          //it does matter if the sets are not same cardinality
82          //we will put this as separate distance measure, and we will learn 
83          //when we do metric learning
84          distance += (maxCostFlow * maximal_possible_distance / 2);
85  
86          return distance;
87      }
88  
89  
90      @Override
91      public String getDistanceName() {
92          return "Greedy matching (sets)";
93      }
94  }
