72          super(par, set);

87              m_nbEx[i] = 0;

98      @Override
99      public double getModelErrorComponent(int i) {
100 
101         //int nb = getNbExamples();
102         int nb = m_nbEx[i];
103 
104         //ClusLogger.info(m_SumErr[i]);
105         double err = nb != 0.0 ? m_SumErr[i] / nb : 0.0;
106         if (m_Weights != null)
107             err *= m_Weights.getWeight(getAttr(i));
108 
109         return err;
110     }

113     @Override
114     public double getModelError() {
115         double ss_tree = 0.0;
116         int nb = 0;
117         for (int i = 0; i < m_Dim; i++) {
118             nb += m_nbEx[i];
119         }
120         if (m_Weights != null) {
121             for (int i = 0; i < m_Dim; i++) {
122                 ss_tree += m_SumErr[i] * m_Weights.getWeight(getAttr(i));
123             }
124             return nb != 0.0 ? ss_tree / nb : 0.0;
125         }
126         else {
127             for (int i = 0; i < m_Dim; i++) {
128                 ss_tree += m_SumErr[i];
129             }
130             return nb != 0.0 ? ss_tree / nb : 0.0;
131         }
132     }

135     @Override
136     public double getModelErrorStandardError() {
137         double sum_err = 0.0;
138         double sum_sq_err = 0.0;
139         for (int i = 0; i < m_Dim; i++) {
140             if (m_Weights != null) {
141                 sum_err += m_SumErr[i];
142                 sum_sq_err += m_SumSqErr[i];
143             }
144             else {
145                 sum_err += m_SumErr[i] * m_Weights.getWeight(getAttr(i));
146                 sum_sq_err += m_SumSqErr[i] * sqr(m_Weights.getWeight(getAttr(i)));
147             }
148         }
149         //double n = getNbExamples() * m_Dim;
150         double n = 0;
151         for (int i = 0; i < m_Dim; i++) {
152             n += m_nbEx[i];
153         }
154 
155         if (n <= 1) {
156             return Double.POSITIVE_INFINITY;
157         }
158         else {
159             double ss_x = (n * sum_sq_err - sqr(sum_err)) / (n * (n - 1));
160             return Math.sqrt(ss_x / n);
161         }
162     }

164 
165     public final static double sqr(double value) {
166         return value * value;
167     }
168

170     @Override
171     public void addExample(double[] real, double[] predicted) {
172         for (int i = 0; i < m_Dim; i++) {
173             double err = sqr(real[i] - predicted[i]);
174             ClusLogger.info(err);
175             if (!Double.isInfinite(err) && !Double.isNaN(err)) {
176                 m_SumErr[i] += err;
177                 m_SumSqErr[i] += sqr(err);
178                 m_nbEx[i]++;
179             }
180         }
181     }
182 
183 
184     private void addExample(Set realSet, Set predicted) throws ClusException {
185 
186         double error = (SetOperations.getIntersectionCardinality(realSet, predicted, innerDistance) + 0.0) / realSet.size();
187         if (realSet.size() == 0) {
188             if (predicted.size() == 0) {
189                 error = 1;
190             }
191             else {
192                 error = 0;
193             }
194         }
195         else {
196             if (predicted.size() == 0) {
197                 error = 0;
198             }
199         }
200 
201         m_SumErr[0] += error;
202         m_nbEx[0]++;
203     }

209         Set realSet = getAttr(0).getSet(tuple);
210         addExample(realSet, predicted);

227     @Override
228     public void add(ClusError other) {
229         Precision oe = (Precision) other;
230         for (int i = 0; i < m_Dim; i++) {
231             m_SumErr[i] += oe.m_SumErr[i];
232             m_SumSqErr[i] += oe.m_SumSqErr[i];
233             m_nbEx[i] += oe.m_nbEx[i];
234         }
235     }

238     @Override
239     public void showModelError(PrintWriter out, int detail) {
240     	ClusNumberFormat fr = getFormat();
241         StringBuffer buf = new StringBuffer();
242         if (m_PrintAllComps) {
243             buf.append("[");
244             for (int i = 0; i < m_Dim; i++) {
245                 if (i != 0)
246                     buf.append(",");
247                 buf.append(fr.format(getModelErrorComponent(i)));
248             }
249             if (m_Dim > 1)
250                 buf.append("]: ");
251             else
252                 buf.append("]");
253         }
254         if (m_Dim > 1 || !m_PrintAllComps) {
255             buf.append(fr.format(getModelError()));
256         }
257         out.println(buf.toString());
258     }
