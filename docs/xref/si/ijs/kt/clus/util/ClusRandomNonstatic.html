4   import java.util.Random;
5   
6   
7   public class ClusRandomNonstatic {
8   
9       public final static int RANDOM_EXTRATREE = 0;
10      public final static int RANDOM_SAMPLE = 1;
11      public final static int RANDOM_INT_RANFOR_TREE_DEPTH = 2;
12      public final static int RANDOM_SELECTION = 3;
13      public final static int RANDOM_SEED = 4;
14  
15      private Random[] m_Random;
16      private int m_Length = 5;
17      private int m_Seed;
18  
19      
20      public ClusRandomNonstatic(int seed) {
21          m_Seed = seed; // remember the initial seed
22          m_Random = new Random[m_Length];
23          Random seedGenerator = new Random(seed);
24          for (int i = 0; i < m_Length; i++) {
25              m_Random[i] = new Random(seedGenerator.nextInt());
26          }
27      }
28  
29  
30      public int nextInt(int which, int max) {
31          return m_Random[which].nextInt(max);
32      }
33  
34  
35      public int nextInt(int which) {
36          return m_Random[which].nextInt();
37      }
38  
39  
40      public double nextDouble(int which) {
41          return m_Random[which].nextDouble();
42      }
43  
44  
45      public Random getRandom(int idx) {
46          return m_Random[idx];
47      }
48      
49      public int getSeed(){
50      	return m_Seed;
51      }
52  
53  }
