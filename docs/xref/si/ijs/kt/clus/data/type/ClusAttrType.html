453     public static Object createDataObject(String object, ClusAttrType attrType) {
454 
455         if (attrType instanceof TupleAttrType) {
456 
457             object = object.substring(1);
458 
459             object = object.substring(0, object.length() - 1);
460             char[] td = object.toCharArray();
461             int cnt = 0;
462             StringBuilder innerObject = new StringBuilder();
463             ArrayList<String> innerObjects = new ArrayList<String>();
464             for (int i = 0; i < td.length; i++) {
465                 if (td[i] == '[') {
466                     cnt++;
467                     innerObject.append(td[i]);
468                 }
469                 else if (td[i] == ']') {
470                     cnt--;
471                     innerObject.append(td[i]);
472                 }
473                 else if (td[i] == ',') {
474                     if (cnt == 0) {
475                         innerObjects.add(innerObject.toString());
476                         innerObject = new StringBuilder();
477                     }
478                     else {
479                         innerObject.append(td[i]);
480                     }
481                 }
482                 else if (td[i] == ' ') {
483 
484                 }
485                 else {
486                     innerObject.append(td[i]);
487                 }
488 
489             }
490             innerObjects.add(innerObject.toString());
491             Object[] innerTupleObjects = new Object[innerObjects.size()];
492             ClusAttrType[] innerTypes = attrType.getInnerTypes();
493             int i = 0;
494             for (String iType : innerObjects) {
495                 innerTupleObjects[i] = createDataObject(iType, innerTypes[i]);
496                 i++;
497             }
498             return new Tuple(innerTupleObjects);
499 
500         }
501         else if (attrType instanceof TimeSeriesAttrType) {
502             if (object.trim().equalsIgnoreCase("?"))
503                 return null;
504             else
505                 return new TimeSeries(object);
506         }
507         else if (attrType instanceof NumericAttrType) { return new Double(object); }
508         return null;
509     }

512     public static ClusAttrType createClusAttrType(String typeDef, String name) {
513 
514         String typeDefinition = typeDef.toUpperCase();
515         typeDefinition = typeDefinition.trim();
516 
517         if (typeDefinition.startsWith("TUPLE[")) {
518 
519             typeDefinition = typeDefinition.substring(6);
520 
521             typeDefinition = typeDefinition.substring(0, typeDefinition.length() - 1);
522             char[] td = typeDefinition.toCharArray();
523             int cnt = 0;
524             StringBuffer innerType = new StringBuffer();
525             ArrayList<String> innerTypes = new ArrayList<String>();
526             for (int i = 0; i < td.length; i++) {
527                 if (td[i] == '[') {
528                     cnt++;
529                     innerType.append(td[i]);
530                 }
531                 else if (td[i] == ']') {
532                     cnt--;
533                     innerType.append(td[i]);
534                 }
535                 else if (td[i] == ',') {
536                     if (cnt == 0) {
537                         innerTypes.add(innerType.toString());
538                         innerType = new StringBuffer();
539                     }
540                     else {
541                         innerType.append(td[i]);
542                     }
543                 }
544                 else if (td[i] == ' ') {
545 
546                 }
547                 else {
548                     innerType.append(td[i]);
549                 }
550             }
551 
552             innerTypes.add(innerType.toString());
553             ClusAttrType[] innerClussAttrTypes = new ClusAttrType[innerTypes.size()];
554             int i = 0;
555             for (String iType : innerTypes) {
556                 innerClussAttrTypes[i++] = createClusAttrType(iType, "inner");
557             }
558             return new TupleAttrType(name, typeDef, innerClussAttrTypes);
559 
560         }
561         else if (typeDefinition.startsWith("TIMESERIES")) {
562             return new TimeSeriesAttrType(name, "TIMESERIES");
563         }
564         else if (typeDefinition.equalsIgnoreCase("NUMERIC")) { return new NumericAttrType(name); }
565 
566         return null;
567     }

570     public static ClusDistance createDistance(ClusAttrType type, Settings settings) {
571         if (type instanceof TupleAttrType) {
572             TupleAttrType tupleType = (TupleAttrType) type;
573             ClusAttrType[] innertypes = tupleType.getInnerTypes();
574 
575             ClusDistance[] innerDistances = new ClusDistance[innertypes.length];
576             int i = 0;
577             for (ClusAttrType iType : innertypes) {
578                 innerDistances[i++] = createDistance(iType, settings);
579             }
580 
581             TupleDistance tupleDist = null;
582             switch (settings.getTree().getTupleDistance()) {
583                 case Euclidean:
584                     tupleDist = new si.ijs.kt.clus.distance.complex.tuples.EuclideanDistance((TupleAttrType) type, innerDistances);
585                     break;
586                 case Minkowski:
587                     tupleDist = new si.ijs.kt.clus.distance.complex.tuples.MinkowskiDistance(3, innerDistances);
588                     break;
589             }
590             return tupleDist;
591         }
592         else if (type instanceof TimeSeriesAttrType) {
593 
594             TimeSeriesDist tsDistance = null;
595             switch (settings.getTree().getTSDistance()) {
596                 case DTW:
597                     tsDistance = new si.ijs.kt.clus.distance.primitive.timeseries.DTWTimeSeriesDist((TimeSeriesAttrType) type);
598                     break;
599 
600                 case QDM:
601                     tsDistance = new si.ijs.kt.clus.distance.primitive.timeseries.QDMTimeSeriesDist((TimeSeriesAttrType) type);
602                     break;
603 
604                 case TSC:
605                     tsDistance = new si.ijs.kt.clus.distance.primitive.timeseries.TSCTimeSeriesDist((TimeSeriesAttrType) type);
606                     break;
607             }
608             return tsDistance;
609         }
610         return null;
611     }

634     public JsonObject getAttributeJSON() {
635         JsonObject elm = new JsonObject();
636 
637         elm.addProperty("attributeName", getName());
638 
639         if (this instanceof NumericAttrType) {
640             elm.addProperty("attributeType", "numeric");
641         }
642         else if (this instanceof NominalAttrType) {
643             elm.addProperty("attributeType", "nominal");
644         }
645         else if (this instanceof TimeSeriesAttrType) {
646             elm.addProperty("attributeType", "timeseries");
647         }
648         else if (this instanceof TupleAttrType) {
649             elm.addProperty("attributeType", "tuple");
650         }
651         else if (this instanceof SetAttrType) {
652             elm.addProperty("attributeType", "set");
653         }
654         return elm;
655     }
