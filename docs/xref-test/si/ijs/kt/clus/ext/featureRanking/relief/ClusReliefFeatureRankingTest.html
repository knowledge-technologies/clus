13  	public static Pair<ArrayList<Triple<String, String, String>>, String[]> generateNames() throws IOException {
14  		String format = ClusFeatureRankingTest.getSubfolder() + "/test%s.%s";
15  		String[] tasks = new String[] { "Mixed", "MLCHammingLoss", "MTR", "TreeHMLC", "DagHMLC-RealWorld", "TreeHMLC-RealWorld", "MLC-RealWorld", "MTR-RealWorld", "STC-RealWorld" };
16  		ArrayList<Triple<String, String, String>> settingsFimps = new ArrayList<>();
17  		for (String task : tasks) {
18  			String sFile = String.format(format, task, "s");
19  			String fimpTruth = String.format(format, task, "fimpTruth");
20  			String fimpNew = sFile.replace(".s", ".fimp");
21  			settingsFimps.add(new Triple<String, String, String>(sFile, fimpTruth, fimpNew));
22  		}
23  		String[] args = new String[] { "-silent", "-relief", "" };
24  		return new Pair<ArrayList<Triple<String, String, String>>, String[]>(settingsFimps, args);
25  	}
