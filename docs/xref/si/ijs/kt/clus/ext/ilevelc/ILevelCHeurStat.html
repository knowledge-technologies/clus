23  package si.ijs.kt.clus.ext.ilevelc;
24  
25  import java.util.ArrayList;
26  
27  import si.ijs.kt.clus.data.rows.DataTuple;
28  import si.ijs.kt.clus.main.settings.Settings;
29  
30  
31  public class ILevelCHeurStat extends ILevelCStatistic {
32  
33      public final static long serialVersionUID = Settings.SERIAL_VERSION_ID;
34  
35      public static final int EXT = 0;
36      public static final int POS = 1;
37      public static final int NEG = 2;
38  
39      protected int m_NbClass;
40      protected ArrayList m_Constraints;
41      protected int[][] m_ConstraintIndex;
42      protected int[] m_IE;
43      protected int[] m_Clusters;
44      protected int[] m_CL;
45      protected int[] m_ML;
46  
47  
48      public ILevelCHeurStat(Settings sett, ILevelCStatistic stat, int nbclass) {
49          super(sett, stat.m_Numeric);
50          m_NbClass = nbclass;
51          m_CL = new int[nbclass];
52          m_ML = new int[nbclass];
53      }
54  
55  
56      @Override
57      public void updateWeighted(DataTuple tuple, double weight) {
58          super.updateWeighted(tuple, weight);
59          int tidx = tuple.getIndex();
60          int[] cidx = m_ConstraintIndex[tidx];
61          if (cidx == null)
62              return;
63          for (int i = 0; i < cidx.length; i++) {
64              ILevelConstraint cons = (ILevelConstraint) m_Constraints.get(cidx[i]);
65              int otidx = cons.getOtherTupleIdx(tuple);
66              if (m_IE[otidx] == EXT) {
67                  int oclass = m_Clusters[otidx];
68                  if (cons.getType() == ILevelConstraint.ILevelCMustLink)
69                      m_ML[oclass]++;
70                  else
71                      m_CL[oclass]++;
72              }
73          }
74      }
75  
76  
77      public void removeWeighted(DataTuple tuple, double weight) {
78          super.updateWeighted(tuple, -1.0 * weight);
79          int tidx = tuple.getIndex();
80          int[] cidx = m_ConstraintIndex[tidx];
81          if (cidx == null)
82              return;
83          for (int i = 0; i < cidx.length; i++) {
84              ILevelConstraint cons = (ILevelConstraint) m_Constraints.get(cidx[i]);
85              int otidx = cons.getOtherTupleIdx(tuple);
86              if (m_IE[otidx] == EXT) {
87                  int oclass = m_Clusters[otidx];
88                  if (cons.getType() == ILevelConstraint.ILevelCMustLink)
89                      m_ML[oclass]--;
90                  else
91                      m_CL[oclass]--;
92              }
93          }
94      }
95  
96  
97      public void setIndices(int[][] considx, ArrayList constr, int[] ie, int[] clusters) {
98          m_IE = ie;
99          m_Constraints = constr;
100         m_ConstraintIndex = considx;
101         m_Clusters = clusters;
102     }
103 
104 
105     public int computeMinimumExtViolated(int ig1, int ig2, boolean allownew) {
106         int totalml = 0;
107         for (int i = 0; i < m_NbClass; i++) {
108             totalml += m_ML[i];
109         }
110         int best = -2;
111         int bestviolated = Integer.MAX_VALUE;
112         for (int i = 0; i < m_NbClass; i++) {
113             if (i != ig1 && i != ig2 && m_ML[i] > 0) {
114                 int violated = m_CL[i] + (totalml - m_ML[i]);
115                 if (violated < bestviolated) {
116                     bestviolated = violated;
117                     best = i;
118                 }
119             }
120         }
121         if (allownew && totalml < bestviolated) {
122             bestviolated = totalml;
123             best = -1;
124         }
125         setClusterID(best);
126         return best == -2 ? -1 : bestviolated;
127     }
128 }
