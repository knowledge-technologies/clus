28      public static void quickSort(MSortable x, int off, int len) {
29          // Insertion sort on smallest arrays
30          if (len < 7) {
31              for (int i = off; i < len + off; i++)
32                  for (int j = i; j > off && x.getDouble(j - 1) <= x.getDouble(j); j--)
33                      x.swap(j, j - 1);
34              return;
35          }
36  
37          // Choose a partition element, v
38          int m = off + len / 2; // Small arrays, middle element
39          if (len > 7) {
40              int l = off;
41              int n = off + len - 1;
42              if (len > 40) { // Big arrays, pseudomedian of 9
43                  int s = len / 8;
44                  l = med3(x, l, l + s, l + 2 * s);
45                  m = med3(x, m - s, m, m + s);
46                  n = med3(x, n - 2 * s, n - s, n);
47              }
48              m = med3(x, l, m, n); // Mid-size, med of 3
49          }
50          double v = x.getDouble(m);
51  
52          // Establish Invariant: v* (<v)* (>v)* v*
53          int a = off, b = a, c = off + len - 1, d = c;
54          while (true) {
55              while (b <= c && x.getDouble(b) > v) {
56                  if (x.getDouble(b) == v)
57                      x.swap(a++, b);
58                  b++;
59              }
60              while (c >= b && x.getDouble(c) < v) {
61                  if (x.getDouble(c) == v)
62                      x.swap(c, d--);
63                  c--;
64              }
65              if (b > c)
66                  break;
67              x.swap(b++, c--);
68          }
69  
70          // Swap partition elements back to middle
71          int s, n = off + len;
72          s = Math.min(a - off, b - a);
73          vecswap(x, off, b - s, s);
74          s = Math.min(d - c, n - d - 1);
75          vecswap(x, b, n - s, s);
76  
77          // Recursively sort non-partition-elements
78          if ((s = b - a) > 1)
79              quickSort(x, off, s);
80          if ((s = d - c) > 1)
81              quickSort(x, n - s, s);
82      }
83  
84  
85      /**
86       * Swaps x[a .. (a+n-1)] with x[b .. (b+n-1)].
87       */
88      private static void vecswap(MSortable x, int a, int b, int n) {
89          for (int i = 0; i < n; i++, a++, b++)
90              x.swap(a, b);
91      }
92  
93  
94      /**
95       * Returns the index of the median of the three indexed doubles.
96       */
97      private static int med3(MSortable x, int a, int b, int c) {
98          double da = x.getDouble(a);
99          double db = x.getDouble(b);
100         double dc = x.getDouble(c);
101         return (da >= db ? (db >= dc ? b : da >= dc ? c : a) : (db <= dc ? b : da <= dc ? c : a));
102     }
103 }
