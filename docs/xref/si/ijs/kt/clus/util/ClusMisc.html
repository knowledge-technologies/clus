13      private static String getClusVersion() {
14          MavenXpp3Reader reader = new MavenXpp3Reader();
15          Model model = null;
16  
17          try {
18              try (InputStreamReader isr = new InputStreamReader(si.ijs.kt.clus.Clus.class.getResourceAsStream("/pom.xml"))) {
19                  model = reader.read(isr);
20              }
21  
22              return model.getVersion();
23          }
24          catch (Exception ex) {
25              return "DEVELOPMENT";
26          }
27      }
